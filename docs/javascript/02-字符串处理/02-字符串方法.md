# 字符串方法

## 查找方法

### indexOf() / lastIndexOf()
```javascript
const str = 'Hello World Hello';

// indexOf() - 从前往后找
console.log(str.indexOf('o'));      // 4
console.log(str.indexOf('o', 5));   // 7（从位置5开始）
console.log(str.indexOf('x'));      // -1（未找到）

// lastIndexOf() - 从后往前找
console.log(str.lastIndexOf('o'));     // 13
console.log(str.lastIndexOf('Hello')); // 12
```

### includes()（ES6）
```javascript
const str = 'Hello World';
console.log(str.includes('World'));    // true
console.log(str.includes('world'));    // false（区分大小写）
console.log(str.includes('World', 6)); // false（从位置6开始）
```

### startsWith() / endsWith()（ES6）
```javascript
const url = 'https://example.com';

console.log(url.startsWith('https://')); // true
console.log(url.startsWith('http', 0));  // true

const filename = 'document.pdf';
console.log(filename.endsWith('.pdf'));  // true
console.log(filename.endsWith('.txt'));  // false
```

### charAt() / charCodeAt() / codePointAt()
```javascript
const str = 'Hello';
console.log(str.charAt(0));       // 'H'
console.log(str.charAt(100));     // ''（空字符串）
console.log(str.charCodeAt(0));   // 72（ASCII 码）
console.log(str.codePointAt(0));  // 72

// 处理 Unicode 字符
const emoji = '😀';
console.log(emoji.charCodeAt(0));  // 55357（代理对前半部分）
console.log(emoji.codePointAt(0)); // 128512（完整的 code point）
```

### at()（ES2022）
```javascript
const str = 'Hello';
console.log(str.at(0));   // 'H'
console.log(str.at(-1));  // 'o'（支持负索引）
console.log(str.at(100)); // undefined
```

## 截取方法

### slice()
```javascript
const str = 'Hello World';

console.log(str.slice(0, 5));   // 'Hello'
console.log(str.slice(6));      // 'World'
console.log(str.slice(-5));     // 'World'（从后往前）
console.log(str.slice(-5, -1)); // 'Worl'
console.log(str.slice(6, 3));   // ''（结束小于开始）
```

### substring()
```javascript
const str = 'Hello World';

console.log(str.substring(0, 5)); // 'Hello'
console.log(str.substring(6));    // 'World'

// 与 slice 的区别：
console.log(str.substring(6, 3)); // 'lo '（自动交换参数）
console.log(str.slice(6, 3));     // ''

// substring 不支持负数
console.log(str.substring(-5));   // 'Hello World'（负数视为0）
console.log(str.slice(-5));       // 'World'
```

### substr()（已废弃）
```javascript
const str = 'Hello World';
console.log(str.substr(6, 5)); // 'World'（从位置6开始取5个字符）
// 不推荐使用，请用 slice 或 substring 代替
```

## 修改方法

### concat()
```javascript
const str1 = 'Hello';
const str2 = 'World';
console.log(str1.concat(' ', str2)); // 'Hello World'
console.log(str1.concat(' ', str2, '!')); // 'Hello World!'

// 更常用 + 或模板字符串
console.log(str1 + ' ' + str2);
console.log(`${str1} ${str2}`);
```

### trim() / trimStart() / trimEnd()
```javascript
const str = '  Hello World  ';
console.log(str.trim());      // 'Hello World'
console.log(str.trimStart()); // 'Hello World  '
console.log(str.trimEnd());   // '  Hello World'

// trimLeft / trimRight 是旧名称
console.log(str.trimLeft());  // 'Hello World  '（同 trimStart）
```

### toLowerCase() / toUpperCase()
```javascript
const str = 'Hello World';
console.log(str.toLowerCase()); // 'hello world'
console.log(str.toUpperCase()); // 'HELLO WORLD'
```

### toLocaleLowerCase() / toLocaleUpperCase()
```javascript
// 处理特定语言的转换
const turkish = 'İstanbul';
console.log(turkish.toLowerCase());          // 'i̇stanbul'
console.log(turkish.toLocaleLowerCase('tr')); // 'istanbul'（土耳其语规则）
```

### padStart() / padEnd()（ES2017）
```javascript
const num = '5';
console.log(num.padStart(3, '0'));  // '005'
console.log(num.padEnd(3, '0'));    // '500'

const str = 'Hello';
console.log(str.padStart(10));      // '     Hello'
console.log(str.padEnd(10, '.'));   // 'Hello.....'

// 实际应用：格式化数字
function formatNumber(n) {
    return n.toString().padStart(2, '0');
}
console.log(formatNumber(5));  // '05'
console.log(formatNumber(12)); // '12'
```

### repeat()（ES6）
```javascript
const str = 'Hello';
console.log(str.repeat(3)); // 'HelloHelloHello'
console.log(str.repeat(0)); // ''
// console.log(str.repeat(-1)); // RangeError
// console.log(str.repeat(Infinity)); // RangeError
```

## 分割与替换

### split()
```javascript
const str = 'Hello World';

// 按空格分割
console.log(str.split(' ')); // ['Hello', 'World']

// 分隔符
const csv = 'a,b,c,d';
console.log(csv.split(',')); // ['a', 'b', 'c', 'd']

// 限制分割数量
console.log(csv.split(',', 2)); // ['a', 'b']

// 分割成字符数组
console.log(str.split('')); // ['H', 'e', 'l', 'l', 'o', ' ', 'W', 'o', 'r', 'l', 'd']

// 使用正则表达式
const text = 'Hello123World456';
console.log(text.split(/\d+/)); // ['Hello', 'World', '']
```

### replace() / replaceAll()（ES2021）
```javascript
const str = 'Hello World World';

// replace 只替换第一个匹配
console.log(str.replace('World', 'JS')); // 'Hello JS World'

// 使用正则全局替换
console.log(str.replace(/World/g, 'JS')); // 'Hello JS JS'

// replaceAll 替换所有匹配（ES2021）
console.log(str.replaceAll('World', 'JS')); // 'Hello JS JS'

// 使用回调函数
const result = str.replace(/World/g, (match, offset) => {
    return `${match}(${offset})`;
});
console.log(result); // 'Hello World(6) World(12)'

// 特殊替换模式
const text2 = 'Hello World';
console.log(text2.replace(/(\w+) (\w+)/, '$2 $1')); // 'World Hello'
// $& 匹配的子串
// $` 匹配子串之前的文本
// $' 匹配子串之后的文本
// $n 第 n 个捕获组
```

## 其他方法

### localeCompare()
```javascript
const str1 = 'apple';
const str2 = 'banana';

console.log(str1.localeCompare(str2)); // -1（str1 < str2）
console.log(str2.localeCompare(str1)); // 1（str2 > str1）
console.log(str1.localeCompare('apple')); // 0（相等）

// 中文排序
const words = ['北京', '上海', '广州', '深圳'];
words.sort((a, b) => a.localeCompare(b, 'zh-CN'));
console.log(words); // ['北京', '广州', '上海', '深圳']

// 选项参数
console.log('a'.localeCompare('A')); // 1
console.log('a'.localeCompare('A', undefined, { sensitivity: 'base' })); // 0
```

### normalize()
```javascript
// Unicode 规范化
const str1 = '\u0041';        // 'A'
const str2 = 'A\u030A';       // 'Å'（A + 组合圆圈）
const str3 = '\u00C5';        // 'Å'（预组合字符）

console.log(str1 === str2);   // false
console.log(str2 === str3);   // false
console.log(str2.normalize() === str3.normalize()); // true

// 规范化形式
// NFC - 规范分解后重新组合
// NFD - 规范分解
// NFKC - 兼容分解后重新组合
// NFKD - 兼容分解
```

### fromCharCode() / fromCodePoint()
```javascript
// 静态方法，通过 String 调用
console.log(String.fromCharCode(65, 66, 67)); // 'ABC'
console.log(String.fromCodePoint(65, 128512)); // 'A😀'
```
