# 创建型模式

## 单例模式

```javascript
class Singleton {
    static instance;
    
    constructor() {
        if (Singleton.instance) {
            return Singleton.instance;
        }
        Singleton.instance = this;
    }
}

// 或使用闭包
const singleton = (function() {
    let instance;
    return function() {
        if (!instance) instance = { /* ... */ };
        return instance;
    };
})();
```

## 工厂模式

```javascript
class Car {
    constructor(type) {
        this.type = type;
    }
}

class CarFactory {
    create(type) {
        return new Car(type);
    }
}

const factory = new CarFactory();
const car = factory.create('sedan');
```

## 建造者模式

```javascript
class QueryBuilder {
    constructor() {
        this.query = {};
    }
    
    select(fields) {
        this.query.select = fields;
        return this;
    }
    
    from(table) {
        this.query.from = table;
        return this;
    }
    
    where(condition) {
        this.query.where = condition;
        return this;
    }
    
    build() {
        return this.query;
    }
}

const query = new QueryBuilder()
    .select('*')
    .from('users')
    .where('id = 1')
    .build();
```
