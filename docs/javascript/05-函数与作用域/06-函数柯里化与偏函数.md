# 函数柯里化与偏函数

> 柯里化（Currying）和偏函数应用（Partial Application）是函数式编程的重要概念，它们可以让函数更加灵活和可复用。

## 学习要点

- 🔍 理解柯里化的概念和实现
- 📦 掌握偏函数应用
- 🔄 区分柯里化和偏函数
- 🎯 学会实际应用场景

---

## 1. 什么是柯里化？

柯里化是将一个接受多个参数的函数转换为一系列接受单个参数的函数的技术。

### 基本概念

```javascript
// 普通函数
function add(a, b, c) {
    return a + b + c;
}

add(1, 2, 3); // 6

// 柯里化后的函数
function curriedAdd(a) {
    return function(b) {
        return function(c) {
            return a + b + c;
        };
    };
}

curriedAdd(1)(2)(3); // 6

// 可以分步调用
const addOne = curriedAdd(1);       // 返回一个函数
const addOneAndTwo = addOne(2);     // 返回一个函数
const result = addOneAndTwo(3);     // 6

// 也可以一次调用
curriedAdd(1)(2)(3); // 6
```

### 柯里化的意义

```javascript
// 1. 参数复用
function createUrl(protocol) {
    return function(domain) {
        return function(path) {
            return `${protocol}://${domain}/${path}`;
        };
    };
}

// 创建一个专门生成 https URL 的函数
const httpsUrl = createUrl('https');
const apiGithub = httpsUrl('api.github.com');
const githubUsers = apiGithub('users'); // 'https://api.github.com/users'

// 2. 延迟执行
function curry(func) {
    return function curried(...args) {
        if (args.length >= func.length) {
            return func.apply(this, args);
        } else {
            return function(...moreArgs) {
                return curried.apply(this, args.concat(moreArgs));
            };
        }
    };
}

function sum(a, b, c) {
    return a + b + c;
}

const curriedSum = curry(sum);

curriedSum(1)(2)(3);    // 6
curriedSum(1, 2)(3);    // 6
curriedSum(1)(2, 3);    // 6
curriedSum(1, 2, 3);    // 6
```

## 2. 柯里化的实现

### 基础实现

```javascript
function curry(fn) {
    return function curried(...args) {
        // 如果参数数量足够，执行原函数
        if (args.length >= fn.length) {
            return fn.apply(this, args);
        }
        // 否则返回一个函数继续收集参数
        return function(...moreArgs) {
            return curried.apply(this, args.concat(moreArgs));
        };
    };
}

// 使用示例
function multiply(a, b, c) {
    return a * b * c;
}

const curriedMultiply = curry(multiply);

console.log(curriedMultiply(2)(3)(4));  // 24
console.log(curriedMultiply(2, 3)(4));  // 24
console.log(curriedMultiply(2)(3, 4));  // 24
```

### 支持占位符的柯里化

```javascript
function curry(fn) {
    const placeholder = Symbol('placeholder');
    
    function curried(...args) {
        // 过滤掉占位符后的有效参数数量
        const validArgs = args.filter(arg => arg !== placeholder);
        
        // 检查是否有足够的参数（不包括占位符）
        if (validArgs.length >= fn.length && !args.includes(placeholder)) {
            return fn.apply(this, args.map(arg => 
                arg === placeholder ? undefined : arg
            ));
        }
        
        return function(...moreArgs) {
            // 合并参数，处理占位符位置
            const finalArgs = args.map(arg => 
                arg === placeholder && moreArgs.length 
                    ? moreArgs.shift() 
                    : arg
            );
            return curried(...finalArgs, ...moreArgs);
        };
    }
    
    curried._ = placeholder;
    return curried;
}

// 使用示例
function greet(greeting, name, punctuation) {
    return `${greeting}, ${name}${punctuation}`;
}

const curriedGreet = curry(greet);

// 使用占位符
console.log(curriedGreet('Hello')(curriedGreet._)('!')('World')); // 'Hello, World!'
```

### 自动柯里化工具

```javascript
// 简洁版柯里化
const curry = (fn) => {
    const curried = (...args) => 
        args.length >= fn.length 
            ? fn(...args) 
            : (...more) => curried(...args, ...more);
    return curried;
};

// 测试
const add = curry((a, b, c) => a + b + c);
console.log(add(1)(2)(3)); // 6
console.log(add(1, 2)(3)); // 6
console.log(add(1)(2, 3)); // 6
```

## 3. 柯里化应用场景

### 日志函数

```javascript
// 普通日志函数
function log(level, time, message) {
    console.log(`[${level}] ${time}: ${message}`);
}

log('ERROR', '2024-01-01', 'Something went wrong');

// 柯里化后
const curriedLog = curry(log);

// 创建特定级别的日志函数
const errorLog = curriedLog('ERROR');
const debugLog = curriedLog('DEBUG');

// 创建带时间戳的日志函数
const errorLogWithTime = errorLog(new Date().toISOString());

// 使用
errorLogWithTime('Database connection failed');
debugLog('User logged in');
```

### 数据处理

```javascript
// 数组处理
const map = curry((fn, arr) => arr.map(fn));
const filter = curry((fn, arr) => arr.filter(fn));
const reduce = curry((fn, initial, arr) => arr.reduce(fn, initial));

// 创建特定功能的函数
const double = map(x => x * 2);
const isEven = filter(x => x % 2 === 0);
const sum = reduce((a, b) => a + b, 0);

// 组合使用
const numbers = [1, 2, 3, 4, 5];

const doubleEvens = arr => double(isEven(arr));
console.log(doubleEvens(numbers)); // [4, 8]

// 或者使用管道
const pipe = (...fns) => x => fns.reduce((v, f) => f(v), x);

const processNumbers = pipe(
    isEven,
    double,
    sum
);

console.log(processNumbers(numbers)); // 12
```

### 事件处理

```javascript
// 创建事件处理器
const handleEvent = curry((action, event) => {
    console.log(`Action: ${action}, Event: ${event.type}`);
    // 执行具体操作
});

const handleClick = handleEvent('click');
const handleSubmit = handleEvent('submit');

document.addEventListener('click', handleClick);
document.addEventListener('submit', handleSubmit);
```

### API 请求

```javascript
// 创建 API 请求函数
const request = curry((method, baseUrl, endpoint, data) => {
    return fetch(`${baseUrl}${endpoint}`, {
        method,
        headers: { 'Content-Type': 'application/json' },
        body: data ? JSON.stringify(data) : undefined
    });
});

// 创建特定基础 URL 的请求函数
const apiRequest = request('GET')('https://api.example.com');

// 创建特定端点的函数
const getUsers = apiRequest('/users');
const getPosts = apiRequest('/posts');

// 使用
getUsers().then(res => res.json());
getPosts().then(res => res.json());

// POST 请求
const postRequest = request('POST')('https://api.example.com');
const createUser = postRequest('/users');

createUser({ name: 'John', email: 'john@example.com' });
```

## 4. 偏函数应用（Partial Application）

偏函数应用是固定一个函数的一个或多个参数，然后返回一个新函数。

### 基本概念

```javascript
// 使用 bind 实现偏函数
function greet(greeting, name) {
    return `${greeting}, ${name}!`;
}

const sayHello = greet.bind(null, 'Hello');
console.log(sayHello('John')); // 'Hello, John!'
console.log(sayHello('Jane')); // 'Hello, Jane!'

const sayHi = greet.bind(null, 'Hi');
console.log(sayHi('Bob')); // 'Hi, Bob!'
```

### 手动实现偏函数

```javascript
function partial(fn, ...presetArgs) {
    return function(...laterArgs) {
        return fn(...presetArgs, ...laterArgs);
    };
}

// 使用示例
function multiply(a, b, c) {
    return a * b * c;
}

const double = partial(multiply, 2);
console.log(double(3, 4)); // 24 (2 * 3 * 4)

const triple = partial(multiply, 3);
console.log(triple(2, 4)); // 24 (3 * 2 * 4)

// 固定多个参数
const alwaysDoubleTriple = partial(multiply, 2, 3);
console.log(alwaysDoubleTriple(4)); // 24 (2 * 3 * 4)
```

### 支持占位符的偏函数

```javascript
function partial(fn, ...presetArgs) {
    const placeholder = Symbol('placeholder');
    
    return function(...laterArgs) {
        const args = presetArgs.map(arg => 
            arg === placeholder ? laterArgs.shift() : arg
        );
        return fn(...args, ...laterArgs);
    };
}

// 使用示例
function greet(greeting, name, punctuation) {
    return `${greeting}, ${name}${punctuation}`;
}

const sayHelloTo = partial(greet, 'Hello', partial._, '!');
console.log(sayHelloTo('John')); // 'Hello, John!'
console.log(sayHelloTo('Jane')); // 'Hello, Jane!'
```

## 5. 柯里化 vs 偏函数

### 核心区别

```javascript
// 柯里化：将多参数函数转换为一系列单参数函数
// 每次只接受一个参数

function curriedAdd(a) {
    return function(b) {
        return function(c) {
            return a + b + c;
        };
    };
}

curriedAdd(1)(2)(3); // 6

// 偏函数：固定部分参数，返回接受剩余参数的函数
// 可以固定多个参数

function partialAdd(a, b) {
    return function(c) {
        return a + b + c;
    };
}

partialAdd(1, 2)(3); // 6
```

### 对比总结

| 特性 | 柯里化 | 偏函数 |
|------|--------|--------|
| 参数数量 | 每次只接受一个参数 | 可以固定多个参数 |
| 返回值 | 嵌套函数 | 单个新函数 |
| 调用方式 | `f(a)(b)(c)` | `f(a, b)(c)` 或 `f(a)(b, c)` |
| 用途 | 函数链式调用、组合 | 预设参数、配置函数 |
| 实现 | 递归嵌套 | 绑定预设参数 |

### 选择建议

```javascript
// 使用柯里化的场景：
// 1. 需要链式调用
// 2. 函数组合
// 3. 高度可配置的函数

const curriedMap = curry((fn, arr) => arr.map(fn));
const double = curriedMap(x => x * 2);
double([1, 2, 3]); // [2, 4, 6]

// 使用偏函数的场景：
// 1. 固定部分配置参数
// 2. 创建特定功能的变体函数
// 3. 简单的参数预设

const log = partial(console.log, '[APP]');
log('User logged in'); // [APP] User logged in
```

## 6. 实际应用示例

### 配置函数工厂

```javascript
// 创建可配置的函数
function createValidator(rules) {
    return function(value) {
        return rules.every(rule => rule(value));
    };
}

const isRequired = value => value !== null && value !== '';
const isEmail = value => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
const minLength = min => value => value.length >= min;

// 使用柯里化组合规则
const validateEmail = createValidator([isRequired, isEmail]);
const validatePassword = createValidator([isRequired, minLength(8)]);

validateEmail('test@example.com'); // true
validateEmail('invalid'); // false
validatePassword('12345678'); // true
validatePassword('123'); // false
```

### React 事件处理

```javascript
// 使用柯里化创建事件处理器
const handleChange = curry((fieldName, event) => {
    this.setState({ [fieldName]: event.target.value });
});

// 在 JSX 中使用
<input onChange={handleChange('username')} />
<input onChange={handleChange('email')} />

// 使用偏函数
const handleClick = partial((action, id) => {
    // 处理点击
}, 'delete');

<button onClick={() => handleClick(item.id)}>删除</button>
```

## 小结

| 概念 | 说明 | 示例 |
|------|------|------|
| **柯里化** | 多参数函数转单参数函数链 | `f(a)(b)(c)` |
| **偏函数** | 固定部分参数返回新函数 | `f(a, b)(c)` |
| **参数复用** | 预设常用参数 | `const add1 = add(1)` |
| **延迟执行** | 分步收集参数 | 配置完成后再执行 |
