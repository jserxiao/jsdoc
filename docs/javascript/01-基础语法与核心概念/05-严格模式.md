# 严格模式

> 严格模式（Strict Mode）是 ES5 引入的一种限制性 JavaScript 变体，它使 JavaScript 在更严格的条件下运行，帮助开发者写出更安全、更优化的代码。

## 学习要点

- 📝 理解严格模式的作用和意义
- 🔧 掌握如何启用严格模式
- ⚠️ 了解严格模式的限制
- 🎯 养成使用严格模式的习惯

---

## 1. 什么是严格模式？

严格模式是对 JavaScript 的一种限制性变体，它：

- 消除了一些 JavaScript 的静默错误，改为抛出异常
- 修复了一些 JavaScript 引擎难以优化的错误
- 禁止了一些不安全的语法
- 为未来的 ECMAScript 版本铺平道路

```javascript
// 非严格模式下的问题代码
x = 10; // 意外创建全局变量，不会报错

// 严格模式下
'use strict';
x = 10; // ReferenceError: x is not defined
```

---

## 2. 如何启用严格模式

### 为整个脚本启用

```javascript
// 在脚本文件的第一行添加
'use strict';

// 后续的所有代码都在严格模式下运行
var x = 10;
```

### 为函数启用

```javascript
// 只在特定函数内启用严格模式
function strict() {
    'use strict';
    // 函数内的代码在严格模式下运行
    return '这是严格模式';
}

function notStrict() {
    // 这里的代码不在严格模式下
    return '这不是严格模式';
}
```

### 为模块启用（自动启用）

```javascript
// ES6 模块自动启用严格模式
// 不需要手动添加 'use strict'
export function add(a, b) {
    return a + b;
}

// 类（class）内部也自动启用严格模式
class MyClass {
    constructor() {
        // 这里是严格模式
    }
}
```

## 3. 严格模式的限制

### 3.1 变量必须先声明

```javascript
'use strict';

// ❌ 未声明的变量
x = 10; // ReferenceError: x is not defined

// ✅ 正确写法
var x = 10;
let y = 20;
const z = 30;

// ❌ 删除变量
delete x; // SyntaxError: Delete of an unqualified identifier

// ❌ 删除不可删除的属性
delete Object.prototype; // TypeError
```

### 3.2 禁止重复参数名

```javascript
'use strict';

// ❌ 重复的参数名
function add(a, a, c) { // SyntaxError: Duplicate parameter name
    return a + b + c;
}

// 非严格模式下，后面的 a 会覆盖前面的 a
```

### 3.3 禁止 with 语句

```javascript
'use strict';

// ❌ with 语句
var obj = { x: 1 };
with (obj) { // SyntaxError: Strict mode code may not include a with statement
    console.log(x);
}
```

### 3.4 禁止八进制字面量

```javascript
'use strict';

// ❌ 八进制字面量（旧语法）
var n = 0100; // SyntaxError: Octal literals are not allowed

// ✅ ES6 八进制语法
var n = 0o100; // 64（正确）
var n = 0O100; // 64（正确）
```

### 3.5 禁止 this 指向全局对象

```javascript
'use strict';

function showThis() {
    console.log(this);
}

showThis(); // undefined（非严格模式下是 window/global）

// 这避免了意外修改全局对象的风险
function Person(name) {
    // 如果忘记使用 new，非严格模式下 this 指向 window
    // 严格模式下 this 是 undefined，更容易发现错误
    this.name = name;
}

// 非严格模式
var p = Person('John'); // window.name = 'John'（污染全局！）

// 严格模式
'use strict';
var p = Person('John'); // TypeError: Cannot set property 'name' of undefined
```

### 3.6 禁止只读属性赋值

```javascript
'use strict';

var obj = {};

Object.defineProperty(obj, 'x', { value: 42, writable: false });

// ❌ 给只读属性赋值
obj.x = 9; // TypeError: Cannot assign to read only property

// ❌ 给不可扩展对象添加属性
Object.preventExtensions(obj);
obj.y = 10; // TypeError: Can't add property y, object is not extensible
```

### 3.7 禁止删除不可配置的属性

```javascript
'use strict';

// ❌ 删除不可配置的属性
delete Object.prototype; // TypeError: Cannot delete property 'prototype'

var obj = {};
Object.defineProperty(obj, 'x', { configurable: false });
delete obj.x; // TypeError
```

### 3.8 禁止重复的对象属性名（ES6 已允许）

```javascript
'use strict';

// ES5 严格模式
var obj = {
    name: 'John',
    name: 'Jane' // SyntaxError（ES5）
};

// ES6+ 已允许重复属性名（后面的覆盖前面的）
// 所以现代浏览器中这不会报错
```

### 3.9 禁止保留字作为变量名

```javascript
'use strict';

// ❌ 使用保留字
var let = 10;     // SyntaxError
var const = 20;   // SyntaxError
var class = 30;   // SyntaxError
var function = 40; // SyntaxError
var static = 50;  // SyntaxError
var yield = 60;   // SyntaxError
```

### 3.10 eval 的限制

```javascript
'use strict';

// eval 创建的变量不会污染外部作用域
eval('var x = 10');
console.log(x); // ReferenceError: x is not defined

// eval 不能被重新赋值
eval = 10; // SyntaxError

// 不能删除 eval
delete eval; // SyntaxError
```

### 3.11 arguments 的限制

```javascript
'use strict';

// arguments 不能被重新赋值
arguments = 10; // SyntaxError

// 不再追踪参数变化
function foo(a) {
    a = 10;
    console.log(arguments[0]); // 非严格模式：10，严格模式：保持原值
}
foo(5);

// arguments.callee 禁止使用
function factorial(n) {
    if (n <= 1) return 1;
    return n * arguments.callee(n - 1); // TypeError
}

// arguments.caller 禁止使用
function outer() {
    inner();
}
function inner() {
    console.log(arguments.caller); // TypeError
}
```

### 3.12 禁止 fun.caller 和 fun.arguments

```javascript
'use strict';

function outer() {
    inner();
}

function inner() {
    // ❌ 访问调用栈
    console.log(inner.caller);     // TypeError
    console.log(inner.arguments);  // TypeError
}
```

## 4. 严格模式的优势

### 4.1 捕获常见错误

```javascript
'use strict';

// 变量拼写错误更容易发现
var username = 'John';
usrname = 'Jane'; // ReferenceError，而不是创建新变量

// 给 NaN 赋值
NaN = 5; // TypeError

// 给 undefined 赋值
undefined = 5; // TypeError
```

### 4.2 安全性提升

```javascript
'use strict';

// this 不会意外指向全局对象
function Point(x, y) {
    this.x = x;
    this.y = y;
}

// 忘记 new 时会报错，而不是污染全局
var p = Point(1, 2); // TypeError
```

### 4.3 性能优化

```javascript
'use strict';

// 严格模式下，编译器可以进行更激进的优化
// 因为：
// 1. 不会意外创建全局变量
// 2. this 不会意外指向全局对象
// 3. arguments 对象不可修改
// 4. eval 不影响外部作用域

function calculate(a, b) {
    // 编译器知道 a 和 b 不会被 arguments 修改
    // 可以进行更好的优化
    return a + b;
}
```

### 4.4 为未来做准备

```javascript
'use strict';

// 使用保留字作为变量名会报错
// 这有助于代码向新版本迁移
var let = 1;    // SyntaxError
var const = 2;  // SyntaxError
var class = 3;  // SyntaxError
```

## 5. 严格模式的影响范围

```javascript
// 全局严格模式
'use strict';

function outer() {
    // 内部函数继承严格模式
    function inner() {
        // 这里也是严格模式
    }
}

// 严格模式不会影响被调用的非严格代码
function strictFn() {
    'use strict';
    // 严格模式
}

function nonStrictFn() {
    // 非严格模式
}

strictFn.call(nonStrictFn); // strictFn 内部仍是严格模式
```

## 6. 最佳实践

### 6.1 总是使用严格模式

```javascript
// ✅ 推荐：在文件开头添加
'use strict';

// 或者在函数内部
(function() {
    'use strict';
    // 你的代码
})();
```

### 6.2 使用 IIFE 包装代码

```javascript
// 避免严格模式影响其他脚本
(function() {
    'use strict';
    
    // 你的代码在这里
})();
```

### 6.3 使用 ES6 模块

```javascript
// ES6 模块自动启用严格模式
// 无需手动添加 'use strict'

// module.js
export function add(a, b) {
    return a + b;
}
```

### 6.4 注意拼接脚本的问题

```javascript
// 问题：如果将严格模式脚本与非严格模式脚本拼接
// 可能会导致非严格模式脚本也变成严格模式

// 解决方案 1：使用 IIFE 包装
(function() {
    'use strict';
    // 严格模式代码
})();

// 解决方案 2：使用 ES6 模块（推荐）
// 每个模块都有自己的严格模式设置
```

## 7. 严格模式检查表

| 限制项 | 非严格模式 | 严格模式 |
|-------|-----------|---------|
| 未声明变量 | 创建全局变量 | ReferenceError |
| 重复参数名 | 允许（后者覆盖） | SyntaxError |
| with 语句 | 允许 | SyntaxError |
| 八进制字面量 | 允许 | SyntaxError |
| this 默认值 | window/global | undefined |
| 只读属性赋值 | 静默失败 | TypeError |
| 删除不可配置属性 | 静默失败 | TypeError |
| arguments.callee | 允许 | TypeError |
| eval 创建变量 | 污染外部作用域 | 不污染 |

## 小结

| 要点 | 说明 |
|------|------|
| **启用方式** | 'use strict' 或使用 ES6 模块 |
| **主要限制** | 变量声明、this 指向、只读属性等 |
| **优势** | 更安全、更易调试、性能更好 |
| **推荐** | 始终使用严格模式 |
