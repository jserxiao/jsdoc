# 运算符与表达式

## 算术运算符

```javascript
// 基本运算
console.log(10 + 5);   // 15 加法
console.log(10 - 5);   // 5  减法
console.log(10 * 5);   // 50 乘法
console.log(10 / 5);   // 2  除法
console.log(10 % 3);   // 1  取余
console.log(10 ** 2);  // 100 幂运算（ES2016）

// 字符串拼接
console.log('Hello' + ' ' + 'World'); // 'Hello World'
console.log('10' + 5);  // '105' - 数字被转换为字符串
console.log(10 + '5');  // '105'

// 一元运算符
let x = 5;
console.log(+x);   // 5  正号（转换为数字）
console.log(-x);   // -5 负号
console.log(++x);  // 6  前置递增
console.log(x++);  // 6  后置递增（返回旧值）
console.log(x);    // 7
```

## 赋值运算符

```javascript
let a = 10;
a += 5;   // a = a + 5  => 15
a -= 3;   // a = a - 3  => 12
a *= 2;   // a = a * 2  => 24
a /= 4;   // a = a / 4  => 6
a %= 4;   // a = a % 4  => 2
a **= 3;  // a = a ** 3 => 8

// ES2021 逻辑赋值运算符
let b = null;
b ??= 'default'; // b = b ?? 'default' => 'default'

let c = false;
c ||= true;      // c = c || true => true

let d = 'hello';
d &&= 'world';   // d = d && 'world' => 'world'
```

## 比较运算符

```javascript
// 相等比较
console.log(5 == '5');   // true  （类型转换后比较）
console.log(5 === '5');  // false （严格比较，不转换类型）
console.log(5 != '5');   // false
console.log(5 !== '5');  // true

// 大小比较
console.log(10 > 5);   // true
console.log(10 < 5);   // false
console.log(10 >= 10); // true
console.log(10 <= 9);  // false

// 字符串比较（按字典序）
console.log('apple' < 'banana'); // true
console.log('a' < 'A');          // false（小写字母 ASCII 码更大）

// 特殊比较
console.log(NaN === NaN);  // false
console.log(Number.isNaN(NaN)); // true - 正确的 NaN 检测

console.log(null == undefined);  // true
console.log(null === undefined); // false
```

### == 转换规则表

| 比较 | 结果 | 说明 |
|------|------|------|
| `1 == '1'` | true | 字符串转数字 |
| `true == 1` | true | 布尔转数字 |
| `false == 0` | true | 布尔转数字 |
| `null == undefined` | true | 特殊规则 |
| `null == 0` | false | null 不转换 |
| `undefined == 0` | false | undefined 不转换 |

## 逻辑运算符

```javascript
// 与（&&）- 短路求值
console.log(true && true);    // true
console.log(true && false);   // false
console.log(false && anything); // false（短路）

// 或（||）- 短路求值
console.log(false || true);   // true
console.log(true || anything); // true（短路）

// 非（!）
console.log(!true);   // false
console.log(!false);  // true
console.log(!!'hello'); // true（双重否定转布尔值）

// 返回实际值（非布尔值）
console.log(0 && 'hello');    // 0（假值，短路返回）
console.log(1 && 'hello');    // 'hello'
console.log('' || 'default'); // 'default'
console.log('hi' || 'default'); // 'hi'

// 空值合并运算符（??）ES2020
console.log(null ?? 'default');    // 'default'
console.log(undefined ?? 'default'); // 'default'
console.log(0 ?? 'default');       // 0（只有 null/undefined 才使用默认值）
console.log('' ?? 'default');      // ''
```

### || vs ?? 的区别

```javascript
const config = {
    count: 0,
    name: ''
};

// || 会将假值视为未定义
const count1 = config.count || 10;  // 10（0 是假值）
const name1 = config.name || 'Guest'; // 'Guest'

// ?? 只将 null/undefined 视为未定义
const count2 = config.count ?? 10;  // 0
const name2 = config.name ?? 'Guest'; // ''
```

## 位运算符

```javascript
// 按位与（&）
console.log(5 & 3);  // 1  (0101 & 0011 = 0001)

// 按位或（|）
console.log(5 | 3);  // 7  (0101 | 0011 = 0111)

// 按位异或（^）
console.log(5 ^ 3);  // 6  (0101 ^ 0011 = 0110)

// 按位非（~）
console.log(~5);     // -6 (~0101 = ...11111010)

// 左移（<<）
console.log(5 << 1); // 10 (0101 << 1 = 1010)

// 右移（>>）保留符号位
console.log(-5 >> 1); // -3

// 无符号右移（>>>）
console.log(-5 >>> 1); // 2147483645（大正数）

// 位运算应用
// 判断奇偶
const isOdd = n => n & 1;
console.log(isOdd(5));  // true
console.log(isOdd(4));  // false

// 取整
const floor = n => n | 0;
console.log(floor(3.7)); // 3

// 交换变量
let a = 1, b = 2;
a = a ^ b;
b = a ^ b;
a = a ^ b;
console.log(a, b); // 2 1
```

## 三元运算符

```javascript
// condition ? expr1 : expr2
const age = 18;
const status = age >= 18 ? '成年' : '未成年';
console.log(status); // '成年'

// 嵌套使用（不推荐，可读性差）
const score = 85;
const grade = score >= 90 ? 'A' :
              score >= 80 ? 'B' :
              score >= 70 ? 'C' : 'D';
console.log(grade); // 'B'

// 实际应用
function greet(name) {
    return name ? `Hello, ${name}!` : 'Hello, Guest!';
}
```

## 运算符优先级

从高到低排列：

| 优先级 | 运算符 | 说明 |
|--------|--------|------|
| 1 | `()` | 分组 |
| 2 | `.` `[]` `new` | 成员访问、数组访问、new |
| 3 | `()` | 函数调用 |
| 4 | `++` `--` | 后置递增/递减 |
| 5 | `!` `~` `+` `-` `++` `--` `typeof` | 一元运算符 |
| 6 | `**` | 幂运算 |
| 7 | `*` `/` `%` | 乘、除、取余 |
| 8 | `+` `-` | 加、减 |
| 9 | `<<` `>>` `>>>` | 位移动 |
| 10 | `<` `<=` `>` `>=` `in` `instanceof` | 比较 |
| 11 | `==` `!=` `===` `!==` | 相等 |
| 12 | `&` | 按位与 |
| 13 | `^` | 按位异或 |
| 14 | `\|` | 按位或 |
| 15 | `&&` | 逻辑与 |
| 16 | `\|\|` `??` | 逻辑或、空值合并 |
| 17 | `?:` | 三元运算符 |
| 18 | `=` `+=` `-=` 等 | 赋值 |
| 19 | `,` | 逗号 |

```javascript
// 示例
console.log(2 + 3 * 4);    // 14（先乘后加）
console.log((2 + 3) * 4);  // 20（括号改变优先级）

console.log(true || false && false); // true（&& 优先级高于 ||）
console.log((true || false) && false); // false
```

## 可选链操作符（?.）

```javascript
const user = {
    name: 'John',
    address: {
        city: 'New York'
    }
};

// 传统方式
const city1 = user && user.address && user.address.city;

// 可选链（ES2020）
const city2 = user?.address?.city;
console.log(city2); // 'New York'

const country = user?.address?.country;
console.log(country); // undefined（而不是报错）

// 函数调用
const result = user.greet?.();
console.log(result); // undefined（方法不存在不会报错）

// 数组访问
const arr = null;
console.log(arr?.[0]); // undefined
```

## 空值合并操作符（??）

```javascript
// 仅当左侧为 null 或 undefined 时返回右侧
const value1 = null ?? 'default';    // 'default'
const value2 = undefined ?? 'default'; // 'default'
const value3 = 0 ?? 'default';       // 0
const value4 = '' ?? 'default';      // ''
const value5 = false ?? 'default';   // false

// 与 || 的区别
const count = 0;
console.log(count || 10);  // 10（0 是假值）
console.log(count ?? 10);  // 0（0 不是 null/undefined）

// 结合可选链使用
const config = {
    timeout: null
};
const timeout = config?.timeout ?? 3000;
console.log(timeout); // 3000
```
