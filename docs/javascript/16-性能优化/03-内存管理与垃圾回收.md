# 内存管理与垃圾回收

> 内存管理是 JavaScript 性能优化的核心内容。《JavaScript高级程序设计》和《你不知道的JavaScript》都强调了理解内存管理的重要性，它是避免内存泄漏和优化性能的关键。

## 学习要点

- 🔍 理解 JavaScript 的内存生命周期
- 📦 掌握垃圾回收机制
- ⚠️ 识别和避免内存泄漏
- 🛠️ 学会内存分析和优化

---

## 1. 内存生命周期

JavaScript 的内存生命周期分为三个阶段：

```
1. 分配内存 → 2. 使用内存 → 3. 释放内存
```

### 内存分配

```javascript
// 1. 值的初始化（自动分配）
let n = 123;            // 为数字分配内存
let s = 'hello';        // 为字符串分配内存
let obj = {             // 为对象及其属性分配内存
    a: 1,
    b: 2
};
let arr = [1, 2, 3];    // 为数组及其元素分配内存

// 2. 函数调用（分配栈帧）
function add(a, b) {
    return a + b;       // 函数调用时分配栈内存
}

// 3. 动态分配
const node = document.createElement('div');
```

### 内存使用

```javascript
// 读取和写入值
let a = 10;    // 写入
console.log(a); // 读取

// 对象属性操作
const obj = { name: 'John' };
obj.name = 'Jane';           // 写入
console.log(obj.name);       // 读取
obj.age = 30;                // 添加属性

// 函数调用传递值
function greet(person) {
    console.log(person.name); // 读取参数
}
greet(obj);
```

### 内存释放

```javascript
// 自动垃圾回收（开发者无需手动释放）
function createPerson() {
    const person = { name: 'John' };
    // person 在函数结束后不再被引用
    // 会被垃圾回收器自动回收
}

// 手动解除引用（帮助垃圾回收）
let globalObj = { data: new Array(10000) };

// 不再需要时
globalObj = null; // 解除引用，允许垃圾回收
```

## 2. 堆内存与栈内存

### 栈内存（Stack）

```javascript
// 栈内存特点：
// 1. 存储基本类型值
// 2. 存储函数调用信息（调用栈）
// 3. 自动分配和释放
// 4. 空间小，速度快

function example() {
    let a = 1;        // 栈
    let b = 'hello';  // 栈
    let c = true;     // 栈
    
    // 函数执行完毕，栈内存自动释放
}
```

### 堆内存（Heap）

```javascript
// 堆内存特点：
// 1. 存储引用类型值
// 2. 动态分配
// 3. 需要垃圾回收器管理
// 4. 空间大，速度较慢

// 对象存储在堆中
const obj = { 
    name: 'John', 
    age: 30 
};

// 数组存储在堆中
const arr = [1, 2, 3, 4, 5];

// 函数存储在堆中
const fn = function() {
    return 'Hello';
};
```

## 3. 垃圾回收算法

### 引用计数法（已废弃）

```javascript
// 引用计数法：跟踪每个对象被引用的次数
// 当引用次数为 0 时，回收内存

let obj = { name: 'John' }; // 引用次数：1
let ref = obj;              // 引用次数：2

obj = null;                 // 引用次数：1
ref = null;                 // 引用次数：0 → 可以回收

// ❌ 问题：无法处理循环引用
function circularRef() {
    const obj1 = {};
    const obj2 = {};
    
    obj1.ref = obj2; // obj2 引用次数 +1
    obj2.ref = obj1; // obj1 引用次数 +1
    
    // 即使函数结束，两个对象的引用次数都不为 0
    // 导致内存泄漏
}
```

### 标记清除法（现代浏览器使用）

```javascript
// 标记清除法：从根对象开始，标记所有可达对象
// 未被标记的对象会被回收

// 根对象包括：
// - 全局变量
// - 当前调用栈中的变量
// - 正在执行的函数中的变量

let global = { name: 'John' }; // 根可达

function example() {
    const local = { data: 1 }; // 栈可达
    
    function nested() {
        const inner = { value: local.data }; // 闭包可达
        console.log(inner);
    }
    
    return nested;
}

// 标记清除过程：
// 1. 从根开始遍历
// 2. 标记所有可达对象
// 3. 清除未标记的对象

// ✅ 可以处理循环引用
function noCircularLeak() {
    const obj1 = {};
    const obj2 = {};
    
    obj1.ref = obj2;
    obj2.ref = obj1;
    
    // 函数结束后，obj1 和 obj2 都不可达
    // 会被正确回收
}
```

### 分代回收

```javascript
// V8 引擎使用分代回收，将堆分为新生代和老生代

/*
新生代（New Generation）：
- 存放生命周期短的对象
- 使用 Scavenge 算法（复制算法）
- 空间小（1-8MB）
- 回收速度快

老生代（Old Generation）：
- 存放生命周期长的对象
- 使用标记清除 + 标记整理
- 空间大
- 回收速度较慢

对象晋升：
- 新生代对象存活一定时间后 → 老生代
- 新生代空间不足时 → 晋升到老生代
*/

// 示例：生命周期短的对象
function shortLived() {
    const temp = { data: new Array(1000) };
    return temp.data.length;
    // temp 立即变为不可达
    // 在新生代中快速回收
}

// 示例：生命周期长的对象
const longLived = { 
    cache: new Map(),
    config: {}
};
// 长期存在，在老生代中
```

## 4. 内存泄漏场景

### 1. 意外的全局变量

```javascript
// ❌ 内存泄漏：意外的全局变量
function leak() {
    bar = 'I am global now!'; // 忘记使用 var/let/const
    this.data = 'also global'; // this 指向 window
}

// ✅ 正确做法
function noLeak() {
    'use strict'; // 使用严格模式
    const bar = 'local variable';
}
```

### 2. 忘记清除的定时器

```javascript
// ❌ 内存泄漏：定时器引用 DOM
const element = document.getElementById('button');

function leak() {
    const data = { heavy: new Array(10000) };
    
    setInterval(() => {
        if (element) {
            console.log(data);
        }
    }, 1000);
}

// ✅ 正确做法
let timerId;

function noLeak() {
    const element = document.getElementById('button');
    const data = { heavy: new Array(10000) };
    
    timerId = setInterval(() => {
        if (element) {
            console.log(data);
        }
    }, 1000);
}

// 不需要时清除
function cleanup() {
    clearInterval(timerId);
}

// 元素移除时清除
element.addEventListener('DOMNodeRemoved', cleanup);
```

### 3. 闭包引起的泄漏

```javascript
// ❌ 内存泄漏：闭包持有大对象
function createClosure() {
    const largeData = new Array(1000000);
    
    return function() {
        console.log(largeData.length); // 闭包持有 largeData
    };
}

const fn = createClosure(); // largeData 无法被回收

// ✅ 正确做法：只保留需要的数据
function createClosureFixed() {
    const largeData = new Array(1000000);
    const length = largeData.length; // 只保留需要的数据
    
    return function() {
        console.log(length);
    };
    // largeData 可以被回收
}
```

### 4. DOM 引用

```javascript
// ❌ 内存泄漏：JavaScript 持有已删除 DOM 的引用
const elements = [];
const button = document.getElementById('button');

elements.push(button);

document.body.removeChild(button); // DOM 删除了
// 但 elements 数组仍持有引用 → 内存泄漏

// ✅ 正确做法
elements.length = 0; // 清空数组

// 或使用 WeakMap
const weakElements = new WeakMap();
const btn = document.getElementById('button');
weakElements.set(btn, { data: 'metadata' });

document.body.removeChild(btn);
// WeakMap 不会阻止垃圾回收
```

### 5. 事件监听器

```javascript
// ❌ 内存泄漏：忘记移除事件监听器
class Component {
    constructor() {
        this.data = new Array(10000);
        this.handleClick = this.handleClick.bind(this);
        document.addEventListener('click', this.handleClick);
    }
    
    handleClick() {
        console.log(this.data);
    }
    
    // 没有 destroy 方法来清理
}

// ✅ 正确做法
class ComponentFixed {
    constructor() {
        this.data = new Array(10000);
        this.handleClick = this.handleClick.bind(this);
        document.addEventListener('click', this.handleClick);
    }
    
    handleClick() {
        console.log(this.data);
    }
    
    destroy() {
        document.removeEventListener('click', this.handleClick);
        this.data = null;
    }
}
```

### 6. Map 和 Set

```javascript
// ❌ 内存泄漏：Map/Set 持有对象引用
const cache = new Map();

function addToCache(id, obj) {
    cache.set(id, obj);
}

const obj = { data: new Array(10000) };
addToCache('key1', obj);

obj = null; // 尝试释放
// 但 cache 中仍持有引用 → 内存泄漏

// ✅ 正确做法：使用 WeakMap
const weakCache = new WeakMap();

function addToWeakCache(obj, value) {
    weakCache.set(obj, value);
}

let obj2 = { data: new Array(10000) };
addToWeakCache(obj2, 'metadata');

obj2 = null; // 可以被回收，WeakMap 不阻止垃圾回收
```

## 5. 内存分析工具

### Chrome DevTools

```javascript
// 1. Memory 面板
// - Heap Snapshot：堆快照
// - Allocation instrumentation on timeline：时间线分配
// - Allocation sampling：分配采样

// 2. Performance 面板
// - 记录内存使用情况
// - 查看 GC 事件

// 3. 代码中标记
console.log('内存使用:', performance.memory);

// 手动触发 GC（DevTools 中可用）
// 点击 "Collect garbage" 按钮
```

### 使用 performance.memory

```javascript
// 浏览器内存信息
function logMemory() {
    const memory = performance.memory;
    console.log({
        usedJSHeapSize: (memory.usedJSHeapSize / 1024 / 1024).toFixed(2) + ' MB',
        totalJSHeapSize: (memory.totalJSHeapSize / 1024 / 1024).toFixed(2) + ' MB',
        jsHeapSizeLimit: (memory.jsHeapSizeLimit / 1024 / 1024).toFixed(2) + ' MB'
    });
}

logMemory();
```

### 检测内存泄漏

```javascript
// 检测内存泄漏的模式
function detectLeak() {
    const before = performance.memory.usedJSHeapSize;
    
    // 执行可能泄漏的操作
    for (let i = 0; i < 100; i++) {
        createObject();
    }
    
    // 强制 GC（仅 DevTools）
    // ...
    
    const after = performance.memory.usedJSHeapSize;
    
    console.log(`内存变化: ${(after - before) / 1024} KB`);
    
    if (after > before * 1.5) {
        console.warn('可能存在内存泄漏');
    }
}
```

## 6. 内存优化技巧

### 1. 及时解除引用

```javascript
// 及时设置为 null
let largeData = loadData();

// 使用完毕后
largeData = null;
```

### 2. 使用池化技术

```javascript
// 对象池
class ObjectPool {
    constructor(factory, size = 10) {
        this.factory = factory;
        this.pool = [];
        
        for (let i = 0; i < size; i++) {
            this.pool.push(factory());
        }
    }
    
    acquire() {
        return this.pool.length > 0 
            ? this.pool.pop() 
            : this.factory();
    }
    
    release(obj) {
        // 重置对象
        for (const key in obj) {
            delete obj[key];
        }
        this.pool.push(obj);
    }
}

// 使用
const pool = new ObjectPool(() => ({ x: 0, y: 0 }));

const point = pool.acquire();
point.x = 10;
point.y = 20;
// 使用后归还
pool.release(point);
```

### 3. 避免创建不必要的对象

```javascript
// ❌ 在循环中创建对象
function bad() {
    for (let i = 0; i < 1000; i++) {
        const point = { x: i, y: i * 2 }; // 每次创建新对象
        process(point);
    }
}

// ✅ 复用对象
function good() {
    const point = { x: 0, y: 0 };
    for (let i = 0; i < 1000; i++) {
        point.x = i;
        point.y = i * 2;
        process(point);
    }
}
```

### 4. 使用 TypedArray

```javascript
// 处理大量数值数据时使用 TypedArray
const regularArray = new Array(10000).fill(0);     // 普通数组
const typedArray = new Float64Array(10000);        // 类型化数组

// TypedArray 内存占用更小，性能更好
```

## 小结

| 概念 | 说明 |
|------|------|
| **栈内存** | 存储基本类型和调用栈，自动管理 |
| **堆内存** | 存储引用类型，垃圾回收管理 |
| **标记清除** | 从根遍历标记可达对象，清除未标记的 |
| **分代回收** | 新生代（快）+ 老生代（慢）|
| **内存泄漏** | 意外的全局变量、定时器、闭包、DOM引用等 |
| **WeakMap/Set** | 不阻止垃圾回收的集合类型 |
