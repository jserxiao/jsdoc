# 语法增强

## 模板字符串

```javascript
const name = 'John';
const greeting = `Hello, ${name}!`;

// 多行
const html = `
<div>
    <h1>${name}</h1>
</div>
`;

// 标签模板
function tag(strings, ...values) {
    console.log(strings, values);
}
tag`Hello ${name}!`;
```

---

## 解构赋值

```javascript
// 数组解构
const [a, b, ...rest] = [1, 2, 3, 4];
const [first, , third] = [1, 2, 3];

// 对象解构
const { name, age } = { name: 'John', age: 30 };
const { name: userName } = { name: 'John' }; // 重命名

// 默认值
const { x = 10 } = {};

// 函数参数解构
function greet({ name = 'Guest' } = {}) {
    console.log(name);
}
```

---

## 扩展运算符

```javascript
// 数组
const arr = [1, 2, 3];
const copy = [...arr];
const merged = [...arr, 4, 5];

// 对象
const obj = { a: 1 };
const copy = { ...obj, b: 2 };
```
