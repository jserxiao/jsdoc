# ES近年新特性

## ES2020

```javascript
// 可选链
obj?.prop?.nested;
arr?.[0];
func?.();

// 空值合并
const value = null ?? 'default';

// BigInt
const big = 9007199254740991n;

// globalThis
console.log(globalThis);

// Promise.allSettled
Promise.allSettled([p1, p2]);
```

---

## ES2021

```javascript
// 逻辑赋值
a ||= b;  // a = a || b
a &&= b;  // a = a && b
a ??= b;  // a = a ?? b

// 数字分隔符
const num = 1_000_000;

// replaceAll
'aaa'.replaceAll('a', 'b'); // 'bbb'

// Promise.any
Promise.any([p1, p2]);
```

---

## ES2022

```javascript
// 顶层 await
const data = await fetch('/api');

// 私有字段
class Person {
    #name;
    constructor(name) { this.#name = name; }
}

// Object.hasOwn
Object.hasOwn(obj, 'prop');

// Array.at
arr.at(-1); // 最后一个元素
```

---

## ES2023

```javascript
// 从后查找
arr.findLast(x => x > 0);
arr.findLastIndex(x => x > 0);

// 非破坏性方法
arr.toSorted();
arr.toReversed();
arr.toSpliced(1, 1);
arr.with(0, 'new');
```
