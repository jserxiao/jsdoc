# 正则方法

## RegExp 方法

```javascript
// test - 测试是否匹配
/^\d+$/.test('123'); // true

// exec - 执行匹配
const result = /(\d+)/g.exec('a1b2c3');
// ['1', '1', index: 1, input: 'a1b2c3', groups: undefined]

// 全局匹配需要多次调用
const regex = /(\d+)/g;
const str = 'a1b2c3';
let match;
while ((match = regex.exec(str)) !== null) {
    console.log(match[1]); // 1, 2, 3
}

// lastIndex 属性（全局匹配时）
const re = /ab/g;
re.lastIndex = 0;
re.test('abab'); // true
console.log(re.lastIndex); // 2
re.test('abab'); // true
console.log(re.lastIndex); // 4
```

## String 方法

```javascript
// match
'abc123'.match(/\d+/);     // ['123']
'abc123'.match(/\d+/g);    // ['123']
'abc123def456'.match(/\d+/g); // ['123', '456']

// matchAll（ES2020）
const matches = 'a1b2c3'.matchAll(/(\d)/g);
for (const match of matches) {
    console.log(match);
}

// search - 返回匹配位置
'hello world'.search(/world/); // 6

// replace
'hello world'.replace(/world/, 'JS'); // 'hello JS'
'aaa'.replace(/a/g, 'b'); // 'bbb'

// replace 回调
'1 2 3'.replace(/\d+/g, match => match * 2); // '2 4 6'

// replaceAll（ES2021）
'a a a'.replaceAll('a', 'b'); // 'b b b'

// split
'a1b2c3'.split(/\d+/); // ['a', 'b', 'c', '']
```

## 正则属性

```javascript
const regex = /pattern/gi;

console.log(regex.source);  // 'pattern'
console.log(regex.flags);   // 'gi'
console.log(regex.global);  // true
console.log(regex.ignoreCase); // true
console.log(regex.multiline);  // false
console.log(regex.lastIndex);  // 0
```
