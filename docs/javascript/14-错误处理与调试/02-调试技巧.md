# 调试技巧

## console 方法

```javascript
console.log('普通日志');
console.warn('警告');
console.error('错误');
console.info('信息');

// 格式化表格
console.table([{ name: 'John' }, { name: 'Jane' }]);

// 分组
console.group('Group');
console.log('Item 1');
console.log('Item 2');
console.groupEnd();

// 计时
console.time('timer');
// ... 代码
console.timeEnd('timer');

// 堆栈跟踪
console.trace('Trace');

// 断言
console.assert(1 === 2, 'Assertion failed');

// 计数
console.count('click');
```

---

## debugger

```javascript
function debug() {
    debugger; // 断点
    console.log('Debugging');
}
```

## 性能分析

```javascript
// Performance API
performance.mark('start');
// ... 代码
performance.mark('end');
performance.measure('myMeasure', 'start', 'end');

const measures = performance.getEntriesByName('myMeasure');
console.log(measures[0].duration);

// console.time
console.time('operation');
// ... 操作
console.timeEnd('operation');
```
