## 类定义

### 基本结构

```typescript
class User {
  // 属性声明
  name: string;
  age: number;

  // 构造函数
  constructor(name: string, age: number) {
    this.name = name;
    this.age = age;
  }

  // 方法
  greet(): string {
    return `Hello, I'm ${this.name}`;
  }
}

const user = new User('Alice', 25);
console.log(user.greet()); // "Hello, I'm Alice"
```

### 属性初始化

```typescript
class User {
  // 声明时初始化
  name: string = 'Anonymous';
  age: number = 0;

  // 可选属性
  email?: string;

  // 构造函数初始化
  constructor(name?: string) {
    if (name) {
      this.name = name;
    }
  }
}

const user1 = new User();
console.log(user1.name); // 'Anonymous'

const user2 = new User('Bob');
console.log(user2.name); // 'Bob'
```

### 类型断言初始化

```typescript
class User {
  // 使用非空断言（确定会被赋值）
  name!: string;

  initialize(name: string) {
    this.name = name;
  }
}
```

## 属性与方法

### 实例属性

```typescript
class User {
  // 实例属性
  name: string;
  private _age: number;

  constructor(name: string, age: number) {
    this.name = name;
    this._age = age;
  }
}
```

### 实例方法

```typescript
class Calculator {
  // 实例方法
  add(a: number, b: number): number {
    return a + b;
  }

  subtract(a: number, b: number): number {
    return a - b;
  }

  // 方法可以是异步的
  async fetchData(url: string): Promise<Response> {
    return fetch(url);
  }
}

const calc = new Calculator();
calc.add(1, 2); // 3
```

### 箭头函数方法

```typescript
class Counter {
  private count = 0;

  // 普通方法：this 可能丢失
  increment() {
    this.count++;
  }

  // 箭头函数：this 永远绑定到实例
  decrement = () => {
    this.count--;
  };

  getCount() {
    return this.count;
  }
}

const counter = new Counter();
const inc = counter.increment;
const dec = counter.decrement;

// inc(); // Error: this 是 undefined
dec();     // OK: 箭头函数绑定正确
```

## 访问修饰符

### public（默认）

```typescript
class User {
  // public 是默认值
  public name: string;

  constructor(name: string) {
    this.name = name;
  }
}

const user = new User('Alice');
console.log(user.name); // ✅ 可以访问
user.name = 'Bob';      // ✅ 可以修改
```

### private

```typescript
class User {
  private name: string;
  private age: number;

  constructor(name: string, age: number) {
    this.name = name;
    this.age = age;
  }

  // 公共方法访问私有属性
  getName(): string {
    return this.name;
  }

  setAge(age: number): void {
    if (age > 0 && age < 150) {
      this.age = age;
    }
  }

  // 同类实例可以访问私有属性
  isSameAge(other: User): boolean {
    return this.age === other.age; // ✅ 允许
  }
}

const user = new User('Alice', 25);
// console.log(user.name); // ❌ Error: 私有属性
// user.age = 26;          // ❌ Error: 私有属性
console.log(user.getName()); // ✅ 'Alice'
```

### protected

```typescript
class Animal {
  protected name: string;

  constructor(name: string) {
    this.name = name;
  }

  protected makeSound(): void {
    console.log('Some sound');
  }
}

class Dog extends Animal {
  private breed: string;

  constructor(name: string, breed: string) {
    super(name);
    this.breed = breed;
  }

  bark(): void {
    console.log(`${this.name} says: Woof!`); // ✅ 子类可访问
    this.makeSound();                         // ✅ 子类可访问
  }
}

const dog = new Dog('Buddy', 'Golden Retriever');
// console.log(dog.name); // ❌ Error: protected
dog.bark(); // 'Buddy says: Woof!'
```

### 修饰符对比

| 修饰符 | 类内部 | 子类 | 类外部 |
|--------|--------|------|--------|
| `public` | ✅ | ✅ | ✅ |
| `protected` | ✅ | ✅ | ❌ |
| `private` | ✅ | ❌ | ❌ |

## readonly 修饰符

### 基本用法

```typescript
class User {
  readonly id: number;
  name: string;

  constructor(id: number, name: string) {
    this.id = id;
    this.name = name;
  }
}

const user = new User(1, 'Alice');
user.name = 'Bob';   // ✅ 可以修改
// user.id = 2;      // ❌ Error: 只读
```

### 组合修饰符

```typescript
class User {
  // readonly + public
  public readonly id: number;

  // readonly + private
  private readonly createdAt: Date;

  constructor(id: number) {
    this.id = id;
    this.createdAt = new Date();
  }
}
```

### readonly 在参数属性中

```typescript
class Configuration {
  constructor(
    public readonly apiUrl: string,
    private readonly apiKey: string,
    readonly timeout: number = 5000
  ) {}
}

const config = new Configuration('https://api.example.com', 'secret');
console.log(config.apiUrl); // ✅ 可以读取
// config.apiUrl = 'other'; // ❌ Error: 只读
```

## 参数属性

### 简化构造函数

```typescript
// 传统写法
class User1 {
  public name: string;
  private age: number;
  readonly id: number;

  constructor(name: string, age: number, id: number) {
    this.name = name;
    this.age = age;
    this.id = id;
  }
}

// 参数属性写法（更简洁）
class User2 {
  constructor(
    public name: string,
    private age: number,
    readonly id: number
  ) {}
}

const user = new User2('Alice', 25, 1);
console.log(user.name); // 'Alice'
console.log(user.id);   // 1
```

### 参数属性最佳实践

```typescript
class User {
  constructor(
    // 必需参数
    public name: string,
    private age: number,

    // 可选参数
    public email?: string,

    // 默认值参数
    public role: string = 'user',

    // readonly 参数
    readonly id: number = Date.now()
  ) {}
}

const user1 = new User('Alice', 25);
const user2 = new User('Bob', 30, 'bob@example.com', 'admin');
```

## 存取器

### getter 和 setter

```typescript
class User {
  private _age: number = 0;
  private _password: string = '';

  // getter
  get age(): number {
    return this._age;
  }

  // setter
  set age(value: number) {
    if (value >= 0 && value < 150) {
      this._age = value;
    } else {
      throw new Error('Invalid age');
    }
  }

  // 只读属性（只有 getter）
  get isAdult(): boolean {
    return this._age >= 18;
  }

  // 密码设置（只写）
  set password(value: string) {
    if (value.length >= 8) {
      this._password = hashPassword(value);
    }
  }
}

const user = new User();
user.age = 25;           // 通过 setter
console.log(user.age);   // 25，通过 getter
console.log(user.isAdult); // true
user.password = 'secure123'; // 通过 setter
```

### 存取器与惰性计算

```typescript
class User {
  private _fullName: string | null = null;

  constructor(
    public firstName: string,
    public lastName: string
  ) {}

  // 惰性计算
  get fullName(): string {
    if (this._fullName === null) {
      this._fullName = `${this.firstName} ${this.lastName}`;
    }
    return this._fullName;
  }
}

const user = new User('Alice', 'Smith');
console.log(user.fullName); // 首次访问时计算
```

## 静态成员

### 静态属性和方法

```typescript
class User {
  // 静态属性
  static count: number = 0;
  static readonly version: string = '1.0.0';

  // 实例属性
  public name: string;

  constructor(name: string) {
    this.name = name;
    User.count++; // 访问静态属性
  }

  // 静态方法
  static getCount(): number {
    return User.count;
  }

  static createAnonymous(): User {
    return new User('Anonymous');
  }
}

const user1 = new User('Alice');
const user2 = new User('Bob');

console.log(User.getCount()); // 2
console.log(User.version);    // '1.0.0'
console.log(User.createAnonymous().name); // 'Anonymous'
```

### 静态块（ES2022）

```typescript
class Configuration {
  static readonly API_URL: string;
  static readonly TIMEOUT: number;

  static {
    // 初始化静态属性
    this.API_URL = process.env.API_URL || 'https://api.example.com';
    this.TIMEOUT = 5000;
    console.log('Configuration initialized');
  }
}
```

### 单例模式

```typescript
class Singleton {
  private static instance: Singleton;

  private constructor() {
    // 私有构造函数
  }

  static getInstance(): Singleton {
    if (!Singleton.instance) {
      Singleton.instance = new Singleton();
    }
    return Singleton.instance;
  }
}

const instance1 = Singleton.getInstance();
const instance2 = Singleton.getInstance();
console.log(instance1 === instance2); // true
```

## 抽象类

### 定义抽象类

```typescript
abstract class Animal {
  // 抽象属性
  abstract name: string;

  // 抽象方法（必须被子类实现）
  abstract makeSound(): void;

  // 具体方法
  move(distance: number): void {
    console.log(`${this.name} moved ${distance} meters`);
  }
}

class Dog extends Animal {
  name: string;

  constructor(name: string) {
    super();
    this.name = name;
  }

  makeSound(): void {
    console.log('Woof!');
  }
}

class Cat extends Animal {
  name: string;

  constructor(name: string) {
    super();
    this.name = name;
  }

  makeSound(): void {
    console.log('Meow!');
  }
}

// const animal = new Animal(); // ❌ Error: 不能实例化抽象类
const dog = new Dog('Buddy');
dog.makeSound(); // 'Woof!'
dog.move(10);    // 'Buddy moved 10 meters'
```

### 抽象类作为类型

```typescript
abstract class Shape {
  abstract getArea(): number;
  abstract getPerimeter(): number;

  describe(): string {
    return `Area: ${this.getArea()}, Perimeter: ${this.getPerimeter()}`;
  }
}

class Rectangle extends Shape {
  constructor(
    private width: number,
    private height: number
  ) {
    super();
  }

  getArea(): number {
    return this.width * this.height;
  }

  getPerimeter(): number {
    return 2 * (this.width + this.height);
  }
}

function printShapeInfo(shape: Shape): void {
  console.log(shape.describe());
}

printShapeInfo(new Rectangle(5, 3));
```

## 泛型类

### 基本泛型类

```typescript
class Container<T> {
  private value: T;

  constructor(value: T) {
    this.value = value;
  }

  getValue(): T {
    return this.value;
  }

  setValue(value: T): void {
    this.value = value;
  }
}

const stringContainer = new Container<string>('hello');
console.log(stringContainer.getValue()); // 'hello'

const numberContainer = new Container(42); // 类型推断
console.log(numberContainer.getValue());   // 42
```

### 泛型约束

```typescript
interface Lengthwise {
  length: number;
}

class Container<T extends Lengthwise> {
  constructor(private value: T) {}

  getLength(): number {
    return this.value.length;
  }
}

const c1 = new Container('hello');    // ✅ string 有 length
const c2 = new Container([1, 2, 3]);  // ✅ array 有 length
// const c3 = new Container(123);     // ❌ number 没有 length
```

### 多类型参数

```typescript
class Pair<K, V> {
  constructor(
    public key: K,
    public value: V
  ) {}

  toString(): string {
    return `${String(this.key)}: ${String(this.value)}`;
  }
}

const pair = new Pair('name', 'Alice');
console.log(pair.toString()); // 'name: Alice'
```

## this 类型

### 类中的 this

```typescript
class Calculator {
  private value: number = 0;

  add(n: number): this {
    this.value += n;
    return this; // 返回 this 支持链式调用
  }

  subtract(n: number): this {
    this.value -= n;
    return this;
  }

  getValue(): number {
    return this.value;
  }
}

const calc = new Calculator();
const result = calc.add(5).subtract(3).getValue(); // 2
```

### this 与继承

```typescript
class Animal {
  constructor(public name: string) {}

  // this 类型会自动变为子类类型
  clone(): this {
    // 返回相同类型
    return new (this.constructor as any)(this.name);
  }
}

class Dog extends Animal {
  breed: string;

  constructor(name: string, breed: string) {
    super(name);
    this.breed = breed;
  }
}

const dog = new Dog('Buddy', 'Golden Retriever');
const clonedDog = dog.clone(); // 类型是 Dog
console.log(clonedDog.breed);  // 'Golden Retriever'
```

## 实战示例

### 状态机

```typescript
type State = 'idle' | 'loading' | 'success' | 'error';

class StateMachine {
  private _state: State = 'idle';
  private listeners: Set<(state: State) => void> = new Set();

  get state(): State {
    return this._state;
  }

  setState(newState: State): void {
    const oldState = this._state;
    this._state = newState;
    this.listeners.forEach(listener => listener(newState));
  }

  subscribe(listener: (state: State) => void): () => void {
    this.listeners.add(listener);
    return () => this.listeners.delete(listener);
  }

  transition(event: 'start' | 'succeed' | 'fail' | 'reset'): void {
    switch (this._state) {
      case 'idle':
        if (event === 'start') this.setState('loading');
        break;
      case 'loading':
        if (event === 'succeed') this.setState('success');
        if (event === 'fail') this.setState('error');
        break;
      case 'success':
      case 'error':
        if (event === 'reset') this.setState('idle');
        break;
    }
  }
}
```

### 事件发射器

```typescript
type EventHandler<T = any> = (data: T) => void;

class EventEmitter<EventMap extends Record<string, any>> {
  private handlers: Map<keyof EventMap, Set<EventHandler>> = new Map();

  on<K extends keyof EventMap>(
    event: K,
    handler: EventHandler<EventMap[K]>
  ): () => void {
    if (!this.handlers.has(event)) {
      this.handlers.set(event, new Set());
    }
    this.handlers.get(event)!.add(handler);

    // 返回取消订阅函数
    return () => this.off(event, handler);
  }

  off<K extends keyof EventMap>(
    event: K,
    handler: EventHandler<EventMap[K]>
  ): void {
    this.handlers.get(event)?.delete(handler);
  }

  emit<K extends keyof EventMap>(event: K, data: EventMap[K]): void {
    this.handlers.get(event)?.forEach(handler => handler(data));
  }

  once<K extends keyof EventMap>(
    event: K,
    handler: EventHandler<EventMap[K]>
  ): void {
    const wrapper: EventHandler<EventMap[K]> = (data) => {
      this.off(event, wrapper);
      handler(data);
    };
    this.on(event, wrapper);
  }
}

// 使用示例
interface MyEvents {
  login: { userId: string };
  logout: void;
  message: { from: string; content: string };
}

const emitter = new EventEmitter<MyEvents>();

emitter.on('login', ({ userId }) => {
  console.log(`User ${userId} logged in`);
});

emitter.emit('login', { userId: '123' });
```

## 最佳实践

### 类设计原则

```typescript
// ✅ 好：单一职责
class UserValidator {
  validate(user: User): boolean { /* ... */ }
}

class UserRepository {
  save(user: User): void { /* ... */ }
}

// ❌ 差：职责过多
class UserService {
  validate() { /* ... */ }
  save() { /* ... */ }
  sendEmail() { /* ... */ }
  logActivity() { /* ... */ }
}
```

### 使用接口定义契约

```typescript
// 定义接口
interface ILogger {
  log(message: string): void;
}

// 实现接口
class ConsoleLogger implements ILogger {
  log(message: string): void {
    console.log(message);
  }
}

class FileLogger implements ILogger {
  constructor(private filename: string) {}

  log(message: string): void {
    // 写入文件
  }
}

// 依赖接口
class Application {
  constructor(private logger: ILogger) {}

  run(): void {
    this.logger.log('Application started');
  }
}
```

### 避免过度使用 private

```typescript
// ❌ 过度私有化
class User {
  private firstName: string;
  private lastName: string;

  getFirstName() { return this.firstName; }
  getLastName() { return this.lastName; }
  setFirstName(name: string) { this.firstName = name; }
  setLastName(name: string) { this.lastName = name; }
}

// ✅ 合理暴露
class User {
  constructor(
    public firstName: string,
    public lastName: string
  ) {}

  get fullName(): string {
    return `${this.firstName} ${this.lastName}`;
  }
}
```

## 小结

| 特性 | 说明 |
|------|------|
| `public` | 默认，公开访问 |
| `private` | 仅类内部可访问 |
| `protected` | 类和子类可访问 |
| `readonly` | 只读属性 |
| `static` | 静态成员 |
| `abstract` | 抽象类/方法 |
| `get/set` | 存取器 |
