## 基本接口实现

使用 `implements` 关键字实现接口。

```typescript
interface Printable {
  print(): void;
}

class Document implements Printable {
  print(): void {
    console.log('Printing document...');
  }
}

const doc = new Document();
doc.print(); // 'Printing document...'
```

### 属性接口

```typescript
interface User {
  id: string;
  name: string;
  email: string;
}

class Person implements User {
  // 必须实现所有属性
  id: string;
  name: string;
  email: string;

  constructor(id: string, name: string, email: string) {
    this.id = id;
    this.name = name;
    this.email = email;
  }
}

// 使用参数属性简化
class Person2 implements User {
  constructor(
    public id: string,
    public name: string,
    public email: string
  ) {}
}
```

### 可选属性

```typescript
interface Config {
  apiUrl: string;
  timeout?: number; // 可选属性
}

class ApiClient implements Config {
  apiUrl: string;
  timeout: number = 5000; // 可以提供默认值

  constructor(apiUrl: string, timeout?: number) {
    this.apiUrl = apiUrl;
    if (timeout !== undefined) {
      this.timeout = timeout;
    }
  }
}
```

## 实现多个接口

一个类可以实现多个接口，用逗号分隔。

```typescript
interface Printable {
  print(): void;
}

interface Loggable {
  log(): void;
}

interface Serializable {
  serialize(): string;
}

class Document implements Printable, Loggable, Serializable {
  constructor(public content: string) {}

  print(): void {
    console.log(`Printing: ${this.content}`);
  }

  log(): void {
    console.log(`Logging: ${this.content}`);
  }

  serialize(): string {
    return JSON.stringify({ content: this.content });
  }
}

const doc = new Document('Hello, TypeScript!');
doc.print();     // 'Printing: Hello, TypeScript!'
doc.log();       // 'Logging: Hello, TypeScript!'
doc.serialize(); // '{"content":"Hello, TypeScript!"}'
```

### 接口冲突处理

当多个接口有同名但类型不同的属性时，需要手动处理冲突：

```typescript
interface A {
  value: string;
}

interface B {
  value: number;
}

// ❌ Error: 类型冲突
// class C implements A, B {
//   value: string; // 不能同时满足 string 和 number
// }

// ✅ 方案1：使用联合类型
class C1 implements A, B {
  value: string | number = ''; // 需要满足两个接口
}

// ✅ 方案2：分开实现
class C2 implements A {
  value: string = '';

  asB(): B {
    return { value: Number(this.value) };
  }
}
```

## 接口继承类

接口可以继承类，这会继承类的成员但不包括其实现。

```typescript
class Control {
  private state: any;

  protected setState(state: any) {
    this.state = state;
  }

  public getState() {
    return this.state;
  }
}

// 接口继承类
interface SelectableControl extends Control {
  select(): void;
}

// 只有 Control 的子类才能实现此接口
class Button extends Control implements SelectableControl {
  select(): void {
    console.log('Button selected');
  }
}

class TextBox extends Control implements SelectableControl {
  select(): void {
    console.log('Text selected');
  }
}

// ❌ Error: ImageControl 没有 state 私有成员
// class ImageControl implements SelectableControl {
//   select(): void {}
// }
```

## 类类型接口

接口可以描述类的构造函数签名。

### 构造函数接口

```typescript
// 描述类的构造函数
interface IConstructor<T> {
  new (...args: any[]): T;
}

// 描述类本身
interface IClass {
  new (name: string): IInstance;
  staticMethod(): void;
}

interface IInstance {
  name: string;
  instanceMethod(): void;
}

class MyClass implements IInstance {
  constructor(public name: string) {}

  instanceMethod(): void {
    console.log(`Instance: ${this.name}`);
  }

  static staticMethod(): void {
    console.log('Static method');
  }
}

// 使用构造函数接口
function createInstance<T>(ctor: IConstructor<T>, ...args: any[]): T {
  return new ctor(...args);
}

const instance = createInstance(MyClass, 'Test');
console.log(instance.name); // 'Test'
```

### 工厂模式

```typescript
interface Animal {
  name: string;
  makeSound(): void;
}

interface AnimalConstructor {
  new (name: string): Animal;
}

class Dog implements Animal {
  constructor(public name: string) {}

  makeSound(): void {
    console.log('Woof!');
  }
}

class Cat implements Animal {
  constructor(public name: string) {}

  makeSound(): void {
    console.log('Meow!');
  }
}

// 工厂函数
function createAnimal(ctor: AnimalConstructor, name: string): Animal {
  return new ctor(name);
}

const dog = createAnimal(Dog, 'Rex');
const cat = createAnimal(Cat, 'Whiskers');

dog.makeSound(); // 'Woof!'
cat.makeSound(); // 'Meow!'
```

### 依赖注入

```typescript
interface ILogger {
  log(message: string): void;
}

interface ILoggerConstructor {
  new (): ILogger;
}

class ConsoleLogger implements ILogger {
  log(message: string): void {
    console.log(message);
  }
}

class FileLogger implements ILogger {
  log(message: string): void {
    // 写入文件
    console.log(`[FILE] ${message}`);
  }
}

class Application {
  private logger: ILogger;

  constructor(LoggerClass: ILoggerConstructor) {
    this.logger = new LoggerClass();
  }

  run(): void {
    this.logger.log('Application started');
  }
}

// 使用不同的日志实现
const app1 = new Application(ConsoleLogger);
const app2 = new Application(FileLogger);
```

## 接口与抽象类对比

### 选择指南

```typescript
// ✅ 使用接口：定义能力/行为契约
interface Flyable {
  fly(): void;
}

interface Swimmable {
  swim(): void;
}

class Duck implements Flyable, Swimmable {
  fly(): void {
    console.log('Duck flying');
  }

  swim(): void {
    console.log('Duck swimming');
  }
}

// ✅ 使用抽象类：共享实现
abstract class Animal {
  constructor(public name: string) {}

  // 共享实现
  breathe(): void {
    console.log(`${this.name} is breathing`);
  }

  // 抽象方法
  abstract makeSound(): void;
}

class Dog extends Animal {
  constructor(name: string) {
    super(name);
  }

  makeSound(): void {
    console.log('Woof!');
  }
}
```

### 混合使用

```typescript
// 抽象类提供基础实现
abstract class Vehicle {
  constructor(
    public brand: string,
    public model: string
  ) {}

  getInfo(): string {
    return `${this.brand} ${this.model}`;
  }

  abstract start(): void;
}

// 接口定义额外能力
interface Electric {
  charge(): void;
  getBatteryLevel(): number;
}

interface Autonomous {
  enableAutoPilot(): void;
  disableAutoPilot(): void;
}

// 继承抽象类并实现多个接口
class TeslaModelS extends Vehicle implements Electric, Autonomous {
  private batteryLevel: number = 100;
  private autoPilotEnabled: boolean = false;

  start(): void {
    console.log(`${this.getInfo()} started silently`);
  }

  charge(): void {
    console.log('Charging...');
    this.batteryLevel = 100;
  }

  getBatteryLevel(): number {
    return this.batteryLevel;
  }

  enableAutoPilot(): void {
    this.autoPilotEnabled = true;
    console.log('AutoPilot enabled');
  }

  disableAutoPilot(): void {
    this.autoPilotEnabled = false;
    console.log('AutoPilot disabled');
  }
}

const car = new TeslaModelS('Tesla', 'Model S');
car.start();             // 'Tesla Model S started silently'
car.charge();            // 'Charging...'
car.enableAutoPilot();   // 'AutoPilot enabled'
```

## 最佳实践

### 1. 接口优先原则

```typescript
// ✅ 推荐：面向接口编程
interface Repository<T> {
  findById(id: string): Promise<T | null>;
  findAll(): Promise<T[]>;
  save(entity: T): Promise<void>;
  delete(id: string): Promise<void>;
}

class UserRepository implements Repository<User> {
  async findById(id: string): Promise<User | null> {
    // 实现细节
    return null;
  }

  async findAll(): Promise<User[]> {
    return [];
  }

  async save(entity: User): Promise<void> {
    // 实现细节
  }

  async delete(id: string): Promise<void> {
    // 实现细节
  }
}
```

### 2. 小接口原则

```typescript
// ❌ 不推荐：大接口
interface Worker {
  work(): void;
  eat(): void;
  sleep(): void;
  code(): void;
  design(): void;
  test(): void;
}

// ✅ 推荐：小接口
interface Workable {
  work(): void;
}

interface Eatable {
  eat(): void;
}

interface Sleepable {
  sleep(): void;
}

interface Coder {
  code(): void;
}

interface Designer {
  design(): void;
}

interface Tester {
  test(): void;
}

class Developer implements Workable, Eatable, Sleepable, Coder, Tester {
  work(): void { console.log('Working'); }
  eat(): void { console.log('Eating'); }
  sleep(): void { console.log('Sleeping'); }
  code(): void { console.log('Coding'); }
  test(): void { console.log('Testing'); }
}
```
