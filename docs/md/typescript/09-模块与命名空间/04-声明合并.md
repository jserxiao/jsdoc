## 接口合并

### 基本合并

```typescript
interface User {
  name: string;
}

interface User {
  age: number;
}

// 合并结果
const user: User = {
  name: 'Alice',
  age: 30
};
```

### 方法合并

```typescript
interface Document {
  createElement(tag: 'div'): HTMLDivElement;
  createElement(tag: 'span'): HTMLSpanElement;
}

interface Document {
  createElement(tag: string): HTMLElement;
}

// 合并后拥有所有重载
document.createElement('div');    // HTMLDivElement
document.createElement('span');   // HTMLSpanElement
document.createElement('custom'); // HTMLElement
```

### 非函数成员

```typescript
interface Config {
  size: number;
}

interface Config {
  size: string;  // Error: 后续声明必须有相同类型
}
```

## 命名空间合并

### 基本合并

```typescript
namespace Animal {
  export class Dog {}
}

namespace Animal {
  export class Cat {}
}

// 合并结果
const dog = new Animal.Dog();
const cat = new Animal.Cat();
```

### 跨文件合并

```typescript
// animal.ts
namespace Animal {
  export interface Dog {
    bark(): void;
  }
}

// animal-extended.ts
namespace Animal {
  export interface Cat {
    meow(): void;
  }
}
```

## 函数与命名空间合并

```typescript
function User(name: string) {
  return { name };
}

namespace User {
  export const defaultName = 'Anonymous';
  export function create() {
    return User(User.defaultName);
  }
}

const user = User('Alice');
const anonymous = User.create();
```

## 类与命名空间合并

```typescript
class Album {
  label: Album.AlbumLabel;

  constructor(public name: string) {}
}

namespace Album {
  export class AlbumLabel {
    constructor(public name: string) {}
  }

  export function create(name: string) {
    return new Album(name);
  }
}

const album = Album.create('Thriller');
```

## 扩展内置类型

### 扩展 Window

```typescript
interface Window {
  myCustomAPI: {
    init(): void;
    version: string;
  };
}

// 使用
window.myCustomAPI.init();
```

### 扩展 Array

```typescript
interface Array<T> {
  first(): T | undefined;
  last(): T | undefined;
}

// 需要实现
Array.prototype.first = function() {
  return this[0];
};

[1, 2, 3].first();  // 1
```

### 扩展模块

```typescript
// 扩展 Express
declare module 'express' {
  interface Request {
    user?: {
      id: string;
      name: string;
    };
  }
}
```
