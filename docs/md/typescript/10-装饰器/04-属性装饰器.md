# 属性装饰器

属性装饰器应用于类的属性，可以修改属性的行为。

## 基本语法

```typescript
function logProperty(target: any, propertyKey: string) {
  console.log(`Property: ${propertyKey}`);
  console.log(`Target:`, target);
}

class User {
  @logProperty
  name: string;

  constructor(name: string) {
    this.name = name;
  }
}
```

## 实际应用

### 只读属性

```typescript
function readonly(target: any, propertyKey: string) {
  Object.defineProperty(target, propertyKey, {
    writable: false
  });
}

class Config {
  @readonly
  apiUrl = 'https://api.example.com';
}

const config = new Config();
// config.apiUrl = 'new';  // Error
```

### 大写转换

```typescript
function uppercase(target: any, propertyKey: string) {
  let value: string;

  Object.defineProperty(target, propertyKey, {
    get() {
      return value;
    },
    set(newValue: string) {
      value = newValue.toUpperCase();
    }
  });
}

class User {
  @uppercase
  name: string = '';
}

const user = new User();
user.name = 'alice';
console.log(user.name);  // 'ALICE'
```

### 范围验证

```typescript
function range(min: number, max: number) {
  return function(target: any, propertyKey: string) {
    let value: number;

    Object.defineProperty(target, propertyKey, {
      get() {
        return value;
      },
      set(newValue: number) {
        if (newValue < min || newValue > max) {
          throw new Error(`${propertyKey} must be between ${min} and ${max}`);
        }
        value = newValue;
      }
    });
  };
}

class Product {
  @range(0, 100)
  stock: number = 0;
}

const product = new Product();
product.stock = 50;  // OK
// product.stock = 150;  // Error
```

### 必填验证

```typescript
function required(target: any, propertyKey: string) {
  let value: any;

  Object.defineProperty(target, propertyKey, {
    get() {
      return value;
    },
    set(newValue: any) {
      if (newValue === undefined || newValue === null) {
        throw new Error(`${propertyKey} is required`);
      }
      value = newValue;
    }
  });
}

class User {
  @required
  name: string = '';
}

const user = new User();
user.name = 'Alice';  // OK
// user.name = null;  // Error
```

### 格式化输出

```typescript
function format(formatter: (value: any) => string) {
  return function(target: any, propertyKey: string) {
    let value: any;

    Object.defineProperty(target, propertyKey, {
      get() {
        return value;
      },
      set(newValue: any) {
        value = newValue;
      }
    });

    // 添加格式化方法
    const formatMethod = `getFormatted${propertyKey.charAt(0).toUpperCase()}${propertyKey.slice(1)}`;
    target[formatMethod] = function() {
      return formatter(value);
    };
  };
}

class User {
  @format(value => `$${value.toFixed(2)}`)
  price: number = 0;
}

const product = new User() as any;
product.price = 99.99;
console.log(product.getFormattedPrice());  // '$99.99'
```
