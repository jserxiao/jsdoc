# 参数装饰器

参数装饰器应用于函数参数，可以记录或修改参数。

## 基本语法

```typescript
function logParameter(
  target: any,
  propertyKey: string,
  parameterIndex: number
) {
  console.log(`Parameter ${parameterIndex} of ${propertyKey}`);
}

class Example {
  method(@logParameter name: string, @logParameter age: number) {}
}
// Parameter 1 of method
// Parameter 0 of method
```

## 实际应用

### 参数验证

```typescript
function validate(
  target: any,
  propertyKey: string,
  parameterIndex: number
) {
  const existingValidators = (target as any).__validators || {};
  const methodValidators = existingValidators[propertyKey] || [];
  methodValidators.push(parameterIndex);
  existingValidators[propertyKey] = methodValidators;
  (target as any).__validators = existingValidators;
}

function validateMethod(
  target: any,
  propertyKey: string,
  descriptor: PropertyDescriptor
) {
  const originalMethod = descriptor.value;
  const validators = (target as any).__validators?.[propertyKey] || [];

  descriptor.value = function(...args: any[]) {
    for (const index of validators) {
      if (args[index] === undefined || args[index] === null) {
        throw new Error(`Parameter at index ${index} is required`);
      }
    }
    return originalMethod.apply(this, args);
  };

  return descriptor;
}

class UserService {
  @validateMethod
  createUser(@validate name: string, @validate email: string) {
    return { name, email };
  }
}
```

### 参数日志

```typescript
const loggedParams = new Map<string, number[]>();

function logParam(
  target: any,
  propertyKey: string,
  parameterIndex: number
) {
  const key = `${target.constructor.name}.${propertyKey}`;
  if (!loggedParams.has(key)) {
    loggedParams.set(key, []);
  }
  loggedParams.get(key)!.push(parameterIndex);
}

function logMethod(
  target: any,
  propertyKey: string,
  descriptor: PropertyDescriptor
) {
  const originalMethod = descriptor.value;
  const key = `${target.constructor.name}.${propertyKey}`;
  const paramsToLog = loggedParams.get(key) || [];

  descriptor.value = function(...args: any[]) {
    paramsToLog.forEach(index => {
      console.log(`Parameter ${index}:`, args[index]);
    });
    return originalMethod.apply(this, args);
  };

  return descriptor;
}

class Calculator {
  @logMethod
  add(@logParam a: number, @logParam b: number) {
    return a + b;
  }
}

const calc = new Calculator();
calc.add(1, 2);
// Parameter 0: 1
// Parameter 1: 2
```

### 依赖注入标记

```typescript
import 'reflect-metadata';

function Inject(token?: string) {
  return function(
    target: any,
    propertyKey: string | undefined,
    parameterIndex: number
  ) {
    const designParamTypes = Reflect.getMetadata('design:paramtypes', target) || [];
    const existingTokens = Reflect.getMetadata('inject:tokens', target) || [];
    existingTokens[parameterIndex] = token || designParamTypes[parameterIndex].name;
    Reflect.defineMetadata('inject:tokens', existingTokens, target);
  };
}

class Database {}
class Logger {}

class UserService {
  constructor(
    @Inject() private db: Database,
    @Inject('Logger') private logger: Logger
  ) {}
}
```
