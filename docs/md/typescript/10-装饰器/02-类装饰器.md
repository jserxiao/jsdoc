# 类装饰器

类装饰器应用于类构造函数，可以修改或替换类定义。

## 基本语法

```typescript
function sealed(constructor: Function) {
  Object.seal(constructor);
  Object.seal(constructor.prototype);
}

@sealed
class Greeter {
  greeting: string;
  constructor(message: string) {
    this.greeting = message;
  }
}
```

## 修改类

### 添加属性

```typescript
function addProperty(target: any) {
  target.prototype.createdAt = new Date();
}

@addProperty
class User {
  name: string;
  
  constructor(name: string) {
    this.name = name;
  }
}

const user = new User('Alice');
console.log((user as any).createdAt);  // Date
```

### 添加方法

```typescript
function addMethod(target: any) {
  target.prototype.getInfo = function() {
    return JSON.stringify(this);
  };
}

@addMethod
class User {
  constructor(public name: string, public age: number) {}
}

const user = new User('Alice', 30);
console.log((user as any).getInfo());  // '{"name":"Alice","age":30}'
```

## 替换类

```typescript
function replaceClass<T extends { new (...args: any[]): {} }>(
  constructor: T
) {
  return class extends constructor {
    extraProperty = 'extra';
    
    extraMethod() {
      return 'extra method';
    }
  };
}

@replaceClass
class User {
  constructor(public name: string) {}
}

const user = new User('Alice') as any;
console.log(user.name);           // 'Alice'
console.log(user.extraProperty);  // 'extra'
console.log(user.extraMethod());  // 'extra method'
```

## 实际应用

### 单例模式

```typescript
function Singleton<T extends { new (...args: any[]): {} }>(constructor: T) {
  let instance: any;
  
  return class extends constructor {
    constructor(...args: any[]) {
      if (instance) {
        return instance;
      }
      super(...args);
      instance = this;
    }
  };
}

@Singleton
class Database {
  constructor(public connectionString: string) {}
}

const db1 = new Database('conn1');
const db2 = new Database('conn2');

console.log(db1 === db2);  // true
console.log(db1.connectionString);  // 'conn1'
```

### 日志装饰器

```typescript
function LogClass(target: any) {
  const original = target;
  
  // 创建新构造函数
  const newConstructor: any = function(...args: any[]) {
    console.log(`Creating instance of ${original.name}`);
    return new original(...args);
  };
  
  newConstructor.prototype = original.prototype;
  return newConstructor;
}

@LogClass
class User {
  constructor(public name: string) {}
}

const user = new User('Alice');  // 'Creating instance of User'
```

### 冻结类

```typescript
function Frozen(target: any) {
  Object.freeze(target);
  Object.freeze(target.prototype);
}

@Frozen
class Config {
  static apiUrl = 'https://api.example.com';
}

// Config.apiUrl = 'new';  // Error: Cannot assign
```
