# 类型守卫

类型守卫（Type Guard）是 TypeScript 中用于在运行时检查类型，从而在编译时缩小类型范围的表达式。通过类型守卫，可以在特定代码块中获得更精确的类型信息。

## typeof 守卫

### 基本用法

```typescript
function process(value: string | number) {
  if (typeof value === 'string') {
    // 这里 value 被缩小为 string
    return value.toUpperCase();
  } else {
    // 这里 value 被缩小为 number
    return value * 2;
  }
}
```

### typeof 支持的类型

```typescript
function checkType(value: unknown) {
  if (typeof value === 'string') {
    return 'string';
  }
  if (typeof value === 'number') {
    return 'number';
  }
  if (typeof value === 'boolean') {
    return 'boolean';
  }
  if (typeof value === 'function') {
    return 'function';
  }
  if (typeof value === 'object') {
    // 注意: null 的 typeof 也是 'object'
    if (value === null) {
      return 'null';
    }
    if (Array.isArray(value)) {
      return 'array';
    }
    return 'object';
  }
  if (typeof value === 'undefined') {
    return 'undefined';
  }
  if (typeof value === 'symbol') {
    return 'symbol';
  }
  if (typeof value === 'bigint') {
    return 'bigint';
  }
}
```

### typeof 与联合类型

```typescript
function formatValue(value: string | number | boolean) {
  switch (typeof value) {
    case 'string':
      return `"${value}"`;
    case 'number':
      return value.toFixed(2);
    case 'boolean':
      return value ? '是' : '否';
  }
}

// 使用 typeof 检查函数
function callIfFunction(value: string | (() => void)) {
  if (typeof value === 'function') {
    value();
  } else {
    console.log(value);
  }
}
```

## instanceof 守卫

### 基本用法

```typescript
function process(value: Date | string) {
  if (value instanceof Date) {
    // value: Date
    return value.toISOString();
  }
  // value: string
  return value.toUpperCase();
}
```

### 自定义类检查

```typescript
class ValidationError extends Error {
  constructor(message: string, public field: string) {
    super(message);
  }
}

class NetworkError extends Error {
  constructor(message: string, public statusCode: number) {
    super(message);
  }
}

function handleError(error: ValidationError | NetworkError) {
  if (error instanceof ValidationError) {
    // error: ValidationError
    console.log(`字段 ${error.field} 验证失败`);
  } else {
    // error: NetworkError
    console.log(`HTTP ${error.statusCode} 错误`);
  }
}
```

### 数组检查

```typescript
function process(value: string | string[]) {
  if (value instanceof Array) {
    // value: string[]
    return value.join(', ');
  }
  // value: string
  return value;
}

// 更推荐使用 Array.isArray
function processSafe(value: string | string[]) {
  if (Array.isArray(value)) {
    return value.join(', ');
  }
  return value;
}
```

### instanceof 的局限性

```typescript
// instanceof 对原始类型无效
const num = 42;
console.log(num instanceof Number); // false

// 使用 typeof 检查原始类型
console.log(typeof num === 'number'); // true

// 跨 iframe 或跨 realm 时 instanceof 可能失败
// 使用 Array.isArray 替代 instanceof Array
```

## in 操作符守卫

### 基本用法

```typescript
interface User {
  name: string;
  email: string;
}

interface Admin {
  name: string;
  permissions: string[];
}

function process(user: User | Admin) {
  if ('email' in user) {
    // user: User
    console.log(user.email);
  } else {
    // user: Admin
    console.log(user.permissions);
  }
}
```

### 检查可选属性

```typescript
interface Config {
  apiUrl: string;
  timeout?: number;
}

function setup(config: Config) {
  console.log(config.apiUrl);

  if ('timeout' in config) {
    // timeout 存在且不为 undefined
    console.log(`Timeout: ${config.timeout}ms`);
  }
}

// 注意: in 操作符不区分 undefined 和不存在
interface Item {
  value?: string;
}

const item: Item = { value: undefined };
console.log('value' in item); // true
```

### 方法存在检查

```typescript
interface SyncStorage {
  getItem(key: string): string | null;
}

interface AsyncStorage {
  getItem(key: string): Promise<string | null>;
}

function get(storage: SyncStorage | AsyncStorage, key: string) {
  const result = storage.getItem(key);

  // 检查返回值是否是 Promise
  if (result instanceof Promise) {
    return result;
  }

  return Promise.resolve(result);
}
```

## 自定义类型守卫

### 类型谓词语法

```typescript
function isType(value: any): value is Type {
  // 返回 boolean
  // 当返回 true 时，TypeScript 将 value 缩小为 Type
}
```

### 基本示例

```typescript
interface User {
  name: string;
  age: number;
}

function isUser(value: unknown): value is User {
  return (
    typeof value === 'object' &&
    value !== null &&
    'name' in value &&
    'age' in value &&
    typeof (value as User).name === 'string' &&
    typeof (value as User).age === 'number'
  );
}

function process(value: unknown) {
  if (isUser(value)) {
    // value: User
    console.log(value.name, value.age);
  } else {
    console.log('不是有效的 User');
  }
}
```

### 复杂类型守卫

```typescript
interface Success<T> {
  success: true;
  data: T;
}

interface Failure {
  success: false;
  error: string;
}

type Result<T> = Success<T> | Failure;

function isSuccess<T>(result: Result<T>): result is Success<T> {
  return result.success === true;
}

function handleResult<T>(result: Result<T>) {
  if (isSuccess(result)) {
    // result: Success<T>
    console.log('Data:', result.data);
  } else {
    // result: Failure
    console.error('Error:', result.error);
  }
}
```

### 断言函数

```typescript
// 断言函数：如果检查失败则抛出错误
function assertIsString(value: unknown): asserts value is string {
  if (typeof value !== 'string') {
    throw new TypeError('Expected a string');
  }
}

function process(value: unknown) {
  assertIsString(value);
  // 这里 value 是 string
  console.log(value.toUpperCase());
}

// 断言条件
function assert(condition: boolean, message: string): asserts condition {
  if (!condition) {
    throw new Error(message);
  }
}

function divide(a: number, b: number): number {
  assert(b !== 0, '除数不能为零');
  return a / b;
}
```

### 泛型类型守卫

```typescript
function isDefined<T>(value: T | null | undefined): value is T {
  return value !== null && value !== undefined;
}

function filterDefined<T>(items: (T | null | undefined)[]): T[] {
  return items.filter(isDefined);
}

const result = filterDefined([1, null, 2, undefined, 3]);
// result: number[]

// 检查数组元素类型
function isStringArray(value: unknown): value is string[] {
  return Array.isArray(value) && value.every(item => typeof item === 'string');
}
```

## 字面量守卫

### 等值检查

```typescript
type Direction = 'up' | 'down' | 'left' | 'right';

function move(dir: Direction) {
  switch (dir) {
    case 'up':
      // dir: 'up'
      console.log('向上移动');
      break;
    case 'down':
      // dir: 'down'
      console.log('向下移动');
      break;
    case 'left':
      // dir: 'left'
      console.log('向左移动');
      break;
    case 'right':
      // dir: 'right'
      console.log('向右移动');
      break;
  }
}
```

### switch 穷尽性检查

```typescript
interface Circle {
  kind: 'circle';
  radius: number;
}

interface Square {
  kind: 'square';
  sideLength: number;
}

interface Triangle {
  kind: 'triangle';
  base: number;
  height: number;
}

type Shape = Circle | Square | Triangle;

function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2;
    case 'square':
      return shape.sideLength ** 2;
    case 'triangle':
      return (shape.base * shape.height) / 2;
    default:
      // 如果所有情况都处理了，shape 是 never
      const _exhaustiveCheck: never = shape;
      throw new Error(`未知的形状: ${_exhaustiveCheck}`);
  }
}
```

## 真值守卫

### 基本用法

```typescript
function process(value: string | null | undefined) {
  if (value) {
    // value: string（排除了 null、undefined、空字符串）
    console.log(value.toUpperCase());
  }
}

function greet(name: string | null) {
  // 使用 ?? 提供默认值
  const displayName = name ?? '访客';
  console.log(`你好，${displayName}！`);
}
```

### 假值列表

```typescript
// JavaScript 中的假值
const falsyValues = [
  false,
  0,
  -0,
  0n,
  '',
  null,
  undefined,
  NaN
];

// 使用真值检查时要注意
function process(value: number | null) {
  if (value) {
    // 这里排除了 0、null、undefined
    console.log(value);
  }
}

// 更精确的检查
function processNumber(value: number | null) {
  if (value !== null) {
    // 这里只排除了 null
    console.log(value); // 包括 0
  }
}
```

## 等值守卫

### 严格相等

```typescript
function process(value: string | number | null) {
  if (value === null) {
    return '空值';
  }

  // value: string | number
  if (value === 'default') {
    // value: 'default'
    return '默认值';
  }

  // value: number | 其他 string
  return value;
}
```

### 松散相等检查 null 和 undefined

```typescript
function process(value: string | null | undefined) {
  // == null 同时检查 null 和 undefined
  if (value == null) {
    return '空值';
  }

  // value: string
  return value.toUpperCase();
}
```

## 类型守卫组合

### && 和 || 组合

```typescript
interface User {
  name: string;
  email: string;
}

interface Admin extends User {
  permissions: string[];
}

function canAccess(user: User | Admin | null) {
  // 组合多个检查
  if (user && 'permissions' in user && user.permissions.includes('admin')) {
    // user: Admin 且有 admin 权限
    return true;
  }
  return false;
}
```

### 使用类型谓词组合

```typescript
function isUser(value: unknown): value is User {
  return (
    typeof value === 'object' &&
    value !== null &&
    'name' in value &&
    'email' in value
  );
}

function isAdmin(value: unknown): value is Admin {
  return isUser(value) && 'permissions' in value;
}

function process(value: unknown) {
  if (isAdmin(value)) {
    // value: Admin
    console.log(value.permissions);
  } else if (isUser(value)) {
    // value: User
    console.log(value.email);
  }
}
```

## 实战示例

### API 响应处理

```typescript
interface ApiSuccess<T> {
  status: 'success';
  data: T;
}

interface ApiError {
  status: 'error';
  error: {
    code: number;
    message: string;
  };
}

type ApiResponse<T> = ApiSuccess<T> | ApiError;

function isSuccess<T>(response: ApiResponse<T>): response is ApiSuccess<T> {
  return response.status === 'success';
}

async function fetchData<T>(url: string): Promise<T> {
  const response: ApiResponse<T> = await fetch(url).then(r => r.json());

  if (isSuccess(response)) {
    return response.data;
  }

  throw new Error(`[${response.error.code}] ${response.error.message}`);
}

// 使用
interface User {
  id: number;
  name: string;
}

const user = await fetchData<User>('/api/users/1');
console.log(user.name);
```

### 事件处理

```typescript
interface MouseEvent {
  type: 'click' | 'mousemove';
  x: number;
  y: number;
}

interface KeyboardEvent {
  type: 'keydown' | 'keyup';
  key: string;
  code: string;
}

type AppEvent = MouseEvent | KeyboardEvent;

function isMouseEvent(event: AppEvent): event is MouseEvent {
  return event.type === 'click' || event.type === 'mousemove';
}

function handleEvent(event: AppEvent) {
  if (isMouseEvent(event)) {
    console.log(`鼠标位置: (${event.x}, ${event.y})`);
  } else {
    console.log(`按键: ${event.key}`);
  }
}
```

### 表单验证

```typescript
interface FormData {
  username: string;
  password: string;
  email?: string;
}

interface ValidationResult {
  valid: boolean;
  errors: string[];
}

function isValidFormData(data: unknown): data is FormData {
  if (typeof data !== 'object' || data === null) {
    return false;
  }

  const obj = data as Record<string, unknown>;

  return (
    typeof obj.username === 'string' &&
    obj.username.length >= 3 &&
    typeof obj.password === 'string' &&
    obj.password.length >= 6 &&
    (obj.email === undefined || typeof obj.email === 'string')
  );
}

function processForm(data: unknown): ValidationResult {
  if (!isValidFormData(data)) {
    return {
      valid: false,
      errors: ['无效的表单数据']
    };
  }

  // data: FormData
  const errors: string[] = [];

  if (data.email && !data.email.includes('@')) {
    errors.push('邮箱格式不正确');
  }

  return {
    valid: errors.length === 0,
    errors
  };
}
```

## 最佳实践

### 优先使用内置守卫

```typescript
// ✅ 好：使用 typeof
function process(value: string | number) {
  if (typeof value === 'string') {
    return value.toUpperCase();
  }
  return value.toFixed(2);
}

// ✅ 好：使用 Array.isArray
function processArray(value: string | string[]) {
  if (Array.isArray(value)) {
    return value.join(', ');
  }
  return value;
}

// ❌ 避免：不必要的自定义守卫
function isString(value: unknown): value is string {
  return typeof value === 'string';
}
```

### 类型守卫要有意义

```typescript
// ✅ 好：有意义的类型守卫
function isAdmin(user: User | Admin): user is Admin {
  return 'permissions' in user;
}

// ❌ 差：过于简单，不如直接用 typeof
function isNumber(value: unknown): value is number {
  return typeof value === 'number';
}
```

### 断言函数与类型谓词的选择

```typescript
// 类型谓词：不抛出错误，返回 boolean
function isUser(value: unknown): value is User {
  return typeof value === 'object' && value !== null && 'name' in value;
}

// 断言函数：失败时抛出错误
function assertIsUser(value: unknown): asserts value is User {
  if (!isUser(value)) {
    throw new TypeError('Expected User');
  }
}

// 使用场景
function processConditional(value: unknown) {
  if (isUser(value)) {
    console.log(value.name);
  }
}

function processRequired(value: unknown) {
  assertIsUser(value); // 如果不是 User 会抛出错误
  console.log(value.name);
}
```

## 小结

| 守卫类型 | 用途 | 示例 |
|---------|------|------|
| `typeof` | 检查原始类型 | `typeof x === 'string'` |
| `instanceof` | 检查类实例 | `x instanceof Date` |
| `in` | 检查属性存在 | `'name' in obj` |
| 类型谓词 | 自定义检查 | `x is Type` |
| 断言函数 | 断言并可能抛错 | `asserts x is Type` |
| 字面量守卫 | 检查具体值 | `x === 'value'` |
| 真值守卫 | 排除假值 | `if (x)` |
