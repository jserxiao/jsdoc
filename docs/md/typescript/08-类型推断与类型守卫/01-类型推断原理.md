# 类型推断原理

TypeScript 的类型推断是其核心特性之一。编译器能够自动推断变量、参数、返回值等的类型，使开发者无需在每个地方都显式注解类型。

## 类型推断基础

### 推断机制概述

TypeScript 的类型推断基于以下原则：

1. **初始化推断**：根据初始化值推断类型
2. **最佳通用类型**：从多个候选类型中选择最通用的
3. **上下文类型**：根据使用位置推断类型
4. **控制流分析**：根据代码逻辑推断类型变化

```typescript
// 基本推断
let x = 10;          // number
let y = 'hello';     // string
let z = true;        // boolean

// 推断发生在编译时
// 运行时类型信息被擦除
```

## 变量类型推断

### 基本类型推断

```typescript
// 数字
let num = 42;              // number
let float = 3.14;          // number
let hex = 0xff;            // number
let binary = 0b1010;       // number
let octal = 0o744;         // number
let bigint = 100n;         // bigint

// 字符串
let str = 'hello';         // string
let template = `value: ${num}`; // string

// 布尔值
let bool = true;           // boolean

// null 和 undefined
let n = null;              // null（严格模式下）
let u = undefined;         // undefined

// symbol
let sym = Symbol('key');   // symbol
```

### 数组推断

```typescript
// 数组推断
let arr1 = [1, 2, 3];              // number[]
let arr2 = ['a', 'b', 'c'];        // string[]
let arr3 = [1, 'two', 3];          // (number | string)[]

// 数组方法推断
let first = arr1[0];               // number
let mapped = arr1.map(n => n * 2); // number[]

// 元组推断
let tuple = [1, 'hello'] as const; // readonly [1, "hello"]
let tuple2: [number, string] = [1, 'hello'];
```

### 对象推断

```typescript
// 对象推断
const obj = {
  name: 'Alice',
  age: 25,
  email: 'alice@example.com'
};
// { name: string; age: number; email: string }

// 访问属性
obj.name;  // string
obj.age;   // number

// const 断言
const objConst = {
  name: 'Alice',
  age: 25
} as const;
// { readonly name: "Alice"; readonly age: 25 }
```

### const vs let 推断

```typescript
// let - 宽泛类型
let a = 'hello';     // string
let b = 42;          // number
let c = true;        // boolean

// const - 字面量类型
const aConst = 'hello';  // "hello"（字面量类型）
const bConst = 42;       // 42（字面量类型）
const cConst = true;     // true（字面量类型）

// 对象的 const 断言
const obj = { x: 10 } as const;
// { readonly x: 10 }

// 数组的 const 断言
const arr = [1, 2, 3] as const;
// readonly [1, 2, 3]
```

## 函数类型推断

### 参数类型推断

```typescript
// 从回调推断
const numbers = [1, 2, 3, 4, 5];

numbers.forEach(n => {
  console.log(n.toFixed(2));  // n: number
});

numbers.map(n => n * 2);      // n: number, 返回 number[]

numbers.filter(n => n > 2);   // n: number, 返回 number[]

// 从类型注解推断
type Handler = (event: MouseEvent) => void;

const handler: Handler = (e) => {
  console.log(e.clientX);     // e: MouseEvent
};

// 从函数签名推断
function onEvent(handler: (e: Event) => void) {}

onEvent((e) => {
  console.log(e.type);        // e: Event
});
```

### 返回值推断

```typescript
// 简单返回值
function add(a: number, b: number) {
  return a + b;               // 返回 number
}

// 对象返回值
function getUser() {
  return {
    id: 1,
    name: 'Alice',
    email: 'alice@example.com'
  };
}
// 返回 { id: number; name: string; email: string }

// 异步函数
async function fetchData() {
  const response = await fetch('/api/data');
  return response.json();     // 返回 Promise<any>
}

// 条件返回
function process(value: string | number) {
  if (typeof value === 'string') {
    return value.toUpperCase(); // string
  }
  return value.toFixed(2);     // string
}
// 返回 string
```

### 泛型函数推断

```typescript
// 泛型推断
function identity<T>(arg: T): T {
  return arg;
}

const a = identity(42);        // number
const b = identity('hello');   // string
const c = identity([1, 2, 3]); // number[]

// 多泛型推断
function pair<T, U>(first: T, second: U): [T, U] {
  return [first, second];
}

const p = pair(1, 'hello');    // [number, string]

// 泛型约束推断
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

const user = { name: 'Alice', age: 25 };
const name = getProperty(user, 'name');  // string
const age = getProperty(user, 'age');    // number
```

## 最佳通用类型

### 类型联合推断

```typescript
// 混合类型数组
let mixed = [1, 'two', 3];     // (string | number)[]

// 从多个类型推断
class Dog { bark() {} }
class Cat { meow() {} }

let animals = [new Dog(), new Cat()];
// (Dog | Cat)[]

// 使用联合类型
type Animal = Dog | Cat;
let animals2: Animal[] = [new Dog(), new Cat()];
```

### 对象类型推断

```typescript
// 属性推断
const config = {
  host: 'localhost',
  port: 3000,
  secure: false
};
// { host: string; port: number; secure: boolean }

// 方法推断
const service = {
  getData() {
    return { id: 1, name: 'test' };
  },
  setData(data: { id: number; name: string }) {
    // ...
  }
};
// { getData: () => { id: number; name: string }; setData: (data: { id: number; name: string }) => void }
```

### 类型超集推断

```typescript
// 数组推断
let arr = [1, 2, 3];
arr = [1, 2, 3, 'four'];  // ❌ 不能将 string 赋值给 number[]

// 使用联合类型
let arr2: (number | string)[] = [1, 2, 3];
arr2 = [1, 2, 3, 'four']; // ✅

// 对象推断
let obj: { x: number; y: number } = { x: 1, y: 2 };
obj = { x: 1, y: 2, z: 3 };  // ✅ 多余属性允许
```

## 上下文类型

### 回调函数推断

```typescript
// DOM 事件处理
document.addEventListener('click', (event) => {
  // event: MouseEvent
  console.log(event.clientX, event.clientY);
});

// 数组方法
const numbers = [1, 2, 3, 4, 5];

numbers.forEach((num, index, arr) => {
  // num: number, index: number, arr: number[]
});

numbers.reduce((acc, cur) => acc + cur, 0);
// acc: number, cur: number

// Promise
Promise.resolve(42).then((value) => {
  // value: number
});
```

### 对象字面量推断

```typescript
interface Options {
  width: number;
  height: number;
}

function createWindow(options: Options) {}

createWindow({
  width: 800,
  height: 600
});

// 回调中的对象
function fetchUser(options: {
  id: number;
  onSuccess: (user: { name: string }) => void;
}) {}

fetchUser({
  id: 1,
  onSuccess(user) {
    // user: { name: string }
  }
});
```

### 函数参数推断

```typescript
type Callback = (error: Error | null, data: string) => void;

function fetchData(callback: Callback) {}

fetchData((err, data) => {
  // err: Error | null
  // data: string
});

// 多参数推断
type Handler = (req: Request, res: Response, next: () => void) => void;

function use(handler: Handler) {}

use((req, res, next) => {
  // req: Request
  // res: Response
  // next: () => void
});
```

## 控制流分析

### 条件类型缩小

```typescript
function process(value: string | number | boolean) {
  // value: string | number | boolean

  if (typeof value === 'string') {
    // value: string
    return value.toUpperCase();
  }

  // value: number | boolean

  if (typeof value === 'number') {
    // value: number
    return value.toFixed(2);
  }

  // value: boolean
  return value ? 'yes' : 'no';
}
```

### 赋值分析

```typescript
let value: string | number;

value = 'hello';
console.log(value.toUpperCase());  // ✅ value: string

value = 42;
console.log(value.toFixed(2));     // ✅ value: number

// 重新赋值后类型更新
value = 'world';
console.log(value.toUpperCase());  // ✅ value: string
```

### 循环分析

```typescript
function findFirst(items: (string | number)[]): string | number | undefined {
  for (const item of items) {
    if (typeof item === 'string') {
      return item.toUpperCase();  // item: string
    }
  }

  for (const item of items) {
    if (typeof item === 'number') {
      return item.toFixed(2);     // item: number
    }
  }

  return undefined;
}
```

### 异常处理

```typescript
function process(value: unknown) {
  if (typeof value !== 'string') {
    throw new TypeError('Expected string');
  }

  // value: string（因为如果不是 string 就抛出）
  console.log(value.toUpperCase());
}
```

## 类型推断的限制

### 无法推断的情况

```typescript
// any 返回值
function parseJSON(str: string) {
  return JSON.parse(str);  // any
}

// 空数组
let arr = [];  // any[]

// 需要显式类型
interface User {
  id: number;
  name: string;
}

async function fetchUsers(): Promise<User[]> {
  const response = await fetch('/api/users');
  return response.json();  // 需要返回类型注解
}
```

### 超出作用域

```typescript
// 类型信息在函数边界丢失
function getLength(value: string | number) {
  if (typeof value === 'string') {
    return value.length;
  }
  return value.toFixed(2).length;
}

const result = getLength('hello');  // number，但不知道输入类型
```

### 复杂泛型

```typescript
// 复杂泛型推断可能不准确
function merge<T, U>(a: T, b: U): T & U {
  return { ...a, ...b };
}

const result = merge({ x: 1 }, { y: 2 });
// { x: number; y: number } ✅ 推断正确

// 更复杂的情况
function deepMerge<T extends object, U extends object>(
  a: T,
  b: U
): DeepMerge<T, U> {
  // ...
}

// 可能需要显式类型参数
```

## 推断技巧与最佳实践

### 使用 as const 获得精确类型

```typescript
// 不使用 as const
const config = {
  endpoint: '/api',
  timeout: 5000,
  retries: 3
};
// { endpoint: string; timeout: number; retries: number }

// 使用 as const
const configConst = {
  endpoint: '/api',
  timeout: 5000,
  retries: 3
} as const;
// { readonly endpoint: "/api"; readonly timeout: 5000; readonly retries: 3 }

// 用于联合类型
const directions = ['up', 'down', 'left', 'right'] as const;
type Direction = typeof directions[number];
// "up" | "down" | "left" | "right"
```

### 使用 satisfies 获得推断和验证

```typescript
interface Colors {
  primary: string;
  secondary: string;
}

const colors = {
  primary: '#ff0000',
  secondary: '#00ff00',
  tertiary: '#0000ff'
} satisfies Colors;
// 类型仍然是 { primary: string; secondary: string; tertiary: string }
// 同时验证满足 Colors 接口
```

### 显式类型注解的时机

```typescript
// ✅ 需要显式注解
async function fetchUser(): Promise<User> {
  const response = await fetch('/api/user');
  return response.json();
}

// ✅ 需要显式注解
function parseValue(str: string): string | number {
  const num = parseFloat(str);
  return isNaN(num) ? str : num;
}

// ❌ 不需要显式注解
function add(a: number, b: number): number {
  return a + b;  // 推断足够
}

// ✅ 函数参数需要注解
function greet(name: string) {  // 必须注解
  return `Hello, ${name}`;
}
```

### 类型推断调试

```typescript
// 使用工具类型查看推断结果
type InferredType = typeof someValue;

// 使用条件类型检查
type CheckNumber = typeof num extends number ? 'yes' : 'no';

// 鼠标悬停查看推断类型
const result = complexFunction();
// 在 IDE 中悬停查看实际推断的类型
```

## 小结

| 推断类型 | 说明 |
|---------|------|
| 初始化推断 | 根据初始值推断类型 |
| 最佳通用类型 | 从多个类型选择最通用的 |
| 上下文类型 | 根据使用位置推断 |
| 控制流分析 | 根据代码逻辑推断类型变化 |

TypeScript 的类型推断非常强大，但并非万能。在以下情况建议显式类型注解：

- 函数返回值（特别是涉及 `any` 的情况）
- 复杂泛型
- 公共 API
- 需要精确字面量类型时
