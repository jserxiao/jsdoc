# 类型缩小

类型缩小（Type Narrowing）是 TypeScript 将类型从较宽泛的联合类型缩小到更具体类型的过程。通过类型守卫和各种检查，TypeScript 能够在特定代码块中确定变量的确切类型。

## typeof 缩小

### 基本用法

```typescript
function process(value: string | number) {
  if (typeof value === 'string') {
    // 这里 value 被缩小为 string
    console.log(value.toUpperCase());
  } else {
    // 这里 value 被缩小为 number
    console.log(value.toFixed(2));
  }
}
```

### typeof 支持的类型

```typescript
function checkType(value: unknown) {
  if (typeof value === 'string') {
    return 'string';
  }
  if (typeof value === 'number') {
    return 'number';
  }
  if (typeof value === 'boolean') {
    return 'boolean';
  }
  if (typeof value === 'function') {
    return 'function';
  }
  if (typeof value === 'object') {
    // 注意: null 的 typeof 也是 'object'
    if (value === null) {
      return 'null';
    }
    return 'object';
  }
  if (typeof value === 'undefined') {
    return 'undefined';
  }
  if (typeof value === 'symbol') {
    return 'symbol';
  }
  if (typeof value === 'bigint') {
    return 'bigint';
  }
}
```

### typeof 与联合类型

```typescript
function formatValue(value: string | number | boolean) {
  switch (typeof value) {
    case 'string':
      return `"${value}"`;
    case 'number':
      return value.toString();
    case 'boolean':
      return value ? 'true' : 'false';
  }
}

// 使用类型守卫后的返回值
function parseValue(input: string | number): string | number {
  if (typeof input === 'string') {
    // input: string
    return input.trim();
  }
  // input: number
  return Math.abs(input);
}
```

## 真值缩小

### 基本概念

```typescript
function process(value: string | null | undefined) {
  if (value) {
    // value: string (排除了 null, undefined, '')
    console.log(value.toUpperCase());
  }
}

function greet(name: string | null) {
  if (name) {
    // name: string (排除了 null 和 '')
    console.log(`Hello, ${name}!`);
  } else {
    console.log('Hello, stranger!');
  }
}
```

### 假值列表

```typescript
// JavaScript 中的假值
const falsyValues = [
  false,
  0,
  -0,
  0n,
  '',
  null,
  undefined,
  NaN
];

// 真值检查
function printLength(str: string | null | undefined) {
  if (str) {
    // str 不是 null, undefined, 或空字符串
    console.log(str.length);
  } else {
    console.log('No value');
  }
}

// 使用逻辑运算符
function getConfig(config: { timeout?: number }) {
  const timeout = config.timeout || 3000;
  // 注意: 如果 timeout 是 0，会被替换为 3000

  // 更安全的写法
  const safeTimeout = config.timeout ?? 3000;
  // 只在 null 或 undefined 时使用默认值
}
```

### !! 和 Boolean

```typescript
function process(value: string | null) {
  // 显式转换为布尔值
  const hasValue = !!value;
  // 或使用 Boolean
  const hasValue2 = Boolean(value);

  if (hasValue) {
    // 这里 value 仍然是 string | null
    // 因为 hasValue 只是布尔值
  }

  // 正确的真值缩小
  if (value) {
    // 这里 value 是 string
    console.log(value);
  }
}
```

## 等值缩小

### 严格相等

```typescript
function process(value: string | number | null | undefined) {
  // 检查 null
  if (value === null) {
    return 'null';
  }

  // 此时 value 是 string | number | undefined
  if (value === undefined) {
    return 'undefined';
  }

  // 此时 value 是 string | number
  if (value === 'specific') {
    // value 是字面量类型 'specific'
    return 'specific value';
  }

  // value 是 number 或其他 string
  return 'other';
}
```

### 松散相等

```typescript
function process(value: string | number | null) {
  // == null 同时检查 null 和 undefined
  if (value == null) {
    // value 是 null 或 undefined
    return 'nullish';
  }

  // value 是 string | number
  console.log(value);
}
```

### switch 语句

```typescript
type Status = 'pending' | 'approved' | 'rejected';

function handleStatus(status: Status) {
  switch (status) {
    case 'pending':
      // status: 'pending'
      return 'Waiting for approval';
    case 'approved':
      // status: 'approved'
      return 'Approved';
    case 'rejected':
      // status: 'rejected'
      return 'Rejected';
  }
}
```

## in 操作符缩小

### 检查属性存在

```typescript
interface Bird {
  fly(): void;
  layEggs(): void;
}

interface Fish {
  swim(): void;
  layEggs(): void;
}

function move(animal: Bird | Fish) {
  if ('fly' in animal) {
    // animal: Bird
    animal.fly();
  } else {
    // animal: Fish
    animal.swim();
  }
}
```

### 复杂类型判断

```typescript
interface Circle {
  kind: 'circle';
  radius: number;
}

interface Rectangle {
  kind: 'rectangle';
  width: number;
  height: number;
}

interface Triangle {
  kind: 'triangle';
  base: number;
  height: number;
}

type Shape = Circle | Rectangle | Triangle;

function getArea(shape: Shape): number {
  if ('radius' in shape) {
    // shape: Circle
    return Math.PI * shape.radius ** 2;
  }
  if ('width' in shape) {
    // shape: Rectangle
    return shape.width * shape.height;
  }
  // shape: Triangle
  return (shape.base * shape.height) / 2;
}
```

### 方法存在检查

```typescript
interface SyncStorage {
  getItem(key: string): string | null;
}

interface AsyncStorage {
  getItem(key: string): Promise<string | null>;
}

function getItem(storage: SyncStorage | AsyncStorage, key: string) {
  const result = storage.getItem(key);

  // 检查返回值是否是 Promise
  if (result instanceof Promise) {
    return result.then(v => v);
  }

  return result;
}
```

## instanceof 缩小

### 基本用法

```typescript
function process(value: Date | string) {
  if (value instanceof Date) {
    // value: Date
    return value.toISOString();
  }
  // value: string
  return value.toUpperCase();
}
```

### 自定义类

```typescript
class ValidationError extends Error {
  constructor(message: string, public field: string) {
    super(message);
  }
}

class NetworkError extends Error {
  constructor(message: string, public statusCode: number) {
    super(message);
  }
}

function handleError(error: ValidationError | NetworkError) {
  if (error instanceof ValidationError) {
    console.log(`Field ${error.field} is invalid`);
  } else {
    console.log(`HTTP ${error.statusCode} error`);
  }
}
```

### 数组检查

```typescript
function process(value: string | string[]) {
  if (value instanceof Array) {
    // value: string[]
    return value.join(', ');
  }
  // value: string
  return value;
}

// 注意: instanceof Array 对于跨 iframe 可能有问题
// 更安全的方式
function isArray(value: unknown): value is unknown[] {
  return Array.isArray(value);
}
```

## 赋值缩小

### 变量赋值

```typescript
let value: string | number;

value = 'hello';
// value: string
console.log(value.toUpperCase());

value = 42;
// value: number
console.log(value.toFixed(2));
```

### 条件赋值

```typescript
function process(input: string | number | null) {
  let value: string | number;

  if (input === null) {
    value = 'default';
  } else {
    value = input;
  }

  // value: string | number
  console.log(value);
}
```

## 控制流分析

### 分支分析

```typescript
function process(value: string | number | null) {
  if (value === null) {
    return 'null';
  }

  // value: string | number
  if (typeof value === 'string') {
    return value.toUpperCase();
  }

  // value: number
  return value.toFixed(2);
}
```

### 循环分析

```typescript
function findFirst(items: (string | number)[]): string | number | undefined {
  for (const item of items) {
    if (typeof item === 'string') {
      return item.toUpperCase(); // item: string
    }
  }
  return undefined;
}
```

### 异常处理

```typescript
function process(value: unknown) {
  if (typeof value !== 'string') {
    throw new Error('Expected string');
  }

  // value: string (TypeScript 知道如果不是 string 就会抛出)
  console.log(value.toUpperCase());
}
```

## 类型谓词

### 自定义类型守卫

```typescript
interface Fish {
  swim: () => void;
}

interface Bird {
  fly: () => void;
}

// 类型谓词: value is Fish
function isFish(value: Fish | Bird): value is Fish {
  return 'swim' in value;
}

function move(animal: Fish | Bird) {
  if (isFish(animal)) {
    // animal: Fish
    animal.swim();
  } else {
    // animal: Bird
    animal.fly();
  }
}
```

### 复杂类型守卫

```typescript
interface User {
  type: 'user';
  id: number;
  name: string;
}

interface Admin {
  type: 'admin';
  id: number;
  name: string;
  permissions: string[];
}

type Person = User | Admin;

function isAdmin(person: Person): person is Admin {
  return person.type === 'admin';
}

function getPermissions(person: Person): string[] {
  if (isAdmin(person)) {
    return person.permissions;
  }
  return [];
}
```

### 泛型类型守卫

```typescript
function isDefined<T>(value: T | null | undefined): value is T {
  return value !== null && value !== undefined;
}

function processItems<T>(items: (T | null | undefined)[]): T[] {
  return items.filter(isDefined);
}

const result = processItems([1, null, 2, undefined, 3]);
// result: number[]
```

## 断言函数

### asserts 语法

```typescript
function assertIsString(value: unknown): asserts value is string {
  if (typeof value !== 'string') {
    throw new Error('Value is not a string');
  }
}

function process(value: unknown) {
  assertIsString(value);
  // value: string
  console.log(value.toUpperCase());
}
```

### 断言非空

```typescript
function assertDefined<T>(value: T | null | undefined): asserts value is T {
  if (value === null || value === undefined) {
    throw new Error('Value is null or undefined');
  }
}

function process(value: string | null) {
  assertDefined(value);
  // value: string
  console.log(value.toUpperCase());
}
```

### 条件断言

```typescript
function assert(condition: boolean, message: string): asserts condition {
  if (!condition) {
    throw new Error(message);
  }
}

function divide(a: number, b: number): number {
  assert(b !== 0, 'Cannot divide by zero');
  return a / b;
}
```

## 可辨识联合

### 基本模式

```typescript
interface Circle {
  kind: 'circle';
  radius: number;
}

interface Square {
  kind: 'square';
  sideLength: number;
}

type Shape = Circle | Square;

function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      // shape: Circle
      return Math.PI * shape.radius ** 2;
    case 'square':
      // shape: Square
      return shape.sideLength ** 2;
  }
}
```

### 完整示例

```typescript
interface Loading {
  status: 'loading';
}

interface Success<T> {
  status: 'success';
  data: T;
}

interface Error {
  status: 'error';
  error: string;
}

type Result<T> = Loading | Success<T> | Error;

function handleResult<T>(result: Result<T>) {
  switch (result.status) {
    case 'loading':
      console.log('Loading...');
      break;
    case 'success':
      // result: Success<T>
      console.log('Data:', result.data);
      break;
    case 'error':
      // result: Error
      console.error('Error:', result.error);
      break;
  }
}
```

## 穷尽性检查

### never 类型

```typescript
interface Circle {
  kind: 'circle';
  radius: number;
}

interface Square {
  kind: 'square';
  sideLength: number;
}

interface Triangle {
  kind: 'triangle';
  base: number;
  height: number;
}

type Shape = Circle | Square | Triangle;

function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2;
    case 'square':
      return shape.sideLength ** 2;
    case 'triangle':
      return (shape.base * shape.height) / 2;
    default:
      // 如果所有情况都处理了，shape 是 never
      const _exhaustiveCheck: never = shape;
      return _exhaustiveCheck;
  }
}
```

### 辅助函数

```typescript
function assertNever(value: never): never {
  throw new Error(`Unexpected value: ${value}`);
}

function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2;
    case 'square':
      return shape.sideLength ** 2;
    case 'triangle':
      return (shape.base * shape.height) / 2;
    default:
      return assertNever(shape);
  }
}
```

## 实战示例

### API 响应处理

```typescript
interface ApiSuccess<T> {
  success: true;
  data: T;
}

interface ApiError {
  success: false;
  error: {
    code: number;
    message: string;
  };
}

type ApiResponse<T> = ApiSuccess<T> | ApiError;

async function fetchData<T>(url: string): Promise<T> {
  const response: ApiResponse<T> = await fetch(url).then(r => r.json());

  if (response.success) {
    return response.data;
  }

  throw new Error(`[${response.error.code}] ${response.error.message}`);
}
```

### 事件处理器

```typescript
interface MouseEvent {
  type: 'click' | 'mousemove';
  x: number;
  y: number;
}

interface KeyboardEvent {
  type: 'keydown' | 'keyup';
  key: string;
  code: string;
}

type AppEvent = MouseEvent | KeyboardEvent;

function handleEvent(event: AppEvent) {
  switch (event.type) {
    case 'click':
      console.log(`Clicked at (${event.x}, ${event.y})`);
      break;
    case 'mousemove':
      console.log(`Mouse moved to (${event.x}, ${event.y})`);
      break;
    case 'keydown':
      console.log(`Key pressed: ${event.key}`);
      break;
    case 'keyup':
      console.log(`Key released: ${event.key}`);
      break;
  }
}
```

## 小结

| 缩小方式 | 用途 |
|---------|------|
| `typeof` | 基本类型检查 |
| 真值检查 | 排除 null/undefined/假值 |
| 等值检查 | 具体值或 null/undefined |
| `in` 操作符 | 检查属性或方法是否存在 |
| `instanceof` | 类实例检查 |
| 类型谓词 | 自定义类型守卫 |
| 断言函数 | 运行时断言类型 |
| 可辨识联合 | 通过公共属性区分类型 |
