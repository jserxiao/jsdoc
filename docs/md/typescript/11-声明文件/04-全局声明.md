## 全局变量声明

### 基本语法

```typescript
// global.d.ts

// 声明全局变量
declare const API_URL: string;
declare const DEBUG: boolean;
declare let currentUser: User | null;

// 声明全局常量对象
declare const config: {
  apiUrl: string;
  version: string;
  environment: 'development' | 'production';
};
```

### Window 对象扩展

```typescript
// window-extensions.d.ts

interface Window {
  // 添加自定义属性
  __APP_VERSION__: string;
  __DEV__: boolean;
  
  // 添加第三方库
  gtag?: (...args: any[]) => void;
  analytics?: {
    track(event: string, data?: object): void;
    identify(userId: string): void;
  };
}

// 使用
window.__APP_VERSION__;
window.gtag?.('event', 'click');
```

## 全局函数声明

### 基本函数

```typescript
// global-functions.d.ts

declare function log(message: string, level?: 'info' | 'warn' | 'error'): void;
declare function debug(...args: any[]): void;

// 带重载的函数
declare function $(selector: string): Element | null;
declare function $(element: Element): Element;
```

### 工具函数

```typescript
// utils.d.ts

declare function debounce<T extends (...args: any[]) => any>(
  fn: T,
  wait: number
): T & { cancel(): void; flush(): void };

declare function throttle<T extends (...args: any[]) => any>(
  fn: T,
  limit: number
): T;
```

## 全局接口扩展

### 扩展内置接口

```typescript
// array-extensions.d.ts

interface Array<T> {
  first(): T | undefined;
  last(): T | undefined;
  unique(): T[];
  flatten(): T extends Array<infer U> ? U[] : T[];
}

// 使用
[1, 2, 3].first();  // 1
[1, 2, 3].last();   // 3
```

### 扩展 Promise

```typescript
// promise-extensions.d.ts

interface Promise<T> {
  timeout(ms: number): Promise<T>;
  retry(attempts: number): Promise<T>;
}

interface PromiseConstructor {
  delay(ms: number): Promise<void>;
}
```

### 扩展 String

```typescript
// string-extensions.d.ts

interface String {
  capitalize(): string;
  truncate(length: number, suffix?: string): string;
  isNullOrEmpty(): boolean;
}

// 使用
'hello'.capitalize();  // 'Hello'
'hello world'.truncate(5);  // 'hello...'
```

## 全局命名空间

### 定义命名空间

```typescript
// namespace.d.ts

declare namespace MyLib {
  interface Options {
    debug?: boolean;
  }
  
  interface User {
    id: string;
    name: string;
  }
  
  function init(options?: Options): void;
  function getUser(): User;
  
  namespace Internal {
    function helper(): void;
  }
}

// 使用
MyLib.init({ debug: true });
const user: MyLib.User = MyLib.getUser();
```

### 命名空间合并

```typescript
// 声明1
declare namespace MyLib {
  interface User {
    name: string;
  }
}

// 声明2 - 自动合并
declare namespace MyLib {
  interface User {
    email: string;  // 合并到 User
  }
  
  interface Config {
    apiUrl: string;
  }
}

// 结果：User 有 name 和 email
```

### 全局类型

```typescript
// global-types.d.ts

// 声明全局类型（不需要 import）
type Nullable<T> = T | null;
type Optional<T> = T | undefined;
type AsyncResult<T, E = Error> = Promise<Result<T, E>>;

interface Result<T, E = Error> {
  ok: boolean;
  value?: T;
  error?: E;
}
```

## declare global

在模块文件中声明全局类型：

```typescript
// my-module.ts

// 扩展全局
declare global {
  interface Array<T> {
    myCustomMethod(): T[];
  }
  
  var myGlobal: {
    version: string;
  };
}

// 必须有导出
export function something() {}
```
