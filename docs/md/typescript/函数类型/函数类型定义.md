## 函数声明

### 基本语法

```typescript
// 基本函数声明
function add(a: number, b: number): number {
  return a + b;
}

// void 返回类型
function log(message: string): void {
  console.log(message);
}

// never 返回类型（永不返回）
function throwError(message: string): never {
  throw new Error(message);
}

function infiniteLoop(): never {
  while (true) {}
}
```

### 返回类型推断

```typescript
// TypeScript 可以自动推断返回类型
function add(a: number, b: number) {
  return a + b; // 推断为 number
}

function greet(name: string) {
  return `Hello, ${name}`; // 推断为 string
}

// 但显式指定返回类型是更好的实践
function multiply(a: number, b: number): number {
  return a * b;
}
```

### 异步函数

```typescript
// async 函数返回 Promise
async function fetchData(url: string): Promise<Response> {
  const response = await fetch(url);
  return response;
}

// 也可以让 TypeScript 推断
async function fetchUser(id: number) {
  const response = await fetch(`/api/users/${id}`);
  return response.json(); // 推断为 Promise<any>
}

// 最佳实践：明确指定泛型
async function fetchTyped<T>(url: string): Promise<T> {
  const response = await fetch(url);
  return response.json();
}

interface User {
  id: number;
  name: string;
}

const user = await fetchTyped<User>('/api/users/1');
```

## 函数表达式

### 普通函数表达式

```typescript
// 匿名函数表达式
const add = function(a: number, b: number): number {
  return a + b;
};

// 命名函数表达式
const subtract = function sub(a: number, b: number): number {
  return a - b;
};
```

### 箭头函数

```typescript
// 单行箭头函数
const add = (a: number, b: number): number => a + b;

// 多行箭头函数
const calculate = (a: number, b: number, op: string): number => {
  switch (op) {
    case '+': return a + b;
    case '-': return a - b;
    case '*': return a * b;
    case '/': return a / b;
    default: throw new Error(`Unknown operator: ${op}`);
  }
};

// 单参数可省略括号
const double = (n: number): number => n * 2;

// 无参数
const getRandom = (): number => Math.random();

// 返回对象字面量需要括号
const createUser = (name: string): { name: string } => ({ name });
```

### 立即执行函数（IIFE）

```typescript
// 传统 IIFE
const result = (function(a: number, b: number): number {
  return a + b;
})(1, 2);

// 箭头函数 IIFE
const result2 = ((a: number, b: number): number => a + b)(1, 2);
```

## 函数类型别名

### 基本类型别名

```typescript
// 简单函数类型
type MathFunction = (a: number, b: number) => number;

// 使用类型别名
const add: MathFunction = (a, b) => a + b;
const subtract: MathFunction = (a, b) => a - b;
const multiply: MathFunction = (a, b) => a * b;

// 类型别名让代码更简洁
function calculate(a: number, b: number, op: MathFunction): number {
  return op(a, b);
}

calculate(10, 5, add);      // 15
calculate(10, 5, subtract); // 5
calculate(10, 5, multiply); // 50
```

### 复杂函数类型

```typescript
// 带可选参数
type GreetFunction = (name: string, greeting?: string) => string;

const greet: GreetFunction = (name, greeting = 'Hello') => {
  return `${greeting}, ${name}!`;
};

// 带剩余参数
type LogFunction = (level: string, ...messages: string[]) => void;

const log: LogFunction = (level, ...messages) => {
  console.log(`[${level}]`, ...messages);
};

// 带 this 参数
type ClickHandler = (this: HTMLButtonElement, event: MouseEvent) => void;

const handler: ClickHandler = function(event) {
  console.log(this.textContent); // this 正确推断为 HTMLButtonElement
};
```

### 泛型函数类型

```typescript
// 泛型函数类型
type Identity<T> = (arg: T) => T;

const identity: Identity<number> = (arg) => arg;
const identityString: Identity<string> = (arg) => arg;

// 更灵活的泛型
type Mapper<T, U> = (value: T) => U;

const stringToNumber: Mapper<string, number> = (s) => s.length;
const numberToString: Mapper<number, string> = (n) => n.toString();

// 多泛型参数
type Reducer<T, U> = (acc: U, curr: T) => U;

const sum: Reducer<number, number> = (acc, curr) => acc + curr;
```

## 接口定义函数

### 调用签名

```typescript
// 接口定义函数类型
interface MathFunction {
  (a: number, b: number): number;
}

const add: MathFunction = (a, b) => a + b;
const subtract: MathFunction = (a, b) => a - b;

// 接口 vs 类型别名
// 接口可以被扩展和实现
interface MathFunction {
  (a: number, b: number): number;
}

// 扩展接口
interface AdvancedMathFunction extends MathFunction {
  description: string;
}

const advancedAdd: AdvancedMathFunction = (a, b) => a + b;
advancedAdd.description = 'Returns the sum of two numbers';
```

### 混合类型接口

```typescript
// 函数也是对象，可以有属性
interface Counter {
  (): number;           // 调用签名
  value: number;        // 属性
  reset: () => void;    // 方法
}

function createCounter(): Counter {
  const counter = function() {
    return ++counter.value;
  } as Counter;

  counter.value = 0;
  counter.reset = function() {
    counter.value = 0;
  };

  return counter;
}

const counter = createCounter();
console.log(counter());    // 1
console.log(counter());    // 2
console.log(counter.value); // 2
counter.reset();
console.log(counter());    // 1
```

### 多调用签名（重载）

```typescript
// 接口可以定义多个调用签名
interface StringOrNumber {
  (value: string): string;
  (value: number): number;
}

const identity: StringOrNumber = (value: any) => value;

const str = identity('hello'); // string
const num = identity(42);      // number
```

## 构造函数类型

### 构造签名

```typescript
// 构造签名使用 new 关键字
interface UserConstructor {
  new (name: string, age: number): User;
}

class User {
  constructor(public name: string, public age: number) {}
}

function createUser(ctor: UserConstructor, name: string, age: number): User {
  return new ctor(name, age);
}

const user = createUser(User, 'Alice', 30);
```

### 同时可调用和可构造

```typescript
// 同时支持 new 和普通调用
interface DateOrString {
  new (value: string): Date;
  (value: string): string;
}

// 实际例子：内置的 Date
const d1 = new Date('2024-01-01'); // Date 对象
const d2 = Date('2024-01-01');     // string
```

### 工厂函数

```typescript
interface AnimalConstructor {
  new (name: string): Animal;
}

interface Animal {
  name: string;
  speak(): void;
}

class Dog implements Animal {
  constructor(public name: string) {}
  speak() {
    console.log(`${this.name} says: Woof!`);
  }
}

class Cat implements Animal {
  constructor(public name: string) {}
  speak() {
    console.log(`${this.name} says: Meow!`);
  }
}

function createAnimal(ctor: AnimalConstructor, name: string): Animal {
  return new ctor(name);
}

const dog = createAnimal(Dog, 'Buddy');
const cat = createAnimal(Cat, 'Whiskers');
```

## 函数重载签名

### 基本重载

```typescript
// 重载签名
function reverse(x: number): number;
function reverse(x: string): string;
// 实现签名
function reverse(x: number | string): number | string {
  if (typeof x === 'number') {
    return Number(x.toString().split('').reverse().join(''));
  } else {
    return x.split('').reverse().join('');
  }
}

reverse(123);   // 321
reverse('abc'); // 'cba'
```

### 多参数重载

```typescript
function createElement(tag: 'a'): HTMLAnchorElement;
function createElement(tag: 'canvas'): HTMLCanvasElement;
function createElement(tag: 'table'): HTMLTableElement;
function createElement(tag: string): HTMLElement;
function createElement(tag: string): HTMLElement {
  return document.createElement(tag);
}

const anchor = createElement('a');    // HTMLAnchorElement
const canvas = createElement('canvas'); // HTMLCanvasElement
```

### 回调函数重载

```typescript
function fetchData(url: string): Promise<Response>;
function fetchData(url: string, callback: (response: Response) => void): void;
function fetchData(url: string, callback?: (response: Response) => void): Promise<Response> | void {
  const promise = fetch(url);

  if (callback) {
    promise.then(callback);
    return;
  }

  return promise;
}

// Promise 风格
const response = await fetchData('/api/data');

// 回调风格
fetchData('/api/data', (res) => {
  console.log(res);
});
```

## 可选参数与默认参数

### 可选参数

```typescript
// 可选参数使用 ?
function greet(name: string, greeting?: string): string {
  return `${greeting || 'Hello'}, ${name}!`;
}

greet('Alice');              // 'Hello, Alice!'
greet('Alice', 'Hi');        // 'Hi, Alice!'
```

### 默认参数

```typescript
// 默认参数
function greet(name: string, greeting: string = 'Hello'): string {
  return `${greeting}, ${name}!`;
}

greet('Alice');              // 'Hello, Alice!'
greet('Alice', 'Hi');        // 'Hi, Alice!'
greet('Alice', undefined);   // 'Hello, Alice!'
```

### 默认参数与类型推断

```typescript
// 默认参数会影响类型推断
function createConfig(options: { timeout?: number; retries?: number } = {}) {
  return {
    timeout: 1000,
    retries: 3,
    ...options
  };
}

// 类型推断会包含默认值
const config = createConfig();
// { timeout: number; retries: number }
```

## this 类型

### 指定 this 类型

```typescript
interface Person {
  name: string;
  greet(this: Person): void;
}

const person: Person = {
  name: 'Alice',
  greet() {
    console.log(`Hello, I'm ${this.name}`);
  }
};

person.greet(); // OK

const greet = person.greet;
// greet(); // Error: this 是 undefined
```

### 箭头函数与 this

```typescript
class Counter {
  private count = 0;

  // 普通方法，this 需要绑定
  increment(): void {
    this.count++;
  }

  // 箭头函数自动绑定 this
  decrement = (): void => {
    this.count--;
  };

  getCount(): number {
    return this.count;
  }
}

const counter = new Counter();
const inc = counter.increment;
const dec = counter.decrement;

// inc();  // Error: this 是 undefined
dec();     // OK: 箭头函数自动绑定
```

## 实战示例

### API 处理函数

```typescript
// API 处理函数类型
type RequestHandler = (
  req: Request,
  res: Response,
  next?: NextFunction
) => void | Promise<void>;

interface Request {
  method: string;
  path: string;
  params: Record<string, string>;
  query: Record<string, string>;
  body: unknown;
  headers: Record<string, string>;
}

interface Response {
  status: (code: number) => Response;
  json: (data: unknown) => void;
  send: (data: string) => void;
}

type NextFunction = (err?: Error) => void;

// 使用示例
const getUsers: RequestHandler = async (req, res) => {
  const users = await fetchUsers();
  res.status(200).json(users);
};

const createUser: RequestHandler = async (req, res) => {
  const user = await createUser(req.body as CreateUserInput);
  res.status(201).json(user);
};
```

### 事件处理函数

```typescript
// 通用事件处理函数
type EventHandler<E extends Event = Event> = (event: E) => void;

// 特定事件类型
type ClickHandler = EventHandler<MouseEvent>;
type KeydownHandler = EventHandler<KeyboardEvent>;
type SubmitHandler = EventHandler<SubmitEvent>;

// 使用
const handleClick: ClickHandler = (e) => {
  console.log(e.clientX, e.clientY);

  if (e.key === 'Enter') {
    console.log('Enter pressed');
  }

```

### 工具函数

```typescript
// 防抖
function debounce<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let timeoutId: ReturnType<typeof setTimeout>;

  return function(this: any, ...args: Parameters<T>) {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => fn.apply(this, args), delay);
  };
}

// 节流
function throttle<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let lastCall = 0;

  return function(this: any, ...args: Parameters<T>) {
    const now = Date.now();
    if (now - lastCall >= delay) {
      lastCall = now;
      fn.apply(this, args);
    }
  };
}

// 柯里化
function curry<T extends (...args: any[]) => any>(
  fn: T
): (...args: Parameters<T>) => ReturnType<T> | any {
  const arity = fn.length;

  return function curried(...args: any[]): any {
    if (args.length >= arity) {
      return fn(...args);
    }
    return (...more: any[]) => curried(...args, ...more);
  };
}
```

## 最佳实践

### 显式返回类型

```typescript
// ✅ 好：显式指定返回类型
function add(a: number, b: number): number {
  return a + b;
}

// ❌ 差：依赖类型推断可能导致意外
function add(a: number, b: number) {
  // 如果这里意外返回了 string，不会报错
  return a + b;
}
```

### 函数重载顺序

```typescript
// ✅ 好：具体类型在前
function process(value: string): string;
function process(value: number): number;
function process(value: string | number): string | number {
  return value;
}

// ❌ 差：泛型签名在前会匹配所有情况
function process<T>(value: T): T;
function process(value: string): string;
// 第二个重载永远不会被匹配
```

### 避免过多重载

```typescript
// ❌ 避免：过多重载
function getValue(obj: any, key: 'a'): string;
function getValue(obj: any, key: 'b'): number;
function getValue(obj: any, key: 'c'): boolean;
// ... 更多重载

// ✅ 更好：使用泛型或映射类型
function getValue<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}
```

## 小结

| 定义方式 | 语法 | 特点 |
|---------|------|------|
| 函数声明 | `function name(): Type` | 最常用，支持重载 |
| 函数表达式 | `const fn = (): Type => {}` | 灵活，适合回调 |
| 类型别名 | `type Fn = (args) => Type` | 可复用，简洁 |
| 接口定义 | `interface Fn { (args): Type }` | 可扩展，可混合 |
| 构造签名 | `new (args) => Type` | 用于类工厂模式 |
