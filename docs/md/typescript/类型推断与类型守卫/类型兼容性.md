# 类型兼容性

TypeScript 使用结构化类型系统（Structural Type System），也称为"鸭子类型"。类型的兼容性基于成员的结构，而不是类型的名称或声明位置。

## 结构化类型系统

### 核心概念

```typescript
// 类型兼容性基于结构，而不是名称
interface Point2D {
  x: number;
  y: number;
}

interface Coordinate {
  x: number;
  y: number;
}

let point: Point2D = { x: 1, y: 2 };
let coord: Coordinate = point; // ✅ 兼容，结构相同

// 结构相同就兼容，即使类型名称不同
interface User {
  name: string;
}

interface Person {
  name: string;
}

let user: User = { name: 'Alice' };
let person: Person = user; // ✅ 兼容
```

### 与标称类型系统的对比

```typescript
// 标称类型系统（如 Java、C#）
// 类型即使结构相同，名称不同也不兼容

// TypeScript 结构化类型
interface ID { value: string; }
interface Name { value: string; }

let id: ID = { value: '123' };
let name: Name = id; // ✅ TypeScript 允许

// 如果需要标称类型行为，可以使用 branding
type BrandedID = ID & { __brand: 'ID' };
type BrandedName = Name & { __brand: 'Name' };

// let brandedId: BrandedID = name; // ❌ 不兼容
```

## 对象类型兼容性

### 多余属性原则

```typescript
interface Point {
  x: number;
  y: number;
}

interface Point3D {
  x: number;
  y: number;
  z: number;
}

let point: Point = { x: 0, y: 0 };
let point3D: Point3D = { x: 0, y: 0, z: 0 };

// ✅ 多余属性可以赋值（源类型是目标类型的子类型）
point = point3D;

// ❌ 缺少属性不能赋值
// point3D = point; // Error: 缺少 z 属性
```

### 成员兼容性检查

```typescript
interface Animal {
  name: string;
  age: number;
}

interface Dog {
  name: string;
  age: number;
  breed: string;
}

let animal: Animal = { name: 'Max', age: 3 };
let dog: Dog = { name: 'Max', age: 3, breed: 'Labrador' };

// ✅ Dog 有 Animal 的所有属性
animal = dog;

// ❌ Animal 缺少 Dog 的 breed 属性
// dog = animal;
```

### 可选属性

```typescript
interface Config {
  host: string;
  port?: number;
}

interface FullConfig {
  host: string;
  port: number;
}

let config: Config = { host: 'localhost' };
let fullConfig: FullConfig = { host: 'localhost', port: 3000 };

// ✅ FullConfig 有 Config 的所有必需属性
config = fullConfig;

// ❌ Config 可能缺少 port
// fullConfig = config;
```

### 索引签名

```typescript
interface StringMap {
  [key: string]: string;
}

interface UserMap {
  [key: string]: {
    name: string;
    email: string;
  };
}

let stringMap: StringMap = { a: 'hello' };
let userMap: UserMap = { a: { name: 'Alice', email: 'a@b.com' } };

// ✅ UserMap 的值是 StringMap 值的子类型（更具体）
stringMap = userMap; // 不行，值类型不兼容

// 正确的例子
interface AnyMap {
  [key: string]: any;
}

let anyMap: AnyMap = userMap; // ✅ any 兼容
```

## 函数兼容性

### 参数数量

```typescript
type Handler1 = (a: number) => void;
type Handler2 = (a: number, b: number) => void;

let h1: Handler1 = a => console.log(a);
let h2: Handler2 = (a, b) => console.log(a, b);

// ✅ 参数少可以赋值给参数多的（忽略多余参数）
h2 = h1; // ✅ h1 的签名可以安全地用于期望 h2 的地方

// ❌ 参数多不能赋值给参数少的
// h1 = h2; // Error: 可能缺少 b 参数
```

### 参数类型（逆变）

```typescript
class Animal {
  name: string = 'animal';
}

class Dog extends Animal {
  breed: string = 'dog';
}

type AnimalHandler = (animal: Animal) => void;
type DogHandler = (dog: Dog) => void;

let handleAnimal: AnimalHandler = (a) => console.log(a.name);
let handleDog: DogHandler = (d) => console.log(d.breed);

// 在严格函数类型模式下（strictFunctionTypes）
// 参数是逆变的

// ❌ Dog 处理器不能赋值给 Animal 处理器
// handleAnimal = handleDog;
// 因为 handleDog 可能使用 Animal 没有的属性

// ✅ Animal 处理器可以赋值给 Dog 处理器
handleDog = handleAnimal;
// 因为 handleAnimal 只使用 Animal 的属性，Dog 都有
```

### 返回值类型（协变）

```typescript
type AnimalCreator = () => Animal;
type DogCreator = () => Dog;

let createAnimal: AnimalCreator = () => new Animal();
let createDog: DogCreator = () => new Dog();

// ✅ 返回 Dog 的函数可以赋值给返回 Animal 的函数
let creator: AnimalCreator = createDog;

// ❌ 返回 Animal 的函数不能赋值给返回 Dog 的函数
// let dogCreator: DogCreator = createAnimal;
```

### 函数重载兼容性

```typescript
type Overloaded = {
  (a: string): string;
  (a: number): number;
};

type StringOnly = (a: string) => string;
type NumberOnly = (a: number) => number;

let overloaded: Overloaded = (a: any) => a;
let stringFn: StringOnly = (a) => a;
let numberFn: NumberOnly = (a) => a;

// 重载函数可以赋值给单一签名
stringFn = overloaded;
numberFn = overloaded;

// 单一签名不能赋值给重载
// overloaded = stringFn; // Error
```

## 类兼容性

### 实例成员

```typescript
class Animal {
  name: string = 'animal';
}

class Dog extends Animal {
  breed: string = 'dog';
}

let animal: Animal = new Animal();
let dog: Dog = new Dog();

// ✅ 子类实例可以赋值给父类类型
animal = dog;

// ❌ 父类实例不能赋值给子类类型
// dog = animal;
```

### 私有成员和受保护成员

```typescript
class Parent {
  private name: string = 'parent';
  protected age: number = 30;
}

class Child extends Parent {
  private name: string = 'child'; // 不同的私有成员
}

class Other {
  private name: string = 'other';
}

let parent: Parent = new Parent();
let child: Child = new Child();
let other: Other = new Other();

// ❌ 私有成员必须在同一声明中
// parent = other; // Error: 类型不兼容

// ✅ 继承的私有成员兼容
parent = child;
```

### 静态成员

```typescript
class StaticA {
  static count: number = 0;
  instance: number = 0;
}

class StaticB {
  static count: number = 0;
  instance: number = 0;
}

let a: StaticA = new StaticA();
let b: StaticB = new StaticB();

// ✅ 静态成员不影响实例兼容性
a = b; // 兼容，因为实例成员相同
```

### 构造函数签名

```typescript
class Animal {
  constructor(public name: string) {}
}

class Dog extends Animal {
  constructor(name: string, public breed: string) {
    super(name);
  }
}

type AnimalConstructor = new (name: string) => Animal;
type DogConstructor = new (name: string, breed: string) => Dog;

let AnimalCtor: AnimalConstructor = Animal;
let DogCtor: DogConstructor = Dog;

// 构造函数的兼容性规则与普通函数类似
```

## 泛型兼容性

### 类型参数未使用

```typescript
interface Empty<T> {}

let e1: Empty<number> = {};
let e2: Empty<string> = {};

// ✅ 类型参数未使用时兼容
e1 = e2;
e2 = e1;
```

### 类型参数被使用

```interface NotEmpty<T> {
  value: T;
}

let n1: NotEmpty<number> = { value: 42 };
let n2: NotEmpty<string> = { value: 'hello' };

// ❌ 类型参数被使用时不兼容
// n1 = n2;
```

### 泛型函数

```typescript
type Identity<T> = (arg: T) => T;

let idNumber: Identity<number> = (arg) => arg;
let idString: Identity<string> = (arg) => arg;

// ❌ 不同类型参数的泛型函数不兼容
// idNumber = idString;
```

### 泛型约束

```typescript
interface Lengthwise {
  length: number;
}

function logLength<T extends Lengthwise>(arg: T): void {
  console.log(arg.length);
}

// 泛型约束不影响兼容性检查
type LogFn = <T extends Lengthwise>(arg: T) => void;
let fn: LogFn = logLength;
```

## 枚举兼容性

### 数字枚举

```typescript
enum Color {
  Red,
  Green,
  Blue
}

let c: Color = Color.Red;

// ✅ 枚举值可以赋值给 number
let n: number = c;

// ❌ 在严格模式下，number 不能赋值给枚举
// let c2: Color = 0;
```

### 字符串枚举

```typescript
enum Direction {
  Up = 'UP',
  Down = 'DOWN'
}

let dir: Direction = Direction.Up;

// ❌ 字符串枚举不与 string 兼容
// let s: string = dir; // 实际上可以
// let d: Direction = 'UP'; // Error
```

### 枚举之间

```typescript
enum Color { Red, Green, Blue }
enum Status { Active, Inactive }

let color: Color = Color.Red;
let status: Status = Status.Active;

// ❌ 不同枚举类型不兼容
// color = status;
```

## 协变与逆变

### 概念解释

```typescript
// 协变（Covariant）：子类型可以赋值给父类型
// 逆变（Contravariant）：父类型可以赋值给子类型
// 不变（Invariant）：只能赋值给完全相同的类型
// 双变（Bivariant）：两个方向都可以

class Animal { name = ''; }
class Dog extends Animal { breed = ''; }

// 协变示例
let dogs: Dog[] = [new Dog()];
let animals: Animal[] = dogs; // ✅ 数组是协变的

// 逆变示例（函数参数）
type AnimalHandler = (animal: Animal) => void;
type DogHandler = (dog: Dog) => void;

let handleAnimal: AnimalHandler = (a) => console.log(a.name);
let handleDog: DogHandler = handleAnimal; // ✅ 参数逆变
```

### 严格函数类型

```typescript
// tsconfig.json
// "strictFunctionTypes": true

// 启用严格函数类型后，函数参数只在必要时逆变
// 禁用时，函数参数是双变的

interface A { a: string; }
interface B extends A { b: string; }

type FnA = (a: A) => void;
type FnB = (b: B) => void;

declare let fnA: FnA;
declare let fnB: FnB;

// 严格模式下
// fnA = fnB; // ❌ 不兼容
fnB = fnA;    // ✅ 兼容
```

## 类型兼容性规则

### 总结表

| 类型 | 兼容性规则 |
|-----|-----------|
| 基本类型 | 类型相同才兼容 |
| 对象 | 目标类型属性是源类型的子集 |
| 函数参数 | 逆变（父类型可以赋值给子类型） |
| 函数返回值 | 协变（子类型可以赋值给父类型） |
| 类实例 | 私有/受保护成员必须同源，其他同对象 |
| 泛型 | 类型参数被使用时不兼容 |
| 枚举 | 数字枚举与 number 单向兼容 |

### never 和 unknown

```typescript
// never 是所有类型的子类型
type NeverIsSubtype = never extends string ? true : false; // true

let n: never;
// let s: string = n; // ✅ never 可以赋值给任何类型

// unknown 是所有类型的父类型
let u: unknown;
// let s: string = u; // ❌ unknown 不能直接赋值

// 类型缩小后可以赋值
if (typeof u === 'string') {
  let s: string = u; // ✅
}
```

## 实战示例

### 回调函数类型

```typescript
interface EventMap {
  click: MouseEvent;
  focus: FocusEvent;
}

function addListener<K extends keyof EventMap>(
  type: K,
  listener: (event: EventMap[K]) => void
): void {
  // 实现
}

// ✅ 使用更通用的参数类型
addListener('click', (event: Event) => {
  console.log(event.type);
});

// ❌ 使用更具体的参数类型
// addListener('click', (event: MouseEvent) => {
//   console.log(event.clientX); // 可能在严格模式下报错
// });
```

### 接口扩展

```typescript
interface RequestOptions {
  url: string;
  method?: 'GET' | 'POST';
  headers?: Record<string, string>;
}

function request(options: RequestOptions): Promise<Response> {
  // 实现
  return fetch(options.url, {
    method: options.method || 'GET',
    headers: options.headers
  });
}

// ✅ 可以传入更多属性
request({
  url: '/api/data',
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ data: 'test' }) // 多余属性被忽略
});
```

### 工厂函数

```typescript
interface Animal {
  speak(): void;
}

class Dog implements Animal {
  speak() { console.log('Woof!'); }
}

class Cat implements Animal {
  speak() { console.log('Meow!'); }
}

type AnimalFactory<T extends Animal> = new () => T;

function createAnimals<T extends Animal>(
  factory: AnimalFactory<T>,
  count: number
): T[] {
  return Array.from({ length: count }, () => new factory());
}

const dogs = createAnimals(Dog, 3); // Dog[]
const cats = createAnimals(Cat, 2); // Cat[]
```

## 小结

| 概念 | 说明 |
|------|------|
| 结构化类型 | 基于成员结构判断兼容性 |
| 协变 | 子类型可以赋值给父类型 |
| 逆变 | 父类型可以赋值给子类型（函数参数） |
| 多余属性 | 源有多余属性可以赋值 |
| 缺少属性 | 源缺少属性不能赋值 |
