# 可辨识联合

可辨识联合（Discriminated Union），也称为标签联合或代数数据类型，是 TypeScript 中一种强大的模式。它使用公共的字面量属性来区分联合类型中的不同成员，使类型缩小变得简单可靠。

## 基本概念

### 什么是可辨识联合

```typescript
// 可辨识联合有三个要素：
// 1. 公共的辨识属性（discriminant）
// 2. 字面量类型的辨识值
// 3. 联合类型

interface Circle {
  kind: 'circle';        // 辨识属性 + 字面量类型
  radius: number;
}

interface Square {
  kind: 'square';        // 相同的辨识属性 + 不同字面量
  sideLength: number;
}

type Shape = Circle | Square;  // 联合类型
```

### 与普通联合类型的区别

```typescript
// 普通联合类型 - 需要使用 in 或 instanceof 缩小
interface Circle {
  radius: number;
}

interface Square {
  sideLength: number;
}

type Shape = Circle | Square;

function getArea(shape: Shape): number {
  if ('radius' in shape) {
    return Math.PI * shape.radius ** 2;
  }
  return shape.sideLength ** 2;
}

// 可辨识联合 - 使用公共属性缩小，更简洁
interface Circle2 {
  kind: 'circle';
  radius: number;
}

interface Square2 {
  kind: 'square';
  sideLength: number;
}

type Shape2 = Circle2 | Square2;

function getArea2(shape: Shape2): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2;
    case 'square':
      return shape.sideLength ** 2;
  }
}
```

## 定义可辨识联合

### 使用接口

```typescript
interface Loading {
  status: 'loading';
  progress: number;
}

interface Success {
  status: 'success';
  data: string;
}

interface Error {
  status: 'error';
  error: string;
}

type Result = Loading | Success | Error;
```

### 使用类型别名

```typescript
type Shape =
  | { kind: 'circle'; radius: number }
  | { kind: 'square'; sideLength: number }
  | { kind: 'triangle'; base: number; height: number };
```

### 使用枚举作为辨识值

```typescript
enum RequestStatus {
  Pending = 'pending',
  Loading = 'loading',
  Success = 'success',
  Error = 'error'
}

interface Pending {
  status: RequestStatus.Pending;
}

interface Loading {
  status: RequestStatus.Loading;
  progress: number;
}

interface Success {
  status: RequestStatus.Success;
  data: string;
}

interface Error {
  status: RequestStatus.Error;
  error: string;
}

type Request = Pending | Loading | Success | Error;
```

## 类型缩小

### switch 语句

```typescript
function handleShape(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      // shape 被缩小为 Circle
      return Math.PI * shape.radius ** 2;
    case 'square':
      // shape 被缩小为 Square
      return shape.sideLength ** 2;
    case 'triangle':
      // shape 被缩小为 Triangle
      return (shape.base * shape.height) / 2;
  }
}
```

### if 语句

```typescript
function handleResult(result: Result): string {
  if (result.status === 'loading') {
    return `Loading: ${result.progress}%`;
  }

  if (result.status === 'success') {
    return `Success: ${result.data}`;
  }

  if (result.status === 'error') {
    return `Error: ${result.error}`;
  }

  return 'Pending';
}
```

### 对象解构

```typescript
function processEvent(event: Event): void {
  const { type } = event;

  switch (type) {
    case 'click':
      // TypeScript 仍然能缩小 event 的类型
      const { x, y } = event as ClickEvent;
      console.log(`Clicked at (${x}, ${y})`);
      break;
    case 'keypress':
      console.log(`Key: ${(event as KeyEvent).key}`);
      break;
  }
}
```

## 穷尽性检查

### 使用 never 类型

```typescript
interface Circle {
  kind: 'circle';
  radius: number;
}

interface Square {
  kind: 'square';
  sideLength: number;
}

type Shape = Circle | Square;

function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2;
    case 'square':
      return shape.sideLength ** 2;
    default:
      // 如果所有情况都已处理，shape 是 never
      const _exhaustiveCheck: never = shape;
      throw new Error(`Unknown shape: ${_exhaustiveCheck}`);
  }
}
```

### 辅助函数

```typescript
function assertNever(value: never): never {
  throw new Error(`Unexpected value: ${value}`);
}

function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2;
    case 'square':
      return shape.sideLength ** 2;
    default:
      return assertNever(shape);
  }
}

// 添加新类型时，TypeScript 会报错提示需要处理
interface Triangle {
  kind: 'triangle';
  base: number;
  height: number;
}

type Shape2 = Circle | Square | Triangle;

function getArea2(shape: Shape2): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2;
    case 'square':
      return shape.sideLength ** 2;
    default:
      return assertNever(shape); // Error: Triangle 不能赋值给 never
  }
}
```

### noFallthroughCasesInSwitch

```typescript
// tsconfig.json
// "noFallthroughCasesInSwitch": true

// 帮助检测 switch 中的遗漏情况
function handle(status: 'a' | 'b' | 'c') {
  switch (status) {
    case 'a':
      console.log('a');
      // 如果没有 return 或 break，会警告
    case 'b':
      console.log('b');
      break;
  }
}
```

## 常见应用场景

### 状态管理

```typescript
// 异步操作状态
type AsyncState<T, E = Error> =
  | { status: 'idle' }
  | { status: 'loading' }
  | { status: 'success'; data: T }
  | { status: 'error'; error: E };

interface User {
  id: number;
  name: string;
}

function renderUserState(state: AsyncState<User>) {
  switch (state.status) {
    case 'idle':
      return <div>点击加载</div>;
    case 'loading':
      return <div>加载中...</div>;
    case 'success':
      return <div>{state.data.name}</div>;
    case 'error':
      return <div>错误: {state.error.message}</div>;
  }
}
```

### Redux Actions

```typescript
// Action 类型定义
interface Increment {
  type: 'INCREMENT';
}

interface Decrement {
  type: 'DECREMENT';
}

interface SetCount {
  type: 'SET_COUNT';
  payload: number;
}

interface AddTodo {
  type: 'ADD_TODO';
  payload: {
    id: number;
    text: string;
  };
}

type CounterAction = Increment | Decrement | SetCount;

function counterReducer(state: number, action: CounterAction): number {
  switch (action.type) {
    case 'INCREMENT':
      return state + 1;
    case 'DECREMENT':
      return state - 1;
    case 'SET_COUNT':
      return action.payload;
  }
}
```

### API 响应

```typescript
interface SuccessResponse<T> {
  success: true;
  data: T;
  message?: string;
}

interface ErrorResponse {
  success: false;
  error: {
    code: string;
    message: string;
  };
}

type ApiResponse<T> = SuccessResponse<T> | ErrorResponse;

async function fetchData<T>(url: string): Promise<T> {
  const response: ApiResponse<T> = await fetch(url).then(r => r.json());

  if (response.success) {
    return response.data;
  }

  throw new Error(`[${response.error.code}] ${response.error.message}`);
}
```

## 高级模式

### 嵌套可辨识联合

```typescript
interface BaseAction {
  type: string;
}

interface UserAction {
  type: 'user';
  action: {
    type: 'create' | 'update' | 'delete';
    userId: string;
    data?: { name: string; email: string };
  };
}

interface PostAction {
  type: 'post';
  action: {
    type: 'publish' | 'unpublish' | 'delete';
    postId: string;
  };
}

type AppAction = UserAction | PostAction;

function handleAction(appAction: AppAction) {
  switch (appAction.type) {
    case 'user':
      switch (appAction.action.type) {
        case 'create':
          console.log('Create user:', appAction.action.data);
          break;
        case 'update':
          console.log('Update user:', appAction.action.userId);
          break;
        case 'delete':
          console.log('Delete user:', appAction.action.userId);
          break;
      }
      break;
    case 'post':
      // ...
      break;
  }
}
```

### 泛型可辨识联合

```typescript
interface Success<T> {
  status: 'success';
  data: T;
}

interface Error<E = Error> {
  status: 'error';
  error: E;
}

type Result<T, E = Error> = Success<T> | Error<E>;

function mapResult<T, U, E>(
  result: Result<T, E>,
  fn: (data: T) => U
): Result<U, E> {
  if (result.status === 'success') {
    return { status: 'success', data: fn(result.data) };
  }
  return result;
}

// 使用
const success: Result<number> = { status: 'success', data: 42 };
const mapped = mapResult(success, n => n.toString());
// mapped: Result<string>
```

### 条件类型与可辨识联合

```typescript
type EventType<T> = T extends { type: infer U } ? U : never;

interface ClickEvent { type: 'click'; x: number; y: number; }
interface KeyEvent { type: 'keypress'; key: string; }

type Event = ClickEvent | KeyEvent;
type EventTypes = EventType<Event>; // 'click' | 'keypress'
```

## 最佳实践

### 命名约定

```typescript
// ✅ 好：使用语义化的辨识属性名
interface User {
  type: 'user';
  name: string;
}

interface Admin {
  type: 'admin';
  name: string;
  permissions: string[];
}

// ✅ 好：状态相关的使用 status
interface Loading {
  status: 'loading';
}

interface Success {
  status: 'success';
}

// ❌ 差：使用不同属性名
interface User2 {
  isUser: true;  // 不同的属性名
}

interface Admin2 {
  isAdmin: true; // 不同的属性名
}
```

### 共享属性

```typescript
// ✅ 好：共享属性放在基础接口
interface BaseAction {
  timestamp: number;
  userId: string;
}

interface AddAction extends BaseAction {
  type: 'add';
  payload: { text: string };
}

interface RemoveAction extends BaseAction {
  type: 'remove';
  payload: { id: number };
}

type Action = AddAction | RemoveAction;

function handleAction(action: Action) {
  // 所有 action 都有 timestamp 和 userId
  console.log(`User ${action.userId} at ${action.timestamp}`);

  switch (action.type) {
    case 'add':
      console.log('Add:', action.payload.text);
      break;
    case 'remove':
      console.log('Remove:', action.payload.id);
      break;
  }
}
```

### 类型安全的事件发射器

```typescript
type EventMap = {
  click: { x: number; y: number };
  focus: { target: HTMLElement };
  keypress: { key: string; code: string };
};

type EventType = keyof EventMap;

type Event<T extends EventType> = {
  type: T;
} & EventMap[T];

class TypedEventEmitter {
  private listeners = new Map<EventType, Set<Function>>();

  on<T extends EventType>(
    type: T,
    listener: (event: Event<T>) => void
  ): void {
    if (!this.listeners.has(type)) {
      this.listeners.set(type, new Set());
    }
    this.listeners.get(type)!.add(listener);
  }

  emit<T extends EventType>(event: Event<T>): void {
    const handlers = this.listeners.get(event.type);
    handlers?.forEach(handler => handler(event));
  }
}
```

## 小结

| 要素 | 说明 |
|------|------|
| 辨识属性 | 公共的字面量类型属性 |
| 联合类型 | 组合所有变体 |
| 类型缩小 | 通过 switch/if 检查辨识属性 |
| 穷尽检查 | 使用 never 类型确保处理所有情况 |

可辨识联合是 TypeScript 中处理复杂状态和事件流的强大工具，结合穷尽性检查可以确保代码的类型安全性。
