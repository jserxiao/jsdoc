## 递归类型基础

### 基本语法

```typescript
interface TreeNode {
  value: number;
  left?: TreeNode;
  right?: TreeNode;
}

const tree: TreeNode = {
  value: 1,
  left: {
    value: 2,
    left: { value: 4 },
    right: { value: 5 }
  },
  right: {
    value: 3
  }
};
```

### 数组的递归定义

```typescript
type NestedArray<T> = T | NestedArray<T>[];

const nested: NestedArray<number> = [1, [2, 3], [[4], 5]];
```

## 深度工具类型

### DeepPartial

```typescript
type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object 
    ? DeepPartial<T[P]> 
    : T[P];
};

interface Config {
  server: {
    host: string;
    port: number;
  };
  database: {
    name: string;
    tables: string[];
  };
}

const config: DeepPartial<Config> = {
  server: {
    // host 可选
  }
};
```

### DeepRequired

```typescript
type DeepRequired<T> = {
  [P in keyof T]-?: T[P] extends object 
    ? DeepRequired<T[P]> 
    : T[P];
};
```

### DeepReadonly

```typescript
type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object 
    ? DeepReadonly<T[P]> 
    : T[P];
};

const config: DeepReadonly<Config> = {
  server: { host: 'localhost', port: 3000 }
};
// config.server.host = 'new'; // Error
```

### DeepMutable

```typescript
type DeepMutable<T> = {
  -readonly [P in keyof T]: T[P] extends object 
    ? DeepMutable<T[P]> 
    : T[P];
};
```

## JSON 类型

### 递归 JSON 类型

```typescript
type JSONValue = 
  | string 
  | number 
  | boolean 
  | null 
  | JSONValue[] 
  | { [key: string]: JSONValue };

function parseJSON(json: string): JSONValue {
  return JSON.parse(json);
}

const data: JSONValue = {
  name: 'Alice',
  age: 30,
  hobbies: ['reading', 'coding'],
  address: {
    city: 'New York'
  }
};
```

### 类型安全的 JSON

```typescript
interface JSONObject {
  [key: string]: JSONValue;
}

type JSONArray = JSONValue[];

type JSONValue = 
  | string 
  | number 
  | boolean 
  | null 
  | JSONObject 
  | JSONArray;
```

## 树结构类型

### 通用树结构

```typescript
interface Tree<T> {
  value: T;
  children?: Tree<T>[];
}

const fileTree: Tree<string> = {
  value: 'root',
  children: [
    { value: 'src', children: [{ value: 'index.ts' }] },
    { value: 'dist' }
  ]
};
```

### 文件系统类型

```typescript
type FileType = 'file' | 'directory';

interface FileSystemNode {
  name: string;
  type: FileType;
  children?: FileSystemNode[];
  content?: string;
}

const fs: FileSystemNode = {
  name: 'project',
  type: 'directory',
  children: [
    {
      name: 'src',
      type: 'directory',
      children: [
        { name: 'index.ts', type: 'file', content: '...' }
      ]
    }
  ]
};
```

### 遍历树

```typescript
function traverse<T>(
  node: Tree<T>,
  callback: (value: T, depth: number) => void,
  depth: number = 0
): void {
  callback(node.value, depth);
  node.children?.forEach(child => traverse(child, callback, depth + 1));
}

traverse(fileTree, (value, depth) => {
  console.log('  '.repeat(depth) + value);
});
```

## 递归限制

TypeScript 对递归深度有限制，过度递归会导致错误：

```typescript
// 可能导致 "Type instantiation is excessively deep"
type DeepArray<T> = T | DeepArray<T>[];

// 解决方案：限制递归深度
type DeepArrayLimited<T, D extends number, C extends any[] = []> = 
  C['length'] extends D 
    ? T 
    : T | DeepArrayLimited<T, D, [...C, 0]>;

type SafeDeepArray<T> = DeepArrayLimited<T, 10>;
```
