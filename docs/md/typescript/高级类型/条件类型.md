## 基本语法

### 条件表达式

```typescript
type IsString<T> = T extends string ? true : false;

type A = IsString<string>;   // true
type B = IsString<number>;   // false
type C = IsString<'hello'>;  // true
```

### 嵌套条件类型

```typescript
type TypeName<T> = 
  T extends string ? 'string' :
  T extends number ? 'number' :
  T extends boolean ? 'boolean' :
  T extends undefined ? 'undefined' :
  T extends Function ? 'function' :
  'object';

type T1 = TypeName<string>;    // 'string'
type T2 = TypeName<number[]>;  // 'object'
type T3 = TypeName<() => void>;// 'function'
```

## 分布式条件类型

### 自动分发

```typescript
type ToArray<T> = T extends any ? T[] : never;

type Result = ToArray<string | number>;
// string[] | number[]
```

### 阻止分发

```typescript
type ToArrayNonDist<T> = [T] extends [any] ? T[] : never;

type Result = ToArrayNonDist<string | number>;
// (string | number)[]
```

### 过滤类型

```typescript
type Filter<T, U> = T extends U ? T : never;

type Result = Filter<string | number | boolean, string | number>;
// string | number

// 内置 Exclude
type Exclude<T, U> = T extends U ? never : T;
type NonString = Exclude<string | number | boolean, string>;
// number | boolean

// 内置 Extract
type Extract<T, U> = T extends U ? T : never;
type StringOrNumber = Extract<string | number | boolean, string | number>;
// string | number
```

### NonNullable

```typescript
type NonNullable<T> = T extends null | undefined ? never : T;

type Result = NonNullable<string | null | undefined>;
// string
```

## infer 关键字

### 提取返回类型

```typescript
type GetReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

type Result = GetReturnType<() => string>;  // string
type Result2 = GetReturnType<(x: number) => boolean>;  // boolean
```

### 提取参数类型

```typescript
type GetParameters<T> = T extends (...args: infer P) => any ? P : never;

type Params = GetParameters<(a: string, b: number) => void>;
// [string, number]
```

### 提取 Promise 值

```typescript
type UnwrapPromise<T> = T extends Promise<infer R> ? R : T;

type A = UnwrapPromise<Promise<string>>;  // string
type B = UnwrapPromise<string>;           // string
```

### 提取数组元素

```typescript
type UnwrapArray<T> = T extends (infer R)[] ? R : T;

type Item = UnwrapArray<string[]>;  // string
```

### 提取元组第一个

```typescript
type First<T extends any[]> = T extends [infer F, ...any[]] ? F : never;

type Head = First<[1, 2, 3]>;  // 1
```

### 提取元组最后一个

```typescript
type Last<T extends any[]> = T extends [...any[], infer L] ? L : never;

type Tail = Last<[1, 2, 3]>;  // 3
```

## 条件类型约束

### 约束 infer 变量

```typescript
type GetProperty<T, K extends string> = 
  T extends { [P in K]: infer V } ? V : never;

interface User {
  name: string;
  age: number;
}

type Name = GetProperty<User, 'name'>;  // string
```

### 多个 infer

```typescript
type Swap<T extends [any, any]> = 
  T extends [infer A, infer B] ? [B, A] : T;

type Result = Swap<[string, number]>;  // [number, string]
```

### 条件返回

```typescript
type FunctionPropertyNames<T> = {
  [K in keyof T]: T[K] extends Function ? K : never
}[keyof T];

interface User {
  name: string;
  greet(): void;
  celebrate(): void;
}

type Methods = FunctionPropertyNames<User>;
// 'greet' | 'celebrate'
```

## 实际应用

### 深度解包

```typescript
type DeepUnwrap<T> = 
  T extends Promise<infer R> ? DeepUnwrap<R> :
  T extends Array<infer R> ? DeepUnwrap<R> :
  T;

type Result = DeepUnwrap<Promise<Promise<string[]>>>;
// string
```

### 类型谓词

```typescript
type IsArray<T> = T extends any[] ? true : false;

type IsTuple<T> = T extends readonly any[] 
  ? number extends T['length'] 
    ? false 
    : true 
  : false;

type A = IsTuple<[1, 2]>;    // true
type B = IsTuple<number[]>;  // false
```
