## 基本推断

### 变量推断

```typescript
let x = 10;           // number
let y = 'hello';      // string
let z = [1, 2, 3];    // number[]
let obj = { a: 1 };   // { a: number }
```

### 函数返回值推断

```typescript
function add(a: number, b: number) {
  return a + b;       // 返回类型推断为 number
}

const result = add(1, 2);  // number
```

### 数组元素推断

```typescript
const arr = [1, 2, 'three'];  // (string | number)[]

// 使用 as const 获得更精确的类型
const tuple = [1, 2, 'three'] as const;  
// readonly [1, 2, 'three']
```

## 最佳通用类型

### 多类型推断

```typescript
const mixed = [1, 'hello', true];  // (string | number | boolean)[]
```

### 对象类型推断

```typescript
const obj = {
  name: 'Alice',
  age: 30,
  greet() {
    return `Hello, ${this.name}`;
  }
};

// 推断类型
// {
//   name: string;
//   age: number;
//   greet(): string;
// }
```

### 类型拓宽

```typescript
let x = 'hello';      // string
const y = 'hello';    // 'hello' (字面量类型)

let arr = [1, 2, 3];  // number[]
const arr2 = [1, 2, 3] as const;  // readonly [1, 2, 3]
```

## 上下文类型

### 函数参数推断

```typescript
const numbers = [1, 2, 3];
numbers.forEach(n => {
  console.log(n.toFixed(2));  // n 自动推断为 number
});
```

### 事件处理推断

```typescript
document.addEventListener('click', (event) => {
  // event 自动推断为 MouseEvent
  console.log(event.clientX);
});
```

### 回调函数推断

```typescript
type Callback = (value: string) => void;

function run(cb: Callback) {
  cb('hello');
}

run((value) => {
  // value 自动推断为 string
  console.log(value.toUpperCase());
});
```

### Promise 链式推断

```typescript
Promise.resolve('hello')
  .then(str => {
    // str: string
    return str.length;
  })
  .then(len => {
    // len: number
    return len.toFixed(2);
  })
  .then(str => {
    // str: string
  });
```

## 类型推断位置

### 函数类型推断

```typescript
// 参数类型需要显式声明
function greet(name: string) {
  return `Hello, ${name}`;
}

// 或者使用上下文类型
type GreetFn = (name: string) => string;
const greet2: GreetFn = (name) => `Hello, ${name}`;
```

### 泛型推断

```typescript
function identity<T>(arg: T): T {
  return arg;
}

// 自动推断 T
const result = identity('hello');  // string

// 显式指定 T
const result2 = identity<string>('hello');  // string
```

### 复杂类型推断

```typescript
function map<T, U>(arr: T[], fn: (item: T) => U): U[] {
  return arr.map(fn);
}

const result = map([1, 2, 3], x => x.toString());
// result: string[]
```

## 推断限制

### 无法推断的情况

```typescript
// 参数需要显式类型
function fn(x) {  // Error: 参数 'x' 隐式具有 'any' 类型
  return x;
}

// 解决方案
function fn(x: unknown) {
  return x;
}
```

### 类型断言

```typescript
let value: unknown = 'hello';

// 使用类型断言
const length = (value as string).length;

// 或使用类型守卫
if (typeof value === 'string') {
  const length = value.length;
}
```

## infer 关键字

### 在条件类型中推断

```typescript
type UnwrapPromise<T> = T extends Promise<infer R> ? R : T;

type Result = UnwrapPromise<Promise<string>>;  // string
```

### 提取函数类型

```typescript
type GetReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

type Result = GetReturnType<() => string>;  // string
```
