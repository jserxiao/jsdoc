# 映射类型

映射类型（Mapped Types）是 TypeScript 中一种强大的类型构造方式，它允许基于旧类型创建新类型。通过映射类型，可以批量转换类型的属性。

## 概述

### 什么是映射类型

映射类型建立在索引签名的语法之上，用于遍历类型的属性并创建新类型。

```typescript
// 索引签名
type StringMap = {
  [key: string]: string;
};

// 映射类型：遍历另一个类型的键
type Readonly<T> = {
  readonly [P in keyof T]: T[P];
};

interface User {
  name: string;
  age: number;
}

type ReadonlyUser = Readonly<User>;
// { readonly name: string; readonly age: number }
```

## 基本语法

### 基本结构

```typescript
type MappedType<T> = {
  [P in keyof T]: T[P];
};

// 示例
interface Person {
  name: string;
  age: number;
  email: string;
}

type PersonCopy = {
  [P in keyof Person]: Person[P];
};
// { name: string; age: number; email: string }
```

### keyof 操作符

```typescript
interface User {
  id: number;
  name: string;
  email: string;
}

type UserKeys = keyof User; // 'id' | 'name' | 'email'

// 遍历键
type KeyOf<T> = {
  [P in keyof T]: P;
};

type UserKeyValues = KeyOf<User>;
// { id: 'id'; name: 'name'; email: 'email' }
```

### 属性值访问

```typescript
interface Config {
  host: string;
  port: number;
  secure: boolean;
}

// T[P] 获取属性类型
type ValueOf<T> = T[keyof T];
type ConfigValues = ValueOf<Config>; // string | number | boolean

// 创建值类型映射
type ValueTypes<T> = {
  [P in keyof T]: T[P] extends string ? 'string' : 'other';
};

type ConfigValueTypes = ValueTypes<Config>;
// { host: 'string'; port: 'other'; secure: 'other' }
```

## 内置映射类型

### Partial

```typescript
// 内置实现
type Partial<T> = {
  [P in keyof T]?: T[P];
};

interface User {
  id: number;
  name: string;
  email: string;
}

type PartialUser = Partial<User>;
// { id?: number; name?: string; email?: string }

// 使用示例
function updateUser(user: User, updates: Partial<User>): User {
  return { ...user, ...updates };
}

updateUser({ id: 1, name: 'Alice', email: 'a@b.com' }, { name: 'Bob' });
```

### Required

```typescript
// 内置实现
type Required<T> = {
  [P in keyof T]-?: T[P];
};

interface Config {
  host?: string;
  port?: number;
}

type RequiredConfig = Required<Config>;
// { host: string; port: number }（所有属性必需）
```

### Readonly

```typescript
// 内置实现
type Readonly<T> = {
  readonly [P in keyof T]: T[P];
};

interface Point {
  x: number;
  y: number;
}

type ReadonlyPoint = Readonly<Point>;
// { readonly x: number; readonly y: number }

const point: ReadonlyPoint = { x: 10, y: 20 };
// point.x = 30; // Error: 无法修改只读属性
```

### Pick

```typescript
// 内置实现
type Pick<T, K extends keyof T> = {
  [P in K]: T[P];
};

interface User {
  id: number;
  name: string;
  email: string;
  password: string;
}

type PublicUser = Pick<User, 'id' | 'name' | 'email'>;
// { id: number; name: string; email: string }
```

### Omit

```typescript
// 内置实现
type Omit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>;

interface User {
  id: number;
  name: string;
  email: string;
  password: string;
}

type SafeUser = Omit<User, 'password'>;
// { id: number; name: string; email: string }
```

### Record

```typescript
// 内置实现
type Record<K extends keyof any, V> = {
  [P in K]: V;
};

type UserRole = 'admin' | 'user' | 'guest';

type RolePermissions = Record<UserRole, string[]>;
// {
//   admin: string[];
//   user: string[];
//   guest: string[];
// }

const permissions: RolePermissions = {
  admin: ['read', 'write', 'delete'],
  user: ['read', 'write'],
  guest: ['read']
};
```

## 修饰符

### 添加修饰符

```typescript
// 添加 readonly
type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object
    ? DeepReadonly<T[P]>
    : T[P];
};

interface Nested {
  user: {
    name: string;
    address: {
      city: string;
    };
  };
}

type ReadonlyNested = DeepReadonly<Nested>;

// 添加可选
type Optional<T> = {
  [P in keyof T]?: T[P];
};
```

### 移除修饰符

```typescript
// 移除 readonly
type Mutable<T> = {
  -readonly [P in keyof T]: T[P];
};

interface ReadonlyUser {
  readonly id: number;
  readonly name: string;
}

type MutableUser = Mutable<ReadonlyUser>;
// { id: number; name: string }（可修改）

// 移除可选
type Required<T> = {
  [P in keyof T]-?: T[P];
};

interface PartialConfig {
  host?: string;
  port?: number;
}

type FullConfig = Required<PartialConfig>;
// { host: string; port: number }
```

### 组合修饰符

```typescript
// 同时添加 readonly 和可选
type ReadonlyPartial<T> = {
  readonly [P in keyof T]?: T[P];
};

// 同时移除 readonly 和可选
type MutableRequired<T> = {
  -readonly [P in keyof T]-?: T[P];
};
```

## 键重映射

### as 子句（TypeScript 4.1+）

```typescript
// 重命名键
type Mapped<T> = {
  [P in keyof T as `get${Capitalize<string & P>}`]: () => T[P];
};

interface User {
  name: string;
  age: number;
}

type Getters = Mapped<User>;
// { getName: () => string; getAge: () => number }
```

### 键过滤

```typescript
// 只保留函数类型属性
type FunctionProperties<T> = {
  [P in keyof T as T[P] extends Function ? P : never]: T[P];
};

// 只保留非函数类型属性
type NonFunctionProperties<T> = {
  [P in keyof T as T[P] extends Function ? never : P]: T[P];
};

interface Service {
  name: string;
  version: number;
  start(): void;
  stop(): void;
}

type ServiceMethods = FunctionProperties<Service>;
// { start: () => void; stop: () => void }

type ServiceData = NonFunctionProperties<Service>;
// { name: string; version: number }
```

### 键转换

```typescript
// 添加前缀
type Prefixed<T, P extends string> = {
  [K in keyof T as `${P}${Capitalize<string & K>}`]: T[K];
};

interface Data {
  id: number;
  name: string;
}

type ApiData = Prefixed<Data, 'api'>;
// { apiId: number; apiName: string }

// 移除前缀
type RemovePrefix<T, P extends string> = {
  [K in keyof T as K extends `${P}${infer Rest}`
    ? Uncapitalize<Rest>
    : K]: T[K];
};
```

## 条件映射

### 基于属性值类型映射

```typescript
// 将所有 string 类型变为可选
type OptionalStrings<T> = {
  [P in keyof T]: T[P] extends string ? T[P] | undefined : T[P];
};

interface Config {
  host: string;
  port: number;
  debug: boolean;
}

type ConfigWithOptionalStrings = OptionalStrings<Config>;
// { host: string | undefined; port: number; debug: boolean }
```

### 类型转换映射

```typescript
// 将所有属性变为字符串
type Stringify<T> = {
  [P in keyof T]: string;
};

interface Numbers {
  a: number;
  b: number;
  c: number;
}

type StringNumbers = Stringify<Numbers>;
// { a: string; b: string; c: string }

// 将所有属性变为数组
type Arrayify<T> = {
  [P in keyof T]: T[P][];
};

type NumberArrays = Arrayify<Numbers>;
// { a: number[]; b: number[]; c: number[] }
```

### 深度映射

```typescript
// 深度只读
type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object
    ? T[P] extends Function
      ? T[P]
      : DeepReadonly<T[P]>
    : T[P];
};

// 深度可选
type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object
    ? T[P] extends Function
      ? T[P]
      : DeepPartial<T[P]>
    : T[P];
};

interface DeepConfig {
  server: {
    host: string;
    port: number;
    ssl: {
      enabled: boolean;
      cert: string;
    };
  };
}

type DeepPartialConfig = DeepPartial<DeepConfig>;
// 所有层级属性都变为可选
```

## 模板字面量映射

### getter/setter 生成

```typescript
type Getters<T> = {
  [P in keyof T as `get${Capitalize<string & P>}`]: () => T[P];
};

type Setters<T> = {
  [P in keyof T as `set${Capitalize<string & P>}`]: (value: T[P]) => void;
};

interface State {
  name: string;
  count: number;
}

type StateGetters = Getters<State>;
// { getName: () => string; getCount: () => number }

type StateSetters = Setters<State>;
// { setName: (value: string) => void; setCount: (value: number) => void }
```

### 事件处理器生成

```typescript
type EventHandlers<T> = {
  [P in keyof T as `on${Capitalize<string & P>}`]: (event: T[P]) => void;
};

interface Events {
  click: { x: number; y: number };
  scroll: { scrollTop: number };
}

type Handlers = EventHandlers<Events>;
// { onClick: (event: { x: number; y: number }) => void; onScroll: (event: { scrollTop: number }) => void }
```

## 实际应用场景

### 表单验证

```typescript
// 验证错误类型
type ValidationErrors<T> = {
  [P in keyof T]?: string[];
};

interface FormData {
  username: string;
  email: string;
  password: string;
}

type FormErrors = ValidationErrors<FormData>;
// { username?: string[]; email?: string[]; password?: string[] }
```

### API 响应类型

```typescript
// API 响应包装
type ApiResponse<T> = {
  [P in keyof T]: {
    data: T[P];
    status: 'success' | 'error';
    timestamp: number;
  };
};

interface Endpoints {
  users: User[];
  posts: Post[];
}

type ApiResponses = ApiResponse<Endpoints>;
```

### 状态管理

```typescript
// 状态类型生成
type State<T> = {
  data: T;
  loading: boolean;
  error: string | null;
};

type States<T extends Record<string, any>> = {
  [P in keyof T]: State<T[P]>;
};

interface DataModels {
  users: User[];
  posts: Post[];
  comments: Comment[];
}

type AppState = States<DataModels>;
// { users: State<User[]>; posts: State<Post[]>; comments: State<Comment[]> }
```

## 高级技巧

### 类型推断与映射

```typescript
// 获取函数返回值类型
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

// 获取函数参数类型
type Parameters<T> = T extends (...args: infer P) => any ? P : never;

// 获取构造函数实例类型
type InstanceType<T> = T extends new (...args: any[]) => infer R ? R : never;

function createUser() {
  return { id: 1, name: 'Alice' };
}

type User = ReturnType<typeof createUser>;
// { id: number; name: string }
```

### 联合类型与映射

```typescript
// 联合类型转交叉类型
type UnionToIntersection<U> = (
  U extends any ? (k: U) => void : never
) extends (k: infer I) => void
  ? I
  : never;

type Union = { a: string } | { b: number };
type Intersection = UnionToIntersection<Union>;
// { a: string } & { b: number }
```

### 递归映射

```typescript
// 递归将所有属性变为可选
type DeepPartial<T> = T extends object
  ? { [P in keyof T]?: DeepPartial<T[P]> }
  : T;

// 递归将所有属性变为只读
type DeepReadonly<T> = T extends object
  ? { readonly [P in keyof T]: DeepReadonly<T[P]> }
  : T;

// 深度合并类型
type DeepMerge<T, U> = {
  [P in keyof T | keyof U]: P extends keyof T
    ? P extends keyof U
      ? T[P] extends object
        ? U[P] extends object
          ? DeepMerge<T[P], U[P]>
          : U[P]
        : U[P]
      : T[P]
    : P extends keyof U
    ? U[P]
    : never;
};
```

## 小结

| 语法 | 说明 |
|------|------|
| `[P in keyof T]` | 遍历 T 的所有属性键 |
| `T[P]` | 获取属性值类型 |
| `readonly` | 添加只读修饰符 |
| `-readonly` | 移除只读修饰符 |
| `?` | 添加可选修饰符 |
| `-?` | 移除可选修饰符 |
| `as NewKey` | 键重映射 |
| `infer` | 条件类型推断 |
