## 品牌类型

### 1. 基本品牌类型

品牌类型（Branded Type）用于在类型层面区分相同基础类型的不同用途。

```typescript
// 基本品牌类型
type Brand<T, B> = T & { __brand: B };

// 定义特定类型
type UserId = Brand<string, 'UserId'>;
type OrderId = Brand<string, 'OrderId'>;
type Email = Brand<string, 'Email'>;
type Url = Brand<string, 'Url'>;

// 使用
function getUser(id: UserId): User { /* ... */ }
function getOrder(id: OrderId): Order { /* ... */ }

const userId = 'user-123' as UserId;
const orderId = 'order-456' as OrderId;

getUser(userId);  // ✅ OK
getUser(orderId); // ❌ Error: OrderId 不能赋值给 UserId
```

### 2. 品牌类型工厂

```typescript
// 创建品牌类型的工厂函数
function createBrand<T, B extends string>() {
  return {
    create(value: T): Brand<T, B> {
      return value as Brand<T, B>;
    },
    is(value: unknown): value is Brand<T, B> {
      return typeof value === typeof (null as unknown as T);
    }
  };
}

// 使用
const UserId = createBrand<string, 'UserId'>();
const OrderId = createBrand<string, 'OrderId'>();

const userId = UserId.create('user-123');
const orderId = OrderId.create('order-456');

if (UserId.is(someValue)) {
  // someValue 是 UserId 类型
}
```

### 3. 带验证的品牌类型

```typescript
// 带验证的品牌类型
type ValidatedBrand<T, B> = T & { __brand: B; __validated: true };

function createEmail(value: string): ValidatedBrand<string, 'Email'> | null {
  if (/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) {
    return value as ValidatedBrand<string, 'Email'>;
  }
  return null;
}

function sendEmail(email: ValidatedBrand<string, 'Email'>, content: string) {
  // email 已经验证过格式
}

// 使用
const email = createEmail('user@example.com');
if (email) {
  sendEmail(email, 'Hello!'); // ✅ OK
}

// sendEmail('invalid', 'Hello!'); // ❌ Error
```

### 4. 整数类型

```typescript
// 创建整数类型
type Int = Brand<number, 'Int'>;

function isInt(value: number): value is Int {
  return Number.isInteger(value);
}

function toInt(value: number): Int {
  if (!isInt(value)) {
    throw new Error('Value is not an integer');
  }
  return value as Int;
}

// 安全的整数运算
function safeDivide(a: Int, b: Int): Int | null {
  if (b === 0) return null;
  return Math.floor(a / b) as Int;
}
```

## 穷尽检查

### 1. never 类型检查

```typescript
// 使用 never 进行穷尽检查
type Shape = 'circle' | 'square' | 'triangle';

function getArea(shape: Shape): number {
  switch (shape) {
    case 'circle':
      return Math.PI * 1 * 1;
    case 'square':
      return 1 * 1;
    case 'triangle':
      return 0.5 * 1 * 1;
    default:
      // 如果漏掉任何 case，这里会报错
      const _exhaustiveCheck: never = shape;
      return _exhaustiveCheck;
  }
}

// 添加新类型后
// type Shape = 'circle' | 'square' | 'triangle' | 'rectangle';
// default 分支会报错，提示需要处理 'rectangle'
```

### 2. 穷尽检查工具函数

```typescript
// 穷尽检查工具
function assertNever(value: never): never {
  throw new Error(`Unexpected value: ${value}`);
}

// 使用
type Action =
  | { type: 'INCREMENT' }
  | { type: 'DECREMENT' }
  | { type: 'RESET' };

function reducer(state: number, action: Action): number {
  switch (action.type) {
    case 'INCREMENT':
      return state + 1;
    case 'DECREMENT':
      return state - 1;
    case 'RESET':
      return 0;
    default:
      return assertNever(action);
  }
}
```

### 3. 对象穷尽检查

```typescript
// 对象类型的穷尽检查
type EventMap = {
  click: { x: number; y: number };
  focus: { target: HTMLElement };
  blur: { target: HTMLElement };
};

type EventKey = keyof EventMap;

function handleEvent<K extends EventKey>(
  type: K,
  event: EventMap[K]
): void {
  switch (type) {
    case 'click':
      // event 类型为 { x: number; y: number }
      console.log(`Clicked at (${event.x}, ${event.y})`);
      break;
    case 'focus':
    case 'blur':
      // event 类型为 { target: HTMLElement }
      console.log(`${type} on ${event.target.tagName}`);
      break;
    default:
      const _check: never = type;
      return _check;
  }
}
```

## 类型守卫进阶

### 1. 自定义类型守卫

```typescript
// 基本类型守卫
function isString(value: unknown): value is string {
  return typeof value === 'string';
}

function isNumber(value: unknown): value is number {
  return typeof value === 'number';
}

function isObject(value: unknown): value is Record<string, unknown> {
  return typeof value === 'object' && value !== null;
}

// 使用
function process(value: unknown) {
  if (isString(value)) {
    console.log(value.toUpperCase());
  } else if (isNumber(value)) {
    console.log(value.toFixed(2));
  }
}
```

### 2. 属性检查守卫

```typescript
// 检查对象是否有特定属性
function hasProperty<T extends string>(
  obj: unknown,
  prop: T
): obj is Record<T, unknown> {
  return typeof obj === 'object' && obj !== null && prop in obj;
}

// 检查多个属性
function hasProperties<T extends string>(
  obj: unknown,
  ...props: T[]
): obj is Record<T, unknown> {
  return (
    typeof obj === 'object' &&
    obj !== null &&
    props.every(prop => prop in obj)
  );
}

// 使用
interface User {
  id: string;
  name: string;
  email: string;
}

function isUser(value: unknown): value is User {
  return hasProperties(value, 'id', 'name', 'email');
}
```

### 3. 可辨识联合守卫

```typescript
// 可辨识联合
type Result<T, E = Error> =
  | { success: true; value: T }
  | { success: false; error: E };

function isSuccess<T, E>(result: Result<T, E>): result is { success: true; value: T } {
  return result.success === true;
}

function isError<T, E>(result: Result<T, E>): result is { success: false; error: E } {
  return result.success === false;
}

// 使用
function divide(a: number, b: number): Result<number, string> {
  if (b === 0) {
    return { success: false, error: 'Division by zero' };
  }
  return { success: true, value: a / b };
}

const result = divide(10, 2);
if (isSuccess(result)) {
  console.log(result.value); // number
} else {
  console.log(result.error); // string
}
```

### 4. 断言函数

```typescript
// 断言函数
function assert(condition: boolean, message?: string): asserts condition {
  if (!condition) {
    throw new Error(message ?? 'Assertion failed');
  }
}

function assertIsString(value: unknown): asserts value is string {
  if (typeof value !== 'string') {
    throw new Error('Value is not a string');
  }
}

function assertDefined<T>(value: T | null | undefined): asserts value is T {
  if (value === null || value === undefined) {
    throw new Error('Value is null or undefined');
  }
}

// 使用
function process(value: unknown) {
  assertIsString(value);
  // value 现在是 string 类型
  console.log(value.toUpperCase());
}

function processArray<T>(arr: T[] | null) {
  assertDefined(arr);
  // arr 现在是 T[] 类型
  console.log(arr.length);
}
```

## 不可变数据

### 1. 深度只读

```typescript
// 深度只读类型
type DeepReadonly<T> = T extends Function
  ? T
  : T extends Array<infer U>
  ? ReadonlyArray<DeepReadonly<U>>
  : T extends object
  ? { readonly [K in keyof T]: DeepReadonly<T[K]> }
  : T;

// 使用
interface Config {
  apiUrl: string;
  settings: {
    theme: string;
    options: {
      darkMode: boolean;
    };
  };
}

const config: DeepReadonly<Config> = {
  apiUrl: 'https://api.example.com',
  settings: {
    theme: 'dark',
    options: {
      darkMode: true
    }
  }
};

// config.apiUrl = 'new-url'; // ❌ Error
// config.settings.theme = 'light'; // ❌ Error
```

### 2. 不可变数组

```typescript
// 不可变数组操作
function immutablePush<T>(arr: ReadonlyArray<T>, item: T): readonly T[] {
  return [...arr, item];
}

function immutableRemove<T>(arr: ReadonlyArray<T>, index: number): readonly T[] {
  return [...arr.slice(0, index), ...arr.slice(index + 1)];
}

function immutableUpdate<T>(arr: ReadonlyArray<T>, index: number, item: T): readonly T[] {
  return [...arr.slice(0, index), item, ...arr.slice(index + 1)];
}

// 使用
const original = [1, 2, 3] as const;
const added = immutablePush(original, 4);     // readonly [1, 2, 3, 4]
const removed = immutableRemove(original, 1); // readonly [1, 3]
const updated = immutableUpdate(original, 1, 5); // readonly [1, 5, 3]
```

### 3. 不可变对象

```typescript
// 不可变对象操作
type ImmutableUpdate<T, K extends keyof T, V extends T[K]> = Omit<T, K> & {
  [P in K]: V;
};

function updateAt<T extends object, K extends keyof T>(
  obj: T,
  key: K,
  value: T[K]
): T {
  return { ...obj, [key]: value };
}

// 使用
interface User {
  id: string;
  name: string;
  age: number;
}

const user: User = { id: '1', name: 'Alice', age: 30 };
const updatedUser = updateAt(user, 'age', 31);
// { id: '1', name: 'Alice', age: 31 }
```

### 4. 冻结工具

```typescript
// 深度冻结
function deepFreeze<T extends object>(obj: T): DeepReadonly<T> {
  Object.keys(obj).forEach(key => {
    const value = (obj as any)[key];
    if (typeof value === 'object' && value !== null) {
      deepFreeze(value);
    }
  });
  return Object.freeze(obj) as DeepReadonly<T>;
}

// 使用
const config = deepFreeze({
  apiUrl: 'https://api.example.com',
  settings: {
    theme: 'dark'
  }
});

// 运行时也会阻止修改
// config.apiUrl = 'new'; // Error in strict mode
```

## 错误处理类型

### 1. Result 类型

```typescript
// Result 类型
type Result<T, E = Error> =
  | { ok: true; value: T }
  | { ok: false; error: E };

// 构造函数
function Ok<T>(value: T): Result<T, never> {
  return { ok: true, value };
}

function Err<E>(error: E): Result<never, E> {
  return { ok: false, error };
}

// 使用
function divide(a: number, b: number): Result<number, string> {
  if (b === 0) {
    return Err('Division by zero');
  }
  return Ok(a / b);
}

const result = divide(10, 2);
if (result.ok) {
  console.log(result.value);
} else {
  console.error(result.error);
}
```

### 2. Try-Catch 类型安全

```typescript
// 类型安全的 try-catch
async function safeAsync<T>(
  fn: () => Promise<T>
): Promise<Result<T, unknown>> {
  try {
    const value = await fn();
    return { ok: true, value };
  } catch (error) {
    return { ok: false, error };
  }
}

// 使用
const result = await safeAsync(() => fetch('/api/users').then(r => r.json()));
if (result.ok) {
  // result.value 是 json 结果
} else {
  // result.error 是错误
}
```

### 3. Option 类型

```typescript
// Option 类型
type Option<T> = Some<T> | None;

interface Some<T> {
  _tag: 'Some';
  value: T;
}

interface None {
  _tag: 'None';
}

// 构造函数
function Some<T>(value: T): Option<T> {
  return { _tag: 'Some', value };
}

const None: None = { _tag: 'None' };

// 使用
function findUser(id: string): Option<User> {
  const user = database.get(id);
  return user ? Some(user) : None;
}

const result = findUser('1');
if (result._tag === 'Some') {
  console.log(result.value);
} else {
  console.log('User not found');
}
```

### 4. Either 类型

```typescript
// Either 类型
type Either<L, R> = Left<L> | Right<R>;

interface Left<L> {
  _tag: 'Left';
  left: L;
}

interface Right<R> {
  _tag: 'Right';
  right: R;
}

function Left<L>(value: L): Either<L, never> {
  return { _tag: 'Left', left: value };
}

function Right<R>(value: R): Either<never, R> {
  return { _tag: 'Right', right: value };
}

// 使用
function validateAge(age: number): Either<string, number> {
  if (age < 0) {
    return Left('Age cannot be negative');
  }
  if (age > 150) {
    return Left('Age is too high');
  }
  return Right(age);
}

const result = validateAge(25);
if (result._tag === 'Right') {
  console.log(`Valid age: ${result.right}`);
} else {
  console.log(`Error: ${result.left}`);
}
```

## 运行时验证

### 1. 类型守卫 + 运行时检查

```typescript
// 运行时类型检查
interface User {
  id: string;
  name: string;
  age: number;
  email: string;
}

function isUser(value: unknown): value is User {
  if (typeof value !== 'object' || value === null) return false;
  const obj = value as Record<string, unknown>;
  return (
    typeof obj.id === 'string' &&
    typeof obj.name === 'string' &&
    typeof obj.age === 'number' &&
    typeof obj.email === 'string'
  );
}

// 安全解析
function parseUser(data: unknown): User | null {
  if (isUser(data)) {
    return data;
  }
  return null;
}
```

### 2. Schema 验证

```typescript
// 简单 Schema 验证
type Schema<T> = {
  [K in keyof T]: {
    type: 'string' | 'number' | 'boolean';
    required?: boolean;
    validate?: (value: T[K]) => boolean;
  };
};

function validate<T>(
  value: unknown,
  schema: Schema<T>
): value is T {
  if (typeof value !== 'object' || value === null) return false;
  const obj = value as Record<string, unknown>;
  
  for (const key in schema) {
    const field = schema[key];
    const fieldValue = obj[key];
    
    if (fieldValue === undefined) {
      if (field.required) return false;
      continue;
    }
    
    if (typeof fieldValue !== field.type) return false;
    if (field.validate && !field.validate(fieldValue as T[typeof key])) {
      return false;
    }
  }
  
  return true;
}

// 使用
const userSchema: Schema<User> = {
  id: { type: 'string', required: true },
  name: { type: 'string', required: true },
  age: { type: 'number', required: true, validate: (v) => v >= 0 },
  email: { type: 'string', required: true }
};

function processUser(data: unknown) {
  if (validate(data, userSchema)) {
    // data 是 User 类型
    console.log(data.name);
  }
}
```

### 3. 与 Zod 集成

```typescript
// 使用 Zod 进行运行时验证
import { z } from 'zod';

// 定义 Schema
const UserSchema = z.object({
  id: z.string(),
  name: z.string().min(1).max(100),
  age: z.number().int().min(0).max(150),
  email: z.string().email()
});

// 自动推导类型
type User = z.infer<typeof UserSchema>;

// 安全解析
function parseUser(data: unknown): User | null {
  const result = UserSchema.safeParse(data);
  if (result.success) {
    return result.data;
  }
  console.error(result.error);
  return null;
}
```

## 总结

### 类型安全模式速查

| 模式 | 用途 | 示例 |
|------|------|------|
| 品牌类型 | 区分相同基础类型 | `type UserId = Brand<string, 'UserId'>` |
| 穷尽检查 | 确保处理所有情况 | `assertNever(value)` |
| 类型守卫 | 运行时类型判断 | `value is T` |
| 不可变数据 | 防止意外修改 | `DeepReadonly<T>` |
| Result 类型 | 安全的错误处理 | `Result<T, E>` |

### 最佳实践

1. **使用品牌类型**：区分不同用途的相同类型
2. **穷尽检查**：处理所有可能的分支
3. **类型守卫**：结合运行时检查
4. **不可变数据**：防止意外修改
5. **安全错误处理**：使用 Result/Option 类型
6. **运行时验证**：外部数据必须验证
