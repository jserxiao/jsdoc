## API 类型安全

### 1. 类型安全的请求函数

```typescript
// 定义 API 端点类型
interface ApiEndpoints {
  '/users': {
    GET: {
      response: User[];
    };
    POST: {
      body: Omit<User, 'id'>;
      response: User;
    };
  };
  '/users/:id': {
    GET: {
      params: { id: string };
      response: User;
    };
    PUT: {
      params: { id: string };
      body: Partial<User>;
      response: User;
    };
    DELETE: {
      params: { id: string };
      response: void;
    };
  };
}

// 提取路径参数类型
type ExtractParams<Path extends string> = Path extends `${infer Prefix}/:${infer Param}/${infer Rest}`
  ? { [K in Param]: string } & ExtractParams<`/${Rest}`>
  : Path extends `${infer Prefix}/:${infer Param}`
  ? { [K in Param]: string }
  : {};

// 构建类型安全的请求函数
type HttpMethod = 'GET' | 'POST' | 'PUT' | 'DELETE';

type ApiRequest<
  Path extends keyof ApiEndpoints,
  Method extends keyof ApiEndpoints[Path]
> = ApiEndpoints[Path][Method] extends { params: infer P }
  ? { params: P } & (ApiEndpoints[Path][Method] extends { body: infer B }
      ? { body: B }
      : {})
  : ApiEndpoints[Path][Method] extends { body: infer B }
  ? { body: B }
  : {};

async function api<
  Path extends keyof ApiEndpoints,
  Method extends keyof ApiEndpoints[Path]
>(
  path: Path,
  method: Method,
  options: ApiRequest<Path, Method>
): Promise<ApiEndpoints[Path][Method] extends { response: infer R } ? R : never> {
  // 实现请求逻辑
  const response = await fetch(path as string, {
    method: method as string,
    body: 'body' in options ? JSON.stringify(options.body) : undefined
  });
  return response.json();
}

// 使用示例
interface User {
  id: string;
  name: string;
  email: string;
}

// 类型安全调用
const users = await api('/users', 'GET', {}); // User[]
const user = await api('/users/:id', 'GET', { params: { id: '1' } }); // User
const newUser = await api('/users', 'POST', { body: { name: 'Alice', email: 'alice@example.com' } }); // User
```

### 2. 响应类型推导

```typescript
// 根据状态码区分响应类型
type ApiResponse<T, E = unknown> =
  | { status: 200; data: T }
  | { status: 201; data: T }
  | { status: 400; error: { message: string; code: string } }
  | { status: 401; error: { message: 'Unauthorized' } }
  | { status: 404; error: { message: 'Not Found' } }
  | { status: 500; error: E };

// 使用
async function fetchUser(id: string): Promise<ApiResponse<User>> {
  const response = await fetch(`/api/users/${id}`);
  const status = response.status as ApiResponse<User>['status'];
  
  if (status === 200 || status === 201) {
    return { status, data: await response.json() };
  }
  
  return { status, error: await response.json() } as ApiResponse<User>;
}

// 类型安全处理
const result = await fetchUser('1');
if (result.status === 200) {
  console.log(result.data.name); // 类型安全
} else if (result.status === 400) {
  console.log(result.error.message); // 类型安全
}
```

### 3. OpenAPI 类型生成

```typescript
// 从 OpenAPI Schema 生成类型
type OpenAPISchema = {
  paths: {
    '/pets': {
      get: {
        responses: {
          200: {
            content: {
              'application/json': {
                schema: {
                  type: 'array';
                  items: { $ref: '#/components/schemas/Pet' };
                };
              };
            };
          };
        };
      };
    };
  };
  components: {
    schemas: {
      Pet: {
        type: 'object';
        properties: {
          id: { type: 'integer' };
          name: { type: 'string' };
          tag: { type: 'string' };
        };
        required: ['id', 'name'];
      };
    };
  };
};

// 提取 Schema 类型
type SchemaToType<T> = T extends { type: 'string' }
  ? string
  : T extends { type: 'number' | 'integer' }
  ? number
  : T extends { type: 'boolean' }
  ? boolean
  : T extends { type: 'array'; items: infer I }
  ? SchemaToType<I>[]
  : T extends { type: 'object'; properties: infer P; required?: infer R }
  ? {
      [K in keyof P as R extends ReadonlyArray<K> ? K : never]: SchemaToType<P[K]>;
    } & {
      [K in keyof P as R extends ReadonlyArray<K> ? never : K]?: SchemaToType<P[K]>;
    }
  : never;
```

## 表单类型推导

### 1. 表单字段类型推导

```typescript
// 表单 Schema 定义
interface FormSchema {
  username: {
    type: 'text';
    required: true;
    validation: (value: string) => boolean;
  };
  email: {
    type: 'email';
    required: true;
    validation: (value: string) => boolean;
  };
  age: {
    type: 'number';
    required: false;
    min: 0;
    max: 150;
  };
  bio: {
    type: 'textarea';
    required: false;
    maxLength: 500;
  };
  acceptTerms: {
    type: 'checkbox';
    required: true;
  };
}

// 提取表单值类型
type FormValues<T extends Record<string, any>> = {
  [K in keyof T]: T[K]['type'] extends 'text' | 'email' | 'textarea'
    ? string
    : T[K]['type'] extends 'number'
    ? number
    : T[K]['type'] extends 'checkbox'
    ? boolean
    : unknown;
};

// 提取必填字段
type RequiredFields<T extends Record<string, any>> = {
  [K in keyof T as T[K]['required'] extends true ? K : never]: FormValues<T>[K];
};

// 提取可选字段
type OptionalFields<T extends Record<string, any>> = {
  [K in keyof T as T[K]['required'] extends true ? never : K]?: FormValues<T>[K];
};

// 合并生成表单类型
type FormType<T extends Record<string, any>> = RequiredFields<T> & OptionalFields<T>;

// 使用
type MyFormValues = FormType<FormSchema>;
// {
//   username: string;
//   email: string;
//   acceptTerms: boolean;
//   age?: number;
//   bio?: string;
// }
```

## 状态管理类型

### 1. Redux 风格状态类型

```typescript
// Action 类型
type Action<T extends string, P = undefined> = P extends undefined
  ? { type: T }
  : { type: T; payload: P };

// Action Creator 类型
type ActionCreators = {
  increment: () => Action<'INCREMENT'>;
  decrement: () => Action<'DECREMENT'>;
  add: (amount: number) => Action<'ADD', number>;
  setName: (name: string) => Action<'SET_NAME', string>;
};

// 自动推导 Action 联合类型
type AllActions<T extends Record<string, (...args: any) => any>> = ReturnType<
  T[keyof T]
>;

type Actions = AllActions<ActionCreators>;
// { type: 'INCREMENT' } | { type: 'DECREMENT' } | { type: 'ADD'; payload: number } | { type: 'SET_NAME'; payload: string }

// Reducer 类型
type Reducer<S, A> = (state: S, action: A) => S;

interface State {
  count: number;
  name: string;
}

function reducer(state: State, action: Actions): State {
  switch (action.type) {
    case 'INCREMENT':
      return { ...state, count: state.count + 1 };
    case 'DECREMENT':
      return { ...state, count: state.count - 1 };
    case 'ADD':
      return { ...state, count: state.count + action.payload };
    case 'SET_NAME':
      return { ...state, name: action.payload };
  }
}
```

```typescript
// 状态机定义
type StateMachine<T extends string, E extends string> = {
  initial: T;
  states: {
    [K in T]: {
      on?: {

      exit?: () => void;

// 提取有效事件
type ValidEvents<
  T extends string,
  E extends string,
  CurrentState extends T,
  Machine extends StateMachine<T, E>

  ? keyof Machine['states'][CurrentState]['on']

    yellow: {

      entry: () => console.log('Caution!')
```

### 3. Store 类型推导

```typescript
// 定义 Store
interface StoreDefinition<S, A> {
  state: S;
  actions: A;
  getters?: {
    [K: string]: (state: S) => any;
  };
}

// 提取 Store 类型
type StoreState<D extends StoreDefinition<any, any>> = D['state'];
type StoreActions<D extends StoreDefinition<any, any>> = D['actions'];
type StoreGetters<D extends StoreDefinition<any, any>> = D['getters'] extends {
  [K: string]: (state: any) => infer R;
}
  ? { [K in keyof D['getters']]: R }
  : {};

// 使用示例
const userStore = {
  state: {
    user: null as User | null,
    isLoggedIn: false
  },
  actions: {
    login(user: User) {
      this.state.user = user;
      this.state.isLoggedIn = true;
    },
    logout() {
      this.state.user = null;
      this.state.isLoggedIn = false;
    }
  },
  getters: {
    userName: (state) => state.user?.name ?? 'Guest'
  }
} satisfies StoreDefinition<{ user: User | null; isLoggedIn: boolean }, typeof userStore.actions>;
```

### 1. 路由定义类型

```typescript
interface Routes {
  '/': {};
  '/users': {};
  '/users/:id': { id: string };
  '/users/:userId/posts/:postId': { userId: string; postId: string };

type RoutePaths = keyof Routes;

function push<Path extends RoutePaths>(

    ? [params: RouteParams<Path>]
    : []
): void {
  // 实现路由跳转
}

// 使用
push('/'); // OK
push('/users'); // OK
push('/users/:id', { id: '123' }); // OK
push('/users/:id'); // Error: 缺少参数
push('/users/:userId/posts/:postId', { userId: '1', postId: '2' }); // OK
```

### 2. 动态路由匹配

```typescript
// 路径参数提取
type ExtractPathParams<Path extends string> = Path extends `${infer Start}/:${infer Param}/${infer Rest}`
  ? { [K in Param]: string } & ExtractPathParams<`/${Rest}`>
  : Path extends `${infer Start}/:${infer Param}`
  ? { [K in Param]: string }
  : {};

// 路径填充
type FillPath<Path extends string, Params extends Record<string, string>> = Path extends `:${infer Param}/${infer Rest}`
  ? `${Params[Param & keyof Params]}${FillPath<`/${Rest}`, Params>}`
  : Path extends `:${infer Param}`
  ? Params[Param & keyof Params]
  : Path extends `${infer Prefix}/${infer Rest}`
  ? `${Prefix}/${FillPath<Rest, Params>}`
  : Path;

// 测试
type Params = ExtractPathParams<'/users/:userId/posts/:postId'>;
// { userId: string; postId: string }

type Filled = FillPath<'/users/:userId/posts/:postId', { userId: '1'; postId: '2' }>;
// '/users/1/posts/2'
```

### 3. 查询参数类型

```typescript
// 查询参数定义
interface QueryParams {
  page?: number;
  limit?: number;
  sort?: 'asc' | 'desc';
  filter?: string;
}

// 构建查询字符串
type QueryString<T extends Record<string, any>> = {
  [K in keyof T]: T[K] extends undefined ? never : `${K & string}=${string}`;
}[keyof T];

// URL 构建器
type UrlBuilder<Path extends string, Query extends Record<string, any>> =
  | Path
  | `${Path}?${QueryString<Query>}`;

function buildUrl<Path extends string, Query extends Record<string, any>>(
  path: Path,
  query?: Query
): UrlBuilder<Path, Query> {
  if (!query) return path as any;
  const queryString = Object.entries(query)
    .filter(([_, v]) => v !== undefined)
    .map(([k, v]) => `${k}=${v}`)
    .join('&');
  return `${path}?${queryString}` as any;
}
```

```typescript
interface ListProps<T> {
  items: T[];
  renderItem: (item: T, index: number) => React.ReactNode;
  keyExtractor: (item: T, index: number) => string;
}

function List<T>({ items, renderItem, keyExtractor }: ListProps<T>) {
  return (
    <ul>
      {items.map((item, index) => (
        <li key={keyExtractor(item, index)}>{renderItem(item, index)}</li>
      ))}
    </ul>
  );
}

// 使用
interface User {
  id: string;
  name: string;
}

<List<User>
  items={[{ id: '1', name: 'Alice' }]}
  renderItem={(user) => <span>{user.name}</span>} // user 类型自动推断
  keyExtractor={(user) => user.id}
/>;
```

## 工具函数类型

### 1. 对象操作类型

```typescript
// 深度合并
type DeepMerge<A, B> = A extends object
  ? B extends object
    ? {
        [K in keyof A | keyof B]: K extends keyof B
          ? K extends keyof A
            ? DeepMerge<A[K], B[K]>
            : B[K]
          : K extends keyof A
          ? A[K]
          : never;
      }
    : B
  : B;

// 深度 Pick
type DeepPick<T, Paths extends string> = Paths extends `${infer K}.${infer Rest}`
  ? K extends keyof T
    ? { [P in K]: DeepPick<T[K], Rest> }
    : never
  : Paths extends keyof T
  ? Pick<T, Paths>
  : never;

// 测试
interface Config {
  database: {
    host: string;
    port: number;
  };
  server: {
    port: number;
  };
}

type DbConfig = DeepPick<Config, 'database.host' | 'database.port'>;
// { database: { host: string; port: number } }
```

```typescript
type Curry<F> = F extends (...args: infer A) => infer R

    ? (arg: First) => Curry<(...args: Rest) => R>
    : R
  : never;
```

### 3. 类型守卫工具

```typescript
// 创建类型守卫
function createTypeGuard<T extends Record<string, any>>(
  discriminant: keyof T
) {
  return <K extends T[keyof T]>(
    value: T,
    expected: K
  ): value is T & Record<typeof discriminant, K> => {
    return value[discriminant] === expected;
  };
}

// 使用
type Event = { type: 'click'; x: number } | { type: 'keydown'; key: string };

const isEventType = createTypeGuard<Event>('type');

const event: Event = { type: 'click', x: 100 };
if (isEventType(event, 'click')) {
  console.log(event.x); // 类型安全
}
```

## 总结

### 实战模式速查

| 场景 | 模式 | 示例 |
|------|------|------|
| API 请求 | 联合类型 + 泛型 | `ApiRequest<Path, Method>` |
| 表单处理 | 条件类型 | `FormType<Schema>` |
| 状态管理 | Action 联合 | `AllActions<Creators>` |
| 路由 | 模板字面量 | `ExtractPathParams<Path>` |
| 组件 | 泛型组件 | `List<T>` |

### 实践建议

1. **渐进式类型**：从简单类型开始，逐步增强
2. **文档化**：复杂类型添加注释
3. **测试类型**：使用 `@ts-expect-error` 测试类型
4. **性能考虑**：避免过于复杂的类型
