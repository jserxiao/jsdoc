## 常见性能问题

### 1. 递归类型过深

```typescript
// ❌ 问题：递归深度过大
type DeepReadonlyBad<T> = {
  readonly [K in keyof T]: T[K] extends object
    ? DeepReadonlyBad<T[K]>
    : T[K];
};

// 深层嵌套对象会导致编译缓慢
interface DeepObject {
  a: { b: { c: { d: { e: { f: { g: { h: string } } } } } } };
}

// 编译时会进行多次递归展开

// ✅ 优化：限制递归深度
type DeepReadonly<T, Depth extends number = 5> = Depth extends 0
  ? T
  : T extends object
  ? { readonly [K in keyof T]: DeepReadonly<T[K], Depth extends 5 ? 4 : Depth> }
  : T;
```

### 2. 复杂的联合类型

```typescript
// ❌ 问题：联合类型成员过多
type TooManyUnion =
  | 'a' | 'b' | 'c' | 'd' | 'e' | 'f' | 'g' | 'h'
  | 'i' | 'j' | 'k' | 'l' | 'm' | 'n' | 'o' | 'p'
  // ... 几十个成员
  ;

// 联合类型运算会检查每个成员

// ✅ 优化：使用更精确的类型
type Letter = string & { __brand: 'Letter' };
```

### 3. 映射类型过多

```typescript
// ❌ 问题：大量映射类型组合
type MappedA<T> = { [K in keyof T]: T[K] };
type MappedB<T> = { [K in keyof T]?: T[K] };
type MappedC<T> = { readonly [K in keyof T]: T[K] };

type Combined<T> = MappedA<MappedB<MappedC<T>>>;

// 每层映射都会创建新类型

// ✅ 优化：合并映射
type CombinedOptimized<T> = {
  readonly [K in keyof T]?: T[K];
};
```

### 4. 条件类型嵌套

```typescript
// ❌ 问题：条件类型嵌套过深
type NestedCondition<T> = T extends string
  ? T extends 'a'
    ? T extends 'a1'
      ? T extends 'a1a'
        ? 'deep'
        : 'a1'
      : 'a'
    : 'string'
  : 'other';

// ✅ 优化：使用映射或对象查找
type ConditionMap = {
  a1a: 'deep';
  a1: 'a1';
  a: 'a';
};

type NestedConditionOptimized<T extends string> = T extends keyof ConditionMap
  ? ConditionMap[T]
  : T extends string
  ? 'string'
  : 'other';
```

### 5. 模板字面量类型

```typescript
// ❌ 问题：复杂的模板字面量
type ComplexTemplate<S extends string> = S extends `${infer A}${infer B}${infer C}${infer D}`
  ? `${A}-${B}-${C}-${D}`
  : S;

// 多个 infer 会创建大量中间类型

// ✅ 优化：简化模板字面量
type SimpleTemplate<S extends string> = S extends `${infer First}${infer Rest}`
  ? `${First}-${Rest}`
  : S;
```

## 优化策略

### 1. 类型缓存

```typescript
// ❌ 每次都重新计算
function process<T extends string>(value: T): Uppercase<T> {
  return value.toUpperCase() as Uppercase<T>;
}

// ✅ 使用类型缓存
type CachedUpper<T extends string> = T extends Uppercase<T> ? T : Uppercase<T>;

// 或者使用接口缓存
interface TypeCache {
  upper: Uppercase<string>;
}
```

### 2. 类型拆分

```typescript
// ❌ 一个大类型
type BigType = {
  // 几十个属性
  prop1: string;
  prop2: number;
  // ...
  prop50: boolean;
};

// ✅ 拆分为小类型
type BaseProps = {
  prop1: string;
  prop2: number;
};

type ExtendedProps = BaseProps & {
  prop3: boolean;
  // ...
};
```

### 3. 延迟计算

```typescript
// ❌ 立即计算所有类型
type AllTypes = Type1 | Type2 | Type3 | Type4 | Type5;

// ✅ 按需计算
type GetType<T extends 'a' | 'b' | 'c'> = T extends 'a'
  ? Type1
  : T extends 'b'
  ? Type2
  : Type3;

// 只有使用时才计算
```

### 4. 减少类型实例化

```typescript
// ❌ 每次都实例化
function getValue<T extends { a: string; b: number }>(obj: T): T['a'] {
  return obj.a;
}

// ✅ 使用更简单的约束
function getValueOptimized<T extends string>(obj: { a: T; b: number }): T {
  return obj.a;
}
```

### 5. 避免类型推断

```typescript
// ❌ 让 TypeScript 推断复杂类型
const config = {
  api: {
    endpoints: {
      users: '/api/users',
      posts: '/api/posts'
    }
  }
};

// ✅ 显式指定类型
interface Config {
  api: {
    endpoints: {
      users: string;
      posts: string;
    };
  };
}

const config: Config = {
  api: {
    endpoints: {
      users: '/api/users',
      posts: '/api/posts'
    }
  }
};
```

## 最佳实践

### 1. 类型导入优化

```typescript
// ❌ 导入整个模块
import * as Utils from './utils';

// ✅ 按需导入
import { formatDate, parseDate } from './utils';

// ✅ 类型导入使用 type 关键字
import type { User } from './types';
```

### 2. 类型别名 vs 接口

```typescript
// ✅ 接口可以合并，性能更好
interface User {
  name: string;
}

interface User {
  age: number;
}

// ❌ 类型别名无法合并，需要重新定义
type User2 = {
  name: string;
};

type User2 = User2 & { age: number }; // 递归引用
```

### 3. 使用 const 断言

```typescript
// ❌ 类型推断为 string[]
const routes = ['home', 'about', 'contact'];

// ✅ 使用 const 断言
const routes = ['home', 'about', 'contact'] as const;
// 类型变为 readonly ['home', 'about', 'contact']

// 减少类型计算
```

### 4. 避免泛型滥用

```typescript
// ❌ 不必要的泛型
function getLength<T extends { length: number }>(arr: T): number {
  return arr.length;
}

// ✅ 使用更简单的类型
function getLengthOptimized(arr: { length: number }): number {
  return arr.length;
}

// 只有真正需要泛型时才使用
function identity<T>(value: T): T {
  return value;
}
```

### 5. 减少类型检查范围

```typescript
// ❌ 对整个对象进行类型检查
function validateUser(user: unknown): user is User {
  return (
    typeof user === 'object' &&
    user !== null &&
    'name' in user &&
    'age' in user
    // ...检查所有属性
  );
}

// ✅ 只检查必要属性
function validateUserOptimized(user: unknown): user is User {
  if (typeof user !== 'object' || user === null) return false;
  const u = user as Record<string, unknown>;
  return typeof u.name === 'string' && typeof u.age === 'number';
}
```

### 6. 项目引用优化

```typescript
// tsconfig.json - 使用项目引用
{
  "references": [
    { "path": "./shared" },
    { "path": "./app" }
  ],
  "compilerOptions": {
    "composite": true,
    "declaration": true,
    "declarationMap": true
  }
}
```

### 7. 跳过类型检查

```typescript
// tsconfig.json - 开发时跳过检查
{
  "compilerOptions": {
    "skipLibCheck": true,      // 跳过 .d.ts 检查
    "noUnusedLocals": false,   // 关闭未使用变量检查
    "noUnusedParameters": false
  }
}
```

type Complexity<T, Acc extends any[] = []> = T extends object

## 性能优化清单

### 编译配置

| 配置项 | 建议 | 说明 |
|--------|------|------|
| `skipLibCheck` | `true` | 跳过库类型检查 |
| `incremental` | `true` | 增量编译 |
| `composite` | `true` | 项目引用 |
| `tsBuildInfoFile` | 设置路径 | 缓存编译信息 |

### 代码层面

| 问题 | 解决方案 |
|------|----------|
| 递归过深 | 限制递归深度 |
| 联合类型过大 | 拆分或使用基础类型 |
| 映射类型过多 | 合并映射 |
| 条件类型嵌套 | 使用映射对象 |
| 泛型滥用 | 只在必要时使用 |

### 监控指标

| 指标 | 正常范围 | 异常信号 |
|------|----------|----------|
| 编译时间 | < 5s | > 30s |
| 内存使用 | < 500MB | > 1GB |
| 类型数量 | < 50000 | > 100000 |

## 总结

### 核心原则

1. **简单优先**：能简单就别复杂
2. **按需计算**：避免不必要的类型运算
3. **适度泛型**：不是所有地方都需要泛型
4. **监控性能**：定期检查编译性能

### 快速优化步骤

1. 启用 `--diagnostics` 诊断
2. 检查递归类型深度
3. 简化复杂的联合类型
4. 合并映射类型
5. 使用项目引用
