## 类作为类型

在 TypeScript 中，类声明会创建两个东西：类的构造函数和类的实例类型。

```typescript
class Point {
  x: number;
  y: number;

  constructor(x: number, y: number) {
    this.x = x;
    this.y = y;
  }

  move(dx: number, dy: number) {
    this.x += dx;
    this.y += dy;
  }
}

// 类作为类型使用（实例类型）
const p: Point = new Point(1, 2);

// 类作为值使用（构造函数）
const PointClass: typeof Point = Point;
const p2 = new PointClass(3, 4);
```

## 类类型与实例类型

### typeof 获取类类型

```typescript
class User {
  constructor(public name: string) {}

  static create(name: string): User {
    return new User(name);
  }

  greet(): string {
    return `Hello, ${this.name}`;
  }
}

// 实例类型
const user: User = new User('Alice');

// 类类型（包含静态成员）
const UserClass: typeof User = User;

// 访问静态成员
const newUser: User = UserClass.create('Bob');

// 创建实例
const anotherUser: User = new UserClass('Charlie');
```

### 接口描述实例类型

```typescript
class Counter {
  private count: number = 0;

  increment(): void {
    this.count++;
  }

  getCount(): number {
    return this.count;
  }
}

// 接口只描述公共成员
interface ICounter {
  increment(): void;
  getCount(): number;
}

// 类实例可以赋值给接口
const counter: ICounter = new Counter();
counter.increment();
console.log(counter.getCount()); // 1
```

### 接口描述类类型

```typescript
// 描述构造函数签名
interface IConstructor<T> {
  new (...args: any[]): T;
}

// 描述具体类的构造函数
interface UserConstructor {
  new (name: string): User;
  create(name: string): User;
}

class User {
  constructor(public name: string) {}

  static create(name: string): User {
    return new User(name);
  }

  greet(): string {
    return `Hello, ${this.name}`;
  }
}

// 使用类类型
function createUser(ctor: UserConstructor, name: string): User {
  return new ctor(name);
}

const user = createUser(User, 'Alice');
console.log(user.greet()); // 'Hello, Alice'
```

## 构造函数类型

### 签名类型

```typescript
// 构造函数签名类型
type Constructor<T = {}> = new (...args: any[]) => T;

// 带参数的构造函数类型
type StringConstructor<T = {}> = new (name: string) => T;

// 使用
class Person {
  constructor(public name: string) {}
}

function createInstance<T>(ctor: Constructor<T>, ...args: any[]): T {
  return new ctor(...args);
}

const person = createInstance(Person, 'Alice');
console.log(person.name); // 'Alice'
```

### 混合构造函数类型

```typescript
// 混合类型：既可以是构造函数，又可以是普通对象
type MixinConstructor<T, U> = {
  new (...args: any[]): T;
} & U;

// 使用
interface StaticMembers {
  staticMethod(): void;
}

class MyClass {
  static staticMethod() {
    console.log('Static method');
  }

  instanceMethod() {
    console.log('Instance method');
  }
}

function useClass(ctor: MixinConstructor<MyClass, StaticMembers>) {
  ctor.staticMethod();
  const instance = new ctor();
  instance.instanceMethod();
}

useClass(MyClass);
```

## 类表达式

类表达式可以创建匿名类或命名类表达式。

```typescript
// 匿名类表达式
const Person = class {
  constructor(public name: string) {}

  greet() {
    return `Hello, ${this.name}`;
  }
};

const person = new Person('Alice');
console.log(person.greet()); // 'Hello, Alice'

// 命名类表达式
const Animal = class Dog {
  constructor(public name: string) {}

  bark() {
    console.log(`${this.name} barks!`);
  }

  static createDog(name: string): Dog {
    return new Dog(name);
  }
};

const dog = new Animal('Rex');
dog.bark(); // 'Rex barks!'
```

### 类表达式与类型

```typescript
interface IAnimal {
  name: string;
  bark(): void;
}

interface IAnimalConstructor {
  new (name: string): IAnimal;
}

// 类表达式实现接口
const Animal: IAnimalConstructor = class implements IAnimal {
  constructor(public name: string) {}

  bark(): void {
    console.log(`${this.name} barks!`);
  }
};

const animal = new Animal('Rex');
animal.bark(); // 'Rex barks!'
```

### 动态创建类

```typescript
function createClass<T extends new (...args: any[]) => any>(
  BaseClass: T
) {
  return class extends BaseClass {
    constructor(...args: any[]) {
      super(...args);
    }

    extraMethod() {
      console.log('Extra method');
    }
  };
}

class Base {
  constructor(public name: string) {}

  baseMethod() {
    console.log('Base method');
  }
}

const ExtendedClass = createClass(Base);
const instance = new ExtendedClass('Test');
instance.baseMethod();   // 'Base method'
instance.extraMethod();  // 'Extra method'
```

## 泛型类

### 基本泛型类

```typescript
class Container<T> {
  private value: T;

  constructor(value: T) {
    this.value = value;
  }

  getValue(): T {
    return this.value;
  }

  setValue(value: T): void {
    this.value = value;
  }
}

const stringContainer = new Container<string>('Hello');
console.log(stringContainer.getValue()); // 'Hello'

const numberContainer = new Container<number>(42);
console.log(numberContainer.getValue()); // 42
```

### 泛型约束

```typescript
interface Comparable {
  compareTo(other: this): number;
}

class SortedList<T extends Comparable> {
  private items: T[] = [];

  add(item: T): void {
    this.items.push(item);
    this.items.sort((a, b) => a.compareTo(b));
  }

  getItems(): T[] {
    return [...this.items];
  }
}

class Person implements Comparable {
  constructor(public name: string, public age: number) {}

  compareTo(other: Person): number {
    return this.age - other.age;
  }
}

const list = new SortedList<Person>();
list.add(new Person('Alice', 30));
list.add(new Person('Bob', 25));
list.add(new Person('Charlie', 35));

console.log(list.getItems().map(p => p.name)); // ['Bob', 'Alice', 'Charlie']
```

### 泛型方法

```typescript
class Utils {
  static identity<T>(value: T): T {
    return value;
  }

  static toArray<T>(value: T): T[] {
    return [value];
  }

  static first<T>(arr: T[]): T | undefined {
    return arr[0];
  }

  static map<T, U>(arr: T[], fn: (value: T) => U): U[] {
    return arr.map(fn);
  }
}

const num = Utils.identity(42);           // 42
const arr = Utils.toArray('hello');       // ['hello']
const first = Utils.first([1, 2, 3]);     // 1
const doubled = Utils.map([1, 2, 3], x => x * 2); // [2, 4, 6]
```

### 泛型继承

```typescript
class BaseContainer<T> {
  constructor(protected value: T) {}

  getValue(): T {
    return this.value;
  }
}

class AdvancedContainer<T, U> extends BaseContainer<T> {
  constructor(value: T, protected extra: U) {
    super(value);
  }

  getExtra(): U {
    return this.extra;
  }

  getBoth(): [T, U] {
    return [this.value, this.extra];
  }
}

const container = new AdvancedContainer('hello', 42);
console.log(container.getValue());  // 'hello'
console.log(container.getExtra());  // 42
console.log(container.getBoth());   // ['hello', 42]
```

## 类型守卫与类

### instanceof 检查

```typescript
class Dog {
  bark() {
    console.log('Woof!');
  }
}

class Cat {
  meow() {
    console.log('Meow!');
  }
}

function makeSound(animal: Dog | Cat) {
  if (animal instanceof Dog) {
    animal.bark(); // animal 是 Dog 类型
  } else {
    animal.meow(); // animal 是 Cat 类型
  }
}
```

### 自定义类型守卫

```typescript
abstract class Animal {
  abstract makeSound(): void;
}

class Dog extends Animal {
  makeSound(): void {
    console.log('Woof!');
  }

  fetch(): void {
    console.log('Fetching...');
  }
}

class Cat extends Animal {
  makeSound(): void {
    console.log('Meow!');
  }

  climb(): void {
    console.log('Climbing...');
  }
}

function isDog(animal: Animal): animal is Dog {
  return animal instanceof Dog;
}

function isCat(animal: Animal): animal is Cat {
  return animal instanceof Cat;
}

function interact(animal: Animal) {
  animal.makeSound();

  if (isDog(animal)) {
    animal.fetch();
  } else if (isCat(animal)) {
    animal.climb();
  }
}
```
