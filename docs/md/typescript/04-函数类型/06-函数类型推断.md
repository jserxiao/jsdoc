# 函数类型推断

## 概述

TypeScript 会自动推断函数的参数和返回值类型。

## 一、参数类型推断

```typescript
// 从使用推断
const numbers = [1, 2, 3, 4, 5]

numbers.forEach(n => {
  console.log(n.toFixed(2))  // n: number (推断)
})

// 上下文类型推断
type Handler = (event: MouseEvent) => void

const handler: Handler = event => {
  console.log(event.clientX, event.clientY)  // event: MouseEvent
}
```

## 二、返回值类型推断

```typescript
function add(a: number, b: number) {
  return a + b  // 返回值推断为 number
}

const result = add(1, 2)  // result: number

// 复杂类型推断
function getUser() {
  return {
    id: 1,
    name: 'Alice',
    email: 'alice@example.com'
  }
}

const user = getUser()
// user: { id: number; name: string; email: string }
```

## 三、泛型推断

```typescript
function identity<T>(arg: T): T {
  return arg
}

const str = identity('hello')  // T: string
const num = identity(42)        // T: number

// 多个参数推断
function pair<T, U>(first: T, second: U): [T, U] {
  return [first, second]
}

const p = pair('hello', 42)  // [string, number]
```

## 四、推断限制

```typescript
// 推断可能不够精确
function getLength(x: string | any[]) {
  return x.length  // number
}

// 显式指定更精确
function getLength2(x: string): number
function getLength2(x: any[]): number
function getLength2(x: string | any[]): number {
  return x.length
}
```

## 五、最佳实践

```typescript
// ✅ 好: 让 TypeScript 推断简单类型
function add(a: number, b: number) {
  return a + b
}

// ✅ 好: 显式指定复杂类型
interface User {
  id: number
  name: string
}

function createUser(id: number, name: string): User {
  return { id, name }
}

// ✅ 好: 泛型让推断更灵活
function map<T, U>(arr: T[], fn: (x: T) => U): U[] {
  return arr.map(fn)
}

const doubled = map([1, 2, 3], x => x * 2)  // number[]
```

## 六、总结

- TypeScript 会自动推断参数和返回值类型
- 上下文类型提供更好的推断
- 复杂场景建议显式指定类型
- 泛型增强推断能力
