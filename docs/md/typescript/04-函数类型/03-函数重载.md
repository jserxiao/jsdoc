# 函数重载

函数重载（Function Overloading）允许同一个函数根据不同的参数类型或数量，提供不同的类型签名和返回类型。这是 TypeScript 实现类型安全多态函数的核心特性。

## 概述

### 什么是函数重载

```typescript
// 没有重载：无法精确推断返回类型
function parse(value: string | number): string | number {
  if (typeof value === 'string') {
    return parseInt(value, 10);
  }
  return value.toString();
}

const a = parse('123'); // string | number（不精确）
const b = parse(123);   // string | number（不精确）

// 使用重载：精确的返回类型推断
function parseOverload(value: string): number;
function parseOverload(value: number): string;
function parseOverload(value: string | number): string | number {
  if (typeof value === 'string') {
    return parseInt(value, 10);
  }
  return value.toString();
}

const c = parseOverload('123'); // number（精确）
const d = parseOverload(123);   // string（精确）
```

### 重载的组成

```typescript
// 1. 重载签名（Overload Signatures）
function func(x: string): string;   // 重载签名 1
function func(x: number): number;   // 重载签名 2

// 2. 实现签名（Implementation Signature）
function func(x: string | number): string | number {
  // 实现代码
  return x;
}
```

## 基本语法

### 基本结构

```typescript
// 重载签名列表
function reverse(x: number): number;
function reverse(x: string): string;

// 实现签名（必须是兼容的）
function reverse(x: number | string): number | string {
  if (typeof x === 'number') {
    return Number(x.toString().split('').reverse().join(''));
  } else {
    return x.split('').reverse().join('');
  }
}

// 使用
const num = reverse(123);     // number: 321
const str = reverse('hello'); // string: 'olleh'
```

### 多个重载签名

```typescript
function convert(value: string): number;
function convert(value: number): string;
function convert(value: boolean): string;
function convert(value: string | number | boolean): string | number {
  if (typeof value === 'string') {
    return parseInt(value, 10);
  } else if (typeof value === 'number') {
    return value.toString();
  } else {
    return value ? 'true' : 'false';
  }
}

convert('123');  // number
convert(123);    // string
convert(true);   // string
```

## 重载签名顺序

### 从具体到通用

TypeScript 按照重载签名的顺序进行匹配，更具体的类型应该放在前面。

```typescript
// ✅ 正确顺序：具体类型在前
function process(value: string): string;
function process(value: number): number;
function process(value: string | number): string | number {
  return value;
}

// ❌ 错误示例：通用类型在前会覆盖具体类型
function process2(value: string | number): string | number;
function process2(value: string): string; // 这个永远不会被匹配！
```

### 子类型优先

```typescript
class Animal {}
class Dog extends Animal {}
class Cat extends Animal {}

// 子类型在前
function getName(animal: Dog): 'dog';
function getName(animal: Cat): 'cat';
function getName(animal: Animal): 'animal';
function getName(animal: Animal): string {
  if (animal instanceof Dog) return 'dog';
  if (animal instanceof Cat) return 'cat';
  return 'animal';
}

getName(new Dog()); // 'dog'
getName(new Cat()); // 'cat'
getName(new Animal()); // 'animal'
```

### 字面量类型优先

```typescript
function create(tag: 'div'): HTMLDivElement;
function create(tag: 'span'): HTMLSpanElement;
function create(tag: 'a'): HTMLAnchorElement;
function create(tag: string): HTMLElement;
function create(tag: string): HTMLElement {
  return document.createElement(tag);
}

const div = create('div');   // HTMLDivElement
const span = create('span'); // HTMLSpanElement
const p = create('p');       // HTMLElement（回退到通用签名）
```

## 不同参数数量的重载

### 可选参数

```typescript
function makeDate(timestamp: number): Date;
function makeDate(year: number, month: number, day: number): Date;
function makeDate(yearOrTimestamp: number, month?: number, day?: number): Date {
  if (month !== undefined && day !== undefined) {
    return new Date(yearOrTimestamp, month - 1, day);
  } else {
    return new Date(yearOrTimestamp);
  }
}

makeDate(1234567890);      // Date（时间戳）
makeDate(2024, 1, 15);     // Date（年月日）
// makeDate(2024, 1);       // Error: 没有匹配的重载
```

### 不同参数类型

```typescript
function format(value: string, format: 'upper' | 'lower'): string;
function format(value: number, decimals: number): string;
function format(value: string | number, options: string | number): string {
  if (typeof value === 'string') {
    return options === 'upper' ? value.toUpperCase() : value.toLowerCase();
  } else {
    return value.toFixed(options as number);
  }
}

format('hello', 'upper');  // 'HELLO'
format(3.14159, 2);        // '3.14'
```

### 回调函数重载

```typescript
function fetch(url: string): Promise<Response>;
function fetch(url: string, callback: (response: Response) => void): void;
function fetch(url: string, callback?: (response: Response) => void): Promise<Response> | void {
  const promise = globalThis.fetch(url);

  if (callback) {
    promise.then(callback);
    return;
  }

  return promise;
}

fetch('/api/data').then(r => r.json());           // Promise<Response>
fetch('/api/data', r => console.log(r.json()));   // void
```

## 与联合类型的对比

### 何时使用重载

```typescript
// 使用重载：当返回类型依赖于参数类型
function process(x: string): number;
function process(x: number): string;
function process(x: string | number): string | number {
  if (typeof x === 'string') return x.length;
  return x.toString();
}

const a = process('hello'); // number
const b = process(123);     // string

// 使用联合类型：当返回类型不依赖参数类型
function format2(value: string | number): string {
  return String(value);
}

const c = format2('hello'); // string
const d = format2(123);     // string
```

### 重载 vs 联合类型

```typescript
// 方案 1：联合类型（简单场景）
function double(x: number | string): number | string {
  if (typeof x === 'number') return x * 2;
  return x + x;
}

const e = double(5);    // number | string（不够精确）

// 方案 2：重载（精确类型）
function doubleOverload(x: number): number;
function doubleOverload(x: string): string;
function doubleOverload(x: number | string): number | string {
  if (typeof x === 'number') return x * 2;
  return x + x;
}

const f = doubleOverload(5);    // number（精确）
const g = doubleOverload('5');  // string（精确）
```

## 方法重载

### 类方法重载

```typescript
class Calculator {
  // 重载签名
  add(a: number, b: number): number;
  add(a: string, b: string): string;
  add(a: number | string, b: number | string): number | string {
    if (typeof a === 'number' && typeof b === 'number') {
      return a + b;
    }
    return String(a) + String(b);
  }
}

const calc = new Calculator();
calc.add(1, 2);       // number: 3
calc.add('1', '2');   // string: '12'
```

### 接口方法重载

```typescript
interface EventEmitter {
  on(event: 'click', handler: (x: number, y: number) => void): void;
  on(event: 'message', handler: (text: string) => void): void;
  on(event: string, handler: (...args: any[]) => void): void;
}

class MyEmitter implements EventEmitter {
  on(event: string, handler: (...args: any[]) => void): void {
    // 实现
  }
}

const emitter = new MyEmitter();
emitter.on('click', (x, y) => console.log(x, y)); // x, y 是 number
emitter.on('message', text => console.log(text)); // text 是 string
```

### 静态方法重载

```typescript
class Factory {
  static create(type: 'user'): User;
  static create(type: 'admin'): Admin;
  static create(type: string): User | Admin {
    if (type === 'admin') {
      return new Admin();
    }
    return new User();
  }
}

const user = Factory.create('user');   // User
const admin = Factory.create('admin'); // Admin
```

## 构造函数重载

### 类构造函数重载

```typescript
class Point {
  x: number;
  y: number;

  // 重载签名
  constructor();
  constructor(x: number, y: number);
  constructor(x?: number, y?: number) {
    this.x = x ?? 0;
    this.y = y ?? 0;
  }
}

const p1 = new Point();       // Point { x: 0, y: 0 }
const p2 = new Point(10, 20); // Point { x: 10, y: 20 }
```

### 工厂函数模式

```typescript
interface User {
  name: string;
  email: string;
}

function createUser(data: string): User;
function createUser(name: string, email: string): User;
function createUser(dataOrName: string, email?: string): User {
  if (email) {
    return { name: dataOrName, email };
  }
  const parsed = JSON.parse(dataOrName);
  return { name: parsed.name, email: parsed.email };
}

createUser('{"name":"John","email":"john@example.com"}');
createUser('John', 'john@example.com');
```

## 最佳实践

### 实现签名必须兼容所有重载

```typescript
// ✅ 正确：实现签名兼容所有重载
function fn(x: string): string;
function fn(x: number): number;
function fn(x: string | number): string | number {
  return x;
}

// ❌ 错误：实现签名不兼容
function fn2(x: string): string;
function fn2(x: number): number;
function fn2(x: boolean): boolean; // Error: 类型不兼容
```

### 避免过度重载

```typescript
// ❌ 过度重载
function getValue(x: 1): string;
function getValue(x: 2): string;
function getValue(x: 3): string;
function getValue(x: 4): string;
// ...

// ✅ 使用联合类型或条件类型
function getValue(x: number): string {
  return String(x);
}

// ✅ 使用映射类型处理大量模式
type TagMap = {
  div: HTMLDivElement;
  span: HTMLSpanElement;
  a: HTMLAnchorElement;
  // ...
};

function createElement<K extends keyof TagMap>(tag: K): TagMap[K];
function createElement(tag: string): HTMLElement;
function createElement(tag: string): HTMLElement {
  return document.createElement(tag);
}
```

### 使用类型守卫简化实现

```typescript
function process(value: string): string[];
function process(value: number): number[];
function process(value: string | number): string[] | number[] {
  if (typeof value === 'string') {
    return value.split('');
  }
  return Array.from(String(value)).map(Number);
}
```

### 文档化重载

```typescript
/**
 * 解析字符串或数字
 * @param value - 字符串，返回解析后的数字
 * @param value - 数字，返回字符串形式
 */
function parse(value: string): number;
function parse(value: number): string;
function parse(value: string | number): string | number {
  if (typeof value === 'string') {
    return parseInt(value, 10);
  }
  return value.toString();
}
```

## 小结

| 要点 | 说明 |
|------|------|
| 语法 | 多个重载签名 + 一个实现签名 |
| 顺序 | 从具体到通用 |
| 兼容性 | 实现签名必须兼容所有重载 |
| 适用场景 | 返回类型依赖于参数类型 |
| 避免 | 过度重载，优先考虑联合类型 |
