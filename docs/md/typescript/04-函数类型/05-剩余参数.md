# 剩余参数

剩余参数（Rest Parameters）允许函数接受不定数量的参数，并将它们收集到一个数组中。在 TypeScript 中，剩余参数有完善的类型支持。

## 基本用法

### 语法

```typescript
// 剩余参数使用 ... 语法
function functionName(...paramName: type[]): returnType {
  // paramName 是一个数组
}

// 示例：计算任意数量数字的和
function sum(...numbers: number[]): number {
  return numbers.reduce((total, n) => total + n, 0);
}

sum(1, 2, 3);       // 6
sum(1, 2, 3, 4, 5); // 15
sum();              // 0（空数组）
```

### 剩余参数的特点

```typescript
// 1. 剩余参数必须是最后一个参数
function example(first: string, ...rest: string[]): void {
  console.log(`First: ${first}`);
  console.log(`Rest: ${rest.join(', ')}`);
}

example('a', 'b', 'c');
// First: a
// Rest: b, c

// ❌ 错误：剩余参数后不能有其他参数
// function wrong(...rest: string[], last: string): void { } // Error!

// 2. 剩余参数是一个数组
function logAll(...items: string[]): void {
  items.forEach(item => console.log(item));
  console.log(`Total items: ${items.length}`);
}

logAll('apple', 'banana', 'orange');
// apple
// banana
// orange
// Total items: 3

// 3. 最多只能有一个剩余参数
// ❌ 错误：不能有多个剩余参数
// function multiple(...first: string[], ...second: number[]): void { } // Error!
```

## 剩余参数类型

### 数组类型

```typescript
// 字符串数组
function concat(...strings: string[]): string {
  return strings.join('');
}

concat('Hello', ' ', 'World'); // 'Hello World'

// 数字数组
function multiplyAll(...numbers: number[]): number {
  return numbers.reduce((acc, n) => acc * n, 1);
}

multiplyAll(1, 2, 3, 4); // 24

// 联合类型数组
function process(...items: (string | number)[]): void {
  items.forEach(item => {
    console.log(typeof item, item);
  });
}

process('hello', 42, 'world', 100);
// string hello
// number 42
// string world
// number 100

// 对象类型数组
interface User {
  name: string;
  age: number;
}

function greetUsers(...users: User[]): void {
  users.forEach(user => {
    console.log(`Hello, ${user.name} (${user.age})`);
  });
}

greetUsers(
  { name: 'Alice', age: 30 },
  { name: 'Bob', age: 25 }
);
```

### 元组类型

```typescript
// 固定长度和类型的元组
function pair(...pair: [string, number]): void {
  const [name, age] = pair;
  console.log(`${name} is ${age} years old`);
}

pair('Alice', 30); // Alice is 30 years old
// pair('Alice');        // Error: 缺少参数
// pair('Alice', 30, 1); // Error: 参数过多

// 更复杂的元组
function createUser(...data: [string, number, boolean]): User {
  const [name, age, isActive] = data;
  return { name, age, isActive };
}

createUser('Charlie', 28, true);

// 带可选元素的元组
function config(...settings: [string, number, boolean?]): void {
  const [host, port, secure] = settings;
  console.log(`${host}:${port}, secure: ${secure ?? false}`);
}

config('localhost', 3000);        // localhost:3000, secure: false
config('localhost', 3000, true);  // localhost:3000, secure: true
```

## 与普通参数组合

### 固定参数 + 剩余参数

```typescript
// 回调函数风格
function debounce<T extends any[]>(
  fn: (...args: T) => void,
  delay: number
): (...args: T) => void {
  let timeoutId: ReturnType<typeof setTimeout>;
  return function(this: any, ...args: T) {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => fn.apply(this, args), delay);
  };
}

// 日志函数
function log(level: string, ...messages: string[]): void {
  const timestamp = new Date().toISOString();
  const message = messages.join(' ');
  console.log(`[${timestamp}] [${level}] ${message}`);
}

log('INFO', 'User', 'logged', 'in');  // [2024-...] [INFO] User logged in
log('ERROR', 'Failed to', 'connect');  // [2024-...] [ERROR] Failed to connect

// 格式化函数
function format(separator: string, first: string, ...rest: string[]): string {
  return [first, ...rest].join(separator);
}

format('-', 'a', 'b', 'c'); // 'a-b-c'
format('|', 'x', 'y', 'z'); // 'x|y|z'
```

### 多个固定参数 + 剩余参数

```typescript
// 路由处理
function handleRoute(
  method: string,
  path: string,
  ...handlers: ((ctx: Context) => void)[]
): void {
  console.log(`Registering ${method} ${path}`);
  handlers.forEach(handler => {
    console.log('Handler registered');
  });
}

handleRoute('GET', '/users',
  (ctx) => ctx.body = 'List users',
  (ctx) => ctx.status = 200
);

// 数据验证
function validate<T>(
  value: T,
  validator: (v: T) => boolean,
  ...additionalValidators: ((v: T) => boolean)[]
): boolean {
  const allValidators = [validator, ...additionalValidators];
  return allValidators.every(v => v(value));
}

validate(
  'hello',
  (s) => s.length > 0,
  (s) => s.length < 100,
  (s) => typeof s === 'string'
);
```

## 泛型剩余参数

### 泛型数组类型

```typescript
// 泛型剩余参数
function wrapInArray<T>(...items: T[]): T[] {
  return items;
}

wrapInArray(1, 2, 3);       // number[]
wrapInArray('a', 'b', 'c'); // string[]
wrapInArray(true, false);   // boolean[]

// 推断联合类型
wrapInArray(1, 'two', true); // (string | number | boolean)[]

// 带约束的泛型
interface Lengthwise {
  length: number;
}

function logLengths<T extends Lengthwise>(...items: T[]): void {
  items.forEach(item => console.log(item.length));
}

logLengths('hello', [1, 2, 3], { length: 10 }); // 5, 3, 10
```

### 泛型元组类型

```typescript
// 泛型元组剩余参数
function tuple<T extends any[]>(...args: T): T {
  return args;
}

const t1 = tuple(1, 'hello', true); // [number, string, boolean]
// t1[0] 是 number，t1[1] 是 string，t2[2] 是 boolean

// 函数柯里化
function curry<T extends any[], R>(
  fn: (...args: T) => R
): (...args: T) => R {
  return fn;
}

const add = (a: number, b: number) => a + b;
const curriedAdd = curry(add);
curriedAdd(1, 2); // 3
```

### 类型推断

```typescript
// 推断剩余参数类型
function call<T extends any[], R>(
  fn: (...args: T) => R,
  ...args: T
): R {
  return fn(...args);
}

call((a: number, b: string) => `${a}-${b}`, 1, 'a'); // '1-a'
// call((a: number, b: string) => `${a}-${b}`, 1, 2); // Error: 类型不匹配

// this 参数与剩余参数
function apply<T, A extends any[], R>(
  this: T,
  fn: (this: T, ...args: A) => R,
  ...args: A
): R {
  return fn.apply(this, args);
}
```

## 元组类型的剩余参数

### 展开 tuple 类型

```typescript
type Args = [string, number, boolean];

function foo(...args: Args): void {
  const [a, b, c] = args;
  console.log(a, b, c);
}

foo('hello', 42, true); // hello 42 true

// 与泛型组合
function typedTuple<T extends any[]>(...args: T): T {
  return args;
}

const tuple1 = typedTuple(1, 'a', true); // [number, string, boolean]
const tuple2 = typedTuple('x', 'y');     // [string, string]
```

### 条件类型中的剩余参数

```typescript
// 获取函数参数类型
type Parameters<T> = T extends (...args: infer P) => any ? P : never;

type Fn = (a: string, b: number) => boolean;
type Params = Parameters<Fn>; // [string, number]

// 获取剩余参数类型
type RestParameters<T> = T extends (first: infer F, ...rest: infer R) => any
  ? [first: F, ...rest: R]
  : never;

type RestParams = RestParameters<(a: string, b: number, c: boolean) => void>;
// [a: string, b: number, c: boolean]
```

## 函数重载中的剩余参数

### 基本重载

```typescript
// 重载签名与剩余参数
function createElement(tag: 'div'): HTMLDivElement;
function createElement(tag: 'span'): HTMLSpanElement;
function createElement(tag: string, ...children: (string | Node)[]): HTMLElement {
  const element = document.createElement(tag);
  children.forEach(child => {
    if (typeof child === 'string') {
      element.appendChild(document.createTextNode(child));
    } else {
      element.appendChild(child);
    }
  });
  return element;
}

createElement('div');                   // HTMLDivElement
createElement('span', 'Hello');         // HTMLSpanElement
createElement('p', 'Hello', ' ', 'World'); // HTMLParagraphElement
```

### 类型安全的事件系统

```typescript
interface EventMap {
  click: [x: number, y: number];
  message: [text: string, timestamp: number];
  data: [payload: unknown];
}

class TypedEventEmitter<T extends Record<string, any[]>> {
  private handlers = new Map<keyof T, Set<(...args: any[]) => void>>();

  on<K extends keyof T>(event: K, handler: (...args: T[K]) => void): void {
    if (!this.handlers.has(event)) {
      this.handlers.set(event, new Set());
    }
    this.handlers.get(event)!.add(handler);
  }

  emit<K extends keyof T>(event: K, ...args: T[K]): void {
    this.handlers.get(event)?.forEach(handler => handler(...args));
  }

  off<K extends keyof T>(event: K, handler: (...args: T[K]) => void): void {
    this.handlers.get(event)?.delete(handler);
  }
}

// 使用
const emitter = new TypedEventEmitter<EventMap>();

emitter.on('click', (x, y) => {
  console.log(`Clicked at (${x}, ${y})`);
});

emitter.emit('click', 100, 200);  // ✅ 类型安全
// emitter.emit('click', 'hello'); // ❌ 类型错误
```

## 实际应用场景

### 工具函数

```typescript
// 合并对象
function merge<T extends object>(...objects: T[]): T {
  return objects.reduce((acc, obj) => ({ ...acc, ...obj }), {} as T);
}

merge({ a: 1 }, { b: 2 }, { c: 3 }); // { a: 1, b: 2, c: 3 }

// 数组扁平化
function flatten<T>(...arrays: T[][]): T[] {
  return arrays.flat();
}

flatten([1, 2], [3, 4], [5, 6]); // [1, 2, 3, 4, 5, 6]

// 条件过滤
function filter<T>(
  array: T[],
  ...predicates: ((item: T) => boolean)[]
): T[] {
  return array.filter(item =>
    predicates.every(predicate => predicate(item))
  );
}

filter(
  [1, 2, 3, 4, 5],
  n => n > 2,
  n => n % 2 === 1
); // [3, 5]
```

### 函数组合

```typescript
// compose 函数
function compose<T>(...fns: ((arg: T) => T)[]): (arg: T) => T {
  return (arg: T) => fns.reduceRight((acc, fn) => fn(acc), arg);
}

const addOne = (x: number) => x + 1;
const double = (x: number) => x * 2;
const square = (x: number) => x * x;

const compute = compose(square, double, addOne);
compute(3); // ((3 + 1) * 2)² = 64

// pipe 函数
function pipe<T>(...fns: ((arg: T) => T)[]): (arg: T) => T {
  return (arg: T) => fns.reduce((acc, fn) => fn(acc), arg);
}

const piped = pipe(addOne, double, square);
piped(3); // 同样返回 64
```

### 装饰器模式

```typescript
// 函数装饰器
function withLogging<T extends any[], R>(
  fn: (...args: T) => R
): (...args: T) => R {
  return function(...args: T): R {
    console.log(`Calling with arguments:`, args);
    const result = fn(...args);
    console.log(`Result:`, result);
    return result;
  };
}

const add = (a: number, b: number) => a + b;
const loggedAdd = withLogging(add);
loggedAdd(1, 2);
// Calling with arguments: [1, 2]
// Result: 3

// 计时装饰器
function withTiming<T extends any[], R>(
  fn: (...args: T) => R
): (...args: T) => R {
  return function(...args: T): R {
    const start = performance.now();
    const result = fn(...args);
    const end = performance.now();
    console.log(`${fn.name} took ${end - start}ms`);
    return result;
  };
}
```

### API 封装

```typescript
// 类型安全的 fetch 封装
type HttpMethod = 'GET' | 'POST' | 'PUT' | 'DELETE';

async function request<T>(
  method: HttpMethod,
  url: string,
  ...options: [
    body?: unknown,
    headers?: Record<string, string>
  ]
): Promise<T> {
  const [body, headers] = options;

  const response = await fetch(url, {
    method,
    headers: {
      'Content-Type': 'application/json',
      ...headers
    },
    body: body ? JSON.stringify(body) : undefined
  });

  return response.json();
}

// 使用
interface User {
  id: number;
  name: string;
}

const users = await request<User[]>('GET', '/api/users');
const created = await request<User>('POST', '/api/users', { name: 'John' });
```

## 小结

| 特性 | 说明 |
|------|------|
| 语法 | `...paramName: type[]` |
| 位置 | 必须是最后一个参数 |
| 数量 | 每个函数最多一个 |
| 类型 | 数组类型或元组类型 |
| 泛型 | 支持泛型剩余参数 |
