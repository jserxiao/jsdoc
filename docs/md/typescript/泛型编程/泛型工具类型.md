## 属性修饰工具

### Partial<T>

将所有属性变为可选：

```typescript
type Partial<T> = {
  [P in keyof T]?: T[P];
};

interface User {
  name: string;
  age: number;
  email: string;
}

type PartialUser = Partial<User>;
// { name?: string; age?: number; email?: string; }
```

### Required<T>

将所有属性变为必需：

```typescript
type Required<T> = {
  [P in keyof T]-?: T[P];
};

interface User {
  name: string;
  age?: number;
}

type RequiredUser = Required<User>;
// { name: string; age: number; }
```

### Readonly<T>

将所有属性变为只读：

```typescript
type Readonly<T> = {
  readonly [P in keyof T]: T[P];
};

interface User {
  name: string;
  age: number;
}

type ReadonlyUser = Readonly<User>;
// { readonly name: string; readonly age: number; }
```

### Mutable<T>

移除只读修饰：

```typescript
type Mutable<T> = {
  -readonly [P in keyof T]: T[P];
};

interface User {
  readonly name: string;
  readonly age: number;
}

type MutableUser = Mutable<User>;
// { name: string; age: number; }
```

## 类型转换工具

### Pick<T, K>

选取部分属性：

```typescript
type Pick<T, K extends keyof T> = {
  [P in K]: T[P];
};

interface User {
  id: string;
  name: string;
  email: string;
  age: number;
}

type UserName = Pick<User, 'name' | 'email'>;
// { name: string; email: string; }
```

### Omit<T, K>

排除部分属性：

```typescript
type Omit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>;

interface User {
  id: string;
  name: string;
  email: string;
  password: string;
}

type PublicUser = Omit<User, 'password'>;
// { id: string; name: string; email: string; }
```

### Record<K, T>

创建键值对类型：

```typescript
type Record<K extends keyof any, T> = {
  [P in K]: T;
};

type UserMap = Record<string, User>;
type Page = 'home' | 'about' | 'contact';
type Routes = Record<Page, string>;
// { home: string; about: string; contact: string; }
```

## 集合操作工具

### Extract<T, U>

提取可赋值给 U 的类型：

```typescript
type Extract<T, U> = T extends U ? T : never;

type T = string | number | boolean;
type StringOrNumber = Extract<T, string | number>;
// string | number
```

### Exclude<T, U>

排除可赋值给 U 的类型：

```typescript
type Exclude<T, U> = T extends U ? never : T;

type T = string | number | boolean;
type NonString = Exclude<T, string>;
// number | boolean
```

### NonNullable<T>

排除 null 和 undefined：

```typescript
type NonNullable<T> = T extends null | undefined ? never : T;

type T = string | null | undefined;
type NonNull = NonNullable<T>;
// string
```

## 函数相关工具

### Parameters<T>

获取函数参数类型：

```typescript
type Parameters<T extends (...args: any) => any> = 
  T extends (...args: infer P) => any ? P : never;

function fn(name: string, age: number): void {}
type Params = Parameters<typeof fn>;
// [string, number]
```

### ReturnType<T>

获取函数返回值类型：

```typescript
type ReturnType<T extends (...args: any) => any> = 
  T extends (...args: any) => infer R ? R : any;

function fn(): string { return 'hello'; }
type Result = ReturnType<typeof fn>;
// string
```

### InstanceType<T>

获取构造函数实例类型：

```typescript
type InstanceType<T extends new (...args: any) => any> = 
  T extends new (...args: any) => infer R ? R : any;

class User {
  constructor(public name: string) {}
}

type UserInstance = InstanceType<typeof User>;
// User
```

## 自定义工具类型

### DeepPartial<T>

深度可选：

```typescript
type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object 
    ? DeepPartial<T[P]> 
    : T[P];
};

interface Config {
  server: {
    host: string;
    port: number;
  };
  database: {
    name: string;
    host: string;
  };
}

type PartialConfig = DeepPartial<Config>;
// server?: { host?: string; port?: number }
// database?: { name?: string; host?: string }
```

### DeepReadonly<T>

深度只读：

```typescript
type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object 
    ? DeepReadonly<T[P]> 
    : T[P];
};
```

### RequiredKeys<T>

获取必需属性的键：

```typescript
type RequiredKeys<T> = {
  [K in keyof T]-?: {} extends Pick<T, K> ? never : K
}[keyof T];

interface User {
  name: string;
  age?: number;
  email: string;
}

type Keys = RequiredKeys<User>;
// 'name' | 'email'
```

### OptionalKeys<T>

获取可选属性的键：

```typescript
type OptionalKeys<T> = {
  [K in keyof T]-?: {} extends Pick<T, K> ? K : never
}[keyof T];
```

### GetRequired<T>

获取必需属性：

```typescript
type GetRequired<T> = Pick<T, RequiredKeys<T>>;

type RequiredUser = GetRequired<User>;
// { name: string; email: string; }
```

### GetOptional<T>

获取可选属性：

```typescript
type GetOptional<T> = Pick<T, OptionalKeys<T>>;

type OptionalUser = GetOptional<User>;
// { age?: number; }
```

### UnionToIntersection<T>

联合类型转交叉类型：

```typescript
type UnionToIntersection<U> = 
  (U extends any ? (k: U) => void : never) extends 
  ((k: infer I) => void) ? I : never;

type T = { a: string } | { b: number };
type Result = UnionToIntersection<T>;
// { a: string } & { b: number }
```
