# 泛型函数

泛型函数是 TypeScript 中最常用的泛型应用场景，允许函数处理多种类型同时保持类型安全。

## 基本语法

### 函数声明

```typescript
function identity<T>(arg: T): T {
  return arg;
}

// 显式指定类型
const str = identity<string>('hello');  // string
const num = identity<number>(42);       // number

// 类型推断
const inferred = identity('world');     // string
```

### 箭头函数

```typescript
// 箭头函数
const identity = <T>(arg: T): T => arg;

// 在 TSX 中需要使用 extends 避免与 JSX 冲突
const identityTSX = <T extends unknown>(arg: T): T => arg;
```

### 函数类型

```typescript
// 泛型函数类型
type Identity<T> = (arg: T) => T;

const identity: Identity<string> = (arg) => arg;
```

## 类型推断

### 自动推断

```typescript
function first<T>(arr: T[]): T | undefined {
  return arr[0];
}

// TypeScript 自动推断 T 为 number
const firstNum = first([1, 2, 3]);      // number | undefined

// 自动推断 T 为 string
const firstStr = first(['a', 'b']);     // string | undefined
```

### 推断限制

```typescript
function loggingIdentity<T>(arg: T): T {
  // console.log(arg.length);  // Error: T 不一定有 length
  return arg;
}

// 解决方案：约束 T
function loggingIdentity2<T extends { length: number }>(arg: T): T {
  console.log(arg.length);  // OK
  return arg;
}
```

## 多类型参数

### 多个泛型参数

```typescript
function pair<T, U>(first: T, second: U): [T, U] {
  return [first, second];
}

const result = pair('hello', 42);  // [string, number]
```

### 交换元组元素

```typescript
function swap<T, U>(tuple: [T, U]): [U, T] {
  return [tuple[1], tuple[0]];
}

const swapped = swap([1, 'hello']);  // ['hello', 1]
```

### 对象映射

```typescript
function mapObject<K extends string, V, U>(
  obj: Record<K, V>,
  fn: (value: V) => U
): Record<K, U> {
  const result = {} as Record<K, U>;
  for (const key in obj) {
    result[key] = fn(obj[key]);
  }
  return result;
}

const obj = { a: 1, b: 2 };
const doubled = mapObject(obj, x => x * 2);  // { a: 2, b: 4 }
```

## 泛型约束

### 接口约束

```typescript
interface Lengthwise {
  length: number;
}

function logLength<T extends Lengthwise>(arg: T): number {
  console.log(arg.length);
  return arg.length;
}

logLength('hello');     // ✅ string 有 length
logLength([1, 2, 3]);   // ✅ array 有 length
logLength({ length: 5 }); // ✅ 对象有 length
// logLength(123);      // ❌ number 没有 length
```

### keyof 约束

```typescript
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

const user = { name: 'Alice', age: 30 };

const name = getProperty(user, 'name');   // string
const age = getProperty(user, 'age');     // number
// getProperty(user, 'gender');           // ❌ 'gender' 不是 user 的属性
```

### 构造函数约束

```typescript
type Constructor<T = {}> = new (...args: any[]) => T;

function createInstance<T>(ctor: Constructor<T>, ...args: any[]): T {
  return new ctor(...args);
}

class Person {
  constructor(public name: string) {}
}

const person = createInstance(Person, 'Alice');
console.log(person.name);  // 'Alice'
```

## 默认类型参数

### 基本用法

```typescript
function createArray<T = string>(length: number, value: T): T[] {
  return Array(length).fill(value);
}

// 使用默认类型 string
const strings = createArray(3, 'x');  // string[]

// 显式指定类型
const numbers = createArray<number>(3, 1);  // number[]
```

### 配合约束

```typescript
interface Options {
  debug?: boolean;
  timeout?: number;
}

function request<T extends Options = Options>(
  url: string,
  options?: T
): Promise<Response> {
  const opts = { ...options } as T & { debug: boolean; timeout: number };
  // ...
  return fetch(url);
}
```

## 函数重载与泛型

### 泛型重载

```typescript
function convert<T extends string>(value: T): T extends 'number' ? number : string;
function convert(value: string): number | string {
  if (value === 'number') {
    return 42;
  }
  return 'result';
}

const num = convert('number');  // number
const str = convert('other');   // string
```

### 数组方法模拟

```typescript
function last<T>(arr: T[]): T | undefined;
function last<T>(arr: T[], n: number): T[];
function last<T>(arr: T[], n?: number): T | T[] | undefined {
  if (n === undefined) {
    return arr[arr.length - 1];
  }
  return arr.slice(-n);
}

const lastOne = last([1, 2, 3]);       // 3
const lastTwo = last([1, 2, 3], 2);    // [2, 3]
```

## 实际应用示例

### 类型安全的 API 请求

```typescript
interface APIResponse<T> {
  data: T;
  status: number;
  message: string;
}

async function fetchAPI<T>(
  url: string,
  options?: RequestInit
): Promise<APIResponse<T>> {
  const response = await fetch(url, options);
  const data = await response.json();
  return {
    data,
    status: response.status,
    message: response.statusText
  };
}

interface User {
  id: string;
  name: string;
}

const result = await fetchAPI<User>('/api/user/1');
console.log(result.data.name);  // 类型安全
```

### 类型安全的事件发射器

```typescript
type EventHandler<T = any> = (data: T) => void;

function createEventEmitter<T extends Record<string, any>>() {
  const handlers = new Map<keyof T, Set<EventHandler>>();

  return {
    on<K extends keyof T>(event: K, handler: EventHandler<T[K]>) {
      if (!handlers.has(event)) {
        handlers.set(event, new Set());
      }
      handlers.get(event)!.add(handler);
    },

    emit<K extends keyof T>(event: K, data: T[K]) {
      const eventHandlers = handlers.get(event);
      eventHandlers?.forEach(handler => handler(data));
    },

    off<K extends keyof T>(event: K, handler: EventHandler<T[K]>) {
      handlers.get(event)?.delete(handler);
    }
  };
}

// 使用
interface Events {
  login: { userId: string };
  logout: void;
  message: { from: string; content: string };
}

const emitter = createEventEmitter<Events>();
emitter.on('login', (data) => {
  console.log(data.userId);  // 类型安全
});
```
