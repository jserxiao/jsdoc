## 声明文件模板

### 基本结构

```typescript
// my-library.d.ts

// 1. 类型定义
type SomeType = string | number;

// 2. 接口定义
interface Options {
  debug?: boolean;
  timeout?: number;
}

// 3. 函数声明
declare function doSomething(value: string): number;

// 4. 类声明
declare class MyClass {
  constructor(options?: Options);
  method(): void;
}

// 5. 模块导出
export { doSomething, MyClass, Options, SomeType };
```

## 函数库声明

### 简单函数

```typescript
// utils.d.ts

// 单个函数
declare function formatDate(date: Date | string, format?: string): string;

// 带选项的函数
declare function request(
  url: string,
  options?: {
    method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
    headers?: Record<string, string>;
    body?: string;
    timeout?: number;
  }
): Promise<Response>;

// 函数重载
declare function createElement(tag: 'div'): HTMLDivElement;
declare function createElement(tag: 'span'): HTMLSpanElement;
declare function createElement(tag: 'a'): HTMLAnchorElement;
declare function createElement(tag: string): HTMLElement;
```

## 类库声明

### 基本类声明

```typescript
// my-class.d.ts

export class User {
  // 属性
  readonly id: string;
  name: string;
  email: string;
  
  // 构造函数
  constructor(data: {
    name: string;
    email: string;
  });
  
  // 实例方法
  update(data: Partial<User>): void;
  delete(): Promise<void>;
  
  // 静态方法
  static findById(id: string): Promise<User>;
  static findAll(): Promise<User[]>;
  static create(data: { name: string; email: string }): Promise<User>;
}
```

### 继承和多态

```typescript
// entities.d.ts

export abstract class Entity {
  readonly id: string;
  createdAt: Date;
  updatedAt: Date;
  
  save(): Promise<void>;
  delete(): Promise<void>;
}

export interface Timestampable {
  createdAt: Date;
  updatedAt: Date;
}

export class User extends Entity implements Timestampable {
  name: string;
  email: string;
  
  // 重写父类方法
  save(): Promise<void>;
}

export class Product extends Entity implements Timestampable {
  name: string;
  price: number;
  stock: number;
  
  save(): Promise<void>;
}
```

### 带事件的类

```typescript
// event-emitter.d.ts

type EventHandler = (...args: any[]) => void;

export class EventEmitter<EventMap extends Record<string, any>> {
  on<K extends keyof EventMap>(
    event: K,
    handler: (data: EventMap[K]) => void
  ): this;
  
  once<K extends keyof EventMap>(
    event: K,
    handler: (data: EventMap[K]) => void
  ): this;
  
  off<K extends keyof EventMap>(
    event: K,
    handler?: (data: EventMap[K]) => void
  ): this;
  
  emit<K extends keyof EventMap>(
    event: K,
    data: EventMap[K]
  ): boolean;
}

// 使用示例
interface UserEvents {
  login: { userId: string; timestamp: Date };
  logout: { userId: string };
  update: { userId: string; changes: Record<string, unknown> };
}

export class UserService extends EventEmitter<UserEvents> {
  login(email: string, password: string): Promise<User>;
  logout(): void;
  updateProfile(data: Partial<User>): Promise<void>;
}
```

## 模块库声明

### ES Module 风格

```typescript
// my-module.d.ts

// 类型导出
export type Result<T, E = Error> = 
  | { ok: true; value: T }
  | { ok: false; error: E };

export interface Config {
  apiUrl: string;
  timeout: number;
  retries: number;
}

// 函数导出
export function initialize(config: Config): void;
export function destroy(): void;

// 类导出
export class Client {
  constructor(config?: Partial<Config>);
  request<T>(endpoint: string, options?: RequestOptions): Promise<Result<T>>;
  get<T>(endpoint: string): Promise<Result<T>>;
  post<T>(endpoint: string, body: unknown): Promise<Result<T>>;
}

// 默认导出
declare const api: {
  version: string;
  initialize(config: Config): void;
  createClient(config?: Partial<Config>): Client;
};

export default api;
```

### CommonJS 风格

```typescript
// my-commonjs.d.ts

declare namespace MyModule {
  interface Options {
    debug?: boolean;
  }
  
  function init(options?: Options): void;
  function doSomething(): void;
  
  class MyClass {
    constructor(name: string);
    greet(): string;
  }
}

declare function MyModule(options?: MyModule.Options): void;

export = MyModule;
```

### 混合导出

```typescript
// 混合导出风格
declare function myLibrary(options?: myLibrary.Options): myLibrary.Instance;

declare namespace myLibrary {
  interface Options {
    debug?: boolean;
  }
  
  interface Instance {
    version: string;
    init(): void;
    destroy(): void;
  }
  
  class Utils {
    static format(value: string): string;
    static parse(value: string): unknown;
  }
}

export = myLibrary;
```

## UMD 库声明

UMD 库可以同时作为全局变量和模块使用。

```typescript
// my-umd.d.ts

// 全局声明
export as namespace MyLibrary;

// 模块导出
export function init(options?: Options): void;
export function destroy(): void;

export interface Options {
  debug?: boolean;
  apiUrl: string;
}

export class Client {
  constructor(config?: Partial<Options>);
  connect(): Promise<void>;
  disconnect(): void;
}

// 默认导出
declare const _default: {
  init(options?: Options): void;
  destroy(): void;
  Client: typeof Client;
};

export default _default;
```

## 插件模式

### jQuery 插件

```typescript
// jquery-plugin.d.ts

import 'jquery';

declare module 'jquery' {
  interface JQuery {
    myPlugin(options?: PluginOptions): JQuery;
    myPlugin(method: 'destroy'): JQuery;
    myPlugin(method: 'update', data: unknown): JQuery;
  }
  
  interface PluginOptions {
    debug?: boolean;
    onInit?: (element: HTMLElement) => void;
    onDestroy?: (element: HTMLElement) => void;
  }
}
```

### Express 插件

```typescript
// express-plugin.d.ts

import 'express';

declare module 'express' {
  interface Request {
    myPlugin?: {
      data: unknown;
      timestamp: Date;
    };
  }
  
  interface Response {
    myPlugin: {
      send(data: unknown): void;
      error(message: string): void;
    };
  }
  
  interface Application {
    myPlugin: {
      init(options?: PluginOptions): void;
    };
  }
  
  interface PluginOptions {
    debug?: boolean;
    prefix?: string;
  }
}
```

### Vue 插件

```typescript
// vue-plugin.d.ts

import Vue from 'vue';

declare module 'vue/types/vue' {
  interface Vue {
    $myPlugin: {
      doSomething(): void;
      getData(): unknown;
    };
  }
  
  interface VueConstructor {
    myPlugin: PluginObject<PluginOptions>;
  }
}

interface PluginOptions {
  debug?: boolean;
  apiKey?: string;
}
```

## 声明文件测试

### 测试文件

```typescript
// my-library.test-d.ts
import { expectType } from 'tsd';
import { User, formatDate, Result } from './my-library';

// 测试函数返回类型
expectType<string>(formatDate(new Date()));

// 测试类实例
const user = new User({ name: 'Alice', email: 'alice@example.com' });
expectType<string>(user.name);
expectType<Promise<void>>(user.delete());

// 测试类型推断
const result: Result<string> = { ok: true, value: 'test' };
if (result.ok) {
  expectType<string>(result.value);
}
```

## 最佳实践

### 1. 提供完整文档

```typescript
/**
 * 格式化日期
 * @param date - 日期对象、字符串或时间戳
 * @param format - 格式字符串，默认 'YYYY-MM-DD'
 * @returns 格式化后的日期字符串
 * @example
 * ```ts
 * formatDate(new Date(), 'YYYY-MM-DD HH:mm:ss')
 * // '2024-01-15 10:30:00'
 * ```
 */
export function formatDate(
  date: Date | string | number,
  format?: string
): string;
```

### 2. 使用严格类型

```typescript
// ✅ 好：精确类型
export function request<T>(
  url: string,
  options?: RequestOptions
): Promise<T>;

// ❌ 差：使用 any
export function request(url: string, options?: any): Promise<any>;
```

### 3. 提供类型工具

```typescript
// 提供类型工具帮助用户
export type ExtractResultType<T> = T extends Promise<infer R> ? R : never;
export type ExtractArrayType<T> = T extends Array<infer R> ? R : never;
```
