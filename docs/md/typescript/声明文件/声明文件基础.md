# 声明文件基础

声明文件（Declaration Files）是 TypeScript 的核心概念之一，用于描述 JavaScript 代码的类型信息。`.d.ts` 文件只包含类型声明，不包含实现代码，类似于 C/C++ 的头文件。

## 概述

### 什么是声明文件

声明文件是 `.d.ts` 后缀的文件，用于描述 JavaScript 代码的类型信息。它们告诉 TypeScript 编译器，某些变量、函数、类等存在，并提供它们的类型信息。

```typescript
// example.d.ts
declare const myGlobal: string;
declare function myFunction(x: number): string;
declare class MyClass {
  constructor(name: string);
  greet(): string;
}
```

### 为什么需要声明文件

1. **为 JavaScript 库提供类型支持**：让 TypeScript 项目能使用 JS 库
2. **类型安全**：在使用第三方库时获得类型检查和智能提示
3. **代码文档化**：声明文件本身就是很好的文档
4. **分离关注点**：类型声明与实现代码分离

### 声明文件的查找

TypeScript 会按以下顺序查找声明文件：

1. `node_modules/@types/` - DefinitelyTyped 提供的类型包
2. `tsconfig.json` 中 `typeRoots` 指定的目录
3. `tsconfig.json` 中 `files` 和 `include` 指定的文件
4. 导入模块时自动查找的 `.d.ts` 文件

## 基本语法

### 变量声明

```typescript
// declare let/var/const 声明全局变量
declare let PI: number;
declare const VERSION: string;
declare var window: Window;

// 声明全局常量
declare const API_URL: string;

// 声明复杂类型
declare const myApp: {
  version: string;
  config: {
    debug: boolean;
    apiUrl: string;
  };
  init(): void;
};
```

### 函数声明

```typescript
// 基本函数声明
declare function fetch(url: string): Promise<Response>;
declare function setTimeout(handler: () => void, timeout: number): number;

// 函数重载
declare function createElement(tag: 'div'): HTMLDivElement;
declare function createElement(tag: 'span'): HTMLSpanElement;
declare function createElement(tag: string): HTMLElement;

// 带可选参数的函数
declare function fetch(
  url: string,
  options?: {
    method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
    headers?: Record<string, string>;
    body?: string;
  }
): Promise<Response>;
```

### 类声明

```typescript
// 基本类声明
declare class User {
  // 属性
  name: string;
  age: number;

  // 构造函数
  constructor(name: string, age: number);

  // 方法
  greet(): string;
  getInfo(): { name: string; age: number };

  // 静态成员
  static create(name: string): User;
}

// 继承
declare class Admin extends User {
  role: 'admin';
  permissions: string[];
}

// 抽象类
declare abstract class Animal {
  abstract speak(): void;
  move(): void;
}
```

### 接口声明

```typescript
// 接口声明不需要 declare
interface User {
  id: number;
  name: string;
  email: string;
}

interface Config {
  apiUrl: string;
  debug?: boolean;
}

// 函数类型接口
interface Callback {
  (data: unknown): void;
}

// 可索引类型
interface StringMap {
  [key: string]: string;
}
```

### 类型别名

```typescript
// 类型别名不需要 declare
type ID = string | number;
type UserCallback = (user: User) => void;
type DeepPartial<T> = {
  [P in keyof T]?: DeepPartial<T[P]>;
};
```

### 枚举声明

```typescript
// 枚举需要 declare
declare enum Direction {
  Up = 'UP',
  Down = 'DOWN',
  Left = 'LEFT',
  Right = 'RIGHT'
}

// const 枚举
declare const enum HttpStatus {
  OK = 200,
  NotFound = 404,
  ServerError = 500
}
```

## 全局声明

### declare global

在模块文件中扩展全局作用域：

```typescript
// 在模块中扩展全局类型
declare global {
  interface Window {
    myAPI: {
      init(): void;
      destroy(): void;
      version: string;
    };
  }

  // 扩展 Array 接口
  interface Array<T> {
    first(): T | undefined;
    last(): T | undefined;
  }

  // 声明全局变量
  var myGlobal: {
    config: Config;
    init(): void;
  };
}

// 必须有导出才能被视为模块
export {};
```

### 全局变量声明文件

```typescript
// global.d.ts
declare var myGlobal: {
  version: string;
  config: Config;
  init(): void;
};

declare function initialize(options?: Options): void;

declare const DEBUG: boolean;
declare const API_BASE_URL: string;
```

### 全局函数声明

```typescript
// utils.d.ts
declare function log(message: string, level?: 'info' | 'warn' | 'error'): void;
declare function debounce<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): T;
declare function throttle<T extends (...args: any[]) => any>(
  fn: T,
  limit: number
): T;
```

## 模块声明

### 基本模块声明

```typescript
// my-lib.d.ts
declare module 'my-lib' {
  // 导出函数
  export function doSomething(value: string): number;
  export function doAnotherThing(value: number): string;

  // 导出接口
  export interface Options {
    debug?: boolean;
    timeout?: number;
  }

  // 导出类型
  export type Result = {
    success: boolean;
    data?: unknown;
    error?: string;
  };

  // 导出常量
  export const VERSION: string;

  // 默认导出
  const main: {
    init(options?: Options): void;
    destroy(): void;
  };
  export default main;
}
```

### 声明文件的模块形式

```typescript
// lodash.d.ts
export function debounce<T extends (...args: any[]) => any>(
  func: T,
  wait?: number,
  options?: {
    leading?: boolean;
    maxWait?: number;
    trailing?: boolean;
  }
): T;

export function throttle<T extends (...args: any[]) => any>(
  func: T,
  wait?: number,
  options?: {
    leading?: boolean;
    trailing?: boolean;
  }
): T;

export function get(object: object, path: string, defaultValue?: any): any;
export function set(object: object, path: string, value: any): object;

export default _.LoDashStatic;
```

### 模块扩展

```typescript
// 扩展现有模块
declare module 'express' {
  interface Request {
    user?: {
      id: string;
      name: string;
      role: string;
    };
  }

  interface Response {
    success(data: any): void;
    error(message: string, code?: number): void;
  }
}
```

### 非JS模块声明

```typescript
// 声明图片模块
declare module '*.png' {
  const src: string;
  export default src;
}

declare module '*.jpg' {
  const src: string;
  export default src;
}

declare module '*.svg' {
  const content: string;
  export default content;
}

// 声明 CSS 模块
declare module '*.css' {
  const classes: { readonly [key: string]: string };
  export default classes;
}

declare module '*.scss' {
  const content: { readonly [key: string]: string };
  export default content;
}

// 声明 JSON 模块
declare module '*.json' {
  const value: unknown;
  export default value;
}
```

## 声明合并

### 接口合并

```typescript
// 原始声明
interface User {
  name: string;
}

// 后续声明会合并
interface User {
  age: number;
}

// 结果：User 有 name 和 age 两个属性
const user: User = {
  name: 'Alice',
  age: 30
};
```

### 类合并

```typescript
// 声明文件
declare class User {
  name: string;
}

// 接口可以扩展类
interface User {
  age: number;
  greet(): string;
}
```

### 命名空间合并

```typescript
declare namespace MyLib {
  interface Options {
    debug: boolean;
  }
}

declare namespace MyLib {
  interface Config {
    apiUrl: string;
  }
}

// 结果：MyLib 包含 Options 和 Config
```

### 函数与命名空间合并

```typescript
declare function MyLib(options: MyLib.Options): void;

declare namespace MyLib {
  interface Options {
    debug: boolean;
  }
  function init(): void;
}

// 使用
MyLib({ debug: true });
MyLib.init();
```

## 扩展第三方类型

### 扩展 Window 接口

```typescript
// global.d.ts
interface Window {
  // 添加自定义属性
  myCustomAPI: {
    init(): void;
    destroy(): void;
  };

  // 添加分析工具
  analytics?: {
    track(event: string, data?: Record<string, unknown>): void;
    identify(userId: string, traits?: Record<string, unknown>): void;
  };
}

// 使用
window.myCustomAPI.init();
window.analytics?.track('page_view');
```

### 扩展 Array 接口

```typescript
// array-extensions.d.ts
interface Array<T> {
  first(): T | undefined;
  last(): T | undefined;
  unique(): T[];
  flatten(): T extends Array<infer U> ? U[] : T[];
}

// 使用
const arr = [1, 2, 2, 3];
arr.first();  // 1
arr.last();   // 3
arr.unique(); // [1, 2, 3]
```

### 扩展 Promise

```typescript
// promise-extensions.d.ts
interface Promise<T> {
  finally(onFinally: () => void): Promise<T>;
  timeout(ms: number): Promise<T>;
}

// 注意：实例方法扩展时需要注意兼容性
```

### 扩展 Express Request

```typescript
// express.d.ts
declare module 'express' {
  interface Request {
    user?: {
      id: string;
      email: string;
      role: 'admin' | 'user';
    };
    session: {
      id: string;
      data: Record<string, unknown>;
    };
  }
}

// 使用
import express from 'express';

const app = express();

app.get('/profile', (req, res) => {
  const userId = req.user?.id; // 类型安全
});
```

## 实际应用场景

### 为 JavaScript 库编写声明

```typescript
// my-utils.d.ts
/**
 * 工具函数库
 */
declare module 'my-utils' {
  /**
   * 深拷贝对象
   */
  export function deepClone<T>(obj: T): T;

  /**
   * 防抖函数
   */
  export function debounce<T extends (...args: any[]) => any>(
    fn: T,
    delay: number,
    options?: {
      leading?: boolean;
      trailing?: boolean;
    }
  ): T;

  /**
   * 节流函数
   */
  export function throttle<T extends (...args: any[]) => any>(
    fn: T,
    interval: number
  ): T;

  /**
   * 格式化日期
   */
  export function formatDate(date: Date | string | number, format: string): string;

  /**
   * 解析 URL 参数
   */
  export function parseQueryString(url: string): Record<string, string>;

  /**
   * 配置选项
   */
  export interface UtilsOptions {
    debug?: boolean;
    logLevel?: 'info' | 'warn' | 'error';
  }

  /**
   * 初始化配置
   */
  export function init(options?: UtilsOptions): void;

  const utils: {
    version: string;
    init(options?: UtilsOptions): void;
  };

  export default utils;
}
```

### jQuery 风格的声明

```typescript
// jquery.d.ts
declare const $: JQueryStatic;

interface JQueryStatic {
  (selector: string): JQuery;
  (element: HTMLElement): JQuery;
  ajax(settings: JQueryAjaxSettings): JQueryXHR;
  get(url: string, data?: object, success?: (data: any) => void): JQueryXHR;
  post(url: string, data?: object, success?: (data: any) => void): JQueryXHR;
}

interface JQuery {
  length: number;
  html(): string;
  html(value: string): JQuery;
  text(): string;
  text(value: string): JQuery;
  attr(name: string): string | undefined;
  attr(name: string, value: string): JQuery;
  on(event: string, handler: (e: Event) => void): JQuery;
  off(event: string, handler?: (e: Event) => void): JQuery;
  css(property: string): string;
  css(property: string, value: string): JQuery;
  addClass(className: string): JQuery;
  removeClass(className: string): JQuery;
  find(selector: string): JQuery;
  each(callback: (index: number, element: HTMLElement) => void): JQuery;
}

interface JQueryAjaxSettings {
  url: string;
  method?: string;
  data?: any;
  headers?: Record<string, string>;
  success?(data: any, textStatus: string, jqXHR: JQueryXHR): void;
  error?(jqXHR: JQueryXHR, textStatus: string, errorThrown: string): void;
}

interface JQueryXHR {
  done(callback: (data: any) => void): JQueryXHR;
  fail(callback: (jqXHR: JQueryXHR, textStatus: string) => void): JQueryXHR;
  always(callback: () => void): JQueryXHR;
}
```

## 最佳实践

### 文件组织

```typescript
// 推荐的目录结构
project/
├── src/
│   ├── index.ts
│   └── types/
│       ├── global.d.ts      // 全局类型声明
│       ├── express.d.ts     // 扩展 express 类型
│       └── my-lib.d.ts      // 第三方库声明
├── types/                    // 项目自定义类型
│   ├── index.d.ts
│   └── utils.d.ts
└── tsconfig.json
```

### 使用 JSDoc 注释

```typescript
/**
 * 发送网络请求
 * @param url - 请求地址
 * @param options - 请求选项
 * @returns 返回 Promise
 * @example
 * ```ts
 * fetch('/api/users', { method: 'GET' })
 *   .then(res => res.json())
 * ```
 */
declare function fetch(
  url: string,
  options?: {
    method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
    headers?: Record<string, string>;
    body?: string;
  }
): Promise<Response>;
```

### 类型完整性

```typescript
// ✅ 好：完整的类型定义
declare function processData(
  data: {
    id: number;
    name: string;
    items: Array<{ value: number; label: string }>;
  },
  options?: {
    validate?: boolean;
    transform?: (item: { value: number; label: string }) => void;
  }
): Promise<{
  success: boolean;
  result?: unknown;
  error?: string;
}>;

// ❌ 差：使用 any
declare function processData(data: any, options?: any): any;
```

### 避免重复声明

```typescript
// ✅ 使用 extends 继承
interface AdvancedOptions extends BasicOptions {
  advanced?: boolean;
}

// ❌ 重复定义
interface AdvancedOptions {
  debug?: boolean;
  advanced?: boolean;
}
```

## 小结

| 语法 | 用途 |
|------|------|
| `declare var/let/const` | 声明全局变量 |
| `declare function` | 声明全局函数 |
| `declare class` | 声明全局类 |
| `declare enum` | 声明枚举 |
| `declare module` | 声明模块 |
| `declare global` | 在模块中扩展全局 |
| `declare namespace` | 声明命名空间 |
