## 模块声明基础

### declare module 语法

```typescript
// 声明一个模块
declare module 'my-library' {
  export function doSomething(): void;
  export const version: string;
}

// 使用
import { doSomething, version } from 'my-library';
```

### 模块声明文件形式

```typescript
// 方式1：使用 declare module
declare module 'my-library' {
  export function init(): void;
  export class Client {
    constructor();
    connect(): Promise<void>;
  }
}

// 方式2：直接导出（更推荐）
// my-library.d.ts
export function init(): void;
export class Client {
  constructor();
  connect(): Promise<void>;
}
```

## 导出类型

### 具名导出

```typescript
// utils.d.ts
export function debounce<T extends (...args: any[]) => any>(
  fn: T,
  wait: number
): T;

export function throttle<T extends (...args: any[]) => any>(
  fn: T,
  limit: number
): T;

export interface Options {
  leading?: boolean;
  trailing?: boolean;
}

export type Result<T> = {
  success: boolean;
  data?: T;
  error?: string;
};
```

### 默认导出

```typescript
// api.d.ts
export interface Config {
  baseUrl: string;
  timeout?: number;
}

export class Client {
  constructor(config: Config);
  get<T>(url: string): Promise<T>;
  post<T>(url: string, body: unknown): Promise<T>;
}

const api: {
  createClient(config: Config): Client;
  version: string;
};

export default api;
```

### 混合导出

```typescript
// library.d.ts
export function helper(): void;

export class Service {
  run(): void;
}

export interface Config {
  debug?: boolean;
}

const defaultExport: {
  helper: typeof helper;
  Service: typeof Service;
  version: string;
};

export default defaultExport;
```

### 重新导出

```typescript
// index.d.ts
export { debounce, throttle } from './functions';
export { Client } from './client';
export type { Options, Result } from './types';
export { default as api } from './api';
```

## 模块扩展

### 扩展现有模块

```typescript
// express-extension.d.ts
import 'express';

declare module 'express' {
  interface Request {
    user?: {
      id: string;
      role: string;
    };
  }
  
  interface Response {
    success(data: unknown): void;
    error(message: string, code?: number): void;
  }
}
```

### 全局扩展

```typescript
// global-extension.d.ts
// 在模块中扩展全局作用域
declare global {
  interface Array<T> {
    first(): T | undefined;
    last(): T | undefined;
  }
  
  namespace NodeJS {
    interface ProcessEnv {
      NODE_ENV: 'development' | 'production' | 'test';
      API_URL: string;
    }
  }
}

export {}; // 必须有导出才能被视为模块
```

### 文件模块声明

```typescript
// 声明非 TypeScript 文件
declare module '*.css' {
  const styles: { [className: string]: string };
  export default styles;
}

declare module '*.scss' {
  const styles: { [className: string]: string };
  export default styles;
}

declare module '*.png' {
  const src: string;
  export default src;
}

declare module '*.jpg' {
  const src: string;
  export default src;
}

declare module '*.svg' {
  const content: string;
  export default content;
}

declare module '*.json' {
  const value: unknown;
  export default value;
}
```

## 声明文件定位

### 查找策略

TypeScript 按以下顺序查找模块的声明文件：

1. 模块自身的 `package.json` 中的 `types` 或 `typings` 字段
2. 模块根目录的 `index.d.ts`
3. `node_modules/@types/` 目录
4. `tsconfig.json` 中 `typeRoots` 指定的目录

### package.json 配置

```json
{
  "name": "my-library",
  "version": "1.0.0",
  "main": "dist/index.js",
  "types": "dist/index.d.ts",
  "exports": {
    ".": {
      "import": "./dist/index.mjs",
      "require": "./dist/index.js",
      "types": "./dist/index.d.ts"
    },
    "./utils": {
      "import": "./dist/utils.mjs",
      "require": "./dist/utils.js",
      "types": "./dist/utils.d.ts"
    }
  }
}
```

### 目录结构

```
my-library/
├── package.json
├── dist/
│   ├── index.js
│   ├── index.d.ts        # 主声明文件
│   ├── utils.js
│   └── utils.d.ts        # 工具函数声明
└── src/
    ├── index.ts
    └── utils.ts
```
