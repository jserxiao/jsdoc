# 函数式组合

函数式组合（Functional Composition）是函数式编程的核心概念，通过将多个简单函数组合成复杂函数，实现代码的复用和可读性提升。

## 组合概念

### 什么是函数组合？

函数组合是将两个或多个函数结合，产生一个新函数的过程。

```javascript
// 数学定义：(f ∘ g)(x) = f(g(x))

// 简单示例
const double = x => x * 2;
const addOne = x => x + 1;

// 不使用组合
const result1 = addOne(double(3));  // 7

// 使用组合
const doubleThenAddOne = x => addOne(double(x));
const result2 = doubleThenAddOne(3);  // 7
```

### 组合的优势

```javascript
// ❌ 命令式：一步一步描述如何做
function processImperative(str) {
  let result = str.trim();
  result = result.toLowerCase();
  result = result.replace(/\s+/g, '-');
  return result;
}

// ✅ 函数式：描述要做什么
const trim = str => str.trim();
const toLowerCase = str => str.toLowerCase();
const slugify = str => str.replace(/\s+/g, '-');

const processFunctional = pipe(trim, toLowerCase, slugify);

console.log(processFunctional('  Hello World  '));  // 'hello-world'
```

## compose 函数

### 基本实现

`compose` 从右到左执行函数（数学顺序）。

```javascript
// 基础版
const compose = (f, g) => x => f(g(x));

// 多函数版本
const compose = (...fns) => x =>
  fns.reduceRight((acc, fn) => fn(acc), x);

// 使用
const double = x => x * 2;
const addOne = x => x + 1;
const square = x => x * x;

const compute = compose(addOne, double, square);
// 执行顺序：square → double → addOne
console.log(compute(3));  // ((3²) * 2) + 1 = 19
```

### 支持多参数

```javascript
const compose = (...fns) => (...args) =>
  fns.reduceRight((acc, fn) => fn(acc), fns.pop()(...args));

// 使用
const sum = (a, b, c) => a + b + c;
const double = x => x * 2;

const compute = compose(double, sum);
console.log(compute(1, 2, 3));  // (1+2+3) * 2 = 12
```

### 实际应用

```javascript
// 数据处理管道
const users = [
  { name: 'John', age: 30, active: true },
  { name: 'Jane', age: 25, active: false },
  { name: 'Bob', age: 35, active: true }
];

const filter = predicate => arr => arr.filter(predicate);
const map = fn => arr => arr.map(fn);
const sort = fn => arr => [...arr].sort(fn);

// 获取活跃用户的名称，按字母排序
const getActiveUserNames = compose(
  arr => arr,
  sort((a, b) => a.localeCompare(b)),
  map(u => u.name),
  filter(u => u.active)
);

console.log(getActiveUserNames(users));  // ['Bob', 'John']

// 使用 point-free 风格
const prop = key => obj => obj[key];
const isActive = prop('active');
const getName = prop('name');

const getNames = compose(
  map(getName),
  filter(isActive)
);
```

## pipe 函数

### 基本实现

`pipe` 从左到右执行函数（更符合直觉）。

```javascript
// 基础版
const pipe = (f, g) => x => g(f(x));

// 多函数版本
const pipe = (...fns) => x =>
  fns.reduce((acc, fn) => fn(acc), x);

// 支持多参数
const pipe = (...fns) => (...args) =>
  fns.reduce((acc, fn) => fn(acc), fns.shift()(...args));

// 使用
const double = x => x * 2;
const addOne = x => x + 1;
const square = x => x * x;

const compute = pipe(double, addOne, square);
// 执行顺序：double → addOne → square
console.log(compute(3));  // ((3 * 2) + 1)² = 49
```

### 实际应用

```javascript
// 字符串处理
const trim = str => str.trim();
const toLowerCase = str => str.toLowerCase();
const split = separator => str => str.split(separator);
const join = separator => arr => arr.join(separator);
const map = fn => arr => arr.map(fn);

const capitalize = str => str.charAt(0).toUpperCase() + str.slice(1);

// 将 "hello world" 转换为 "Hello-World"
const toSlug = pipe(
  trim,
  toLowerCase,
  split(' '),
  map(capitalize),
  join('-')
);

console.log(toSlug('  hello world  '));  // 'Hello-World'

// 数据验证管道
const isRequired = value => value !== null && value !== undefined && value !== '';
const isEmail = value => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
const isLength = min => value => value.length >= min;

const validateEmail = pipe(
  value => ({ value, valid: true, errors: [] }),
  result => result.value && isRequired(result.value)
    ? result
    : { ...result, valid: false, errors: [...result.errors, 'Required'] },
  result => result.value && isEmail(result.value)
    ? result
    : { ...result, valid: false, errors: [...result.errors, 'Invalid email'] }
);
```

### compose vs pipe

```javascript
// compose：从右到左（数学风格）
const compute1 = compose(fn3, fn2, fn1);
// 等同于 fn3(fn2(fn1(x)))

// pipe：从左到右（Unix 风格）
const compute2 = pipe(fn1, fn2, fn3);
// 等同于 fn3(fn2(fn1(x)))

// 两者可以互相转换
const pipeFromCompose = (...fns) => compose(...fns.reverse());
const composeFromPipe = (...fns) => pipe(...fns.reverse());
```

## 柯里化

### 什么是柯里化？

将接受多个参数的函数转换为一系列接受单个参数的函数。

```javascript
// 普通函数
const add = (a, b, c) => a + b + c;
add(1, 2, 3);  // 6

// 柯里化后
const curriedAdd = a => b => c => a + b + c;
curriedAdd(1)(2)(3);  // 6
```

### 柯里化实现

```javascript
// 基础版
const curry = (fn) => {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    }
    return (...moreArgs) => curried.apply(this, args.concat(moreArgs));
  };
};

// 使用
const add = (a, b, c) => a + b + c;
const curriedAdd = curry(add);

console.log(curriedAdd(1)(2)(3));  // 6
console.log(curriedAdd(1, 2)(3));  // 6
console.log(curriedAdd(1)(2, 3));  // 6
console.log(curriedAdd(1, 2, 3));  // 6

// 柯里化的 map
const curriedMap = curry((fn, arr) => arr.map(fn));

const double = x => x * 2;
const mapDouble = curriedMap(double);

console.log(mapDouble([1, 2, 3]));  // [2, 4, 6]
```

### 柯里化应用

```javascript
// 创建可复用的函数
const formatCurrency = curry((symbol, decimals, number) =>
  symbol + number.toFixed(decimals)
);

const formatUSD = formatCurrency('$', 2);
const formatEUR = formatCurrency('€', 2);

console.log(formatUSD(123.456));  // '$123.46'
console.log(formatEUR(123.456));  // '€123.46'

// 过滤器工厂
const filterBy = curry((key, value, arr) =>
  arr.filter(item => item[key] === value)
);

const users = [
  { name: 'John', role: 'admin' },
  { name: 'Jane', role: 'user' },
  { name: 'Bob', role: 'admin' }
];

const filterAdmins = filterBy('role')('admin');
console.log(filterAdmins(users));  // [{ name: 'John', ... }, { name: 'Bob', ... }]

// API 请求工厂
const request = curry((method, baseURL, endpoint, options = {}) =>
  fetch(`${baseURL}${endpoint}`, { method, ...options })
);

const apiRequest = request('GET')('https://api.example.com');
const getUsers = apiRequest('/users');
const getPosts = apiRequest('/posts');
```

### 占位符柯里化

```javascript
// 支持占位符的柯里化
const curry = (fn) => {
  const _ = Symbol('placeholder');

  const curried = (...args) => {
    if (args.length >= fn.length && !args.includes(_)) {
      return fn(...args);
    }

    return (...moreArgs) => {
      const newArgs = args.map(arg =>
        arg === _ && moreArgs.length ? moreArgs.shift() : arg
      );
      return curried(...newArgs, ...moreArgs);
    };
  };

  curried._ = _;
  return curried;
};

// 使用占位符
const { _ } = curry;

const slice = curry((start, end, arr) => arr.slice(start, end));
const takeFirst3 = slice(0, _, _)(3, _);  // 先固定 start = 0, end = 3

console.log(takeFirst3([1, 2, 3, 4, 5]));  // [1, 2, 3]
```

## 偏函数应用

### 什么是偏函数？

预先设置部分参数，返回一个接受剩余参数的新函数。

```javascript
// bind 实现偏函数
const add = (a, b, c) => a + b + c;
const add5 = add.bind(null, 5);

console.log(add5(3, 2));  // 10

// 手动实现
const partial = (fn, ...presetArgs) => (...laterArgs) =>
  fn(...presetArgs, ...laterArgs);

const add10 = partial(add, 10);
console.log(add10(5, 3));  // 18

// 偏函数 vs 柯里化
// 柯里化：每次只接受一个参数
// 偏函数：可以预设任意数量的参数
```

### 实际应用

```javascript
// 日志函数
const log = (level, timestamp, message) =>
  `[${level}] ${timestamp}: ${message}`;

const errorLog = partial(log, 'ERROR', new Date().toISOString());
const infoLog = partial(log, 'INFO', new Date().toISOString());

console.log(errorLog('Something went wrong'));
// [ERROR] 2024-01-15T10:30:00.000Z: Something went wrong

// 事件处理器
const handleEvent = curry((action, event, data) => {
  console.log(`Action: ${action}, Event: ${event.type}, Data:`, data);
});

const handleClick = handleEvent('click');
element.addEventListener('click', (e) => handleClick(e)(someData));
```

## 函子

### 什么是函子？

函子（Functor）是一个实现了 `map` 方法的容器，可以对容器内的值进行变换而不改变容器本身。

### Identity 函子

```javascript
class Identity {
  constructor(value) {
    this.$value = value;
  }

  static of(value) {
    return new Identity(value);
  }

  map(fn) {
    return Identity.of(fn(this.$value));
  }

  // 用于调试
  inspect() {
    return `Identity(${this.$value})`;
  }
}

// 使用
const result = Identity.of(5)
  .map(x => x + 1)
  .map(x => x * 2);

console.log(result.$value);  // 12
```

### Maybe 函子

```javascript
class Maybe {
  constructor(value) {
    this.$value = value;
  }

  static of(value) {
    return new Maybe(value);
  }

  isNothing() {
    return this.$value === null || this.$value === undefined;
  }

  map(fn) {
    return this.isNothing() ? Maybe.of(null) : Maybe.of(fn(this.$value));
  }

  // 提供默认值
  getOrElse(defaultValue) {
    return this.isNothing() ? defaultValue : this.$value;
  }
}

// 使用：安全地访问嵌套属性
const user = { profile: { name: 'John' } };

const getUserName = user =>
  Maybe.of(user)
    .map(u => u.profile)
    .map(p => p.name)
    .getOrElse('Anonymous');

console.log(getUserName(user));  // 'John'
console.log(getUserName({}));    // 'Anonymous'
```

### Either 函子

```javascript
class Either {
  static of(value) {
    return new Right(value);
  }
}

class Left {
  constructor(value) {
    this.$value = value;
  }

  map(fn) {
    return this;  // 左值不执行 map
  }

  getOrElse(defaultValue) {
    return defaultValue;
  }

  getOrElseThrow(error) {
    throw error;
  }
}

class Right {
  constructor(value) {
    this.$value = value;
  }

  map(fn) {
    return Either.of(fn(this.$value));
  }

  getOrElse(defaultValue) {
    return this.$value;
  }
}

// 使用：处理可能失败的操作
const safeDivide = (a, b) =>
  b === 0
    ? new Left('Division by zero')
    : new Right(a / b);

const result = safeDivide(10, 2)
  .map(x => x * 2)
  .getOrElse(NaN);

console.log(result);  // 10

const error = safeDivide(10, 0)
  .map(x => x * 2)
  .getOrElse(NaN);

console.log(error);  // NaN
```

### IO 函子

```javascript
class IO {
  constructor(fn) {
    this.unsafePerformIO = fn;
  }

  static of(value) {
    return new IO(() => value);
  }

  map(fn) {
    return new IO(() => fn(this.unsafePerformIO()));
  }

  // 链式调用
  chain(fn) {
    return new IO(() => fn(this.unsafePerformIO()).unsafePerformIO());
  }

  // 执行副作用
  run() {
    return this.unsafePerformIO();
  }
}

// 使用：延迟执行副作用
const readDOM = selector => new IO(() => document.querySelector(selector));
const writeDOM = (selector) => (content) =>
  new IO(() => {
    document.querySelector(selector).textContent = content;
  });

const program = readDOM('#input')
  .map(el => el.value)
  .map(str => str.toUpperCase())
  .chain(writeDOM('#output'));

// 延迟到需要时执行
program.run();
```

## 实用工具函数

### 常用组合工具

```javascript
// identity：返回输入值
const identity = x => x;

// always：返回常量函数
const always = x => () => x;

// once：只执行一次
const once = fn => {
  let called = false;
  let result;
  return (...args) => {
    if (!called) {
      called = true;
      result = fn(...args);
    }
    return result;
  };
};

// memoize：记忆化
const memoize = fn => {
  const cache = new Map();
  return (...args) => {
    const key = JSON.stringify(args);
    if (!cache.has(key)) {
      cache.set(key, fn(...args));
    }
    return cache.get(key);
  };
};

// flip：翻转前两个参数
const flip = fn => (a, b, ...args) => fn(b, a, ...args);

// 使用
const subtract = (a, b) => a - b;
const flippedSubtract = flip(subtract);
console.log(subtract(5, 3));      // 2
console.log(flippedSubtract(5, 3));  // -2
```

### 数组操作工具

```javascript
// map
const map = fn => arr => arr.map(fn);

// filter
const filter = predicate => arr => arr.filter(predicate);

// reduce
const reduce = (fn, initial) => arr => arr.reduce(fn, initial);

// find
const find = predicate => arr => arr.find(predicate);

// every
const every = predicate => arr => arr.every(predicate);

// some
const some = predicate => arr => arr.some(predicate);

// flatMap
const flatMap = fn => arr => arr.flatMap(fn);

// 使用管道
const process = pipe(
  filter(x => x > 0),
  map(x => x * 2),
  reduce((acc, x) => acc + x, 0)
);

console.log(process([-1, 2, 3, -4, 5]));  // 20
```

### 对象操作工具

```javascript
// prop：安全获取属性
const prop = key => obj => obj?.[key];

// assoc：设置属性（不可变）
const assoc = (key, value) => obj => ({ ...obj, [key]: value });

// dissoc：删除属性（不可变）
const dissoc = key => ({ [key]: _, ...rest }) => rest;

// evolve：深层更新
const evolve = transformations => obj => {
  const result = {};
  for (const key in obj) {
    if (key in transformations) {
      result[key] = typeof transformations[key] === 'function'
        ? transformations[key](obj[key])
        : evolve(transformations[key])(obj[key]);
    } else {
      result[key] = obj[key];
    }
  }
  return result;
};

// 使用
const user = { name: 'John', age: 30 };

const updatedUser = pipe(
  assoc('name', 'Jane'),
  assoc('email', 'jane@example.com'),
  dissoc('age')
)(user);

console.log(updatedUser);  // { name: 'Jane', email: 'jane@example.com' }
```

## 最佳实践

### 1. 保持函数纯粹

```javascript
// ❌ 不纯函数：依赖外部状态
let counter = 0;
const increment = () => ++counter;

// ✅ 纯函数：只依赖输入
const increment = counter => counter + 1;
```

### 2. 使用 point-free 风格

```javascript
// ❌ 显式参数
const getNames = users => users.map(user => user.name);

// ✅ Point-free
const getNames = map(prop('name'));
```

### 3. 组合而非嵌套

```javascript
// ❌ 嵌套调用
const result = encodeURIComponent(
  toLowerCase(
    trim(input)
  )
);

// ✅ 组合
const sanitize = pipe(trim, toLowerCase, encodeURIComponent);
const result = sanitize(input);
```

### 4. 单一职责

```javascript
// ❌ 大而全的函数
function processUser(user) {
  const trimmed = user.name.trim();
  const lower = trimmed.toLowerCase();
  const validated = lower.length > 0;
  // ...
}

// ✅ 小而专的函数
const trimName = user => ({ ...user, name: user.name.trim() });
const lowerName = user => ({ ...user, name: user.name.toLowerCase() });
const validateName = user => ({ ...user, valid: user.name.length > 0 });

const processUser = pipe(trimName, lowerName, validateName);
```
