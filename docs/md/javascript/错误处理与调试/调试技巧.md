## console API

### 基本输出方法

```javascript
// 基本日志
console.log('普通日志');
console.warn('警告信息');
console.error('错误信息');
console.info('提示信息');

// 带样式的日志
console.log('%c红色文字', 'color: red; font-size: 20px;');
console.log('%c背景%c文字', 'background: yellow;', 'color: blue;');

// 带图标的日志
console.log('✅ 成功');
console.log('❌ 失败');
console.log('⚠️ 警告');
console.log('🔍 查找');
```

### 格式化输出

```javascript
// 字符串格式化
const name = 'John';
const age = 30;
console.log('姓名: %s, 年龄: %d', name, age);
// 姓名: John, 年龄: 30

// 格式化说明符
// %s - 字符串
// %d 或 %i - 整数
// %f - 浮点数
// %o - 对象（可展开）
// %O - 对象（详细）
// %c - CSS 样式

// 对象输出
const user = { name: 'John', age: 30 };
console.log('用户: %o', user);

// DOM 元素输出
const div = document.querySelector('div');
console.log('DOM: %o', div);
```

### 表格输出

```javascript
// 对象数组表格
const users = [
  { name: 'John', age: 30, city: 'New York' },
  { name: 'Jane', age: 25, city: 'London' },
  { name: 'Bob', age: 35, city: 'Paris' }
];
console.table(users);

// 指定列
console.table(users, ['name', 'age']);

// 对象表格
const data = {
  user1: { name: 'John', age: 30 },
  user2: { name: 'Jane', age: 25 }
};
console.table(data);
```

### 分组输出

```javascript
// 折叠分组
console.group('用户信息');
console.log('姓名: John');
console.log('年龄: 30');
console.groupEnd();

// 默认展开的分组
console.groupCollapsed('详细信息');
console.log('详细1');
console.log('详细2');
console.groupEnd();

// 嵌套分组
console.group('外层');
console.group('内层1');
console.log('内容1');
console.groupEnd();
console.group('内层2');
console.log('内容2');
console.groupEnd();
console.groupEnd();
```

### 计时与计数

```javascript
// 计时
console.time('数据处理');
// ... 执行代码
console.timeEnd('数据处理'); // 输出耗时

// 多个计时器
console.time('API请求');
fetch('/api/data').then(() => console.timeEnd('API请求'));

// 计数
function handleClick() {
  console.count('点击次数');
}
handleClick(); // 点击次数: 1
handleClick(); // 点击次数: 2

// 重置计数
console.countReset('点击次数');

// 带标签的计数
console.count('按钮A');
console.count('按钮B');
console.count('按钮A');
```

### 断言与跟踪

```javascript
// 断言：条件为 false 时输出
console.assert(1 === 2, '1 不等于 2'); // 输出错误
console.assert(1 === 1, '这不会输出'); // 无输出

// 堆栈跟踪
function foo() {
  function bar() {
    console.trace('调用堆栈');
  }
  bar();
}
foo();
// 输出：
// console.trace() 调用堆栈
//     at bar (...)
//     at foo (...)
//     at ...
```

### 清空控制台

```javascript
console.clear(); // 清空控制台
```

### 输出对象详情

```javascript
const obj = { a: 1, b: { c: 2 } };

// dir: 显示对象所有属性
console.dir(obj);

// dirxml: 显示 XML/HTML 元素树
console.dirxml(document.body);
```

## 断点调试

### debugger 语句

```javascript
function calculate(a, b) {
  const result = a + b;
  debugger; // 代码在此暂停

}

calculate(1, 2);
```

2. 右键断点 → Edit breakpoint
3. 输入条件表达式

```javascript
// 只在特定条件下断点
for (let i = 0; i < 100; i++) {
  // 在开发者工具中设置条件断点：i === 50
  console.log(i);
}
```

### 日志点（Logpoint）

```javascript
// 不暂停执行，只输出日志
// 在开发者工具中右键行号 → Add logpoint
// 输入：'当前值：' + i

  process(i);
}
```

## 性能分析

### Performance API

```javascript
// 创建时间标记
performance.mark('start');

// 执行代码
for (let i = 0; i < 1000000; i++) {
  // 复杂计算
}

performance.mark('end');

// 测量时间
performance.measure('计算耗时', 'start', 'end');

// 获取测量结果
const measures = performance.getEntriesByName('计算耗时');
console.log(`耗时: ${measures[0].duration}ms`);

// 获取所有性能条目
performance.getEntriesByType('measure').forEach(entry => {
  console.log(`${entry.name}: ${entry.duration}ms`);
});

// 清除标记和测量
performance.clearMarks();
performance.clearMeasures();
```

### console.time 时间分析

```javascript
// 简单计时
console.time('循环');
for (let i = 0; i < 1000000; i++) {}
console.timeEnd('循环'); // 循环: 5.123ms

// 多个计时器对比
console.time('方法A');
methodA();
console.timeEnd('方法A');

console.time('方法B');
methodB();
console.timeEnd('方法B');
```

### Profile 性能分析

```javascript
// 开始性能分析
console.profile('我的分析');

// 执行代码
for (let i = 0; i < 10000; i++) {
  Math.random();
}

// 结束分析
console.profileEnd('我的分析');
// 结果会显示在 Profiles 面板中
```

### requestAnimationFrame 时间检测

```javascript
// 检测帧率
let lastTime = performance.now();
let frameCount = 0;

function measureFPS() {
  frameCount++;
  const now = performance.now();

  if (now - lastTime >= 1000) {
    console.log(`FPS: ${frameCount}`);
    frameCount = 0;
    lastTime = now;
  }

  requestAnimationFrame(measureFPS);
}

measureFPS();
```

### 长任务检测

```javascript
// PerformanceObserver 检测长任务
const observer = new PerformanceObserver((list) => {
  list.getEntries().forEach(entry => {
    console.log('长任务:', {
      名称: entry.name,
      耗时: entry.duration + 'ms',
      开始时间: entry.startTime
    });
  });
});

observer.observe({ entryTypes: ['longtask'] });

// 长任务：执行时间超过 50ms 的任务
```

## 网络调试

### Fetch 调试

```javascript
// 封装带日志的 fetch
function debugFetch(url, options = {}) {
  const startTime = performance.now();

  console.log(`🚀 请求开始: ${options.method || 'GET'} ${url}`);
  console.log('📤 请求参数:', options);

  return fetch(url, options)
    .then(response => {
      const duration = performance.now() - startTime;
      console.log(`✅ 请求成功: ${url} (${duration.toFixed(2)}ms)`);
      return response;
    })
    .catch(error => {
      const duration = performance.now() - startTime;
      console.error(`❌ 请求失败: ${url} (${duration.toFixed(2)}ms)`, error);
      throw error;
    });
}
```

### 拦截网络请求

```javascript
// 拦截 fetch
const originalFetch = window.fetch;
window.fetch = function(...args) {
  console.log('🌐 Fetch 请求:', args[0]);
  return originalFetch.apply(this, args)
    .then(response => {
      console.log('📥 Fetch 响应:', args[0], response.status);
      return response;
    });
};

// 拦截 XMLHttpRequest
const originalXHR = window.XMLHttpRequest;
window.XMLHttpRequest = function() {
  const xhr = new originalXHR();
  const originalOpen = xhr.open;
  xhr.open = function(method, url) {
    console.log(`🌐 XHR 请求: ${method} ${url}`);
    return originalOpen.apply(this, arguments);
  };
  return xhr;
};
```

### 内存泄漏检测

```javascript
// 使用 WeakRef 检测对象是否被回收
class LeakDetector {
  constructor() {
    this.refs = [];
  }

  track(obj) {
    this.refs.push(new WeakRef(obj));
  }

  check() {
    const alive = this.refs.filter(ref => ref.deref() !== undefined);
    console.log(`存活对象: ${alive.length}/${this.refs.length}`);
  }
}

const detector = new LeakDetector();
detector.track({ data: 'test' });
detector.check();
```

### 内存使用监控

```javascript
// performance.memory（仅 Chrome）
function logMemory() {
  if (performance.memory) {
    console.log({
      已用JS堆大小: (performance.memory.usedJSHeapSize / 1024 / 1024).toFixed(2) + 'MB',
      总JS堆大小: (performance.memory.totalJSHeapSize / 1024 / 1024).toFixed(2) + 'MB',
      堆大小限制: (performance.memory.jsHeapSizeLimit / 1024 / 1024).toFixed(2) + 'MB'
    });
  }
}

// 定期监控
setInterval(logMemory, 5000);
```

## 调试最佳实践

### 调试模式开关

```javascript
// 环境变量控制
const DEBUG = process.env.NODE_ENV === 'development';

const debug = {
  log: (...args) => DEBUG && console.log('[DEBUG]', ...args),
  warn: (...args) => DEBUG && console.warn('[WARN]', ...args),
  error: (...args) => DEBUG && console.error('[ERROR]', ...args),
  time: (label) => DEBUG && console.time(label),
  timeEnd: (label) => DEBUG && console.timeEnd(label)
};

// 使用
debug.log('用户数据:', userData);
debug.time('处理耗时');
// ... 代码
debug.timeEnd('处理耗时');
```

### 日志工具封装

```javascript
class Logger {
  constructor(prefix = '') {
    this.prefix = prefix;
    this.enabled = true;
  }

  format(level, ...args) {
    const timestamp = new Date().toISOString();
    return [`[${timestamp}] [${level}] ${this.prefix}`, ...args];
  }

  log(...args) {
    this.enabled && console.log(...this.format('LOG', ...args));
  }

  warn(...args) {
    this.enabled && console.warn(...this.format('WARN', ...args));
  }

  error(...args) {
    this.enabled && console.error(...this.format('ERROR', ...args));
  }

  enable() { this.enabled = true; }
  disable() { this.enabled = false; }
}

const logger = new Logger('App');
logger.log('应用启动');
logger.error('发生错误');
```
