## 错误处理概述

### 为什么需要错误处理？

- **程序健壮性**：防止程序因意外情况崩溃
- **用户体验**：提供友好的错误提示
- **调试效率**：快速定位问题
- **系统稳定性**：避免错误扩散

### JavaScript 错误类型

JavaScript 中的错误分为三类：

1. **语法错误（Syntax Error）**：代码解析阶段发现，无法执行
2. **运行时错误（Runtime Error）**：代码执行阶段发生
3. **逻辑错误（Logic Error）**：代码执行但结果不正确

## try/catch/finally

### 基本语法

```javascript
try {
  // 可能出错的代码
} catch (error) {
  // 错误处理
} finally {
  // 无论成功失败都执行（可选）
}
```

### 执行流程

```javascript
function demonstrateTryCatch() {
  try {
    console.log('1. Try block start');
    throw new Error('Something went wrong');
    console.log('2. This will not execute');
  } catch (error) {
    console.log('3. Catch block:', error.message);
  } finally {
    console.log('4. Finally block always executes');
  }
  console.log('5. After try-catch-finally');
}

demonstrateTryCatch();
// 1. Try block start
// 3. Catch block: Something went wrong
// 4. Finally block always executes
// 5. After try-catch-finally
```

### try 块

```javascript
// try 块中发生的错误会被捕获
try {
  const data = JSON.parse('invalid json');
} catch (error) {
  console.error('Parse error:', error.message);
}

// try 块中声明的变量是块级作用域
try {
  let blockScoped = 'only in try';
  var functionScoped = 'accessible outside';
} catch (e) {
  // ...
}
console.log(functionScoped);  // 'accessible outside'
// console.log(blockScoped);  // ReferenceError
```

### catch 块

```javascript
// catch 接收错误对象
try {
  undefinedVariable;
} catch (error) {
  console.log('Error name:', error.name);      // ReferenceError
  console.log('Error message:', error.message); // undefinedVariable is not defined
  console.log('Error stack:', error.stack);     // 堆栈信息
}

// ES2019: catch 可省略参数
try {
  throw new Error('Oops');
} catch {
  console.log('Error occurred, but we don\'t need the error object');
}

// 条件捕获
try {
  // 可能抛出多种错误的代码
} catch (error) {
  if (error instanceof TypeError) {
    console.log('Type error:', error.message);
  } else if (error instanceof RangeError) {
    console.log('Range error:', error.message);
  } else {
    throw error; // 重新抛出不处理的错误
  }
}
```

### finally 块

```javascript
// finally 总是执行
function withFinally() {
  try {
    console.log('Try');
    return 'from try';  // 即使有 return
  } finally {
    console.log('Finally');  // 仍然会执行
  }
}

console.log(withFinally());
// Try
// Finally
// from try

// finally 中的 return 会覆盖 try 中的 return
function overrideReturn() {
  try {
    return 'from try';
  } finally {
    return 'from finally';  // 覆盖上面的 return
  }
}

console.log(overrideReturn());  // 'from finally'

// finally 用于资源清理
function readFile(path) {
  const file = openFile(path);
  try {
    return processFile(file);
  } finally {
    closeFile(file);  // 确保资源被释放
  }
}
```

### 嵌套 try-catch

```javascript
try {
  try {
    throw new Error('Inner error');
  } catch (innerError) {
    console.log('Inner catch:', innerError.message);
    throw new Error('Outer error');  // 重新抛出
  }
} catch (outerError) {
  console.log('Outer catch:', outerError.message);
}
// Inner catch: Inner error
// Outer catch: Outer error
```

## Error 对象

### Error 构造函数

```javascript
// 创建错误对象
const error = new Error('Something went wrong');

// 不使用 new 也可以
const error2 = Error('Another error');

console.log(error instanceof Error);  // true
```

### Error 实例属性

```javascript
const error = new Error('Test error');

// name: 错误名称
console.log(error.name);     // 'Error'

// message: 错误消息
console.log(error.message);  // 'Test error'

// stack: 堆栈信息（非标准但广泛支持）
console.log(error.stack);
// Error: Test error
//     at <anonymous>:1:15

// cause: 错误原因（ES2022）
const originalError = new Error('Original');
const wrappedError = new Error('Wrapped', { cause: originalError });
console.log(wrappedError.cause);  // Error: Original
```

### 错误链（Error Cause）

```javascript
// ES2022 错误链
async function fetchUserData(userId) {
  try {
    const response = await fetch(`/api/users/${userId}`);
    return await response.json();
  } catch (error) {
    // 包装原始错误，保留原因链
    throw new Error(`Failed to fetch user ${userId}`, { cause: error });
  }
}

// 捕获并追溯错误链
try {
  await fetchUserData(123);
} catch (error) {
  let current = error;
  while (current) {
    console.log('Error:', current.message);
    current = current.cause;
  }
}
```

## 内置错误类型

### Error

所有错误的基类。

```javascript
const error = new Error('Generic error');
console.log(error.name);     // 'Error'
console.log(error.message);  // 'Generic error'
```

### TypeError

类型错误，当值的类型不是预期类型时抛出。

```javascript
// 调用非函数
const notAFunction = 'hello';
notAFunction();  // TypeError: notAFunction is not a function

// 访问 null/undefined 的属性
const obj = null;
obj.property;  // TypeError: Cannot read property 'property' of null

// 修改只读属性
const frozen = Object.freeze({ prop: 1 });
frozen.prop = 2;  // TypeError: Cannot assign to read only property

// 类型错误示例
function expectString(str) {
  if (typeof str !== 'string') {
    throw new TypeError(`Expected string, got ${typeof str}`);
  }
  return str.toUpperCase();
}

expectString(123);  // TypeError: Expected string, got number
```

### ReferenceError

引用错误，当访问不存在的变量时抛出。

```javascript
// 访问未声明的变量
console.log(undefinedVariable);  // ReferenceError: undefinedVariable is not defined

// 在暂时性死区访问 let/const 变量
{
  console.log(x);  // ReferenceError: Cannot access 'x' before initialization
  let x = 1;
}

// 严格模式下对未声明变量赋值
'use strict';
undeclared = 1;  // ReferenceError: undeclared is not defined
```

### RangeError

范围错误，当值超出有效范围时抛出。

```javascript
// 递归溢出
function infiniteRecursion() {
  return infiniteRecursion();
}
infiniteRecursion();  // RangeError: Maximum call stack size exceeded

// 无效的数组长度
new Array(-1);  // RangeError: Invalid array length

// 无效的精度
(1.23).toFixed(101);  // RangeError: toFixed() digits argument must be between 0 and 100

// 自定义范围错误
function createArray(size) {
  if (size < 0 || size > 1000000) {
    throw new RangeError('Array size must be between 0 and 1000000');
  }
  return new Array(size);
}
```

### SyntaxError

语法错误，当代码语法不正确时抛出。

```javascript
// 无效语法
eval('for (');  // SyntaxError: Unexpected end of input

// 无效 JSON
JSON.parse('invalid');  // SyntaxError: Unexpected token i in JSON

// 变量名包含非法字符
const 123var = 1;  // SyntaxError: Invalid or unexpected token
```

### URIError

URI 错误，当 URI 相关函数参数无效时抛出。

```javascript
// 无效的 URI 编码
decodeURIComponent('%');  // URIError: URI malformed

// 无效的 URI 组件
encodeURI('\uD800');  // URIError: URI malformed
```

### AggregateError（ES2021）

聚合错误，表示多个错误的集合。

```javascript
// Promise.any 失败时抛出
Promise.any([
  Promise.reject(new Error('Error 1')),
  Promise.reject(new Error('Error 2'))
]).catch(error => {
  console.log(error instanceof AggregateError);  // true
  console.log(error.errors);  // [Error: Error 1, Error: Error 2]
});

// 手动创建 AggregateError
const errors = [
  new Error('First error'),
  new Error('Second error'),
  new Error('Third error')
];
const aggregateError = new AggregateError(errors, 'Multiple errors occurred');

console.log(aggregateError.message);  // 'Multiple errors occurred'
console.log(aggregateError.errors);   // errors 数组
```

### 错误类型对照表

| 错误类型 | 触发场景 | 示例 |
|---------|---------|------|
| Error | 通用错误 | `throw new Error()` |
| TypeError | 类型错误 | 调用非函数、访问 null 属性 |
| ReferenceError | 引用错误 | 访问未声明变量 |
| RangeError | 范围错误 | 栈溢出、无效长度 |
| SyntaxError | 语法错误 | 解析失败 |
| URIError | URI 错误 | decodeURIComponent 参数无效 |
| AggregateError | 聚合错误 | Promise.any 全部失败 |

### 继承 Error

```javascript
    super(message);
    this.name = 'CustomError';
  }

try {
```

### 带额外属性的错误

```javascript
class ValidationError extends Error {
  constructor(message, field, value) {
    super(message);
    this.name = 'ValidationError';
    this.field = field;
    this.value = value;
    this.timestamp = new Date();
  }

  toJSON() {
    return {
      name: this.name,
      message: this.message,
      field: this.field,
      value: this.value,
      timestamp: this.timestamp
    };
  }
}

// 使用
function validateUser(user) {
  if (!user.email) {
    throw new ValidationError('Email is required', 'email', user.email);
  }
  if (!user.age || user.age < 0) {
    throw new ValidationError('Invalid age', 'age', user.age);
  }
}

try {
  validateUser({ name: 'John', age: -1 });
} catch (error) {
  if (error instanceof ValidationError) {
    console.log(`Validation failed: ${error.field} - ${error.message}`);
  }
}
```

### 错误上报

```javascript
class ErrorReporter {
  constructor(options = {}) {
    this.endpoint = options.endpoint || '/api/errors';
    this.environment = options.environment || 'production';
    this.enabled = options.enabled !== false;
  }

  report(error, context = {}) {
    if (!this.enabled) return;

    const payload = {
      message: error.message,
      name: error.name,
      stack: error.stack,
      timestamp: new Date().toISOString(),
      environment: this.environment,
      url: window.location.href,
      userAgent: navigator.userAgent,
      ...context
    };

    // 使用 sendBeacon 确保请求被发送
    if (navigator.sendBeacon) {
      navigator.sendBeacon(this.endpoint, JSON.stringify(payload));
    } else {
      fetch(this.endpoint, {
        method: 'POST',
        body: JSON.stringify(payload),
        keepalive: true
      }).catch(() => {
        // 忽略上报失败
      });
    }
  }
}

// 初始化全局错误处理
const reporter = new ErrorReporter({
  endpoint: '/api/errors',
  environment: 'production'
});

window.addEventListener('error', (event) => {
  reporter.report(event.error || new Error(event.message));
});

window.addEventListener('unhandledrejection', (event) => {
  const error = event.reason instanceof Error
    ? event.reason
    : new Error(String(event.reason));
  reporter.report(error);
});
```

```javascript
  console.log('Data:', data);
```

### 重试机制

```javascript
async function retry(fn, options = {}) {
  const {
    maxRetries = 3,
    delay = 1000,
    backoff = 2
  } = options;

  let lastError;

  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error;

      if (attempt < maxRetries) {
        const waitTime = delay * Math.pow(backoff, attempt);
        await new Promise(resolve => setTimeout(resolve, waitTime));
      }
    }
  }

  throw lastError;
}

// 使用
const data = await retry(
  () => fetch('/api/data').then(r => r.json()),
  { maxRetries: 3, delay: 1000, backoff: 2 }
);
```

```javascript
class CircuitBreaker {
  constructor(fn, options = {}) {
    this.fn = fn;
    this.failureThreshold = options.failureThreshold || 5;
    this.resetTimeout = options.resetTimeout || 30000;

    this.failures = 0;
    this.state = 'CLOSED';  // CLOSED, OPEN, HALF_OPEN
    this.nextAttempt = Date.now();
  }

  async execute(...args) {
    if (this.state === 'OPEN') {
      if (Date.now() < this.nextAttempt) {
        throw new Error('Circuit breaker is OPEN');
      }
      this.state = 'HALF_OPEN';
    }

    try {
      const result = await this.fn(...args);
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  onSuccess() {
    this.failures = 0;
    this.state = 'CLOSED';
  }

  onFailure() {
    this.failures++;
    if (this.failures >= this.failureThreshold) {
      this.state = 'OPEN';
      this.nextAttempt = Date.now() + this.resetTimeout;
    }
  }
}
```

## 最佳实践

### 1. 使用具体的错误类型

```javascript
// ❌ 不推荐
throw new Error('Invalid input');

// ✅ 推荐
throw new ValidationError('Invalid input', 'field', value);
```

### 2. 提供有意义的错误消息

```javascript
// ❌ 不推荐
throw new Error('Error');

// ✅ 推荐
throw new Error(
  `Failed to parse configuration file at ${path}: ` +
  `expected JSON format but got ${typeof content}`
);
```

### 3. 不要吞没错误

```javascript
// ❌ 不推荐
try {
  doSomething();
} catch (e) {
  // 什么都不做
}

// ✅ 推荐
try {
  doSomething();
} catch (error) {
  logger.error('Failed to do something', error);
  // 决定是否重新抛出
  throw error;
}
```

### 4. 使用错误链保留上下文

```javascript
// ❌ 不推荐
try {
  await fetchData();
} catch (error) {
  throw new Error('Failed to fetch');
}

// ✅ 推荐
try {
  await fetchData();
} catch (error) {
  throw new Error('Failed to fetch', { cause: error });
}
```

### 5. 区分可恢复和不可恢复错误

```javascript
async function processRequest(request) {
  // 可恢复错误：返回错误响应
  try {
    const user = await authenticate(request);
  } catch (error) {
    return { status: 401, error: 'Unauthorized' };
  }

  // 不可恢复错误：让进程崩溃（由全局处理器记录）
  const criticalConfig = loadCriticalConfig();
  if (!criticalConfig) {
    throw new Error('Critical configuration missing');
  }
}
```

### 6. 日志与错误处理分离

```javascript
// ❌ 混合
try {
  await processOrder(order);
} catch (error) {
  console.error('Order processing failed:', error);
  sendEmail(`Error: ${error.message}`);
  throw error;
}

// ✅ 分离
class OrderService {
  constructor(logger, emailService) {
    this.logger = logger;
    this.emailService = emailService;
  }

  async process(order) {
    try {
      await this.doProcess(order);
    } catch (error) {
      this.logger.error('Order processing failed', { order, error });
      await this.emailService.sendAlert(error);
      throw error;
    }
  }
}
```

### 7. 清理资源

```javascript
// 使用 try-finally
function processFile(path) {
  const file = openFile(path);
  try {
    return process(file);
  } finally {
    closeFile(file);
  }
}

// 使用 Symbol.dispose（ES2023）
function processFile(path) {
  using file = openFile(path);  // 自动清理
  return process(file);
}
```
