## 测试基础概念

### 什么是单元测试

```javascript
/**
 * 单元测试的定义：
 * - 测试最小可测试单元（通常是函数或方法）
 * - 隔离测试，不依赖外部系统
 * - 快速执行，自动化运行
 * - 提供即时反馈
 */

// 被测试的函数
function add(a, b) {
  return a + b;
}

// 简单的测试用例
function testAdd() {
  const result = add(1, 2);
  const expected = 3;

  if (result === expected) {
    console.log('✓ 测试通过');
  } else {
    console.error(`✗ 测试失败: 期望 ${expected}, 实际 ${result}`);
  }
}

testAdd();
```

### 测试金字塔

```javascript
/**
 * 测试金字塔（从下到上）：
 *
 *        /\
 *       /  \  E2E 测试（端到端）
 *      /----\  - 最慢，最少数量
 *     /      \
 *    /--------\ 集成测试
 *   /          \ - 中等速度，中等数量
 *  /____________\
 *   单元测试
 *   - 最快，最多数量
 */

// 单元测试示例：测试单个函数
test('formatDate returns formatted string', () => {
  expect(formatDate(new Date('2024-01-01'))).toBe('2024-01-01');
});

// 集成测试示例：测试多个模块协作
test('UserService creates user with valid data', async () => {
  const service = new UserService(new Database());
  const user = await service.create({ name: 'John' });
  expect(user.id).toBeDefined();
});

// E2E 测试示例：测试完整流程
test('user can complete checkout flow', async () => {
  await page.goto('/cart');
  await page.click('[data-testid="checkout"]');
  await expect(page).toHaveURL('/order-success');
});
```

### AAA 模式

```javascript
/**
 * AAA 模式：Arrange-Act-Assert
 * - Arrange（准备）：设置测试数据和条件
 * - Act（执行）：执行被测试的代码
 * - Assert（断言）：验证结果
 */

test('calculateTotal applies discount correctly', () => {
  // Arrange
  const items = [
    { price: 100, quantity: 2 },
    { price: 50, quantity: 1 }
  ];
  const discount = 0.1; // 10% 折扣

  // Act
  const total = calculateTotal(items, discount);

  // Assert
  expect(total).toBe(225); // (200 + 50) * 0.9
});

// 带注释的完整示例
test('user can withdraw money if balance is sufficient', () => {
  // Arrange - 准备
  const account = new BankAccount(1000);
  const withdrawAmount = 500;

  // Act - 执行
  const result = account.withdraw(withdrawAmount);

  // Assert - 断言
  expect(result.success).toBe(true);
  expect(account.balance).toBe(500);
});
```

### 基本语法

```javascript
// 基本测试
test('基本测试', () => {
  expect(1 + 1).toBe(2);
});

// 使用 it（test 的别名）
it('也可以使用 it', () => {
  expect(true).toBeTruthy();
});

// 跳过测试
test.skip('跳过此测试', () => {
  // 不会执行
});

// 只运行特定测试
test.only('只运行此测试', () => {
  expect(1).toBe(1);
});

// 待定测试
test.todo('待实现的测试');

// 并发测试
test.concurrent('并发测试', async () => {
  const result = await fetchData();
  expect(result).toBeDefined();
});

// 每个测试的超时时间
test('长时间测试', async () => {
  // ...
}, 10000); // 10秒超时
```

## 匹配器 Matchers

### 相等性匹配器

```javascript
describe('相等性匹配器', () => {
  test('toBe - 严格相等（引用相等）', () => {
    expect(1).toBe(1);
    expect('hello').toBe('hello');
    expect(true).toBe(true);

    // 对象比较引用
    const obj = { a: 1 };
    expect(obj).toBe(obj);
    expect({ a: 1 }).not.toBe({ a: 1 }); // 不同引用
  });

  test('toEqual - 深度相等', () => {
    expect({ a: 1 }).toEqual({ a: 1 });
    expect([1, 2, 3]).toEqual([1, 2, 3]);
    expect({
      name: 'John',
      address: { city: 'NYC' }
    }).toEqual({
      name: 'John',
      address: { city: 'NYC' }
    });
  });

  test('toStrictEqual - 严格深度相等', () => {
    // toStrictEqual 比 toEqual 更严格
    // 会检查 undefined 属性和数组稀疏性

    expect({ a: 1, b: undefined }).not.toStrictEqual({ a: 1 });
    expect({ a: 1 }).toStrictEqual({ a: 1 });
  });
});
```

### 真值匹配器

```javascript
describe('真值匹配器', () => {
  test('toBeTruthy / toBeFalsy', () => {
    expect(true).toBeTruthy();
    expect(1).toBeTruthy();
    expect('hello').toBeTruthy();
    expect({}).toBeTruthy();
    expect([1, 2]).toBeTruthy();

    expect(false).toBeFalsy();
    expect(0).toBeFalsy();
    expect('').toBeFalsy();
    expect(null).toBeFalsy();
    expect(undefined).toBeFalsy();
  });

  test('toBeNull / toBeUndefined / toBeDefined', () => {
    expect(null).toBeNull();
    expect(undefined).toBeUndefined();
    expect(1).toBeDefined();
    expect(null).toBeDefined(); // null 是已定义的
  });

  test('toBeNaN', () => {
    expect(NaN).toBeNaN();
    expect(1).not.toBeNaN();
  });
});
```

### 数字匹配器

```javascript
describe('数字匹配器', () => {
  test('大于小于', () => {
    expect(10).toBeGreaterThan(5);
    expect(10).toBeGreaterThanOrEqual(10);
    expect(5).toBeLessThan(10);
    expect(5).toBeLessThanOrEqual(5);
  });

  test('toBeCloseTo - 浮点数比较', () => {
    // 避免浮点数精度问题
    expect(0.1 + 0.2).toBeCloseTo(0.3);
    expect(0.1 + 0.2).not.toBe(0.3); // 直接比较会失败

    // 指定精度位数
    expect(0.1 + 0.2).toBeCloseTo(0.3, 5); // 5位小数
  });
});
```

```javascript
describe('对象匹配器', () => {
```

### 异常匹配器

```javascript
describe('异常匹配器', () => {
  test('toThrow', () => {
    const throwError = () => {
      throw new Error('Something went wrong');
    };

    expect(throwError).toThrow();
    expect(throwError).toThrow(Error);
    expect(throwError).toThrow('Something went wrong');
    expect(throwError).toThrow(/wrong/);
  });

  test('toThrow 匹配特定错误', () => {
    class CustomError extends Error {
      constructor(message, code) {
        super(message);
        this.code = code;
      }
    }

    const throwCustomError = () => {
      throw new CustomError('Custom error', 500);
    };

    expect(throwCustomError).toThrow(CustomError);
    expect(throwCustomError).toThrow(expect.objectContaining({
      message: 'Custom error',
      code: 500
    }));
  });

  test('异步异常', async () => {
    const asyncError = async () => {
      throw new Error('Async error');
    };

    await expect(asyncError()).rejects.toThrow('Async error');
  });
});
```

## Mock 和 Spy

### Jest Mock 函数

```javascript
describe('Mock 函数', () => {
  test('基本 Mock', () => {
    const mockFn = jest.fn();

    mockFn('hello');

    expect(mockFn).toHaveBeenCalled();
    expect(mockFn).toHaveBeenCalledWith('hello');
    expect(mockFn).toHaveBeenCalledTimes(1);
  });

  test('Mock 返回值', () => {
    const mockFn = jest.fn();

    // 固定返回值
    mockFn.mockReturnValue(42);
    expect(mockFn()).toBe(42);

    // 连续返回值
    mockFn.mockReturnValueOnce(1);
    mockFn.mockReturnValueOnce(2);
    mockFn.mockReturnValue(3);

    expect(mockFn()).toBe(1);
    expect(mockFn()).toBe(2);
    expect(mockFn()).toBe(3);
    expect(mockFn()).toBe(3);
  });

  test('Mock 实现', () => {
    const mockFn = jest.fn(x => x * 2);

    expect(mockFn(5)).toBe(10);

    // 一次性实现
    mockFn.mockImplementationOnce(x => x * 3);
    expect(mockFn(5)).toBe(15);
    expect(mockFn(5)).toBe(10);
  });

  test('Mock this', () => {
    const mockFn = jest.fn(function() {
      return this.name;
    });

    const context = { name: 'John', fn: mockFn };
    context.fn();

    expect(mockFn.mock.contexts[0]).toBe(context);
  });

  test('Mock 返回值链', () => {
    const mockFn = jest.fn();

    mockFn
      .mockReturnValueOnce(1)
      .mockReturnValueOnce(2)
      .mockReturnValue(3);

    expect(mockFn()).toBe(1);
    expect(mockFn()).toBe(2);
    expect(mockFn()).toBe(3);
    expect(mockFn()).toBe(3);
  });

  test('Mock 属性', () => {
    const mockFn = jest.fn();
    mockFn.mockName('myMockFn');

    expect(mockFn.getMockName()).toBe('myMockFn');
  });
});
```

### Mock Timer

```javascript
describe('Mock Timer', () => {
  beforeEach(() => {
    jest.useFakeTimers();
  });

  afterEach(() => {
    jest.useRealTimers();
  });

  test('setTimeout', () => {
    const callback = jest.fn();

    setTimeout(callback, 1000);

    expect(callback).not.toHaveBeenCalled();

    // 快进时间
    jest.advanceTimersByTime(1000);

    expect(callback).toHaveBeenCalled();
  });

  test('setInterval', () => {
    const callback = jest.fn();

    setInterval(callback, 100);

    jest.advanceTimersByTime(350);

    expect(callback).toHaveBeenCalledTimes(3);
  });

  test('runAllTimers', () => {
    const callback = jest.fn();

    setTimeout(callback, 1000);
    setTimeout(callback, 2000);

    jest.runAllTimers();

    expect(callback).toHaveBeenCalledTimes(2);
  });

  test('runOnlyPendingTimers', () => {
    const callback = jest.fn();

    setTimeout(callback, 1000);
    setTimeout(() => {
      setTimeout(callback, 1000);
    }, 500);

    jest.runOnlyPendingTimers();

    expect(callback).toHaveBeenCalledTimes(1);
  });

  test('清除定时器', () => {
    const callback = jest.fn();

    const timerId = setTimeout(callback, 1000);
```

### 覆盖率指标

```javascript
/**
 * 覆盖率指标：
 *
 * 1. Line Coverage（行覆盖率）
 *    - 已执行的代码行占总代码行的比例
 *
 * 2. Branch Coverage（分支覆盖率）
 *    - 已执行的条件分支占总分支的比例
 *    - if/else, switch, 三元运算符等
 *
 * 3. Function Coverage（函数覆盖率）
 *    - 已调用的函数占总函数的比例
 *
 * 4. Statement Coverage（语句覆盖率）
 *    - 已执行的语句占总语句的比例
 */

// 示例：分支覆盖率
function getGrade(score) {
  if (score >= 90) {      // 分支 1
    return 'A';
  } else if (score >= 80) { // 分支 2
    return 'B';
  } else {                  // 分支 3
    return 'C';
  }
}

// 测试用例
test('getGrade returns correct grades', () => {
  expect(getGrade(95)).toBe('A');  // 覆盖分支 1
  expect(getGrade(85)).toBe('B');  // 覆盖分支 2
  expect(getGrade(70)).toBe('C');  // 覆盖分支 3
});
```

## 测试最佳实践

### 测试原则

```javascript
/**
 * FIRST 原则：
 * - Fast（快速）：测试应该快速执行
 * - Independent（独立）：测试之间不应有依赖
 * - Repeatable（可重复）：结果应该一致
 * - Self-validating（自我验证）：自动判断通过/失败
 * - Timely（及时）：与代码同步编写
 */

// 好的测试示例
describe('isValidEmail', () => {
  test('returns true for valid email', () => {
    expect(isValidEmail('test@example.com')).toBe(true);
  });

  test('returns false for invalid email', () => {
    expect(isValidEmail('invalid')).toBe(false);
    expect(isValidEmail('test@')).toBe(false);
    expect(isValidEmail('@example.com')).toBe(false);
  });

  test('handles edge cases', () => {
    expect(isValidEmail('')).toBe(false);
    expect(isValidEmail(null)).toBe(false);
    expect(isValidEmail(undefined)).toBe(false);
  });
});

// 避免的测试模式
describe('Bad examples', () => {
  // 错误：测试之间有依赖
  let sharedState = 0;
  test('test 1', () => {
    sharedState = 10;
  });
  test('test 2 depends on test 1', () => {
    expect(sharedState).toBe(10); // 不可靠
  });

  // 错误：测试了实现细节
  test('internal variable is set correctly', () => {
    const obj = new MyClass();
    expect(obj._internalVar).toBe(1); // 不应测试私有变量
  });

  // 错误：测试包含逻辑
  test('complex test', () => {
    const data = [1, 2, 3];
    for (let i = 0; i < data.length; i++) {
      if (data[i] > 0) {
        expect(process(data[i])).toBeTruthy();
      }
    }
  });
});
```
