# 浏览器存储

> 浏览器提供了多种客户端存储机制，每种都有其特定的使用场景、容量限制和特性。选择合适的存储方案对于 Web 应用的性能和用户体验至关重要。

## 学习要点

- 📦 掌握各种浏览器存储机制的特性和差异
- 🔧 理解不同存储的使用场景和限制
- 💾 熟练使用 IndexedDB 进行大量数据存储
- 🔐 了解存储安全性和隐私保护

## 2. Cookie 详解

### 2.1 Cookie 基础

```javascript
// Cookie 的特点：
// 1. 自动随 HTTP 请求发送到服务器
// 2. 大小限制约 4KB
// 3. 可设置过期时间
// 4. 支持域名、路径、安全属性

// ────────────────────────────────────────────────────────────────────────────────────
// 基本操作
// ────────────────────────────────────────────────────────────────────────────────────

// 设置 Cookie
document.cookie = 'name=John';

// 设置带属性的 Cookie
document.cookie = 'token=abc123; max-age=3600; path=/; domain=example.com';

// 读取 Cookie
console.log(document.cookie); // 'name=John; token=abc123'

// 解析 Cookie
function getCookie(name) {
    const cookies = document.cookie.split('; ');
    for (const cookie of cookies) {
        const [key, value] = cookie.split('=');
        if (key === name) return decodeURIComponent(value);
    }
    return null;
}

// 删除 Cookie（设置过期时间为过去）
document.cookie = 'name=; max-age=0; path=/';
```

### 2.2 Cookie 属性详解

```javascript
// ────────────────────────────────────────────────────────────────────────────────────
// Cookie 属性详解
// ────────────────────────────────────────────────────────────────────────────────────

/*
┌─────────────────┬─────────────────────────────────────────────────────────────────────┐
│  属性            │  说明                                                                 │
├─────────────────┼─────────────────────────────────────────────────────────────────────┤
│  Name=Value     │  Cookie 名称和值（必需）                                             │
│                 │  • 名称区分大小写                                                    │
│                 │  • 值需要 URL 编码                                                   │
├─────────────────┼─────────────────────────────────────────────────────────────────────┤
│  Expires        │  过期时间（绝对时间）                                                 │
│                 │  • 格式：Wdy, DD Mon YYYY HH:MM:SS GMT                               │
│                 │  • 不设置则为 Session Cookie（关闭浏览器后删除）                       │
├─────────────────┼─────────────────────────────────────────────────────────────────────┤
│  Max-Age        │  有效期（秒）                                                         │
│                 │  • 优先于 Expires                                                    │
│                 │  • 正数：指定秒数后过期 / 0 或负数：立即删除                          │
├─────────────────┼─────────────────────────────────────────────────────────────────────┤
│  Domain         │  生效域名                                                             │
│                 │  • 默认为当前主机名（不含子域名）                                     │
│                 │  • Domain=.example.com 包含所有子域名                                │
├─────────────────┼─────────────────────────────────────────────────────────────────────┤
│  Path           │  生效路径（默认为当前路径，Path=/ 对全站生效）                        │
├─────────────────┼─────────────────────────────────────────────────────────────────────┤
│  Secure         │  仅 HTTPS 传输                                                        │
├─────────────────┼─────────────────────────────────────────────────────────────────────┤
│  HttpOnly       │  禁止 JavaScript 访问（防止 XSS 攻击窃取 Cookie）                     │
├─────────────────┼─────────────────────────────────────────────────────────────────────┤
│  SameSite       │  跨站请求策略                                                         │
│                 │  • Strict：完全禁止跨站发送                                          │
│                 │  • Lax：允许安全的跨站请求（默认）                                    │
│                 │  • None：允许所有跨站请求（需配合 Secure）                            │
└─────────────────┴─────────────────────────────────────────────────────────────────────┘
*/

// Cookie 工具类
class CookieUtil {
    static set(name, value, options = {}) {
        let cookie = `${encodeURIComponent(name)}=${encodeURIComponent(value)}`;
        if (options.maxAge !== undefined) cookie += `; max-age=${options.maxAge}`;
        if (options.expires) cookie += `; expires=${options.expires.toUTCString()}`;
        if (options.path) cookie += `; path=${options.path}`;
        if (options.domain) cookie += `; domain=${options.domain}`;
        if (options.secure) cookie += '; secure';
        if (options.httpOnly) cookie += '; HttpOnly';
        if (options.sameSite) cookie += `; SameSite=${options.sameSite}`;
        document.cookie = cookie;
    }
    
    static get(name) {
        const cookies = document.cookie.split('; ');
        for (const cookie of cookies) {
            const [key, value] = cookie.split('=');
            if (decodeURIComponent(key) === name) return decodeURIComponent(value);
        }
        return null;
    }
    
    static remove(name, options = {}) {
        this.set(name, '', { ...options, maxAge: 0 });
    }
}

// 使用示例
CookieUtil.set('user', 'John', { maxAge: 86400, path: '/', secure: true, sameSite: 'Lax' });
```

## 3. Web Storage（localStorage / sessionStorage）

### 3.1 基础用法

```javascript
// ────────────────────────────────────────────────────────────────────────────────────
// localStorage - 持久化存储
// ────────────────────────────────────────────────────────────────────────────────────

localStorage.setItem('key', 'value');   // 存储
localStorage.getItem('key');            // 获取
localStorage.removeItem('key');         // 删除
localStorage.clear();                   // 清空所有
localStorage.length;                    // 存储数量
localStorage.key(0);                    // 获取第 n 个键名

// 存储对象需要序列化
localStorage.setItem('user', JSON.stringify({ name: 'John', age: 30 }));
const user = JSON.parse(localStorage.getItem('user'));

// ────────────────────────────────────────────────────────────────────────────────────
// sessionStorage - 会话存储（关闭标签页清除）
// ────────────────────────────────────────────────────────────────────────────────────

sessionStorage.setItem('tempData', 'value');
sessionStorage.getItem('tempData');
sessionStorage.removeItem('tempData');
sessionStorage.clear();

// 特点：
// 1. 数据仅在当前标签页有效
// 2. 关闭标签页后数据清除
// 3. 不同标签页数据隔离
// 4. 页面刷新后数据保留
```

### 3.2 Storage 事件与跨标签页通信

```javascript
// ────────────────────────────────────────────────────────────────────────────────────
// storage 事件 - 跨标签页通信
// ────────────────────────────────────────────────────────────────────────────────────

window.addEventListener('storage', (event) => {
    console.log('Key:', event.key);           // 变化的键
    console.log('Old Value:', event.oldValue); // 旧值
    console.log('New Value:', event.newValue); // 新值
    console.log('URL:', event.url);           // 触发变化的页面 URL
});

// 实现跨标签页通信
class TabMessenger {
    constructor(channel = 'default') {
        this.channel = channel;
        this.listeners = new Map();
        this._id = `tab_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
        
        window.addEventListener('storage', (event) => {
            if (event.key === `channel_${channel}` && event.newValue) {
                try {
                    const data = JSON.parse(event.newValue);
                    if (data.sender !== this._id) {
                        this.listeners.forEach(callback => callback(data.message, data));
                    }
                } catch (e) { console.error('Parse error:', e); }
            }
        });
    }
    
    send(message) {
        const data = { sender: this._id, message, timestamp: Date.now() };
        localStorage.setItem(`channel_${this.channel}`, JSON.stringify(data));
        setTimeout(() => localStorage.removeItem(`channel_${this.channel}`), 100);
    }
    
    on(callback) {
        const id = Symbol();
        this.listeners.set(id, callback);
        return () => this.listeners.delete(id);
    }
}

// 使用示例
const messenger = new TabMessenger('app_channel');
messenger.send({ type: 'USER_LOGIN', userId: 123 });
messenger.on((message) => console.log('Received:', message));
```

### 3.3 封装工具类（支持过期时间）

```javascript
class StorageUtil {
    constructor(storage = localStorage, prefix = '') {
        this.storage = storage;
        this.prefix = prefix;
    }
    
    _getKey(key) {
        return this.prefix ? `${this.prefix}_${key}` : key;
    }
    
    // 设置值（支持过期时间，单位：秒）
    set(key, value, maxAge = null) {
        const data = {
            value,
            createdAt: Date.now(),
            expiresAt: maxAge ? Date.now() + maxAge * 1000 : null
        };
        this.storage.setItem(this._getKey(key), JSON.stringify(data));
    }
    
    // 获取值（自动检查过期）
    get(key, defaultValue = null) {
        const raw = this.storage.getItem(this._getKey(key));
        if (!raw) return defaultValue;
        
        try {
            const data = JSON.parse(raw);
            if (data.expiresAt && Date.now() > data.expiresAt) {
                this.storage.removeItem(this._getKey(key));
                return defaultValue;
            }
            return data.value;
        } catch (e) {
            return raw;
        }
    }
    
    remove(key) { this.storage.removeItem(this._getKey(key)); }
    has(key) { return this.get(key) !== null; }
    
    // 获取存储大小（字节）
    getSize() {
        let size = 0;
        for (let i = 0; i < this.storage.length; i++) {
            const key = this.storage.key(i);
            size += key.length + (this.storage.getItem(key)?.length || 0);
        }
        return size * 2; // UTF-16 编码
    }
}

// 使用示例
const myStorage = new StorageUtil(localStorage, 'myApp');
myStorage.set('user', { id: 1, name: 'John' }, 7 * 24 * 3600); // 7天过期
console.log(myStorage.get('user'));
```

```javascript
// 检测存储可用性
function isStorageAvailable(storage) {
    try {
        const testKey = '__test__';
        storage.setItem(testKey, testKey);
        storage.removeItem(testKey);
        return true;
    } catch (e) {
        return false;
    }
}

// 安全存储（处理溢出）
function safeSetItem(key, value, storage = localStorage) {
    try {
        storage.setItem(key, value);
        return true;
    } catch (e) {
        if (e.name === 'QuotaExceededError') {
            console.error('Storage quota exceeded');
            return false;
        }
        throw e;
    }
}
```

## 4. IndexedDB 详解

### 4.1 IndexedDB 概述

```
┌───────────────────────────────────────────────────────────────────────────────────────┐
│  IndexedDB 特性                                                                        │
├───────────────────────────────────────────────────────────────────────────────────────┤
│  优势                                                                                  │
│  • 大容量存储：可存储大量数据（50MB-无上限）                                            │
│  • 结构化数据：支持对象、数组、Date、Blob 等                                            │
│  • 异步 API：不阻塞主线程                                                              │
│  • 事务支持：ACID 事务保证数据一致性                                                    │
│  • 索引查询：支持高效的范围查询和索引                                                   │
│  • 持久化：数据永久保存，除非手动删除                                                   │
│  • 跨标签页：同源页面共享数据                                                          │
│  • Service Worker：可在 Worker 中访问                                                  │
├───────────────────────────────────────────────────────────────────────────────────────┤
│  核心概念                                                                              │
│  ─────────────────────────────────────────────────────────────────────────────────────│
│  1. 数据库（Database）- 每个域名可创建多个数据库，有名称和版本号                        │
│  2. 对象存储（Object Store）- 类似关系数据库的表，存储键值对形式的记录                  │
│  3. 索引（Index）- 用于快速查找记录，支持唯一索引和非唯一索引                           │
│  4. 事务（Transaction）- 所有操作都在事务中执行，保证原子性和一致性                     │
│  5. 游标（Cursor）- 用于遍历对象存储或索引中的记录                                      │
└───────────────────────────────────────────────────────────────────────────────────────┘
```

### 4.2 IndexedDB 完整封装类

```javascript
// ────────────────────────────────────────────────────────────────────────────────────
// IndexedDB 工具类
// ────────────────────────────────────────────────────────────────────────────────────

class IndexedDBHelper {
    constructor(dbName, version, upgradeCallback) {
        this.dbName = dbName;
        this.version = version;
        this.upgradeCallback = upgradeCallback;
        this.db = null;
    }
    
    // 打开数据库
    async open() {
        return new Promise((resolve, reject) => {
            const request = indexedDB.open(this.dbName, this.version);
            
            request.onupgradeneeded = (event) => {
                const db = event.target.result;
                if (this.upgradeCallback) {
                    this.upgradeCallback(db, event.oldVersion, event.newVersion);
                }
            };
            
            request.onsuccess = (event) => {
                this.db = event.target.result;
                resolve(this.db);
            };
            
            request.onerror = (event) => reject(event.target.error);
            request.onblocked = () => console.warn('Database blocked, close other tabs');
        });
    }
    
    close() {
        if (this.db) { this.db.close(); this.db = null; }
    }
    
    // 添加数据
    async add(storeName, data) {
        return new Promise((resolve, reject) => {
            const transaction = this.db.transaction(storeName, 'readwrite');
            const store = transaction.objectStore(storeName);
            const request = store.add(data);
            request.onsuccess = () => resolve(request.result);
            request.onerror = () => reject(request.error);
        });
    }
    
    // 更新数据（不存在则添加）
    async put(storeName, data) {
        return new Promise((resolve, reject) => {
            const transaction = this.db.transaction(storeName, 'readwrite');
            const store = transaction.objectStore(storeName);
            const request = store.put(data);
            request.onsuccess = () => resolve(request.result);
            request.onerror = () => reject(request.error);
        });
    }
    
    // 获取数据
    async get(storeName, key) {
        return new Promise((resolve, reject) => {
            const transaction = this.db.transaction(storeName, 'readonly');
            const store = transaction.objectStore(storeName);

    async queryRange(storeName, options = {}) {
        const { index, lower, upper, lowerOpen = false, upperOpen = false, direction = 'next', limit } = options;
        
        let range;
        if (lower !== undefined && upper !== undefined) {
            range = IDBKeyRange.bound(lower, upper, lowerOpen, upperOpen);
        } else if (lower !== undefined) {
            range = IDBKeyRange.lowerBound(lower, lowerOpen);
        } else if (upper !== undefined) {
            range = IDBKeyRange.upperBound(upper, upperOpen);
        }
        
        return new Promise((resolve, reject) => {
            const transaction = this.db.transaction(storeName, 'readonly');
            const store = transaction.objectStore(storeName);
            const source = index ? store.index(index) : store;
            const request = source.openCursor(range, direction);
            
            const results = [];
            request.onsuccess = (event) => {
                const cursor = event.target.result;
                if (cursor) {
                    results.push(cursor.value);
                    if (limit && results.length >= limit) resolve(results);
                    else cursor.continue();
                } else resolve(results);
            };
            request.onerror = () => reject(request.error);
        });
    }
    
    // 游标遍历
    async cursor(storeName, callback, options = {}) {
        return new Promise((resolve, reject) => {
            const transaction = this.db.transaction(storeName, options.mode || 'readonly');
            const store = transaction.objectStore(storeName);
            const source = options.index ? store.index(options.index) : store;
            const request = source.openCursor(options.range, options.direction || 'next');
            
            request.onsuccess = (event) => {
                const cursor = event.target.result;
                if (cursor) {
                    const shouldContinue = callback(cursor.value, cursor);
                    if (shouldContinue !== false) cursor.continue();
                    else resolve();
                } else resolve();
            };
            request.onerror = () => reject(request.error);
        });
    }
```

```javascript
// 批量更新
async function bulkUpdate(db, storeName, updateFn) {
    return new Promise((resolve, reject) => {
        const transaction = db.transaction(storeName, 'readwrite');
        const store = transaction.objectStore(storeName);
        
        let updated = 0;
        const request = store.openCursor();
        
        request.onsuccess = (event) => {
            const cursor = event.target.result;
            if (cursor) {
                const updatedData = updateFn(cursor.value);
                if (updatedData) {
                    cursor.update(updatedData);
                    updated++;
                }
                cursor.continue();
            }
        };
        
        transaction.oncomplete = () => resolve({ updated });
        transaction.onerror = () => reject(transaction.error);
    });
}

// 跨表事务
async function transferFunds(db, fromId, toId, amount) {
    return new Promise((resolve, reject) => {
        const transaction = db.transaction(['accounts', 'transactions'], 'readwrite');
        const accountStore = transaction.objectStore('accounts');
        const txStore = transaction.objectStore('transactions');
        
        // 获取账户
        const fromReq = accountStore.get(fromId);
        fromReq.onsuccess = () => {
            const from = fromReq.result;
            if (!from || from.balance < amount) {
                transaction.abort();
                reject(new Error('Insufficient funds'));
                return;
            }
            
            const toReq = accountStore.get(toId);
            toReq.onsuccess = () => {
                const to = toReq.result;
                if (!to) {
                    transaction.abort();
                    reject(new Error('Account not found'));
                    return;
                }
                
                from.balance -= amount;
                to.balance += amount;
                accountStore.put(from);
                accountStore.put(to);
                txStore.add({ from: fromId, to: toId, amount, timestamp: new Date() });
            };
        };
        
        transaction.oncomplete = () => resolve();
        transaction.onerror = () => reject(transaction.error);
    });
}
```

### 5.2 缓存策略

```javascript
// ────────────────────────────────────────────────────────────────────────────────────
// 常用缓存策略
// ────────────────────────────────────────────────────────────────────────────────────

/*
┌────────────────────────────┬─────────────────────────┬──────────────────────────────┐
│  策略                       │  说明                    │  适用场景                    │
├────────────────────────────┼─────────────────────────┼──────────────────────────────┤
│  Cache First               │  优先使用缓存            │  静态资源、不常更新的内容    │
│  Network First             │  优先使用网络            │  频繁更新的内容、API 请求    │
│  Stale While Revalidate    │  使用缓存同时更新缓存    │  允许短暂过期的内容          │
│  Network Only              │  只使用网络              │  实时数据、敏感操作          │
│  Cache Only                │  只使用缓存              │  离线优先的静态内容          │
└────────────────────────────┴─────────────────────────┴──────────────────────────────┘
*/

// 1. Cache First - 适合静态资源
async function cacheFirst(request) {
    const cache = await caches.open('static-cache');
    const cached = await cache.match(request);
    if (cached) return cached;
    
    const response = await fetch(request);
    if (response.ok) cache.put(request, response.clone());
    return response;
}

// 2. Network First - 适合频繁更新的内容
async function networkFirst(request) {
    const cache = await caches.open('dynamic-cache');
    
    try {
        const response = await fetch(request);
        if (response.ok) cache.put(request, response.clone());
        return response;
    } catch (error) {
        const cached = await cache.match(request);
        if (cached) return cached;
        throw error;
    }
}

// 3. Stale While Revalidate - 使用缓存同时后台更新
async function staleWhileRevalidate(request) {
    const cache = await caches.open('swr-cache');
    const cached = await cache.match(request);
    
    // 后台更新缓存
    const networkPromise = fetch(request).then(response => {
        if (response.ok) cache.put(request, response.clone());
        return response;
    }).catch(() => null);
    
    // 立即返回缓存，否则等待网络
    return cached || networkPromise;
}

// 4. Service Worker 中的完整示例
self.addEventListener('fetch', (event) => {
    const { request } = event;
    const url = new URL(request.url);
    
    // 静态资源 - Cache First
    if (url.pathname.match(/\.(css|js|png|jpg|jpeg|gif|svg|woff2?)$/)) {
        event.respondWith(cacheFirst(request));
        return;
    }
    
    // API 请求 - Network First
    if (url.pathname.startsWith('/api/')) {
        event.respondWith(networkFirst(request));
        return;
    }
    
    // HTML 页面 - Network First
    if (request.headers.get('accept')?.includes('text/html')) {
        event.respondWith(networkFirst(request));
        return;
    }
    
    // 其他 - Stale While Revalidate
    event.respondWith(staleWhileRevalidate(request));
});
```

```javascript
// 简单的数据加密（前端加密仅增加破解难度，不能替代服务端安全）
class SecureStorage {
    constructor(storage, secretKey) {
        this.storage = storage;
        this.secretKey = secretKey;
    }
    
    // 简单加密（仅作示例，生产环境应使用 Web Crypto API）
    encrypt(text) {
        let result = '';
        for (let i = 0; i < text.length; i++) {
            result += String.fromCharCode(
                text.charCodeAt(i) ^ this.secretKey.charCodeAt(i % this.secretKey.length)
            );
        }
        return btoa(result);
    }
    
    decrypt(encoded) {
        try {
            const text = atob(encoded);
            let result = '';
            for (let i = 0; i < text.length; i++) {
                result += String.fromCharCode(
                    text.charCodeAt(i) ^ this.secretKey.charCodeAt(i % this.secretKey.length)
                );
            }
            return result;
        } catch (e) {
            return null;
        }
    }
    
    set(key, value) {
        const encrypted = this.encrypt(JSON.stringify(value));
        this.storage.setItem(key, encrypted);
    }
    
    get(key, defaultValue = null) {
        const encrypted = this.storage.getItem(key);
        if (!encrypted) return defaultValue;
        
        const decrypted = this.decrypt(encrypted);
        if (!decrypted) return defaultValue;
        
        try {
            return JSON.parse(decrypted);
        } catch (e) {
            return defaultValue;
        }
    }
    
    remove(key) {
        this.storage.removeItem(key);
    }
}

// 使用 Web Crypto API 的安全存储
class CryptoStorage {
    constructor(storage) {
        this.storage = storage;
        this.key = null;
    }
    
    async init(password) {
        // 从密码派生密钥
        const encoder = new TextEncoder();
        const keyMaterial = await crypto.subtle.importKey(
            'raw',
            encoder.encode(password),
            'PBKDF2',
            false,
            ['deriveKey']
        );
        
        this.key = await crypto.subtle.deriveKey(
            {
                name: 'PBKDF2',
                salt: encoder.encode('salt'),
                iterations: 100000,
                hash: 'SHA-256'
            },
            keyMaterial,
            { name: 'AES-GCM', length: 256 },
            false,
            ['encrypt', 'decrypt']
        );
    }
    
    async encrypt(data) {
        const encoder = new TextEncoder();
        const iv = crypto.getRandomValues(new Uint8Array(12));
        const encrypted = await crypto.subtle.encrypt(
            { name: 'AES-GCM', iv },
            this.key,
            encoder.encode(JSON.stringify(data))
        );
        return { iv: Array.from(iv), data: Array.from(new Uint8Array(encrypted)) };
    }
    
    async decrypt(encrypted) {
        const decoder = new TextDecoder();
        const decrypted = await crypto.subtle.decrypt(
            { name: 'AES-GCM', iv: new Uint8Array(encrypted.iv) },
            this.key,
            new Uint8Array(encrypted.data)
        );
        return JSON.parse(decoder.decode(decrypted));
    }
    
    async set(key, value) {
        const encrypted = await this.encrypt(value);
        this.storage.setItem(key, JSON.stringify(encrypted));
    }
    
    async get(key, defaultValue = null) {
        const raw = this.storage.getItem(key);
        if (!raw) return defaultValue;
        
        try {
            const encrypted = JSON.parse(raw);
            return await this.decrypt(encrypted);
        } catch (e) {
            return defaultValue;
        }
    }
}
```
