# 类与面向对象

JavaScript 在 ES6 中引入了 `class` 语法，使面向对象编程更加直观。但需要注意的是，JavaScript 的类本质上是基于原型的继承机制的语法糖。

## 类的基础语法

### 类的声明

```javascript
// 类声明
class Person {
  constructor(name) {
    this.name = name;
  }

  // 实例方法
  greet() {
    console.log(`Hello, I'm ${this.name}`);
  }
}

// 类表达式
const Animal = class {
  constructor(name) {
    this.name = name;
  }
};

// 命名类表达式
const Machine = class Machine {
  constructor(type) {
    this.type = type;
  }
};
```

### 类的特点

```javascript
// 1. 类不会提升
// const p = new Person(); // ReferenceError: Person is not defined
// class Person {}

// 2. 类内部是严格模式
class Example {
  test() {
    // 默认启用严格模式
    // var undeclared = 42; // 严格模式下会报错
  }
}

// 3. 类的方法不可枚举
class Point {
  constructor(x, y) {
    this.x = x;
    this.y = y;
  }

  toString() {
    return `(${this.x}, ${this.y})`;
  }
}

const p = new Point(1, 2);
console.log(Object.keys(p)); // ['x', 'y']，没有 toString

// 4. 类必须使用 new 调用
// Person('John'); // TypeError: Class constructor Person cannot be invoked without 'new'
```

## 构造函数与实例属性

### constructor 方法

```javascript
class Person {
  // constructor 是类的默认方法
  // 通过 new 命令生成对象实例时自动调用
  constructor(name, age) {
    // this 指向新创建的实例
    this.name = name;
    this.age = age;
    this.createdAt = new Date();
  }

  introduce() {
    return `I'm ${this.name}, ${this.age} years old`;
  }
}

const person = new Person('Alice', 25);
console.log(person.introduce()); // "I'm Alice, 25 years old"

// 默认构造函数
class Empty {}
// 等同于
class EmptyDefault {
  constructor() {}
}

// 子类的默认构造函数
class Child extends Parent {
  // 等同于
  constructor(...args) {
    super(...args);
  }
}
```

### 实例属性

```javascript
class Person {
  // 公共实例字段（ES2022）
  name = 'Unknown';  // 带默认值
  age;               // 不带默认值
  
  constructor(name, age) {
    this.name = name;
    this.age = age;
    // 动态添加属性
    this.id = Date.now();
  }
}

// 实例属性在原型链上
const p1 = new Person('Alice', 25);
const p2 = new Person('Bob', 30);

console.log(p1.name); // 'Alice'
console.log(p2.name); // 'Bob'

// 每个实例有自己的属性
console.log(p1.hasOwnProperty('name')); // true
console.log(p2.hasOwnProperty('name')); // true
```

### 实例方法

```javascript
class Calculator {
  constructor(initial = 0) {
    this.value = initial;
  }

  // 实例方法定义在原型上
  add(n) {
    this.value += n;
    return this; // 支持链式调用
  }

  subtract(n) {
    this.value -= n;
    return this;
  }

  getValue() {
    return this.value;
  }
}

const calc = new Calculator(10);
calc.add(5).subtract(3);
console.log(calc.getValue()); // 12

// 方法在原型上共享
const calc2 = new Calculator();
console.log(calc.add === calc2.add); // true
```

## 静态成员

### 静态方法

```javascript
class MathUtils {
  // 静态方法
  static square(x) {
    return x * x;
  }

  static cube(x) {
    return x * x * x;
  }

  // 静态方法可以互相调用
  static sumOfSquares(a, b) {
    return this.square(a) + this.square(b);
  }
}

// 通过类名调用
console.log(MathUtils.square(4)); // 16
console.log(MathUtils.sumOfSquares(3, 4)); // 25

// 静态方法不能通过实例调用
const utils = new MathUtils();
// utils.square(4); // TypeError: utils.square is not a function
```

### 静态属性

```javascript
class Config {
  // 静态属性（ES2022）
  static version = '1.0.0';
  static defaultSettings = {
    theme: 'light',
    language: 'en'
  };

  static getSettings() {
    return this.defaultSettings;
  }
}

console.log(Config.version); // '1.0.0'
console.log(Config.getSettings()); // { theme: 'light', language: 'en' }

// 静态属性在类本身上
console.log(Config.hasOwnProperty('version')); // true
```

### 静态成员的应用场景

```javascript
// 1. 工厂方法
class User {
  constructor(name, email) {
    this.name = name;
    this.email = email;
  }

  // 工厂方法
  static createGuest() {
    return new User('Guest', 'guest@example.com');
  }

  static fromJSON(json) {
    const data = JSON.parse(json);
    return new User(data.name, data.email);
  }
}

const guest = User.createGuest();
const user = User.fromJSON('{"name":"Alice","email":"alice@example.com"}');

// 2. 单例模式
class Database {
  static instance = null;

  constructor() {
    if (Database.instance) {
      return Database.instance;
    }
    Database.instance = this;
  }

  static getInstance() {
    if (!Database.instance) {
      Database.instance = new Database();
    }
    return Database.instance;
  }
}

// 3. 常量定义
class HttpStatus {
  static OK = 200;
  static CREATED = 201;
  static BAD_REQUEST = 400;
  static UNAUTHORIZED = 401;
  static NOT_FOUND = 404;
  static INTERNAL_SERVER_ERROR = 500;
}
```

## 继承与多态

### extends 关键字

```javascript
class Animal {
  constructor(name) {
    this.name = name;
  }

  speak() {
    console.log(`${this.name} makes a sound`);
  }

  move() {
    console.log(`${this.name} moves`);
  }
}

class Dog extends Animal {
  constructor(name, breed) {
    // super 必须在 this 之前调用
    super(name);  // 调用父类构造函数
    this.breed = breed;
  }

  // 重写父类方法
  speak() {
    console.log(`${this.name} barks`);
  }

  // 新增方法
  fetch() {
    console.log(`${this.name} fetches the ball`);
  }
}

const dog = new Dog('Buddy', 'Golden Retriever');
dog.speak(); // "Buddy barks"
dog.move();  // "Buddy moves"（继承自父类）
dog.fetch(); // "Buddy fetches the ball"
```

### super 关键字

```javascript
class Parent {
  constructor(value) {
    this.value = value;
  }

  method() {
    return 'parent method';
  }

  static staticMethod() {
    return 'static parent';
  }
}

class Child extends Parent {
  constructor(value, extra) {
    super(value);  // 调用父类构造函数
    this.extra = extra;
  }

  method() {
    // 调用父类的实例方法
    const parentResult = super.method();
    return `${parentResult}, child override`;
  }

  static staticMethod() {
    // 调用父类的静态方法
    return `${super.staticMethod()}, static child`;
  }
}

const child = new Child(10, 20);
console.log(child.method()); // "parent method, child override"
console.log(Child.staticMethod()); // "static parent, static child"
```

### 多态实现

```javascript
class Shape {
  constructor(name) {
    this.name = name;
  }

  // 父类可以定义接口（或抛出错误）
  area() {
    throw new Error('Subclass must implement area method');
  }

  perimeter() {
    throw new Error('Subclass must implement perimeter method');
  }

  describe() {
    return `${this.name}: area=${this.area()}, perimeter=${this.perimeter()}`;
  }
}

class Rectangle extends Shape {
  constructor(width, height) {
    super('Rectangle');
    this.width = width;
    this.height = height;
  }

  area() {
    return this.width * this.height;
  }

  perimeter() {
    return 2 * (this.width + this.height);
  }
}

class Circle extends Shape {
  constructor(radius) {
    super('Circle');
    this.radius = radius;
  }

  area() {
    return Math.PI * this.radius ** 2;
  }

  perimeter() {
    return 2 * Math.PI * this.radius;
  }
}

// 多态使用
const shapes = [
  new Rectangle(3, 4),
  new Circle(5),
  new Rectangle(2, 6)
];

shapes.forEach(shape => {
  console.log(shape.describe());
});
// "Rectangle: area=12, perimeter=14"
// "Circle: area=78.54..., perimeter=31.41..."
// "Rectangle: area=12, perimeter=16"
```

## 访问器属性

### getter 和 setter

```javascript
class Person {
  constructor(firstName, lastName) {
    this._firstName = firstName;
    this._lastName = lastName;
  }

  // getter
  get fullName() {
    return `${this._firstName} ${this._lastName}`;
  }

  // setter
  set fullName(value) {
    const [first, last] = value.split(' ');
    this._firstName = first;
    this._lastName = last;
  }

  get firstName() {
    return this._firstName;
  }

  set firstName(value) {
    if (value.length < 2) {
      throw new Error('Name too short');
    }
    this._firstName = value;
  }
}

const person = new Person('John', 'Doe');
console.log(person.fullName); // "John Doe"

person.fullName = 'Jane Smith';
console.log(person.firstName); // "Jane"
console.log(person.lastName); // undefined（需要 getter）

// person.firstName = 'A'; // Error: Name too short
```

### 只读属性

```javascript
class Circle {
  constructor(radius) {
    this._radius = radius;
  }

  // 只读属性
  get radius() {
    return this._radius;
  }

  // 没有 setter，属性只读
  get area() {
    return Math.PI * this._radius ** 2;
  }

  get diameter() {
    return this._radius * 2;
  }
}

const circle = new Circle(5);
console.log(circle.radius); // 5
console.log(circle.area); // 78.53...

// circle.radius = 10; // 没有效果（严格模式报错）
// circle.area = 100; // TypeError
```

### 惰性计算

```javascript
class DataProcessor {
  constructor(data) {
    this._data = data;
    this._cachedResult = null;
  }

  get result() {
    if (this._cachedResult === null) {
      console.log('Computing...');
      this._cachedResult = this._data.reduce((sum, n) => sum + n, 0);
    }
    return this._cachedResult;
  }

  set data(value) {
    this._data = value;
    this._cachedResult = null; // 清除缓存
  }
}

const processor = new DataProcessor([1, 2, 3, 4, 5]);
console.log(processor.result); // "Computing..." then 15
console.log(processor.result); // 15（使用缓存）

processor.data = [10, 20];
console.log(processor.result); // "Computing..." then 30
```

## 私有属性与方法

### 私有字段（ES2022）

```javascript
class Person {
  // 私有字段以 # 开头
  #name;
  #age;

  constructor(name, age) {
    this.#name = name;
    this.#age = age;
  }

  // 公共方法访问私有字段
  getName() {
    return this.#name;
  }

  getAge() {
    return this.#age;
  }

  // 私有方法
  #validateAge(age) {
    return age > 0 && age < 150;
  }

  setAge(age) {
    if (this.#validateAge(age)) {
      this.#age = age;
    } else {
      throw new Error('Invalid age');
    }
  }

  // 静态私有字段
  static #instanceCount = 0;

  static getInstanceCount() {
    return Person.#instanceCount;
  }

  constructor(name, age) {
    this.#name = name;
    this.#age = age;
    Person.#instanceCount++;
  }
}

const person = new Person('Alice', 25);
console.log(person.getName()); // "Alice"
// console.log(person.#name); // SyntaxError: Private field '#name'
// console.log(person.#validateAge(30)); // SyntaxError
```

### 私有字段与继承

```javascript
class Parent {
  #privateField = 'parent private';

  getPrivateField() {
    return this.#privateField;
  }
}

class Child extends Parent {
  // 子类无法访问父类的私有字段
  tryAccess() {
    // return this.#privateField; // SyntaxError
    return this.getPrivateField(); // 通过公共方法访问
  }
}

const child = new Child();
console.log(child.tryAccess()); // "parent private"
console.log(child.getPrivateField()); // "parent private"
```

### 模拟私有（传统方式）

```javascript
// 1. 约定命名（不真正的私有）
class Person {
  constructor(name) {
    this._name = name; // 约定：下划线开头表示私有
  }
}

// 2. WeakMap 实现真正的私有
const privateData = new WeakMap();

class Secret {
  constructor(secret) {
    privateData.set(this, { secret });
  }

  getSecret() {
    return privateData.get(this).secret;
  }
}

const s = new Secret('my secret');
console.log(s.getSecret()); // "my secret"
console.log(s.secret); // undefined

// 3. 闭包实现
function createPerson(name) {
  let privateName = name;
  
  return {
    getName() {
      return privateName;
    },
    setName(value) {
      privateName = value;
    }
  };
}
```

### 静态初始化块

```javascript
class Config {
  static settings = {};
  static version;
  
  // 静态初始化块（ES2022）
  static {
    // 类加载时执行
    console.log('Initializing Config...');

      // 可以执行复杂初始化逻辑

    return this.settings[key];
  }
}
```

### 字段初始化顺序

## 类的本质

### 类是函数的语法糖

```javascript
class Person {
  constructor(name) {
    this.name = name;
  }

  greet() {
    console.log(`Hello, ${this.name}`);
  }
}

// typeof Person === 'function'
console.log(typeof Person); // 'function'

// 类本质是构造函数
console.log(Person === Person.prototype.constructor); // true

// 方法定义在原型上
console.log(Person.prototype.greet); // [Function: greet]

// 等价的 ES5 写法
function PersonES5(name) {
  this.name = name;
}

PersonES5.prototype.greet = function() {
  console.log(`Hello, ${this.name}`);
};
```

### 原型链结构

```javascript
class Parent {}
class Child extends Parent {}

const child = new Child();

// 原型链
console.log(child.__proto__ === Child.prototype); // true
console.log(Child.prototype.__proto__ === Parent.prototype); // true
console.log(Parent.prototype.__proto__ === Object.prototype); // true
console.log(Object.prototype.__proto__ === null); // true

// 类的继承链
console.log(Child.__proto__ === Parent); // true
console.log(Parent.__proto__ === Function.prototype); // true

// 图示：
// child -> Child.prototype -> Parent.prototype -> Object.prototype -> null
// Child -> Parent -> Function.prototype -> Object.prototype -> null
```

### babel 编译结果


## 面向对象设计原则

### 单一职责原则

```javascript
// ❌ 违反单一职责
class User {
  constructor(name) {
    this.name = name;
  }

  save() {
    // 保存到数据库
  }

  sendEmail() {
    // 发送邮件
  }

  generateReport() {
    // 生成报告
  }
}

// ✅ 遵循单一职责
class User {
  constructor(name) {
    this.name = name;
  }
}

class UserRepository {
  save(user) {
    // 保存用户
  }

  findById(id) {
    // 查找用户
  }
}

class EmailService {
  send(user, message) {
    // 发送邮件
  }
}

class ReportGenerator {
  generate(users) {
    // 生成报告
  }
}
```

### 开闭原则

```javascript
// ✅ 对扩展开放，对修改关闭
class PaymentProcessor {
  process(payment) {
    payment.pay();
  }
}

// 支付方式基类
class Payment {
  pay() {
    throw new Error('Must implement pay method');
  }
}

// 具体支付方式
class CreditCardPayment extends Payment {
  constructor(cardNumber, amount) {
    super();
    this.cardNumber = cardNumber;
    this.amount = amount;
  }

  pay() {
    console.log(`Paid ${this.amount} via credit card`);
  }
}

class PayPalPayment extends Payment {
  constructor(email, amount) {
    super();
    this.email = email;
    this.amount = amount;
  }

  pay() {
    console.log(`Paid ${this.amount} via PayPal`);
  }
}

// 新增支付方式无需修改 PaymentProcessor
class CryptoPayment extends Payment {
  constructor(wallet, amount) {
    super();
    this.wallet = wallet;
    this.amount = amount;
  }

  pay() {
    console.log(`Paid ${this.amount} via Crypto`);
  }
}
```

### 组合优于继承

```javascript
// ❌ 继承导致的问题
class Animal {
  walk() {}
}
class FlyingAnimal extends Animal {
  fly() {}
}
class SwimmingAnimal extends Animal {
  swim() {}
}
// FlyingFish 怎么办？多继承？

// ✅ 组合模式
const canWalk = {
  walk() {
    console.log(`${this.name} walks`);
  }
};

const canFly = {
  fly() {
    console.log(`${this.name} flies`);
  }
};

  }

fish.walk(); // "Fish walks"
fish.swim(); // "Fish swims"
```

## 小结

| 特性 | 说明 | ES 版本 |
|-----|------|---------|
| `class` 声明 | 类的基本语法 | ES6 |
| `extends` | 实现继承 | ES6 |
| `super` | 调用父类方法/构造函数 | ES6 |
| `static` 方法 | 静态方法 | ES6 |
| `get/set` | 访问器属性 | ES6 |
| `#` 私有字段 | 真正的私有属性 | ES2022 |
| 公共类字段 | 类字段语法 | ES2022 |
| `static {}` | 静态初始化块 | ES2022 |
| 私有静态字段 | `static #field` | ES2022 |

### 关键要点

1. **类是语法糖**：本质仍是基于原型的继承
2. **必须 new 调用**：不能像普通函数一样调用
3. **严格模式**：类内部自动启用严格模式
4. **super 的使用**：子类构造函数必须先调用 super
5. **私有字段**：使用 `#` 前缀实现真正的私有
