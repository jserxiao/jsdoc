## 对象创建

### 对象字面量

```javascript
// 基本语法
const obj = {
  name: 'John',
  age: 30,
  greet() {
    return `Hello, ${this.name}`;
  }
};

// ES6 增强语法
const name = 'John';
const age = 30;

// 属性简写
const obj1 = { name, age };  // { name: 'John', age: 30 }

// 方法简写
const obj2 = {
  greet() { },           // 普通方法
  *generator() { },      // 生成器方法
  async asyncMethod() { }, // 异步方法
  ['computed' + 'Key']: 'value'  // 计算属性名
};

// Symbol 作为键
const sym = Symbol('secret');
const obj3 = {
  [sym]: 'hidden value',
  [Symbol.iterator]: function* () {
    yield 1;
    yield 2;
  }
};
```

### new Object()

```javascript
// 基本用法
const obj = new Object();
obj.name = 'John';

// 等同于
const obj2 = { name: 'John' };

// new Object 可以接收参数
const strObj = new Object('hello');    // String 对象
const numObj = new Object(123);        // Number 对象
const boolObj = new Object(true);      // Boolean 对象

// 判断
console.log(typeof strObj);  // 'object'
console.log(strObj instanceof String);  // true
```

### Object.create()

```javascript
// 创建没有原型的对象
const nullProto = Object.create(null);
console.log(nullProto.toString);  // undefined（没有继承的方法）

// 指定原型
const proto = {
  greet() {
    console.log('Hello');
  }
};

const obj = Object.create(proto);
obj.greet();  // 'Hello'

// 第二个参数：属性描述符
const obj2 = Object.create(proto, {
  name: {
    value: 'John',
    writable: true,
    enumerable: true,
    configurable: true
  }
});

// 创建纯净的字典对象
const dict = Object.create(null);
dict.key = 'value';
// 没有 toString, hasOwnProperty 等方法

// Object.create 实现继承
function Animal(name) {
  this.name = name;
}

Animal.prototype.speak = function() {
  console.log(`${this.name} speaks`);
};

function Dog(name, breed) {
  Animal.call(this, name);
  this.breed = breed;
}

Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.constructor = Dog;

Dog.prototype.bark = function() {
  console.log(`${this.name} barks`);
};
```

### 构造函数

```javascript
// 传统构造函数
function Person(name, age) {
  this.name = name;
  this.age = age;
}

Person.prototype.greet = function() {
  console.log(`Hello, I'm ${this.name}`);
};

const person = new Person('John', 30);

// ES6 类语法
class PersonClass {
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }

  greet() {
    console.log(`Hello, I'm ${this.name}`);
  }

  static create(name, age) {
    return new PersonClass(name, age);
  }
}

const person2 = new PersonClass('Jane', 25);
const person3 = PersonClass.create('Bob', 40);
```

## 属性访问

### 点表示法与方括号表示法

```javascript
const obj = {
  name: 'John',
  'full-name': 'John Doe',
  123: 'number key',
  'my key': 'space key'
};

// 点表示法
console.log(obj.name);  // 'John'

// 方括号表示法
console.log(obj['name']);       // 'John'
console.log(obj['full-name']);  // 'John Doe'
console.log(obj[123]);          // 'number key'（自动转为字符串）
console.log(obj['my key']);     // 'space key'

// 动态属性名
const key = 'name';
console.log(obj[key]);  // 'John'

// 访问不存在的属性
console.log(obj.nonexistent);  // undefined（不会报错）

// 可选链访问（ES2020）
const user = { profile: { name: 'John' } };
console.log(user?.profile?.name);  // 'John'
console.log(user?.address?.city);  // undefined
console.log(user?.profile?.getName?.());  // undefined
```

### 属性检查

```javascript
const obj = { a: 1, b: undefined };

// in 操作符（检查原型链）
console.log('a' in obj);           // true
console.log('b' in obj);           // true
console.log('toString' in obj);    // true（来自原型）
console.log('nonexistent' in obj); // false

// hasOwnProperty（只检查自身）
console.log(obj.hasOwnProperty('a'));        // true
console.log(obj.hasOwnProperty('toString')); // false

// Object.hasOwn（ES2022，推荐）
console.log(Object.hasOwn(obj, 'a'));        // true
console.log(Object.hasOwn(obj, 'toString')); // false

// 避免原型链问题
const nullProto = Object.create(null);
nullProto.key = 'value';
// nullProto.hasOwnProperty('key')  // TypeError
Object.hasOwn(nullProto, 'key');    // true

// 检查 undefined vs 不存在
console.log('b' in obj);       // true（存在，值为 undefined）
console.log('c' in obj);       // false（不存在）
console.log(obj.b === undefined);  // true
console.log(obj.c === undefined);  // true（无法区分）
```

### 属性删除

```javascript
const obj = { a: 1, b: 2, c: 3 };

// delete 操作符
delete obj.a;
console.log(obj);  // { b: 2, c: 3 }

// delete 返回值
console.log(delete obj.b);       // true
console.log(delete obj.nonexistent); // true（总是返回 true）

// 不可配置属性无法删除
const obj2 = {};
Object.defineProperty(obj2, 'a', {
  value: 1,
  configurable: false
});

console.log(delete obj2.a);  // false（严格模式抛出错误）

// 删除所有属性
function clearObject(obj) {
  for (const key of Object.keys(obj)) {
    delete obj[key];
  }
}
```

## 属性描述符

### 描述符属性

每个属性都有四个特性（attributes）：

| 特性 | 默认值 | 说明 |
|------|--------|------|
| value | undefined | 属性值 |
| writable | false | 是否可修改 |
| enumerable | false | 是否可枚举 |
| configurable | false | 是否可删除/修改描述符 |

### Object.defineProperty

```javascript
const obj = {};

Object.defineProperty(obj, 'name', {
  value: 'John',
  writable: true,
  enumerable: true,
  configurable: true
});

// 读取描述符
const descriptor = Object.getOwnPropertyDescriptor(obj, 'name');
console.log(descriptor);
// { value: 'John', writable: true, enumerable: true, configurable: true }

// 批量定义
Object.defineProperties(obj, {
  age: {
    value: 30,
    writable: true,
    enumerable: true,
    configurable: true
  },
  id: {
    value: 1,
    writable: false,
    enumerable: false,
    configurable: false
  }
});
```

### writable（可写性）

```javascript
const obj = {};

Object.defineProperty(obj, 'name', {
  value: 'John',
  writable: false  // 只读
});

obj.name = 'Jane';  // 静默失败（严格模式报错）
console.log(obj.name);  // 'John'

// 注意：对象属性仍可变
Object.defineProperty(obj, 'data', {
  value: { count: 0 },
  writable: false
});

obj.data.count = 1;  // 有效！
console.log(obj.data.count);  // 1

// 完全不可变
Object.defineProperty(obj, 'immutableData', {
  value: Object.freeze({ count: 0 }),
  writable: false
});
```

### enumerable（可枚举性）

```javascript
const obj = { a: 1, b: 2 };

Object.defineProperty(obj, 'secret', {
  value: 'hidden',
  enumerable: false
});

console.log(obj.secret);  // 'hidden'（可访问）
console.log(Object.keys(obj));  // ['a', 'b']（不可枚举的被排除）
console.log('secret' in obj);   // true（仍存在）

for (const key in obj) {
  console.log(key);  // 'a', 'b'
}

// JSON.stringify 忽略不可枚举属性
console.log(JSON.stringify(obj));  // '{"a":1,"b":2}'
```

### configurable（可配置性）

```javascript
const obj = {};

Object.defineProperty(obj, 'name', {
  value: 'John',
  configurable: false  // 不可配置
});

// 无法删除
delete obj.name;  // 无效

// 无法修改描述符类型
Object.defineProperty(obj, 'name', {
  get() { return 'Jane'; }  // TypeError
});

// writable 可以从 true 改为 false
Object.defineProperty(obj, 'age', {
  value: 30,
  writable: true,
  configurable: false
});

Object.defineProperty(obj, 'age', {
  writable: false  // 允许
});

obj.age = 31;  // 无效
```

### 获取所有描述符

```javascript
const obj = {
  a: 1,
  get name() { return 'John'; }
};

Object.defineProperty(obj, 'secret', {
  value: 'hidden',
  enumerable: false
});

// 获取所有自身属性的描述符
const descriptors = Object.getOwnPropertyDescriptors(obj);
console.log(Object.keys(descriptors));  // ['a', 'name', 'secret']

// 克隆对象（包括 getter/setter）
const clone = Object.create(
  Object.getPrototypeOf(obj),
  Object.getOwnPropertyDescriptors(obj)
);
```

## getter 与 setter

### 对象字面量中的 getter/setter

```javascript
const obj = {
  _name: 'John',
  _age: 30,

  get name() {
    console.log('Getting name');
    return this._name;
  },

  set name(value) {
    console.log('Setting name');
    if (typeof value === 'string' && value.length > 0) {
      this._name = value;
    }
  },

  get age() {
    return this._age;
  },

  set age(value) {
    if (value >= 0 && value < 150) {
      this._age = value;
    }
  }
};

obj.name = 'Jane';       // Setting name
console.log(obj.name);   // Getting name, 'Jane'
```

### Object.defineProperty 中的 getter/setter

```javascript
const obj = {};
let _value = 0;

Object.defineProperty(obj, 'value', {
  get() {
    console.log('Getting value');
    return _value;
  },
  set(newValue) {
    console.log('Setting value');
    _value = newValue;
  },
  enumerable: true,
  configurable: true
});

obj.value = 10;
console.log(obj.value);  // 10
```

### 计算属性

```javascript
const circle = {
  _radius: 1,

  get radius() {
    return this._radius;
  },

  set radius(value) {
    if (value > 0) {
      this._radius = value;
    }
  },

  get area() {
    return Math.PI * this._radius ** 2;
  },

  get circumference() {
    return 2 * Math.PI * this._radius;
  }
};

circle.radius = 5;
console.log(circle.area);        // 78.54...
console.log(circle.circumference);  // 31.42...
```

### 私有属性模式

```javascript
// 使用闭包实现私有属性
function createPerson(name, age) {
  let _name = name;
  let _age = age;

  return {
    get name() { return _name; },
    set name(value) { _name = value; },
    get age() { return _age; },
    set age(value) {
      if (value >= 0) _age = value;
    }
  };
}

// 使用 WeakMap 实现私有属性
const privateData = new WeakMap();

class Person {
  constructor(name) {
    privateData.set(this, { name });
  }

  get name() {
    return privateData.get(this).name;
  }

  set name(value) {
    privateData.get(this).name = value;
  }
}

// ES2022 私有字段（推荐）
class PersonNew {
  #name;
  #age;

  constructor(name, age) {
    this.#name = name;
    this.#age = age;
  }

  get name() { return this.#name; }
  get age() { return this.#age; }
}
```

## 对象遍历

### Object.keys / values / entries

```javascript
const obj = { a: 1, b: 2, c: 3 };

// 获取键数组
console.log(Object.keys(obj));     // ['a', 'b', 'c']

// 获取值数组
console.log(Object.values(obj));   // [1, 2, 3]

// 获取键值对数组
console.log(Object.entries(obj));  // [['a', 1], ['b', 2], ['c', 3]]

// 遍历
Object.keys(obj).forEach(key => {
  console.log(key, obj[key]);
});

// entries 配合解构
for (const [key, value] of Object.entries(obj)) {
  console.log(key, value);
}

// 转换回对象
const newObj = Object.fromEntries([
  ['x', 1],
  ['y', 2]
]);
console.log(newObj);  // { x: 1, y: 2 }

// 对象 map 操作
const doubled = Object.fromEntries(
  Object.entries(obj).map(([k, v]) => [k, v * 2])
);
console.log(doubled);  // { a: 2, b: 4, c: 6 }
```

### for...in

```javascript
const obj = { a: 1, b: 2, c: 3 };

// for...in 遍历可枚举属性（包括继承的）
for (const key in obj) {
  console.log(key, obj[key]);
}

// 只遍历自身属性
for (const key in obj) {
  if (Object.hasOwn(obj, key)) {
    console.log(key, obj[key]);
  }
}

// 遍历顺序规则
// 1. 整数键按数值升序
// 2. 字符串键按创建顺序
// 3. Symbol 键被忽略

const ordered = {
  2: 'two',
  1: 'one',
  b: 'b',
  a: 'a'
};

for (const key in ordered) {
  console.log(key);  // 1, 2, b, a
}
```

### Object.getOwnPropertyNames

```javascript
const obj = { a: 1, b: 2 };

Object.defineProperty(obj, 'secret', {
  value: 'hidden',
  enumerable: false
});

// 获取所有自身属性名（包括不可枚举）
console.log(Object.getOwnPropertyNames(obj));  // ['a', 'b', 'secret']

// 只获取可枚举
console.log(Object.keys(obj));  // ['a', 'b']

// 获取 Symbol 属性
const sym = Symbol('key');
obj[sym] = 'symbol value';

console.log(Object.getOwnPropertySymbols(obj));  // [Symbol(key)]
```

### Reflect.ownKeys

```javascript
const obj = { a: 1, b: 2 };
const sym = Symbol('key');
obj[sym] = 'value';

Object.defineProperty(obj, 'hidden', {
  value: 'secret',
  enumerable: false
});

// 获取所有自身属性（包括 Symbol 和不可枚举）
console.log(Reflect.ownKeys(obj));
// ['a', 'b', 'hidden', Symbol(key)]

// 遍历所有属性
Reflect.ownKeys(obj).forEach(key => {
  console.log(key, obj[key]);
});
```

## 对象复制

### 浅拷贝

```javascript
const original = { a: 1, b: { c: 2 } };

// Object.assign
const copy1 = Object.assign({}, original);

// 展开运算符
const copy2 = { ...original };

// 修改嵌套对象会影响原对象
copy1.b.c = 3;
console.log(original.b.c);  // 3

// 数组的 slice 也是浅拷贝
const arr = [{ a: 1 }];
const arrCopy = arr.slice();
arrCopy[0].a = 2;
console.log(arr[0].a);  // 2
```

### 深拷贝

```javascript
// JSON 方法（有限制）
const deepCopy1 = JSON.parse(JSON.stringify(original));

// 限制：
// 1. 忽略函数、Symbol、undefined
// 2. 忽略循环引用
// 3. Date 变成字符串
// 4. RegExp 变成空对象
// 5. Map/Set 变成空对象

// structuredClone（现代浏览器）
const deepCopy2 = structuredClone(original);
// 支持循环引用、Date、RegExp、Map、Set、ArrayBuffer 等
// 仍不支持函数、Symbol、DOM 节点

// 递归实现
function deepClone(obj, hash = new WeakMap()) {
  // 处理 null 和非对象
  if (obj === null || typeof obj !== 'object') {
    return obj;
  }

  // 处理循环引用
  if (hash.has(obj)) {
    return hash.get(obj);
  }

  // 处理特殊对象
  if (obj instanceof Date) return new Date(obj);
  if (obj instanceof RegExp) return new RegExp(obj);
  if (obj instanceof Map) {
    const map = new Map();
    hash.set(obj, map);
    obj.forEach((v, k) => map.set(deepClone(k, hash), deepClone(v, hash)));
    return map;
  }
  if (obj instanceof Set) {
    const set = new Set();
    hash.set(obj, set);
    obj.forEach(v => set.add(deepClone(v, hash)));
    return set;
  }

  // 处理数组或普通对象
  const clone = Array.isArray(obj) ? [] : {};
  hash.set(obj, clone);

  // 复制所有属性（包括 Symbol）
  Reflect.ownKeys(obj).forEach(key => {
    clone[key] = deepClone(obj[key], hash);
  });

  return clone;
}
```

## 对象合并

### Object.assign

```javascript
const target = { a: 1, b: 2 };
const source1 = { b: 3, c: 4 };
const source2 = { c: 5, d: 6 };

// 合并到 target（修改 target）
const result = Object.assign(target, source1, source2);
console.log(target);  // { a: 1, b: 3, c: 5, d: 6 }
console.log(result === target);  // true

// 创建新对象
const merged = Object.assign({}, source1, source2);

// 问题：会触发 getter
const source = {
  get value() {
    console.log('getter called');
    return 1;
  }
};
Object.assign({}, source);  // getter called

// Object.assign 不复制继承属性
const proto = { inherited: 1 };
const source = Object.create(proto);
source.own = 2;
Object.assign({}, source);  // { own: 2 }
```

### 展开运算符

```javascript
const obj1 = { a: 1, b: 2 };
const obj2 = { b: 3, c: 4 };

// 合并（后面的覆盖前面）
const merged = { ...obj1, ...obj2 };
console.log(merged);  // { a: 1, b: 3, c: 4 }

// 添加新属性
const extended = {
  ...obj1,
  d: 5,
  ...obj2,
  e: 6
};

// 条件合并
const enabled = true;
const config = {
  base: 1,
  ...(enabled && { extra: 2 })
};
console.log(config);  // { base: 1, extra: 2 }

// 展开运算符会复制 getter 的值
const source = {
  get value() { return 1; }
};
const copy = { ...source };
console.log(copy.value);  // 1（值已固定）
```

## 对象冻结与密封

### Object.freeze（冻结）

```javascript
const obj = { a: 1, b: { c: 2 } };

Object.freeze(obj);

// 不能添加属性
obj.c = 3;  // 无效

// 不能删除属性
delete obj.a;  // 无效

// 不能修改属性值
obj.a = 10;  // 无效

// 不能修改描述符
Object.defineProperty(obj, 'a', { writable: true });  // TypeError

// 注意：浅冻结
obj.b.c = 3;  // 有效！
console.log(obj.b.c);  // 3

// 深度冻结
function deepFreeze(obj) {
  Object.freeze(obj);

  Object.keys(obj).forEach(key => {
    const value = obj[key];
    if (value && typeof value === 'object') {
      deepFreeze(value);
    }
  });

  return obj;
}
```

### Object.seal（密封）

```javascript
const obj = { a: 1, b: 2 };

Object.seal(obj);

// 不能添加属性
obj.c = 3;  // 无效

// 不能删除属性
delete obj.a;  // 无效

// 可以修改属性值
obj.a = 10;  // 有效
console.log(obj.a);  // 10

// 描述符可能限制修改
Object.defineProperty(obj, 'c', {
  value: 1,
  writable: false,
  configurable: false
});
Object.seal(obj);
obj.c = 2;  // 无效（因为 writable: false）
```

### Object.preventExtensions（阻止扩展）

```javascript
const obj = { a: 1, b: 2 };

Object.preventExtensions(obj);

// 不能添加属性
obj.c = 3;  // 无效

// 可以删除属性
delete obj.a;  // 有效

// 可以修改属性值
obj.b = 20;  // 有效

// 判断方法
console.log(Object.isExtensible(obj));  // false
console.log(Object.isSealed(obj));      // false（有属性可删除）
console.log(Object.isFrozen(obj));      // false
```

### 对比总结

| 方法 | 添加 | 删除 | 修改 | 特点 |
|------|------|------|------|------|
| preventExtensions | ❌ | ✅ | ✅ | 最宽松 |
| seal | ❌ | ❌ | ✅ | 中等 |
| freeze | ❌ | ❌ | ❌ | 最严格 |

```javascript
const obj = { a: 1, b: { c: 2 } };

// 判断方法
console.log(Object.isExtensible(obj));  // true
console.log(Object.isSealed(obj));      // false
console.log(Object.isFrozen(obj));      // false

Object.preventExtensions(obj);
console.log(Object.isExtensible(obj));  // false

Object.seal(obj);
console.log(Object.isSealed(obj));      // true

Object.freeze(obj);
console.log(Object.isFrozen(obj));      // true
```

## 对象比较

### 浅比较

```javascript
// 引用比较
const a = { x: 1 };
const b = { x: 1 };
console.log(a === b);  // false

const c = a;
console.log(a === c);  // true

// 浅比较实现
function shallowEqual(obj1, obj2) {
  if (obj1 === obj2) return true;

  if (typeof obj1 !== 'object' || obj1 === null ||
      typeof obj2 !== 'object' || obj2 === null) {
    return false;
  }

  const keys1 = Object.keys(obj1);
  const keys2 = Object.keys(obj2);

  if (keys1.length !== keys2.length) return false;

  for (const key of keys1) {
    if (!Object.hasOwn(obj2, key) || obj1[key] !== obj2[key]) {
      return false;
    }
  }

  return true;
}
```

### 深比较

```javascript
function deepEqual(a, b, seen = new WeakMap()) {
  // 严格相等
  if (a === b) return true;

  // null 或非对象
  if (a === null || b === null ||
      typeof a !== 'object' || typeof b !== 'object') {
    return false;
  }

  // 循环引用
  if (seen.has(a) || seen.has(b)) {
    return seen.get(a) === b;
  }
  seen.set(a, b);

  // 检查构造函数
  if (a.constructor !== b.constructor) return false;

  // 特殊对象
  if (a instanceof Date) return a.getTime() === b.getTime();
  if (a instanceof RegExp) return a.source === b.source && a.flags === b.flags;

  // Map
  if (a instanceof Map) {
    if (a.size !== b.size) return false;
    for (const [key, val] of a) {
      if (!b.has(key) || !deepEqual(val, b.get(key), seen)) {
        return false;
      }
    }
    return true;
  }

  // Set
  if (a instanceof Set) {
    if (a.size !== b.size) return false;
    for (const val of a) {
      if (!b.has(val)) return false;
    }
    return true;
  }

  // 数组
  if (Array.isArray(a)) {
    if (a.length !== b.length) return false;
    for (let i = 0; i < a.length; i++) {
      if (!deepEqual(a[i], b[i], seen)) return false;
    }
    return true;
  }

  // 普通对象
  const keysA = Reflect.ownKeys(a);
  const keysB = Reflect.ownKeys(b);

  if (keysA.length !== keysB.length) return false;

  for (const key of keysA) {
    if (!Object.hasOwn(b, key) || !deepEqual(a[key], b[key], seen)) {
      return false;
    }
  }

  return true;
}
```

## 对象方法

### Object.keys / values / entries

```javascript
const obj = { a: 1, b: 2, c: 3 };

Object.keys(obj);     // ['a', 'b', 'c']
Object.values(obj);   // [1, 2, 3]
Object.entries(obj);  // [['a', 1], ['b', 2], ['c', 3]]
```

### Object.fromEntries

```javascript
// 从键值对数组创建对象
const entries = [['a', 1], ['b', 2]];
const obj = Object.fromEntries(entries);
// { a: 1, b: 2 }

// Map 转对象
const map = new Map([['x', 1], ['y', 2]]);
const obj = Object.fromEntries(map);
// { x: 1, y: 2 }

// 过滤对象属性
const filtered = Object.fromEntries(
  Object.entries(obj).filter(([k, v]) => v > 0)
);
```

### Object.assign

```javascript
const target = { a: 1 };
const source = { b: 2 };
Object.assign(target, source);  // { a: 1, b: 2 }
```

### Object.getOwnPropertyNames / Symbols

```javascript
const obj = { a: 1 };
const sym = Symbol('key');
obj[sym] = 'value';

Object.getOwnPropertyNames(obj);  // ['a']
Object.getOwnPropertySymbols(obj);  // [Symbol(key)]
```

### Object.getPrototypeOf / setPrototypeOf

```javascript
const obj = {};
const proto = { greet() { console.log('Hello'); } };

Object.setPrototypeOf(obj, proto);
obj.greet();  // 'Hello'

Object.getPrototypeOf(obj) === proto;  // true
```
