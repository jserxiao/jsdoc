# 属性描述符

> 属性描述符是 ES5 引入的特性，它允许精细控制对象属性的行为。《JavaScript高级程序设计》将此作为理解对象的核心内容。

## 学习要点

- 🔍 理解属性描述符的概念
- 📝 掌握 Object.defineProperty() 的使用
- 🔒 理解数据属性和访问器属性的区别
- 🛠️ 学会使用 Object.getOwnPropertyDescriptor()

---

## 1. 什么是属性描述符？

属性描述符是一个对象，描述了一个属性的特性（特性是属性内部的值）。

### 属性描述符的类型

```javascript
// 1. 数据属性描述符
{
    value: undefined,        // 属性值
    writable: true,          // 是否可写
    enumerable: true,        // 是否可枚举
    configurable: true       // 是否可配置（删除、修改特性）
}

// 2. 访问器属性描述符
{
    get: undefined,          // getter 函数
    set: undefined,          // setter 函数
    enumerable: true,        // 是否可枚举
    configurable: true       // 是否可配置
}
```

## 2. Object.defineProperty()

### 基本用法

```javascript
const obj = {};

// 定义属性
Object.defineProperty(obj, 'name', {
    value: 'John',
    writable: true,
    enumerable: true,
    configurable: true
});

console.log(obj.name); // 'John'
```

### 定义多个属性

```javascript
const obj = {};

Object.defineProperties(obj, {
    name: {
        value: 'John',
        writable: true,
        enumerable: true,
        configurable: true
    },
    age: {
        value: 30,
        writable: false,
        enumerable: true,
        configurable: true
    }
});

console.log(obj.name); // 'John'
console.log(obj.age);  // 30
obj.age = 31;          // 静默失败（严格模式报错）
console.log(obj.age);  // 30
```

## 3. 数据属性

数据属性包含一个数据值的位置，可以读取和写入值。

### value

```javascript
const obj = {};

Object.defineProperty(obj, 'name', {
    value: 'John'
});

console.log(obj.name); // 'John'

// value 默认是 undefined
const obj2 = {};
Object.defineProperty(obj2, 'empty', {});
console.log(obj2.empty); // undefined
```

### writable

```javascript
// writable: true（可写，默认值）
const obj1 = {};
Object.defineProperty(obj1, 'name', {
    value: 'John',
    writable: true
});
obj1.name = 'Jane'; // 可以修改
console.log(obj1.name); // 'Jane'

// writable: false（只读）
const obj2 = {};
Object.defineProperty(obj2, 'name', {
    value: 'John',
    writable: false
});
obj2.name = 'Jane'; // 静默失败
console.log(obj2.name); // 'John'

// 严格模式下会报错
'use strict';
obj2.name = 'Jane'; // TypeError: Cannot assign to read only property
```

### enumerable

```javascript
// enumerable: true（可枚举，默认值）
const obj1 = {};
Object.defineProperty(obj1, 'name', {
    value: 'John',
    enumerable: true
});
console.log(Object.keys(obj1)); // ['name']
for (let key in obj1) { console.log(key); } // 'name'

// enumerable: false（不可枚举）
const obj2 = {
    a: 1,
    b: 2
};
Object.defineProperty(obj2, 'hidden', {
    value: 'secret',
    enumerable: false
});

console.log(Object.keys(obj2)); // ['a', 'b']
for (let key in obj2) { console.log(key); } // 'a', 'b'
console.log(obj2.hidden);       // 'secret'（可以访问）
console.log('hidden' in obj2);  // true（in 操作符可检测）

// JSON.stringify 会忽略不可枚举属性
console.log(JSON.stringify(obj2)); // '{"a":1,"b":2}'
```

### configurable

```javascript
// configurable: true（可配置，默认值）
const obj1 = {};
Object.defineProperty(obj1, 'name', {
    value: 'John',
    configurable: true
});
delete obj1.name; // 可以删除
console.log(obj1.name); // undefined

// configurable: false（不可配置）
const obj2 = {};
Object.defineProperty(obj2, 'name', {
    value: 'John',
    configurable: false
});
delete obj2.name; // 静默失败
console.log(obj2.name); // 'John'

// 不能再次修改描述符（除了 writable: true → false）
Object.defineProperty(obj2, 'name', {
    enumerable: false
}); // TypeError: Cannot redefine property

// 特殊情况：可以将 writable 从 true 改为 false
Object.defineProperty(obj2, 'name', {
    writable: false
}); // 允许
```

## 4. 访问器属性

访问器属性不包含数据值，而是包含 getter 和 setter 函数。

### getter 和 setter

```javascript
const obj = {
    _name: 'John', // 约定：下划线前缀表示私有属性
    _age: 30
};

Object.defineProperty(obj, 'name', {
    get() {
        console.log('获取 name');
        return this._name;
    },
    set(value) {
        console.log('设置 name');
        this._name = value;
    },
    enumerable: true,
    configurable: true
});

console.log(obj.name); // '获取 name' → 'John'
obj.name = 'Jane';     // '设置 name'
console.log(obj.name); // '获取 name' → 'Jane'
```

### 只读属性（只有 getter）

```javascript
const obj = {
    _age: 30
};

Object.defineProperty(obj, 'age', {
    get() {
        return this._age;
    },
    // 没有 setter
    enumerable: true,
    configurable: true
});

console.log(obj.age); // 30
obj.age = 31; // 静默失败（严格模式报错）
```

### 只写属性（只有 setter）

```javascript
const obj = {
    _password: ''
};

Object.defineProperty(obj, 'password', {
    set(value) {
        this._password = value;
    },
    // 没有 getter
    enumerable: true,
    configurable: true
});

obj.password = 'secret123';
console.log(obj.password); // undefined（无法读取）
console.log(obj._password); // 'secret123'
```

### 计算属性

```javascript
const person = {
    firstName: 'John',
    lastName: 'Doe'
};

Object.defineProperty(person, 'fullName', {
    get() {
        return `${this.firstName} ${this.lastName}`;
    },
    set(value) {
        const [first, last] = value.split(' ');
        this.firstName = first;
        this.lastName = last;
    }
});

console.log(person.fullName); // 'John Doe'
person.fullName = 'Jane Smith';
console.log(person.firstName); // 'Jane'
console.log(person.lastName);  // 'Smith'
```

### 使用对象字面量定义访问器

```javascript
const obj = {
    _value: 0,
    
    // getter
    get value() {
        return this._value;
    },
    
    // setter
    set value(val) {
        if (val >= 0) {
            this._value = val;
        }
    }
};

obj.value = 10;
console.log(obj.value); // 10

obj.value = -5; // 无效（小于 0）
console.log(obj.value); // 10
```

## 5. Object.getOwnPropertyDescriptor()

### 获取属性描述符

```javascript
const obj = {
    name: 'John'
};

const descriptor = Object.getOwnPropertyDescriptor(obj, 'name');

console.log(descriptor);
// {
//     value: 'John',
//     writable: true,
//     enumerable: true,
//     configurable: true
// }

// 访问器属性
const obj2 = {
    get name() { return 'John'; }
};

const descriptor2 = Object.getOwnPropertyDescriptor(obj2, 'name');

console.log(descriptor2);
// {
//     get: [Function: get name],
//     set: undefined,
//     enumerable: true,
//     configurable: true
// }
```

### 获取所有属性描述符

```javascript
const obj = {
    name: 'John',
    age: 30
};

Object.defineProperty(obj, 'hidden', {
    value: 'secret',
    enumerable: false
});

const descriptors = Object.getOwnPropertyDescriptors(obj);

console.log(descriptors);
// {
//     name: { value: 'John', writable: true, enumerable: true, configurable: true },
//     age: { value: 30, writable: true, enumerable: true, configurable: true },
//     hidden: { value: 'secret', writable: false, enumerable: false, configurable: false }
// }
```

## 6. 对象状态方法

### Object.preventExtensions()

```javascript
// 阻止对象扩展（不能添加新属性）
const obj = { name: 'John' };

Object.preventExtensions(obj);

obj.age = 30;      // 静默失败
console.log(obj);  // { name: 'John' }

delete obj.name;   // 可以删除
console.log(obj);  // {}

// 检查是否可扩展
console.log(Object.isExtensible(obj)); // false
```

### Object.seal()

```javascript
// 密封对象（不能添加/删除属性，但可以修改现有属性值）
const obj = { name: 'John' };

Object.seal(obj);

obj.age = 30;       // 静默失败（不能添加）
delete obj.name;    // 静默失败（不能删除）
obj.name = 'Jane';  // 可以修改

console.log(obj);   // { name: 'Jane' }

// 检查是否密封
console.log(Object.isSealed(obj)); // true
```

### Object.freeze()

```javascript
// 冻结对象（不能添加/删除/修改属性）
const obj = { name: 'John', info: { age: 30 } };

Object.freeze(obj);

obj.age = 30;       // 静默失败
delete obj.name;    // 静默失败
obj.name = 'Jane';  // 静默失败

console.log(obj);   // { name: 'John', info: { age: 30 } }

// ⚠️ 浅冻结：嵌套对象仍可修改
obj.info.age = 31;
console.log(obj.info.age); // 31

// 深冻结
function deepFreeze(obj) {
    const propNames = Object.getOwnPropertyNames(obj);
    
    for (const name of propNames) {
        const value = obj[name];
        if (value && typeof value === 'object') {
            deepFreeze(value);
        }
    }
    
    return Object.freeze(obj);
}

const obj2 = { info: { age: 30 } };
deepFreeze(obj2);
obj2.info.age = 31; // 静默失败

// 检查是否冻结
console.log(Object.isFrozen(obj)); // true
```

### 对象状态对比

| 方法 | 添加属性 | 删除属性 | 修改属性值 | 修改描述符 |
|------|---------|---------|-----------|-----------|
| 普通对象 | ✅ | ✅ | ✅ | ✅ |
| preventExtensions | ❌ | ✅ | ✅ | ✅ |
| seal | ❌ | ❌ | ✅ | ❌ |
| freeze | ❌ | ❌ | ❌ | ❌ |

## 7. 实际应用场景

### 实现响应式数据

```javascript
// 简易响应式系统
function reactive(target) {
    const handlers = {};
    
    return new Proxy(target, {
        get(obj, key) {
            track(obj, key);
            return obj[key];
        },
        set(obj, key, value) {
            obj[key] = value;
            trigger(obj, key);
            return true;
        }
    });
}

// 或者使用 defineProperty（Vue 2 的方式）
function defineReactive(obj, key, val) {
    const dep = new Set();
    
    Object.defineProperty(obj, key, {
        get() {
            // 依赖收集
            if (activeEffect) {
                dep.add(activeEffect);
            }
            return val;
        },
        set(newVal) {
            if (newVal === val) return;
            val = newVal;
            // 触发更新
            dep.forEach(fn => fn());
        }
    });
}
```

### 私有属性

```javascript
// 使用闭包 + 描述符实现私有属性
function createPerson(name) {
    let _name = name; // 真正的私有变量
    
    return {
        get name() {
            return _name;
        },
        set name(value) {
            if (value.length > 0) {
                _name = value;
            }
        }
    };
}

const person = createPerson('John');
console.log(person.name); // 'John'
person.name = 'Jane';
console.log(person.name); // 'Jane'
```

### 常量定义

```javascript
// 定义不可变的常量
const Config = {};

Object.defineProperties(Config, {
    API_URL: {
        value: 'https://api.example.com',
        writable: false,
        enumerable: true,
        configurable: false
    },
    VERSION: {
        value: '1.0.0',
        writable: false,
        enumerable: true,
        configurable: false
    }
});

Config.API_URL = 'changed'; // 静默失败
delete Config.VERSION;      // 静默失败
```

### 属性验证

```javascript
function createValidatedObject() {
    const data = {};
    
    return {
        setProperty(key, value, validator) {
            let _value = value;
            
            Object.defineProperty(this, key, {
                get() {
                    return _value;
                },
                set(newValue) {
                    if (validator(newValue)) {
                        _value = newValue;
                    } else {
                        throw new Error(`Invalid value for ${key}`);
                    }
                },
                enumerable: true,
                configurable: true
            });
        }
    };
}

const obj = createValidatedObject();
obj.setProperty('age', 25, val => Number.isInteger(val) && val >= 0);

obj.age = 30;  // OK
obj.age = -5;  // Error: Invalid value for age
obj.age = 3.5; // Error: Invalid value for age
```

## 小结

| 特性 | 数据属性 | 访问器属性 |
|------|---------|-----------|
| **value** | ✅ | ❌ |
| **writable** | ✅ | ❌ |
| **get** | ❌ | ✅ |
| **set** | ❌ | ✅ |
| **enumerable** | ✅ | ✅ |
| **configurable** | ✅ | ✅ |

| 方法 | 作用 |
|------|------|
| **defineProperty** | 定义单个属性 |
| **defineProperties** | 定义多个属性 |
