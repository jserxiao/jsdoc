## 什么是原型

### 核心概念

每个 JavaScript 对象都有一个内部链接指向另一个对象，这个对象就是它的**原型**。对象从原型继承属性和方法。

```
对象 ←── 原型 ←── 原型的原型 ←── ... ←── null

这条链叫做"原型链"
```

### 最简单的示例

```javascript
const person = {
  greet() {
    console.log('Hello!');
  }
};

const john = Object.create(person); // john 的原型是 person
john.greet(); // 'Hello!'（从原型继承的方法）

console.log(john.hasOwnProperty('greet')); // false
console.log(Object.getPrototypeOf(john) === person); // true
```

### 原型相关属性

```javascript
const obj = {};

// __proto__（已废弃，不推荐使用）
console.log(obj.__proto__ === Object.prototype); // true

// 推荐使用的方法
console.log(Object.getPrototypeOf(obj) === Object.prototype); // true

// 设置原型
const proto = { greet() { console.log('Hi'); } };
Object.setPrototypeOf(obj, proto);
obj.greet(); // 'Hi'

// 获取对象自身的属性描述符
console.log(Object.getOwnPropertyDescriptor(obj, 'greet'));
```

## 原型对象详解

### 构造函数与原型

```javascript
// 构造函数
function Person(name) {
  this.name = name;
}

// 在原型上添加方法（所有实例共享）
Person.prototype.greet = function() {
  console.log(`Hello, I'm ${this.name}!`);
};

Person.prototype.species = 'human';

// 创建实例
const john = new Person('John');
const mary = new Person('Mary');

john.greet();  // "Hello, I'm John!"
mary.greet();  // "Hello, I'm Mary!"

console.log(john.species); // 'human'（从原型继承）
console.log(john.name);    // 'John'（自有属性）
```

### 原型链图解

```
john 实例
┌─────────────────────┐
│ name: 'John'        │  ← 自有属性
│ __proto__ ──────────┼──→ Person.prototype
└─────────────────────┘    ┌─────────────────────┐
                           │ greet: ƒ()          │
                           │ species: 'human'    │
                           │ constructor: Person │
                           │ __proto__ ──────────┼──→ Object.prototype
                           └─────────────────────┘    ┌─────────────────────┐
                                                      │ toString: ƒ()       │
                                                      │ hasOwnProperty: ƒ() │
                                                      │ constructor: Object │
                                                      │ __proto__ ──────────┼──→ null
                                                      └─────────────────────┘
```

### 属性查找规则

```javascript
// 查找属性时，JavaScript 会沿着原型链向上查找
console.log(john.name);        // 'John'（自有属性）
console.log(john.species);     // 'human'（原型上的属性）
console.log(john.toString);    // [Function]（Object.prototype 上的方法）
console.log(john.foo);         // undefined（找不到，返回 undefined）

// 检查属性是自有的还是继承的
console.log(john.hasOwnProperty('name'));     // true
console.log(john.hasOwnProperty('species'));  // false
console.log(john.hasOwnProperty('toString')); // false

// in 操作符检查整个原型链
console.log('name' in john);     // true
console.log('species' in john);  // true（继承的也算）

// for...in 遍历可枚举属性（包括继承的）
for (const key in john) {
  console.log(key); // name, species, greet（不包括 toString，因为不可枚举）
}

// 只遍历自有属性
Object.keys(john).forEach(key => {
  console.log(key); // 只有 name
});
```

### constructor 属性

```javascript
function Person(name) {
  this.name = name;
}

const john = new Person('John');

// constructor 指向构造函数
console.log(john.constructor === Person); // true
console.log(Person.prototype.constructor === Person); // true

// 通过 constructor 创建新实例
const mary = new john.constructor('Mary');
console.log(mary instanceof Person); // true

// 注意：constructor 可以被修改
Person.prototype = {};
console.log(john.constructor === Person); // false
console.log(john.constructor === Object); // true
```

## 原型链继承

### ES5 组合继承

```javascript
// 父类
function Animal(name) {
  this.name = name;
}
Animal.prototype.speak = function() {
  console.log(`${this.name} makes a sound`);
};

// 子类
function Dog(name, breed) {
  Animal.call(this, name);  // 继承属性（借用构造函数）
  this.breed = breed;       // 子类自己的属性
}

// 继承原型方法
Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.constructor = Dog; // 修复 constructor 指向

// 子类自己的方法
Dog.prototype.bark = function() {
  console.log(`${this.name} barks: Woof!`);
};

// 使用
const dog = new Dog('Buddy', 'Golden Retriever');
dog.speak(); // "Buddy makes a sound"（继承的方法）
dog.bark();  // "Buddy barks: Woof!"（自己的方法）

console.log(dog instanceof Dog);    // true
console.log(dog instanceof Animal); // true
console.log(dog instanceof Object); // true
```

### 继承步骤详解

```javascript
// 步骤 1：借用构造函数继承属性
function Child(name) {
  Parent.call(this, name); // 调用父类构造函数
}

// 步骤 2：继承原型方法
Child.prototype = Object.create(Parent.prototype);

// 步骤 3：修复 constructor 指向
Child.prototype.constructor = Child;

// 步骤 4：添加子类自己的方法
Child.prototype.ownMethod = function() {};
```

### 寄生组合继承（最优方案）

```javascript
// 父类
function Animal(name) {
  this.name = name;
}
Animal.prototype.speak = function() {
  console.log(`${this.name} speaks`);
};

// 子类
function Dog(name, breed) {
  Animal.call(this, name);
  this.breed = breed;
}

// 寄生组合继承的核心
function inherit(Child, Parent) {
  // 创建一个以 Parent.prototype 为原型的对象
  const prototype = Object.create(Parent.prototype);
  // 修复 constructor
  prototype.constructor = Child;
  // 设置子类的原型
  Child.prototype = prototype;
}

inherit(Dog, Animal);

Dog.prototype.bark = function() {
  console.log(`${this.name} barks`);
};

const dog = new Dog('Buddy', 'Golden');
dog.speak(); // "Buddy speaks"
dog.bark();  // "Buddy barks"
```

### 原型链继承的问题

```javascript
// 问题 1：引用类型属性共享
function Parent() {
  this.children = ['Alice', 'Bob'];
}

function Child() {}
Child.prototype = new Parent();

const child1 = new Child();
const child2 = new Child();

child1.children.push('Charlie');
console.log(child2.children); // ['Alice', 'Bob', 'Charlie'] - 被污染了！

// 解决方案：使用组合继承
function BetterParent() {
  this.children = ['Alice', 'Bob'];
}

function BetterChild() {
  Parent.call(this); // 每个实例有自己的 children
}
BetterChild.prototype = Object.create(BetterParent.prototype);
BetterChild.prototype.constructor = BetterChild;

const c1 = new BetterChild();
const c2 = new BetterChild();
c1.children.push('Charlie');
console.log(c2.children); // ['Alice', 'Bob'] - 没问题
```

## Object.create()

### 基本用法

```javascript
// 创建一个以指定对象为原型的对象
const proto = {
  greet() {
    console.log('Hello!');
  },
  name: 'default'
};

const obj = Object.create(proto);
obj.greet(); // 'Hello!'
console.log(obj.name); // 'default'

console.log(Object.getPrototypeOf(obj) === proto); // true
```

### 第二个参数：属性描述符

```javascript
const obj = Object.create(proto, {
  name: {
    value: 'John',
    writable: true,
    enumerable: true,
    configurable: true
  },
  age: {
    value: 30,
    writable: false // 只读
  }
});

console.log(obj.name); // 'John'
console.log(obj.age);  // 30
obj.age = 31;          // 静默失败（严格模式报错）
```

### 创建"纯净"对象

```javascript
// 没有原型的对象（不继承 Object.prototype）
const pureObj = Object.create(null);
pureObj.hasOwnProperty; // undefined
pureObj.toString;       // undefined

// 用作字典（避免原型污染）
const dict = Object.create(null);
dict.toString = 'my value'; // 不会与原型方法冲突

// 安全的属性检查
console.log('toString' in {});        // true（继承的）
console.log('toString' in pureObj);   // false
```

### 实现 Object.create

```javascript
function myCreate(proto) {
  // proto 必须是对象或 null
  if (typeof proto !== 'object' && proto !== null) {
    throw new TypeError('Object prototype may only be an Object or null');
  }

  // 创建临时构造函数
  function F() {}
  F.prototype = proto;

  // 返回新实例
  return new F();
}

// 测试
const obj = myCreate({ x: 1 });
console.log(obj.x); // 1
console.log(Object.getPrototypeOf(obj)); // { x: 1 }
```

## instanceof 原理

### 工作原理

```javascript
// instanceof 检查对象的原型链上是否有构造函数的 prototype
obj instanceof Constructor

// 等价于
// 检查 Constructor.prototype 是否在 obj 的原型链上
```

### 手写 instanceof

```javascript
function myInstanceof(obj, Constructor) {
  // 基本类型直接返回 false
  if (obj === null || typeof obj !== 'object') {
    return false;
  }

  // 获取对象的原型
  let proto = Object.getPrototypeOf(obj);

  // 沿着原型链查找
  while (proto !== null) {
    if (proto === Constructor.prototype) {
      return true;
    }
    proto = Object.getPrototypeOf(proto);
  }

  return false;
}

// 测试
console.log(myInstanceof([], Array));    // true
console.log(myInstanceof([], Object));   // true
console.log(myInstanceof({}, Array));    // false
```

### 与 typeof 的区别

```javascript
// typeof 只能判断基本类型和 function
typeof []        // 'object'
typeof {}        // 'object'
typeof null      // 'object'
typeof (() => {}) // 'function'

// instanceof 可以判断具体的构造函数
[] instanceof Array   // true
[] instanceof Object  // true
({}) instanceof Array // false
```

### isPrototypeOf 方法

```javascript
// 检查一个对象是否在另一个对象的原型链上
const proto = {};
const obj = Object.create(proto);

console.log(proto.isPrototypeOf(obj)); // true
console.log(Object.prototype.isPrototypeOf(obj)); // true

// 比 instanceof 更灵活
console.log(Array.prototype.isPrototypeOf([])); // true
```

## ES6 class 语法

### 基本用法

```javascript
// ES6 class 是原型继承的语法糖
class Animal {
  constructor(name) {
    this.name = name;
  }

  speak() {
    console.log(`${this.name} makes a sound`);
  }

  // 静态方法
  static create(name) {
    return new Animal(name);
  }
}

class Dog extends Animal {
  constructor(name, breed) {
    super(name);     // 调用父类构造函数
    this.breed = breed;
  }

  speak() {
    console.log(`${this.name} barks`);
  }

  bark() {
    console.log(`${this.name} says Woof!`);
  }
}

const dog = new Dog('Buddy', 'Golden');
dog.speak(); // "Buddy barks"（重写的方法）
dog.bark();  // "Buddy says Woof!"（自己的方法）
```

### class 的本质

```javascript
class Animal {
  constructor(name) {
    this.name = name;
  }

  speak() {
    console.log(`${this.name} speaks`);
  }
}

// class 本质是函数
console.log(typeof Animal); // 'function'

// 方法在原型上
console.log(Animal.prototype.speak); // [Function: speak]

// 实例的结构
const animal = new Animal('Max');
console.log(animal.hasOwnProperty('name'));  // true
console.log(animal.hasOwnProperty('speak')); // false
console.log('speak' in animal);              // true
```

### getter 和 setter

```javascript
class Person {
  constructor(firstName, lastName) {
    this.firstName = firstName;
    this.lastName = lastName;
  }

  get fullName() {
    return `${this.firstName} ${this.lastName}`;
  }

  set fullName(name) {
    [this.firstName, this.lastName] = name.split(' ');
  }
}

const person = new Person('John', 'Doe');
console.log(person.fullName); // 'John Doe'
person.fullName = 'Jane Smith';
console.log(person.firstName); // 'Jane'
```

## 继承模式对比

| 模式 | 优点 | 缺点 |
|------|------|------|
| 原型链继承 | 简单，共享方法 | 引用类型共享问题 |
| 构造函数继承 | 避免引用共享 | 方法无法复用 |
| 组合继承 | 结合两者优点 | 父类构造函数调用两次 |
| 寄生组合继承 | 最优方案 | 实现稍复杂 |
| ES6 class | 语法简洁 | 需要转译支持 |

### 各种继承实现

```javascript
// 1. 原型链继承
function inheritPrototype(Child, Parent) {
  Child.prototype = new Parent();
}

// 2. 构造函数继承
function inheritConstructor(Child, Parent) {
  Child = function(...args) {
    Parent.apply(this, args);
  };
}

// 3. 组合继承
function inheritCombo(Child, Parent) {
  Parent.call(this);
  Child.prototype = new Parent();
  Child.prototype.constructor = Child;
}

// 4. 寄生组合继承
function inheritParasitic(Child, Parent) {
  const proto = Object.create(Parent.prototype);
  proto.constructor = Child;
  Child.prototype = proto;
}
```

## 原型污染与防护

### 原型污染问题

```javascript
// 危险：扩展内置原型
Array.prototype.first = function() {
  return this[0];
};

[1, 2, 3].first(); // 1

// 问题：可能与其他库冲突
// 问题：影响 for...in 遍历
for (const key in []) {
  console.log(key); // 可能输出 'first'
}

// 更危险：Object.prototype 污染
Object.prototype.evil = 'hacked';
const obj = { x: 1 };
console.log(obj.evil); // 'hacked' - 所有对象都被污染！
```

### 防护措施

```javascript
// 1. 使用 Object.create(null) 创建纯净对象
const safeDict = Object.create(null);
safeDict.toString = 'safe';

// 2. 使用 hasOwnProperty 检查自有属性
const obj = {};
if (obj.hasOwnProperty('key')) {
  // 安全
}

// 3. 使用 Object.prototype.hasOwnProperty.call
// 防止对象重写 hasOwnProperty
Object.prototype.hasOwnProperty.call(obj, 'key');

// 4. 使用 Object.keys/Object.getOwnPropertyNames
// 只获取自有属性
Object.keys(obj);
Object.getOwnPropertyNames(obj);

// 5. 使用 Map 代替普通对象
const map = new Map();
map.set('toString', 'safe');
console.log(map.get('toString')); // 'safe' - 不会与原型冲突
```

### 深拷贝防护

```javascript
// 安全的深拷贝函数
function safeDeepClone(obj, hash = new WeakMap()) {
  // 处理基本类型和 null
  if (obj === null || typeof obj !== 'object') {
    return obj;
  }

  // 循环引用
  if (hash.has(obj)) {
    return hash.get(obj);
  }

  // 使用 Object.create(null) 创建纯净对象
  const clone = Object.create(Object.getPrototypeOf(obj));
  hash.set(obj, clone);

  // 复制所有自有属性
  Reflect.ownKeys(obj).forEach(key => {
    clone[key] = safeDeepClone(obj[key], hash);
  });

  return clone;
}
```

## 实战应用

### 实现类式继承库

```javascript
function extend(Child, Parent) {
  // 继承原型
  Child.prototype = Object.create(Parent.prototype);
  Child.prototype.constructor = Child;

  // 添加 super 方法
  Child.super = function(...args) {
    Parent.apply(this, args);
  };

  // 保存父类引用
  Child.parent = Parent;

  return Child;
}

// 使用
function Animal(name) {
  this.name = name;
}
Animal.prototype.speak = function() {
  console.log(`${this.name} makes a sound`);
};

function Dog(name, breed) {
  Dog.super(name);
  this.breed = breed;
}
extend(Dog, Animal);

Dog.prototype.speak = function() {
  console.log(`${this.name} barks`);
};

const dog = new Dog('Buddy', 'Golden');
dog.speak(); // "Buddy barks"
```

### 实现混入（Mixin）

```javascript
// 对象混入
function mixin(target, ...sources) {
  sources.forEach(source => {
    Object.getOwnPropertyNames(source).forEach(key => {
      if (key !== 'constructor') {
        const descriptor = Object.getOwnPropertyDescriptor(source, key);
        Object.defineProperty(target, key, descriptor);
      }
    });
  });
  return target;
}

// 使用
const CanSwim = {
  swim() {
    console.log(`${this.name} is swimming`);
  }
};

const CanFly = {
  fly() {
    console.log(`${this.name} is flying`);
  }
};

class Duck {
  constructor(name) {
    this.name = name;
  }
}

mixin(Duck.prototype, CanSwim, CanFly);

const duck = new Duck('Donald');
duck.swim(); // "Donald is swimming"
duck.fly();  // "Donald is flying"
```

### 实现观察者模式

```javascript
function Observable() {
  this.observers = [];
}

Observable.prototype.addObserver = function(observer) {
  this.observers.push(observer);
};

Observable.prototype.removeObserver = function(observer) {
  const index = this.observers.indexOf(observer);
  if (index > -1) {
    this.observers.splice(index, 1);
  }
};

Observable.prototype.notify = function(data) {
  this.observers.forEach(observer => observer(data));
};

// 继承 Observable
function Model(data) {
  Observable.call(this);
  this.data = data;
}
Model.prototype = Object.create(Observable.prototype);
Model.prototype.constructor = Model;

Model.prototype.set = function(key, value) {
  this.data[key] = value;
  this.notify({ key, value });
};

Model.prototype.get = function(key) {
  return this.data[key];
};

// 使用
const model = new Model({ name: 'initial' });
model.addObserver(data => {
  console.log(`Changed: ${data.key} = ${data.value}`);
});

model.set('name', 'updated'); // "Changed: name = updated"
```

## 常见问题

### Q1: `__proto__` 和 `prototype` 的区别？

```javascript
// __proto__ 是对象的属性，指向其原型
const obj = {};
console.log(obj.__proto__ === Object.prototype); // true

// prototype 是函数的属性，是用 new 创建的对象的原型
function Person() {}
console.log(Person.prototype); // Person {}

const person = new Person();
console.log(person.__proto__ === Person.prototype); // true
```

### Q2: 如何判断对象是否有某属性？

```javascript
const obj = { a: 1 };

// 1. in 操作符（包括继承）
'a' in obj;           // true
'toString' in obj;    // true

// 2. hasOwnProperty（只检查自有属性）
obj.hasOwnProperty('a');        // true
obj.hasOwnProperty('toString'); // false

// 3. Object.hasOwn（ES2022，推荐）
Object.hasOwn(obj, 'a');        // true
Object.hasOwn(obj, 'toString'); // false

// 4. 属性访问 + undefined 检查（不可靠）
obj.a !== undefined;     // true
obj.b !== undefined;     // false（但如果 obj.b = undefined 呢？）
```

### Q3: 如何创建没有原型的对象？

```javascript
const pureObj = Object.create(null);
console.log(pureObj.__proto__); // undefined
console.log(pureObj.toString);  // undefined

// 适合用作字典
const dict = Object.create(null);
dict['key'] = 'value';
// 不会被 Object.prototype 的属性干扰
```
