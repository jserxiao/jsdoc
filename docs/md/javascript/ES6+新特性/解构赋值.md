# 解构赋值

> 解构赋值是 ES6 最重要的语法增强之一，它允许我们从数组或对象中提取值，并赋给变量。

## 学习要点

- 📝 掌握数组解构的用法
- 📦 掌握对象解构的用法
- 🔄 理解默认值和嵌套解构
- 🎯 学会实际应用场景

---

## 1. 数组解构

### 基本用法

```javascript
// 基本解构
const [a, b, c] = [1, 2, 3];
console.log(a, b, c); // 1 2 3

// 跳过元素
const [x, , z] = [1, 2, 3];
console.log(x, z); // 1 3

// 剩余元素
const [first, ...rest] = [1, 2, 3, 4, 5];
console.log(first); // 1
console.log(rest);  // [2, 3, 4, 5]

// 默认值
const [m = 10, n = 20] = [1];
console.log(m, n); // 1 20
```

### 交换变量

```javascript
// 传统方式
let a = 1, b = 2;
let temp = a;
a = b;
b = temp;

// 使用解构
let x = 1, y = 2;
[x, y] = [y, x];
console.log(x, y); // 2 1
```

### 从函数返回多个值

```javascript
function getCoordinates() {
    return [100, 200];
}

const [x, y] = getCoordinates();
console.log(x, y); // 100 200

// 带默认值
function getPoint() {
    return [10];
}

const [a, b = 0] = getPoint();
console.log(a, b); // 10 0
```

### 嵌套数组解构

```javascript
const [a, [b, c]] = [1, [2, 3]];
console.log(a, b, c); // 1 2 3

// 更复杂的嵌套
const [x, [y, [z]]] = [1, [2, [3]]];
console.log(x, y, z); // 1 2 3
```

## 2. 对象解构

### 基本用法

```javascript
const person = {
    name: 'John',
    age: 30,
    city: 'New York'
};

const { name, age, city } = person;
console.log(name, age, city); // John 30 New York

// 顺序无关
const { city: c, name: n } = person;
console.log(n, c); // John New York
```

### 重命名变量

```javascript
const person = { name: 'John', age: 30 };

// 将 name 重命名为 userName
const { name: userName, age: userAge } = person;
console.log(userName); // John
console.log(userAge);  // 30
```

### 默认值

```javascript
const person = { name: 'John' };

const { name, age = 25 } = person;
console.log(name); // John
console.log(age);  // 25（使用默认值）

// 与重命名结合
const { name: n = 'Default', gender: g = 'male' } = person;
console.log(n); // John
console.log(g); // male
```

### 嵌套对象解构

```javascript
const company = {
    name: 'Tech Corp',
    address: {
        city: 'San Francisco',
        country: 'USA'
    },
    employees: [
        { name: 'John', role: 'Developer' }
    ]
};

// 解构嵌套对象
const { 
    name: companyName,
    address: { city, country },
    employees: [{ name: employeeName }]
} = company;

console.log(companyName);  // Tech Corp
console.log(city);         // San Francisco
console.log(employeeName); // John
```

## 3. 函数参数解构

### 对象参数解构

```javascript
// 传统方式
function greet(person) {
    console.log(`Hello, ${person.name}!`);
    console.log(`You are ${person.age} years old.`);
}

// 使用解构
function greetDestructured({ name, age = 25 }) {
    console.log(`Hello, ${name}!`);
    console.log(`You are ${age} years old.`);
}

greetDestructured({ name: 'John', age: 30 });
greetDestructured({ name: 'Jane' }); // age 使用默认值 25

// 带整体默认值
function greetWithDefault({ name = 'Guest', age = 0 } = {}) {
    console.log(`Hello, ${name}! Age: ${age}`);
}

greetWithDefault(); // 不传参数也安全
```

### 数组参数解构

```javascript
function sum([a, b, c = 0]) {
    return a + b + c;
}

console.log(sum([1, 2]));    // 3
console.log(sum([1, 2, 3])); // 6
```

### 混合解构

```javascript
function process({ name, scores: [math, english] }) {
    console.log(`${name}: Math=${math}, English=${english}`);
}

process({ 
    name: 'John', 
    scores: [90, 85] 
});
// John: Math=90, English=85
```

## 4. 解构的高级用法

### 解构与展开运算符

```javascript
// 数组
const [first, ...others] = [1, 2, 3, 4, 5];
console.log(first);  // 1
console.log(others); // [2, 3, 4, 5]

// 对象
const { name, ...rest } = { name: 'John', age: 30, city: 'NYC' };
console.log(name); // John
console.log(rest); // { age: 30, city: 'NYC' }
```

### 解构与循环

```javascript
const people = [
    { name: 'John', age: 30 },
    { name: 'Jane', age: 25 },
    { name: 'Bob', age: 35 }
];

// 在 for...of 中解构
for (const { name, age } of people) {
    console.log(`${name} is ${age}`);
}

// 在 forEach 中解构
people.forEach(({ name, age }) => {
    console.log(`${name} - ${age}`);
});

// 解构数组项
const coordinates = [[1, 2], [3, 4], [5, 6]];

for (const [x, y] of coordinates) {
    console.log(`(${x}, ${y})`);
}
```

### 动态属性名解构

```javascript
const key = 'name';
const person = { name: 'John', age: 30 };

// 动态属性名解构
const { [key]: value } = person;
console.log(value); // John
```

### 解构与模块导入

```javascript
// 导入时解构
import { useState, useEffect } from 'react';

// 导入时重命名
import { useState as useStateHook } from 'react';
```

## 5. 实际应用场景

### 提取 JSON 数据

```javascript
const response = {
    data: {
        user: {
            id: 1,
            profile: {
                name: 'John',
                email: 'john@example.com'
            }
        },
        token: 'abc123'
    },
    status: 200
};

// 快速提取所需数据
const {
    data: {
        user: { 
            id, 
            profile: { name, email } 
        },
        token
    },
    status
} = response;

console.log(id, name, email, token, status);
```

### 函数配置选项

```javascript
function createRequest({
    url,
    method = 'GET',
    headers = {},
    body = null,
    timeout = 5000
}) {
    console.log(`${method} ${url}`);
    console.log('Headers:', headers);
    console.log('Body:', body);
    console.log('Timeout:', timeout);
}

// 使用
createRequest({
    url: '/api/users',
    method: 'POST',
    body: { name: 'John' }
});
```

### 状态管理

```javascript
// React 中常用的解构模式
function useUser() {
    const [user, setUser] = useState(null);
    const [loading, setLoading] = useState(false);
    const [error, setError] = useState(null);
    
    return { user, setUser, loading, setLoading, error, setError };
}

// 使用时解构
const { user, loading, error, setUser } = useUser();
```

### 批量声明变量

```javascript
// 不使用解构
const width = 100;
const height = 200;
const color = 'red';
const border = '1px solid black';

// 使用解构
const [width, height, color, border] = [100, 200, 'red', '1px solid black'];

// 或从对象解构
const { w: width, h: height, c: color, b: border } = config;
```

## 6. 解构的注意事项

### 已声明变量的对象解构

```javascript
let name, age;

// ❌ 错误写法（会被解析为代码块）
// { name, age } = { name: 'John', age: 30 };

// ✅ 正确写法（用括号包裹）
({ name, age } = { name: 'John', age: 30 });
console.log(name, age); // John 30
```

### 数组的特殊值处理

```javascript
const arr = [1, , 3]; // 稀疏数组

const [a, b, c] = arr;
console.log(a); // 1
console.log(b); // undefined
console.log(c); // 3

// 使用默认值
const [x = 0, y = 0, z = 0] = arr;
console.log(x, y, z); // 1 0 3
```

### 解构 null 和 undefined

```javascript
// ❌ 解构 null 或 undefined 会报错
// const { name } = null;    // TypeError
// const { name } = undefined; // TypeError

// ✅ 提供默认值
const { name } = null || {};
console.log(name); // undefined

// 函数参数默认值
function foo({ name } = {}) {
    console.log(name);
}

foo(null); // 会报错
foo();     // undefined（安全）
```

## 小结

| 解构类型 | 语法 | 特点 |
|---------|------|------|
| 数组解构 | `[a, b] = arr` | 按位置匹配 |
| 对象解构 | `{ a, b } = obj` | 按属性名匹配 |
| 默认值 | `a = default` | 匹配失败时使用 |
| 重命名 | `{ oldName: newName }` | 创建新变量名 |
| 剩余元素 | `...rest` | 收集剩余部分 |
| 嵌套解构 | `{ a: { b } }` | 深层提取值 |
