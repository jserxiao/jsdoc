## 定时器

### setTimeout

在指定延迟后执行一次。

```javascript
// 基本用法
const timeoutId = setTimeout(() => {
  console.log('Delayed execution');
}, 1000);

// 取消定时器
clearTimeout(timeoutId);

// 带参数的回调
setTimeout((message, count) => {
  console.log(message, count);
}, 1000, 'Hello', 42);

// 立即执行（实际上是 4ms）
setTimeout(() => {
  console.log('Almost immediate');
}, 0);

// 嵌套 setTimeout 实现延迟累积
function delayedSequence() {
  console.log('Start');

  setTimeout(() => {
    console.log('Step 1');

    setTimeout(() => {
      console.log('Step 2');

      setTimeout(() => {
        console.log('Step 3');
      }, 1000);
    }, 1000);
  }, 1000);
}
```

### setTimeout 的坑

```javascript
// 问题 1：最小延迟是 4ms
let start = Date.now();
setTimeout(() => {
  console.log(Date.now() - start);  // 实际延迟 >= 4ms，不是 0ms
}, 0);

// 问题 2：this 绑定丢失
const obj = {
  name: 'Object',
  greet() {
    setTimeout(function() {
      console.log(this.name);  // undefined（非严格模式是 window）
    }, 100);
  }
};
obj.greet();

// 解决：箭头函数或 bind
const obj2 = {
  name: 'Object',
  greet() {
    setTimeout(() => {
      console.log(this.name);  // 'Object'
    }, 100);
  }
};

// 问题 3：后台标签页延迟
// 浏览器会对后台标签页的定时器进行节流
// 最小延迟可能变成 1000ms
```

### setInterval

按照指定间隔重复执行。

```javascript
// 基本用法
let count = 0;
const intervalId = setInterval(() => {
  console.log(`Count: ${++count}`);
  if (count >= 5) {
    clearInterval(intervalId);
  }
}, 1000);

// 问题：实际间隔可能大于设定值
const start = Date.now();
setInterval(() => {
  console.log(Date.now() - start);
  // 如果回调执行时间超过间隔，实际间隔会变长
}, 100);
```

### setInterval 的问题

```javascript
// 问题：回调执行时间可能超过间隔
setInterval(() => {
  // 假设这个操作需要 200ms
  const start = Date.now();
  while (Date.now() - start < 200) {}

  console.log('Tick');
}, 100);
// 实际执行间隔：200ms（不是 100ms）

// 问题：错误不会停止定时器
setInterval(() => {
  throw new Error('Oops');  // 错误抛出，但定时器继续
}, 1000);
```

### 递归 setTimeout 替代 setInterval

```javascript
// 使用递归 setTimeout 实现精准间隔
function preciseInterval(callback, delay) {
  let expected = Date.now() + delay;
  let stopped = false;

  function step() {
    if (stopped) return;

    const drift = Date.now() - expected;
    callback();

    expected += delay;
    setTimeout(step, Math.max(0, delay - drift));
  }

  setTimeout(step, delay);

  return () => {
    stopped = true;
  };
}

// 使用
const stop = preciseInterval(() => {
  console.log('Tick at:', new Date().toISOString());
}, 1000);

// 停止
stop();
```

### 倒计时实现

```javascript
class Countdown {
  constructor(seconds, onTick, onComplete) {
    this.remaining = seconds;
    this.onTick = onTick;
    this.onComplete = onComplete;
    this.timerId = null;
  }

  start() {
    this.timerId = setInterval(() => {
      this.remaining--;
      this.onTick(this.remaining);

      if (this.remaining <= 0) {
        this.stop();
        this.onComplete();
      }
    }, 1000);
    return this;
  }

  stop() {
    if (this.timerId) {
      clearInterval(this.timerId);
      this.timerId = null;
    }
    return this;
  }

  reset(seconds) {
    this.stop();
    this.remaining = seconds;
    return this;
  }
}

// 使用
const countdown = new Countdown(
  10,
  remaining => console.log(`${remaining}s`),
  () => console.log('Done!')
);
countdown.start();
```

### 超时控制

```javascript
// 带超时的异步操作
function withTimeout(promise, ms) {
  return Promise.race([
    promise,
    new Promise((_, reject) =>
      setTimeout(() => reject(new Error('Timeout')), ms)
    )
  ]);
}

// 使用
try {
  const data = await withTimeout(
    fetch('/api/data'),
    5000  // 5秒超时
  );
} catch (error) {
  if (error.message === 'Timeout') {
    console.log('请求超时');
  }
}

// 带取消的超时
function withAbortableTimeout(promise, ms, signal) {
  return new Promise((resolve, reject) => {
    const timeoutId = setTimeout(() => {
      reject(new Error('Timeout'));
    }, ms);

    const abortHandler = () => {
      clearTimeout(timeoutId);
      reject(new Error('Aborted'));
    };

    signal.addEventListener('abort', abortHandler);

    promise
      .then(resolve)
      .catch(reject)
      .finally(() => {
        clearTimeout(timeoutId);
        signal.removeEventListener('abort', abortHandler);
      });
  });
}
```

## requestAnimationFrame

### 基本用法

`requestAnimationFrame`（rAF）用于在浏览器下一次重绘之前调用回调，适合动画。

```javascript
// 基本用法
let animationId = null;

function animate() {
  // 更新动画状态
  updateAnimation();

  // 继续下一帧
  animationId = requestAnimationFrame(animate);
}

// 开始动画
animationId = requestAnimationFrame(animate);

// 停止动画
cancelAnimationFrame(animationId);
```

### 时间戳参数

```javascript
// rAF 回调接收时间戳参数（DOMHighResTimeStamp）
let start = null;

function step(timestamp) {
  if (!start) start = timestamp;

  const progress = timestamp - start;
  console.log(`Elapsed: ${progress}ms`);

  if (progress < 2000) {
    requestAnimationFrame(step);
  }
}

requestAnimationFrame(step);
```

### 动画示例

```javascript
// 平滑移动元素
function animate(element, property, from, to, duration) {
  const start = performance.now();

  function step(timestamp) {
    const elapsed = timestamp - start;
    const progress = Math.min(elapsed / duration, 1);

    // 缓动函数（ease-out）
    const eased = 1 - Math.pow(1 - progress, 3);

    const current = from + (to - from) * eased;
    element.style[property] = current + 'px';

    if (progress < 1) {
      requestAnimationFrame(step);
    }
  }

  requestAnimationFrame(step);
}

// 使用
animate(document.getElementById('box'), 'left', 0, 500, 1000);
```

### rAF vs setTimeout

```javascript
// setTimeout 动画（不推荐）
function badAnimate() {
  // 问题：
  // 1. 间隔不精确
  // 2. 后台标签页继续运行，浪费资源
  // 3. 不与浏览器刷新率同步
  // 4. 可能导致丢帧或卡顿

  setTimeout(() => {
    updateAnimation();
    badAnimate();
  }, 16);  // ~60fps
}

// rAF 动画（推荐）
function goodAnimate() {
  // 优势：
  // 1. 与浏览器刷新率同步（通常 60fps）
  // 2. 后台标签页自动暂停
  // 3. 更精确的时间控制
  // 4. 更好的性能

  updateAnimation();
  requestAnimationFrame(goodAnimate);
}
```

### 帧率计算

```javascript
class FPSMonitor {
  constructor() {
    this.frames = [];
    this.lastTime = performance.now();
  }

  tick() {
    const now = performance.now();
    const delta = now - this.lastTime;
    this.lastTime = now;

    this.frames.push(delta);

    // 保留最近 60 帧
    if (this.frames.length > 60) {
      this.frames.shift();
    }

    requestAnimationFrame(() => this.tick());
  }

  getFPS() {
    if (this.frames.length === 0) return 0;

    const avgDelta = this.frames.reduce((a, b) => a + b) / this.frames.length;
    return Math.round(1000 / avgDelta);
  }

  start() {
    this.tick();
  }
}

// 使用
const monitor = new FPSMonitor();
monitor.start();

setInterval(() => {
  console.log(`FPS: ${monitor.getFPS()}`);
}, 1000);
```

### 节流动画

```javascript
// 限制动画帧率
function throttleRAF(callback, fps = 30) {
  const interval = 1000 / fps;
  let lastTime = 0;

  return function animate(timestamp) {
    const delta = timestamp - lastTime;

    if (delta >= interval) {
      lastTime = timestamp - (delta % interval);
      callback(timestamp);
    }

    requestAnimationFrame(animate);
  };
}

// 使用：限制为 30fps
const throttledAnimate = throttleRAF((timestamp) => {
  updateAnimation(timestamp);
}, 30);

requestAnimationFrame(throttledAnimate);
```

## queueMicrotask

### 基本用法

`queueMicrotask` 将回调添加到微任务队列。

```javascript
// 基本用法
queueMicrotask(() => {
  console.log('Microtask');
});

// 等同于
Promise.resolve().then(() => {
  console.log('Microtask');
});
```

### 使用场景

```javascript
// 场景 1：在当前任务完成后执行
function processData(data) {
  // 同步处理
  const processed = data.map(item => transform(item));

  // 异步通知，但不使用宏任务
  queueMicrotask(() => {
    notifyListeners(processed);
  });

  return processed;
}

// 场景 2：批量处理
class BatchProcessor {
  constructor() {
    this.queue = [];
    this.scheduled = false;
  }

  add(item) {
    this.queue.push(item);

    if (!this.scheduled) {
      this.scheduled = true;
      queueMicrotask(() => this.process());
    }
  }

  process() {
    const items = this.queue;
    this.queue = [];
    this.scheduled = false;

    // 批量处理
    console.log('Processing:', items);
  }
}

// 场景 3：确保 Promise 链完整
async function example() {
  console.log('1');

  queueMicrotask(() => {
    console.log('2');  // 在 await 后、setTimeout 前执行
  });

  await Promise.resolve();
  console.log('3');
}

example();
console.log('4');
// 输出：1, 4, 2, 3
```

### queueMicrotask vs setTimeout

```javascript
console.log('Script start');

queueMicrotask(() => {
  console.log('Microtask');

setTimeout(() => {
  console.log('setTimeout');
}, 0);

console.log('Script end');

// Script start
// Script end

// queueMicrotask 在当前任务后立即执行

```

## 异步队列

### 基本异步队列

```javascript
class AsyncQueue {
  constructor() {
    this.queue = [];
    this.running = false;
  }

  async add(task) {
    return new Promise((resolve, reject) => {
      this.queue.push({ task, resolve, reject });
      this.run();
    });
  }

  async run() {
    if (this.running) return;
    this.running = true;

    while (this.queue.length) {
      const { task, resolve, reject } = this.queue.shift();
      try {
        const result = await task();
        resolve(result);
      } catch (error) {
        reject(error);
      }
    }

    this.running = false;
  }

  clear() {
    this.queue = [];
  }

  get length() {
    return this.queue.length;
  }
}

// 使用
const queue = new AsyncQueue();

queue.add(() => fetch('/api/1')).then(console.log);
queue.add(() => fetch('/api/2')).then(console.log);
queue.add(() => fetch('/api/3')).then(console.log);
// 按顺序执行，每个任务完成后才开始下一个
```

### 带优先级的队列

```javascript
class PriorityAsyncQueue {
  constructor() {
    this.queue = [];
    this.running = false;
  }

  add(task, priority = 0) {
    return new Promise((resolve, reject) => {
      const item = { task, priority, resolve, reject };

      // 按优先级插入（优先级高的在前）
      let i = this.queue.length;
      while (i > 0 && this.queue[i - 1].priority < priority) {
        i--;
      }
      this.queue.splice(i, 0, item);

      this.run();
    });
  }

  async run() {
    if (this.running) return;
    this.running = true;

    while (this.queue.length) {
      const { task, resolve, reject } = this.queue.shift();
      try {
        const result = await task();
        resolve(result);
      } catch (error) {
        reject(error);
      }
    }

    this.running = false;
  }
}

// 使用
const priorityQueue = new PriorityAsyncQueue();

priorityQueue.add(() => log('Low priority task'), 0);
priorityQueue.add(() => log('High priority task'), 10);
priorityQueue.add(() => log('Medium priority task'), 5);
// 执行顺序：High, Medium, Low
```

### 带重试的队列

```javascript
class RetryQueue {
  constructor(maxRetries = 3, retryDelay = 1000) {
    this.queue = [];
    this.running = false;
    this.maxRetries = maxRetries;
    this.retryDelay = retryDelay;
  }

  add(task, retries = this.maxRetries) {
    return new Promise((resolve, reject) => {
      this.queue.push({ task, retries, resolve, reject });
      this.run();
    });
  }

  async run() {
    if (this.running) return;
    this.running = true;

    while (this.queue.length) {
      const item = this.queue.shift();
      const { task, retries, resolve, reject } = item;

      try {
        const result = await task();
        resolve(result);
      } catch (error) {
        if (retries > 0) {
          // 重试
          console.log(`Retrying... (${retries} attempts left)`);
          await this.delay(this.retryDelay);
          this.queue.unshift({ ...item, retries: retries - 1 });
        } else {
          reject(error);
        }
      }
    }

    this.running = false;
  }

  delay(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

## 并发控制

### Promise 并发限制

```javascript
class ConcurrencyPool {
  constructor(maxConcurrency) {
    this.maxConcurrency = maxConcurrency;
    this.running = 0;
    this.queue = [];
  }

  async run(task) {
    return new Promise((resolve, reject) => {
      const execute = async () => {
        this.running++;

        try {
          const result = await task();
          resolve(result);
        } catch (error) {
          reject(error);
        } finally {
          this.running--;
          this.next();
        }
      };

      if (this.running < this.maxConcurrency) {
        execute();
      } else {
        this.queue.push(execute);
      }
    });
  }

  next() {
    if (this.queue.length > 0 && this.running < this.maxConcurrency) {
      const execute = this.queue.shift();
      execute();
    }
  }
}

// 使用
const pool = new ConcurrencyPool(3);  // 最多 3 个并发

const urls = ['/api/1', '/api/2', '/api/3', '/api/4', '/api/5'];

const promises = urls.map(url =>
  pool.run(() => fetch(url))
);

const results = await Promise.all(promises);
```

### 并发工具函数

```javascript
// 限制并发数
async function limitConcurrency(tasks, limit) {
  const results = [];
  const executing = [];

  for (const task of tasks) {
    const promise = Promise.resolve().then(() => task());
    results.push(promise);

    if (limit <= tasks.length) {
      const exec = promise.then(() => {
        executing.splice(executing.indexOf(exec), 1);
      });
      executing.push(exec);

      if (executing.length >= limit) {
        await Promise.race(executing);
      }
    }
  }

  return Promise.all(results);
}

// 使用
const tasks = urls.map(url => () => fetch(url));
const results = await limitConcurrency(tasks, 3);
```

### p-limit 风格实现

```javascript
function pLimit(concurrency) {
  const queue = [];
  let activeCount = 0;

  const next = () => {
    activeCount--;

    if (queue.length > 0) {
      queue.shift()();
    }
  };

  const run = async (fn, resolve, reject) => {
    activeCount++;

    try {
      const result = await fn();
      resolve(result);
    } catch (error) {
      reject(error);
    }

    next();
  };

  const enqueue = (fn, resolve, reject) => {
    queue.push(run.bind(null, fn, resolve, reject));

    (async () => {
      await Promise.resolve();

      if (activeCount < concurrency && queue.length > 0) {
        queue.shift()();
      }
    })();
  };

  const generator = (fn) => new Promise((resolve, reject) => {
    enqueue(fn, resolve, reject);
  });

  Object.defineProperties(generator, {
    activeCount: {
      get: () => activeCount
    },
    pendingCount: {
      get: () => queue.length
    }
  });

  return generator;
}

// 使用
const limit = pLimit(2);

const tasks = [
  limit(() => fetch('/api/1')),
  limit(() => fetch('/api/2')),
  limit(() => fetch('/api/3')),
  limit(() => fetch('/api/4'))
];

const results = await Promise.all(tasks);
```

### 重试机制

```javascript
async function retry(fn, options = {}) {
  const {
    maxRetries = 3,
    delay = 1000,
    backoff = 2,
    shouldRetry = () => true
  } = options;

  let lastError;

  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error;

      if (attempt < maxRetries && shouldRetry(error)) {
        const waitTime = delay * Math.pow(backoff, attempt);
        await new Promise(resolve => setTimeout(resolve, waitTime));
      }
    }
  }

  throw lastError;
}

// 使用
const data = await retry(
  () => fetch('/api/data').then(r => r.json()),
  {
    maxRetries: 3,
    delay: 1000,
    shouldRetry: error => error.status >= 500  // 只重试服务器错误
  }
);
```

### 轮询

```javascript
async function poll(fn, options = {}) {
  const {
    interval = 1000,
    maxAttempts = Infinity,
    shouldContinue = () => true,
    onSuccess,
    onFailure
  } = options;

  let attempts = 0;

  while (attempts < maxAttempts) {
    attempts++;

    try {
      const result = await fn();

      if (!shouldContinue(result)) {
        onSuccess?.(result);
        return result;
      }
    } catch (error) {
      onFailure?.(error, attempts);
    }

    await new Promise(resolve => setTimeout(resolve, interval));
  }

  throw new Error('Max attempts reached');
}

// 使用：等待任务完成
const result = await poll(
  () => fetch('/api/task/123').then(r => r.json()),
  {
    interval: 2000,
    shouldContinue: data => data.status !== 'completed',
    maxAttempts: 30
  }
);
```

function debounce(fn, delay) {
  let timeoutId;

  return function (...args) {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => fn.apply(this, args), delay);
  };
}

### 节流滚动

```javascript
function throttle(fn, delay) {
  let lastTime = 0;
  let timeoutId;

  return function (...args) {
    const now = Date.now();
    const remaining = delay - (now - lastTime);

    if (remaining <= 0) {
      lastTime = now;
      fn.apply(this, args);
    } else {
      clearTimeout(timeoutId);
      timeoutId = setTimeout(() => {
        lastTime = Date.now();
        fn.apply(this, args);
      }, remaining);
    }
  };
}

// 使用
const throttledScroll = throttle(() => {
  console.log('Scroll position:', window.scrollY);
}, 100);

window.addEventListener('scroll', throttledScroll);
```

### 无限滚动加载

```javascript
class InfiniteScroll {
  constructor(options) {
    this.loader = options.loader;
    this.onLoad = options.onLoad;
    this.threshold = options.threshold || 200;
    this.loading = false;
    this.hasMore = true;

    this.init();
  }

  init() {
    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting && !this.loading && this.hasMore) {
          this.load();
        }
      },
      { rootMargin: `${this.threshold}px` }
    );

    observer.observe(this.loader);
  }

  async load() {
    this.loading = true;
    this.loader.classList.add('loading');

    try {
      const hasMore = await this.onLoad();
      this.hasMore = hasMore;
    } catch (error) {
      console.error('Load error:', error);
    } finally {
      this.loading = false;
      this.loader.classList.remove('loading');
    }
  }
}

// 使用
const infiniteScroll = new InfiniteScroll({
  loader: document.getElementById('loader'),
  threshold: 200,
  onLoad: async () => {
    const items = await fetchMoreItems();
    renderItems(items);
    return items.length > 0;
  }
});
```

    const formData = new FormData();

    formData.append('file', file);

    xhr.upload.addEventListener('progress', (e) => {
      if (e.lengthComputable) {
        const percent = Math.round((e.loaded / e.total) * 100);
        onProgress?.(percent);
      }
    });

    xhr.addEventListener('load', () => {
      if (xhr.status >= 200 && xhr.status < 300) {
        const response = JSON.parse(xhr.responseText);
        onSuccess?.(response);
        resolve(response);
      } else {
        const error = new Error('Upload failed');
        onError?.(error);
        reject(error);
      }
    });

    xhr.addEventListener('error', () => {
      const error = new Error('Network error');
      onError?.(error);
      reject(error);
    });

    xhr.open('POST', url);
    xhr.send(formData);
  });
}

// 使用
await uploadFile(file, {
  url: '/api/upload',
  onProgress: (percent) => {
    console.log(`Upload progress: ${percent}%`);
  },
  onSuccess: (response) => {
    console.log('Upload success:', response);
  }
});
