# 语句与控制流

> 语句是 JavaScript 程序的基本执行单位，控制流决定了代码的执行顺序。掌握各种语句和控制结构是编写复杂程序的基础。

## 学习要点

- 📝 掌握条件语句的使用
- 🔄 理解循环语句的特点
- 🏷️ 了解标签语句的用法
- ⚡ 学会 break 和 continue 的正确使用

---

## 1. 条件语句

### if 语句

```javascript
// 基本 if 语句
let age = 18;

if (age >= 18) {
    console.log('成年人');
}

// if-else 语句
if (age >= 18) {
    console.log('成年人');
} else {
    console.log('未成年人');
}

// if-else if-else 语句
let score = 85;

if (score >= 90) {
    console.log('优秀');
} else if (score >= 80) {
    console.log('良好');
} else if (score >= 60) {
    console.log('及格');
} else {
    console.log('不及格');
}
```

### if 语句的最佳实践

```javascript
// ✅ 推荐：使用块语句，即使只有一行代码
if (condition) {
    doSomething();
}

// ❌ 不推荐：省略花括号（容易出错）
if (condition)
    doSomething();
    doAnotherThing(); // 这行不在 if 块内！

// ✅ 推荐：条件表达式简写
let message = age >= 18 ? '成年人' : '未成年人';

// ✅ 推荐：多重条件使用逻辑运算符
if (user && user.isLoggedIn && user.hasPermission) {
    // 安全访问
}
```

### switch 语句

```javascript
let day = 3;
let dayName;

switch (day) {
    case 1:
        dayName = '星期一';
        break;
    case 2:
        dayName = '星期二';
        break;
    case 3:
        dayName = '星期三';
        break;
    case 4:
        dayName = '星期四';
        break;
    case 5:
        dayName = '星期五';
        break;
    case 6:
    case 7:
        dayName = '周末';
        break;
    default:
        dayName = '无效日期';
}

console.log(dayName); // '星期三'
```

### switch 的特点

```javascript
// switch 使用严格相等（===）比较
let x = '10';

switch (x) {
    case 10:
        console.log('数字 10');
        break;
    case '10':
        console.log('字符串 10'); // ✅ 匹配这个
        break;
}

// 穿透现象（fall-through）
let num = 1;

switch (num) {
    case 1:
        console.log('一');
        // 没有 break，会继续执行下一个 case
    case 2:
        console.log('二');
        break;
    default:
        console.log('其他');
}
// 输出：一 二

// 利用穿透实现多条件匹配
let grade = 'B';

switch (grade) {
    case 'A':
    case 'B':
    case 'C':
        console.log('通过');
        break;
    case 'D':
    case 'F':
        console.log('不通过');
        break;
}
```

### switch 表达式（ES2023）

```javascript
// ES2023 支持更简洁的写法
const getDayType = (day) => {
    switch (day) {
        case 1:
        case 2:
        case 3:
        case 4:
        case 5:
            return '工作日';
        case 6:
        case 7:
            return '周末';
        default:
            return '无效';
    }
};
```

## 2. 循环语句

### while 循环

```javascript
// 基本语法
while (condition) {
    // 循环体
    // 必须有改变条件的代码，否则会无限循环
}

// 示例
let i = 0;
while (i < 5) {
    console.log(i);
    i++;
}
// 输出：0, 1, 2, 3, 4

// 使用场景：不确定循环次数
let sum = 0;
while (sum < 100) {
    sum += Math.floor(Math.random() * 10);
}
console.log(sum);
```

### do-while 循环

```javascript
// 至少执行一次
let input;

do {
    input = prompt('请输入一个大于 10 的数字');
    input = Number(input);
} while (input <= 10 || isNaN(input));

console.log('你输入的数字是：', input);

// 常见用途：菜单选择
let choice;
do {
    console.log('1. 新游戏');
    console.log('2. 继续游戏');
    console.log('3. 退出');
    choice = getMenuChoice();
} while (choice !== 3);
```

### for 循环

```javascript
// 基本语法
for (初始化; 条件; 迭代) {
    // 循环体
}

// 标准用法
for (let i = 0; i < 5; i++) {
    console.log(i);
}

// 遍历数组
const arr = ['a', 'b', 'c'];
for (let i = 0; i < arr.length; i++) {
    console.log(arr[i]);
}

// 优化：缓存长度
for (let i = 0, len = arr.length; i < len; i++) {
    console.log(arr[i]);
}

// 倒序遍历
for (let i = arr.length - 1; i >= 0; i--) {
    console.log(arr[i]);
}

// 省略部分表达式
let j = 0;
for (; j < 5; ) {
    console.log(j);
    j++;
}

// 无限循环（需要用 break 退出）
for (;;) {
    // 无限循环
    if (shouldBreak) break;
}
```

### for-in 循环

```javascript
// 遍历对象的可枚举属性
const obj = { a: 1, b: 2, c: 3 };

for (const key in obj) {
    console.log(key, obj[key]);
}
// 输出：a 1, b 2, c 3

// ⚠️ 遍历数组不推荐用 for-in
const arr = [10, 20, 30];
for (const index in arr) {
    console.log(index, arr[index]); // index 是字符串 '0', '1', '2'
}

// for-in 会遍历原型链上的属性
function Person(name) {
    this.name = name;
}
Person.prototype.age = 25;

const person = new Person('John');

for (const key in person) {
    console.log(key); // name, age
}

// 只遍历自身属性
for (const key in person) {
    if (person.hasOwnProperty(key)) {
        console.log(key); // 只有 name
    }
}
```

### for-of 循环（ES6）

```javascript
// 遍历可迭代对象（Array, String, Map, Set, arguments 等）
const arr = ['a', 'b', 'c'];

for (const value of arr) {
    console.log(value); // a, b, c
}

// 遍历字符串
for (const char of 'hello') {
    console.log(char); // h, e, l, l, o
}

// 遍历 Map
const map = new Map([['a', 1], ['b', 2]]);
for (const [key, value] of map) {
    console.log(key, value);
}

// 遍历 Set
const set = new Set([1, 2, 3]);
for (const value of set) {
    console.log(value);
}

// 获取索引（使用 entries()）
for (const [index, value] of arr.entries()) {
    console.log(index, value);
}

// for-of 与 for-in 的区别
const arr2 = [10, 20, 30];
arr2.foo = 'bar';

for (const i in arr2) {
    console.log(i); // '0', '1', '2', 'foo'
}

for (const v of arr2) {
    console.log(v); // 10, 20, 30
}
```

## 3. 跳转语句

### break 语句

```javascript
// 跳出循环
for (let i = 0; i < 10; i++) {
    if (i === 5) {
        break; // 循环在 i=5 时终止
    }
    console.log(i);
}
// 输出：0, 1, 2, 3, 4

// 查找元素时使用 break
const arr = [1, 2, 3, 4, 5];
let found = -1;

for (let i = 0; i < arr.length; i++) {
    if (arr[i] === 3) {
        found = i;
        break; // 找到后立即退出
    }
}

// switch 中的 break
let x = 1;
switch (x) {
    case 1:
        console.log('一');
        break; // 必须有 break
    case 2:
        console.log('二');
        break;
}
```

### continue 语句

```javascript
// 跳过当前迭代，继续下一次
for (let i = 0; i < 10; i++) {
    if (i % 2 === 0) {
        continue; // 跳过偶数
    }
    console.log(i);
}
// 输出：1, 3, 5, 7, 9

// 过滤数组元素
const arr = [1, -2, 3, -4, 5];
const positives = [];

for (const num of arr) {
    if (num < 0) {
        continue;
    }
    positives.push(num);
}
console.log(positives); // [1, 3, 5]

// while 中的 continue
let i = 0;
while (i < 5) {
    i++;
    if (i === 3) {
        continue;
    }
    console.log(i);
}
// 输出：1, 2, 4, 5
```

### return 语句

```javascript
// 函数返回值
function add(a, b) {
    return a + b; // 返回结果并退出函数
}

// 提前返回
function greet(name) {
    if (!name) {
        return; // 提前退出（相当于 return undefined）
    }
    console.log('Hello, ' + name);
}

// 使用 return 代替复杂的条件嵌套
function processUser(user) {
    // 提前返回模式（guard clauses）
    if (!user) return null;
    if (!user.isActive) return '用户未激活';
    if (user.isBanned) return '用户已被封禁';
    
    // 正常处理
    return processActiveUser(user);
}
```

## 4. 标签语句

### 标签的定义与使用

```javascript
// 语法：labelName: statement

outerLoop: for (let i = 0; i < 3; i++) {
    for (let j = 0; j < 3; j++) {
        if (i === 1 && j === 1) {
            break outerLoop; // 跳出外层循环
        }
        console.log(`i=${i}, j=${j}`);
    }
}
// 输出：i=0, j=0 / i=0, j=1 / i=0, j=2 / i=1, j=0

// 使用 continue 跳到标签位置
outerLoop: for (let i = 0; i < 3; i++) {
    for (let j = 0; j < 3; j++) {
        if (i === 1 && j === 1) {
            continue outerLoop; // 跳到外层循环的下一次迭代
        }
        console.log(`i=${i}, j=${j}`);
    }
}
// 输出：i=0, j=0 / i=0, j=1 / i=0, j=2 / i=1, j=0 / i=2, j=0 / i=2, j=1 / i=2, j=2
```

### 标签的实际应用

```javascript
// 在二维数组中查找元素
const matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
];

function findValue(matrix, target) {
    let result = null;
    
    search: for (let i = 0; i < matrix.length; i++) {
        for (let j = 0; j < matrix[i].length; j++) {
            if (matrix[i][j] === target) {
                result = { row: i, col: j, value: target };
                break search;
            }
        }
    }
    
    return result;
}

findValue(matrix, 5); // { row: 1, col: 1, value: 5 }
```

### 标签语句的注意事项

```javascript
// ❌ 标签不能跨函数使用
label: {
    function inner() {
        break label; // SyntaxError!
    }
    inner();
}

// ✅ 标签只能在同一作用域内使用
label: {
    if (condition) {
        break label; // 正确
    }
}

// 标签可以用于任何语句
myLabel: {
    console.log('开始');
    break myLabel;
    console.log('这行不会执行');
}
console.log('结束');
```

## 5. 异常处理语句

### try-catch-finally

```javascript
// 基本语法
try {
    // 可能出错的代码
} catch (error) {
    // 处理错误
} finally {
    // 无论是否出错都会执行
}

// 示例
function divide(a, b) {
    try {
        if (b === 0) {
            throw new Error('除数不能为 0');
        }
        return a / b;
    } catch (error) {
        console.error('错误：', error.message);
        return null;
    } finally {
        console.log('计算完成');
    }
}

divide(10, 2); // 输出：计算完成，返回 5
divide(10, 0); // 输出：错误：除数不能为 0，计算完成，返回 null
```

### throw 语句

```javascript
// 抛出错误
throw new Error('出错了');

// 抛出其他类型的值
throw '出错了'; // 字符串
throw { code: 500, message: '服务器错误' }; // 对象

// 自定义错误类型
class ValidationError extends Error {
    constructor(message, field) {
        super(message);
        this.name = 'ValidationError';
        this.field = field;
    }
}

function validateUser(user) {
    if (!user.name) {
        throw new ValidationError('用户名不能为空', 'name');
    }
    if (!user.email) {
        throw new ValidationError('邮箱不能为空', 'email');
    }
}

// 使用 try-catch 处理
try {
    validateUser({ name: '', email: 'test@example.com' });
} catch (error) {
    if (error instanceof ValidationError) {
        console.log(`字段 ${error.field} 验证失败：${error.message}`);
    }
}
```

## 6. 其他语句

### with 语句（不推荐）

```javascript
// with 语句：扩展作用域链（严格模式下禁止使用）
const obj = { a: 1, b: 2, c: 3 };

// 不使用 with
console.log(obj.a, obj.b, obj.c);

// 使用 with（不推荐）
with (obj) {
    console.log(a, b, c);
}

// ⚠️ with 的问题：
// 1. 代码可读性差
// 2. 性能问题
// 3. 严格模式下会报错
// 4. 可能导致意外的变量引用
```

### debugger 语句

```javascript
// 用于调试，在开发者工具中设置断点
function complexCalculation(x) {
    let result = x * 2;
    
    debugger; // 代码会在此处暂停（如果开发者工具打开）
    
    result += 10;
    return result;
}

// 常用于调试复杂逻辑
function debugExample() {
    let data = fetchData();
    
    if (data.someCondition) {
        debugger; // 在这里检查 data 的值
        processData(data);
    }
}
```

### 空语句

```javascript
// 空语句：只有一个分号
;

// 用途：循环体为空时
let i = 0;
while (i++ < 10); // 空循环体，只做递增

// 或者
for (let i = 0; i < arr.length; arr[i++] = 0); // 初始化数组为 0

// ⚠️ 容易造成困惑，建议添加注释
while (condition) {
    /* 空循环体 */
}
```

## 7. 语句对比与选择

| 语句类型 | 使用场景 | 特点 |
|---------|---------|------|
| `if-else` | 条件判断，少量分支 | 灵活，条件可以是任意表达式 |
| `switch` | 多分支，等值匹配 | 更清晰，适合固定值匹配 |
| `while` | 不确定循环次数 | 先判断后执行 |
| `do-while` | 至少执行一次 | 先执行后判断 |
| `for` | 确定循环次数 | 初始化、条件、迭代集中管理 |
| `for-in` | 遍历对象属性 | 遍历可枚举属性（含原型链） |
| `for-of` | 遍历可迭代对象 | ES6 推荐，获取值而非索引 |
| `forEach` | 数组遍历 | 数组方法，无法 break |

## 小结

| 语句 | 核心要点 |
|------|----------|
| **if-else** | 条件判断，使用块语句 |
| **switch** | 多分支，注意 break 穿透 |
| **while** | 先判断后执行 |
| **do-while** | 至少执行一次 |
| **for** | 确定次数的循环 |
| **for-in** | 遍历对象属性（字符串键） |
| **for-of** | 遍历可迭代对象的值（推荐） |
| **break** | 跳出循环或 switch |
| **continue** | 跳过当前迭代 |
| **标签** | 配合 break/continue 跳出嵌套循环 |
