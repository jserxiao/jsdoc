# 字符串基础

## 字符串创建方式

### 字符串字面量
```javascript
let str1 = 'Hello';       // 单引号
let str2 = "World";       // 双引号
let str3 = `Hello World`; // 反引号（模板字符串）
```

### String 构造函数
```javascript
let str4 = String('Hello');     // 返回原始字符串
let str5 = new String('Hello'); // 返回 String 对象

console.log(typeof str4); // 'string'
console.log(typeof str5); // 'object'

// 不推荐使用 new String()
console.log(str4 == str5); // true
console.log(str4 === str5); // false
```

## 模板字符串（Template Literals）

### 基本用法
```javascript
const name = 'John';
const age = 30;

// 变量插值
const greeting = `Hello, I'm ${name} and I'm ${age} years old.`;
console.log(greeting); // "Hello, I'm John and I'm 30 years old."

// 表达式计算
const price = 100;
const quantity = 3;
const total = `Total: $${price * quantity}`;
console.log(total); // "Total: $300"
```

### 多行字符串
```javascript
const html = `
<div class="card">
    <h2>${name}</h2>
    <p>Age: ${age}</p>
</div>
`;
console.log(html);
```

### 嵌套模板
```javascript
const isLargeScreen = true;
const item = {
    isCollapsed: false
};

const classes = `header ${
    isLargeScreen ? '' : `icon-${item.isCollapsed ? 'expander' : 'collapser'}`
}`;
```

## 标签模板（Tagged Templates）

```javascript
function highlight(strings, ...values) {
    let result = '';
    strings.forEach((str, i) => {
        result += str;
        if (i < values.length) {
            result += `<mark>${values[i]}</mark>`;
        }
    });
    return result;
}

const name = 'John';
const age = 30;
const output = highlight`Name: ${name}, Age: ${age}`;
console.log(output); // "Name: <mark>John</mark>, Age: <mark>30</mark>"

// 实际应用：防止 XSS 攻击
function safeHtml(strings, ...values) {
    let result = '';
    strings.forEach((str, i) => {
        result += str;
        if (i < values.length) {
            result += String(values[i])
                .replace(/&/g, '&amp;')
                .replace(/</g, '&lt;')
                .replace(/>/g, '&gt;');
        }
    });
    return result;
}
```

### 原始字符串
```javascript
function tag(strings, ...values) {
    console.log(strings.raw[0]); // 包含转义字符的原始字符串
}

tag`Line 1\nLine 2`;
// strings.raw[0] 是 "Line 1\\nLine 2"

// String.raw 方法
const raw = String.raw`Line 1\nLine 2`;
console.log(raw); // "Line 1\nLine 2"（不转义）
```

## 字符串不可变性

```javascript
let str = 'Hello';
str[0] = 'h'; // 无效，不会改变字符串
console.log(str); // 'Hello'

// "修改"字符串实际上是创建新字符串
str = str.toLowerCase();
console.log(str); // 'hello'

// 字符串拼接也是创建新字符串
let a = 'Hello';
let b = a;
a = a + ' World';
console.log(a); // 'Hello World'
console.log(b); // 'Hello'
```

## 字符串遍历

### for...of 循环（推荐）
```javascript
const str = 'Hello';
for (const char of str) {
    console.log(char);
}
// H
// e
// l
// l
// o
```

### for 循环
```javascript
const str = 'Hello';
for (let i = 0; i < str.length; i++) {
    console.log(str[i]); // 或 str.charAt(i)
}
```

### 转换为数组
```javascript
const str = 'Hello';

// spread 运算符
const arr1 = [...str]; // ['H', 'e', 'l', 'l', 'o']

// Array.from
const arr2 = Array.from(str);

// split
const arr3 = str.split('');
```

## 字符串长度与索引

```javascript
const str = 'Hello';

// 长度
console.log(str.length); // 5

// 索引访问
console.log(str[0]);     // 'H'
console.log(str[4]);     // 'o'
console.log(str[5]);     // undefined
console.log(str[-1]);    // undefined
console.log(str.charAt(0)); // 'H'
console.log(str.charAt(5)); // ''（空字符串）

// at() 方法（ES2022）支持负索引
console.log(str.at(0));  // 'H'
console.log(str.at(-1)); // 'o'（最后一个字符）
```

## Unicode 与字符编码

### 字符编码
```javascript
const str = 'A';

// 获取字符编码
console.log(str.charCodeAt(0)); // 65
console.log(str.codePointAt(0)); // 65

// Unicode 字符
const heart = '❤';
console.log(heart.length); // 1

// 代理对字符（如 emoji）
const emoji = '😀';
console.log(emoji.length); // 2（代理对）
console.log([...emoji].length); // 1

// codePointAt 正确处理代理对
console.log(emoji.codePointAt(0)); // 128512
```

### 从编码创建字符串
```javascript
// fromCharCode（不支持代理对）
console.log(String.fromCharCode(65)); // 'A'

// fromCodePoint（支持代理对，ES6）
console.log(String.fromCodePoint(128512)); // '😀'
console.log(String.fromCodePoint(0x1F600)); // '😀'
```

### Unicode 转义
```javascript
// \uXXXX 形式
console.log('\u0041'); // 'A'

// \u{XXXXXX} 形式（ES6，支持更多字符）
console.log('\u{1F600}'); // '😀'
```
