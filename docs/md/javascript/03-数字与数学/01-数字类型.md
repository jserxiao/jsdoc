# 数字类型

## 数字表示法

### 整数与浮点数
```javascript
// 整数
let int = 42;

// 浮点数
let float = 3.14;
let float2 = .5;      // 0.5（省略前导零）
let float3 = 5.;      // 5（省略小数部分）

// JavaScript 使用 IEEE 754 双精度浮点数
// 所有数字都是浮点数（没有真正的整数类型）
console.log(1 === 1.0); // true
```

### 科学计数法
```javascript
let a = 2.5e6;   // 2,500,000（2.5 × 10^6）
let b = 2.5e-6;  // 0.0000025（2.5 × 10^-6）
let c = 1e3;     // 1000
```

### 不同进制表示
```javascript
// 二进制（ES6）0b 或 0B
console.log(0b10);   // 2
console.log(0b1111); // 15

// 八进制（ES6）0o 或 0O
console.log(0o10);   // 8
console.log(0o777);  // 511

// 十六进制 0x 或 0X
console.log(0x10);   // 16
console.log(0xFF);   // 255
console.log(0x1A);   // 26
```

### 数字分隔符（ES2021）
```javascript
// 使用 _ 分隔数字，提高可读性
let billion = 1_000_000_000;
let bytes = 0xFF_FF_FF_FF;
let bits = 0b1010_0001_1000_0101;
let fractions = 0.000_001;

console.log(billion); // 1000000000
```

## 特殊数值

### NaN（Not a Number）
```javascript
// 产生 NaN 的情况
console.log(Number('abc'));     // NaN
console.log(0 / 0);             // NaN
console.log(Math.sqrt(-1));     // NaN
console.log(NaN + 1);           // NaN
console.log(NaN === NaN);       // false！（NaN 不等于自身）

// 正确检测 NaN
console.log(Number.isNaN(NaN));       // true
console.log(Number.isNaN('hello'));   // false
console.log(isNaN('hello'));          // true（先转换，不推荐）

// Number.isNaN 实现
Number.isNaN = Number.isNaN || function(value) {
    return typeof value === 'number' && value !== value;
};
```

### Infinity
```javascript
// 正无穷和负无穷
console.log(1 / 0);   // Infinity
console.log(-1 / 0);  // -Infinity
console.log(Infinity); // Infinity
console.log(-Infinity); // -Infinity

// Infinity 运算
console.log(Infinity + 1);     // Infinity
console.log(Infinity * 2);     // Infinity
console.log(Infinity / Infinity); // NaN
console.log(1 / Infinity);     // 0

// 检测 Infinity
console.log(Number.isFinite(Infinity)); // false
console.log(Number.isFinite(123));      // true
console.log(isFinite(Infinity));        // false
```

### Number 常量
```javascript
// 最大/最小数值
console.log(Number.MAX_VALUE);           // 1.7976931348623157e+308
console.log(Number.MIN_VALUE);           // 5e-324（最小正数）

// 安全整数范围
console.log(Number.MAX_SAFE_INTEGER);    // 9007199254740991（2^53 - 1）
console.log(Number.MIN_SAFE_INTEGER);    // -9007199254740991

// 精度限制
console.log(Number.EPSILON);             // 2.220446049250313e-16

// 检查是否在安全整数范围内
console.log(Number.isSafeInteger(9007199254740991)); // true
console.log(Number.isSafeInteger(9007199254740992)); // false
```

### 精度问题
```javascript
// 经典的精度问题
console.log(0.1 + 0.2);        // 0.30000000000000004
console.log(0.1 + 0.2 === 0.3); // false

// 解决方法：使用 EPSILON
function isEqual(a, b) {
    return Math.abs(a - b) < Number.EPSILON;
}
console.log(isEqual(0.1 + 0.2, 0.3)); // true

// 或使用 toFixed 后转数字
console.log(Number((0.1 + 0.2).toFixed(10)) === 0.3); // true

// 整数运算更精确
console.log((1 + 2) / 10); // 0.3（先放大再缩小）
```

## 数字方法

### parseInt() / parseFloat()
```javascript
// parseInt - 解析整数
console.log(parseInt('42'));        // 42
console.log(parseInt('42px'));      // 42
console.log(parseInt('3.14'));      // 3
console.log(parseInt('  42  '));    // 42（忽略空格）
console.log(parseInt('abc'));       // NaN

// 指定进制
console.log(parseInt('10', 2));     // 2（二进制）
console.log(parseInt('10', 8));     // 8（八进制）
console.log(parseInt('10', 16));    // 16（十六进制）

// parseFloat - 解析浮点数
console.log(parseFloat('3.14'));    // 3.14
console.log(parseFloat('3.14abc')); // 3.14
console.log(parseFloat('.5'));      // 0.5
console.log(parseFloat('3.14.15')); // 3.14
```

### isNaN() / Number.isNaN()
```javascript
// 全局 isNaN（会先转换类型）
console.log(isNaN('hello'));     // true（不推荐）
console.log(isNaN(undefined));   // true（不推荐）

// Number.isNaN（不会转换类型）
console.log(Number.isNaN('hello'));    // false
console.log(Number.isNaN(NaN));        // true（推荐）
```

### isFinite() / Number.isFinite()
```javascript
// 全局 isFinite（会先转换类型）
console.log(isFinite('42'));     // true
console.log(isFinite(Infinity)); // false

// Number.isFinite（不会转换类型）
console.log(Number.isFinite('42'));     // false
console.log(Number.isFinite(42));       // true
console.log(Number.isFinite(Infinity)); // false
```

### Number.isInteger()
```javascript
console.log(Number.isInteger(42));       // true
console.log(Number.isInteger(42.0));     // true（42.0 === 42）
console.log(Number.isInteger(42.5));     // false
console.log(Number.isInteger('42'));     // false
console.log(Number.isInteger(null));     // false
```

### Number.isSafeInteger()
```javascript
console.log(Number.isSafeInteger(1));                    // true
console.log(Number.isSafeInteger(Math.pow(2, 53)));      // false
console.log(Number.isSafeInteger(Math.pow(2, 53) - 1));  // true
```

### toFixed()
```javascript
const num = 123.456789;

console.log(num.toFixed(2));   // '123.46'（四舍五入）
console.log(num.toFixed(0));   // '123'
console.log(num.toFixed(10));  // '123.4567890000'

const num2 = 123;
console.log(num2.toFixed(2));  // '123.00'

// 注意：返回字符串
console.log(typeof num.toFixed(2)); // 'string'
```

### toPrecision()
```javascript
const num = 123.456;

// 指定有效数字位数
console.log(num.toPrecision(3));  // '123'
console.log(num.toPrecision(4));  // '123.5'
console.log(num.toPrecision(6));  // '123.456'
console.log(num.toPrecision(2));  // '1.2e+2'

// 注意：返回字符串
console.log(typeof num.toPrecision(3)); // 'string'
```

### toString(radix)
```javascript
const num = 255;

// 转换为不同进制字符串
console.log(num.toString(2));   // '11111111'（二进制）
console.log(num.toString(8));   // '377'（八进制）
console.log(num.toString(16));  // 'ff'（十六进制）
console.log(num.toString(10));  // '255'（十进制）
console.log(num.toString());    // '255'（默认十进制）

// 浮点数
console.log((3.14).toString()); // '3.14'
```
