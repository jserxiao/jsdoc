## Performance API

### performance.now()

高精度时间测量，精度可达微秒级。

```javascript
// 基本使用
const start = performance.now();

// 执行需要测量的代码
for (let i = 0; i < 1000000; i++) {
  // ...
}

const end = performance.now();
console.log(`执行时间: ${(end - start).toFixed(2)}ms`);

// 封装测量函数
function measureTime(name, fn) {
  const start = performance.now();
  const result = fn();
  const end = performance.now();
  console.log(`${name}: ${(end - start).toFixed(2)}ms`);
  return result;
}

// 使用
measureTime('数组排序', () => {
  const arr = Array.from({ length: 100000 }, () => Math.random());
  arr.sort((a, b) => a - b);
});

// 异步函数测量
async function measureAsync(name, fn) {
  const start = performance.now();
  const result = await fn();
  const end = performance.now();
  console.log(`${name}: ${(end - start).toFixed(2)}ms`);
  return result;
}
```

### performance.mark() 和 measure()

创建命名时间戳和测量区间。

```javascript
// 创建标记
performance.mark('task-start');

// 执行任务
doHeavyTask();

// 创建结束标记
performance.mark('task-end');

// 创建测量
performance.measure('task-duration', 'task-start', 'task-end');

// 获取测量结果
const measures = performance.getEntriesByName('task-duration');
console.log(`任务耗时: ${measures[0].duration}ms`);

// 清除标记和测量
performance.clearMarks();
performance.clearMeasures();

// 完整示例：测量多个阶段
function measureWorkflow() {
  performance.mark('workflow-start');

  // 阶段1
  performance.mark('phase1-start');
  fetchData();
  performance.mark('phase1-end');
  performance.measure('phase1', 'phase1-start', 'phase1-end');

  // 阶段2
  performance.mark('phase2-start');
  processData();
  performance.mark('phase2-end');
  performance.measure('phase2', 'phase2-start', 'phase2-end');

  // 阶段3
  performance.mark('phase3-start');
  renderUI();
  performance.mark('phase3-end');
  performance.measure('phase3', 'phase3-start', 'phase3-end');

  performance.mark('workflow-end');
  performance.measure('workflow', 'workflow-start', 'workflow-end');

  // 输出所有测量
  const entries = performance.getEntriesByType('measure');
  entries.forEach(entry => {
    console.log(`${entry.name}: ${entry.duration.toFixed(2)}ms`);
  });
}
```

## Web Vitals

Web Vitals 是 Google 定义的核心性能指标，用于衡量用户体验。

### LCP（Largest Contentful Paint）

最大内容绘制时间，衡量加载性能。

```javascript
// 监听 LCP
new PerformanceObserver((entryList) => {
  const entries = entryList.getEntries();
  const lastEntry = entries[entries.length - 1];

  console.log('LCP:', lastEntry.startTime.toFixed(2), 'ms');
  console.log('LCP 元素:', lastEntry.element);
  console.log('LCP URL:', lastEntry.url);
  console.log('LCP 大小:', lastEntry.size);
}).observe({
  type: 'largest-contentful-paint',
  buffered: true
});

// LCP 标准：
// - 好: ≤ 2.5s
// - 需改进: 2.5s - 4s
// - 差: > 4s

// 上报 LCP
function reportLCP() {
  return new Promise((resolve) => {
    const observer = new PerformanceObserver((list) => {
      const entries = list.getEntries();
      const lastEntry = entries[entries.length - 1];

      observer.disconnect();
      resolve({
        name: 'LCP',
        value: lastEntry.startTime,
        element: lastEntry.element?.tagName,
        url: lastEntry.url
      });
    });

    observer.observe({
      type: 'largest-contentful-paint',
      buffered: true
    });
  });
}
```

### FID（First Input Delay）

首次输入延迟，衡量交互性。

```javascript
// 监听 FID
new PerformanceObserver((entryList) => {
  const entries = entryList.getEntries();
  const firstInput = entries[0];

  const fid = firstInput.processingStart - firstInput.startTime;
  console.log('FID:', fid.toFixed(2), 'ms');
  console.log('事件类型:', firstInput.name);
  console.log('目标元素:', firstInput.target);
}).observe({
  type: 'first-input',
  buffered: true
});

// FID 标准：
// - 好: ≤ 100ms
// - 需改进: 100ms - 300ms
// - 差: > 300ms

// 计算 FID
function getFID() {
  return new Promise((resolve) => {
    const observer = new PerformanceObserver((list) => {
      const entries = list.getEntries();
      const firstInput = entries[0];

      observer.disconnect();
      resolve({
        name: 'FID',
        value: firstInput.processingStart - firstInput.startTime,
        eventType: firstInput.name,
        target: firstInput.target?.tagName
      });
    });

    observer.observe({
      type: 'first-input',
      buffered: true
    });

    // 如果用户没有交互，设置超时
    setTimeout(() => {
      observer.disconnect();
      resolve({ name: 'FID', value: null });
    }, 10000);
  });
}
```

### CLS（Cumulative Layout Shift）

累积布局偏移，衡量视觉稳定性。

```javascript
// 监听 CLS
let clsValue = 0;
let clsEntries = [];

new PerformanceObserver((entryList) => {
  for (const entry of entryList.getEntries()) {
    // 只计算非用户输入引起的偏移
    if (!entry.hadRecentInput) {
      clsValue += entry.value;
      clsEntries.push(entry);
    }
  }
  console.log('当前 CLS:', clsValue.toFixed(4));
}).observe({
  type: 'layout-shift',
  buffered: true
});

// CLS 标准：
// - 好: ≤ 0.1
// - 需改进: 0.1 - 0.25
// - 差: > 0.25

// 完整 CLS 监控
function measureCLS() {
  let clsValue = 0;
  let sessionValue = 0;
  let sessionEntries = [];

  const observer = new PerformanceObserver((list) => {
    for (const entry of list.getEntries()) {
      if (entry.hadRecentInput) continue;

      const firstSessionEntry = sessionEntries[0];
      const lastSessionEntry = sessionEntries[sessionEntries.length - 1];

      // 如果距离上次偏移小于 1 秒，且距离首次偏移小于 5 秒，属于同一会话
      if (sessionValue &&
          entry.startTime - lastSessionEntry.startTime < 1000 &&
          entry.startTime - firstSessionEntry.startTime < 5000) {
        sessionValue += entry.value;
        sessionEntries.push(entry);
      } else {
        sessionValue = entry.value;
        sessionEntries = [entry];
      }

      if (sessionValue > clsValue) {
        clsValue = sessionValue;
      }
    }
  });

  observer.observe({ type: 'layout-shift', buffered: true });

  return () => {
    observer.disconnect();
    return clsValue;
  };
}
```

### 其他重要指标

```javascript
// FCP（First Contentful Paint）- 首次内容绘制
new PerformanceObserver((entryList) => {
  const entries = entryList.getEntries();
  console.log('FCP:', entries[0].startTime.toFixed(2), 'ms');
}).observe({
  type: 'paint',
  buffered: true
});

// TTFB（Time to First Byte）- 首字节时间
const [navigationEntry] = performance.getEntriesByType('navigation');
if (navigationEntry) {
  const ttfb = navigationEntry.responseStart - navigationEntry.requestStart;
  console.log('TTFB:', ttfb.toFixed(2), 'ms');
}

// INP（Interaction to Next Paint）- 交互到下一次绘制
let inpValue = 0;
new PerformanceObserver((list) => {
  for (const entry of list.getEntries()) {
    if (entry.interactionId) {
      const duration = entry.duration;
      if (duration > inpValue) {
        inpValue = duration;
      }
    }
  }
}).observe({
  type: 'event',
  durationThreshold: 16,
  buffered: true
});

// TTI（Time to Interactive）- 可交互时间
function measureTTI() {
  const navigationEntry = performance.getEntriesByType('navigation')[0];
  if (navigationEntry) {
    // 简化计算，实际更复杂
    const domContentLoaded = navigationEntry.domContentLoadedEventEnd;
    console.log('TTI（简化）:', domContentLoaded.toFixed(2), 'ms');
  }
}
```

页面加载性能时间线。

```javascript
// 获取导航计时信息
const [navigationEntry] = performance.getEntriesByType('navigation');

if (navigationEntry) {
  const timing = {
    // DNS 查询
    DNS查询: navigationEntry.domainLookupEnd - navigationEntry.domainLookupStart,

    // TCP 连接
    TCP连接: navigationEntry.connectEnd - navigationEntry.connectStart,

    // SSL 握手
    SSL握手: navigationEntry.secureConnectionStart > 0
      ? navigationEntry.connectEnd - navigationEntry.secureConnectionStart
      : 0,

    // TTFB

    // 内容下载
    内容下载: navigationEntry.responseEnd - navigationEntry.responseStart,

    // DOM 解析
    DOM解析: navigationEntry.domComplete - navigationEntry.domInteractive,

    // DOMContentLoaded
    DOMContentLoaded: navigationEntry.domContentLoadedEventEnd - navigationEntry.startTime,

    // 页面加载完成
    页面加载完成: navigationEntry.loadEventEnd - navigationEntry.startTime
  };

  console.table(timing);
}

// 完整的页面加载分析
function analyzePageLoad() {
  const entry = performance.getEntriesByType('navigation')[0];
  if (!entry) return null;

  return {
    // 关键时间点
    startTime: entry.startTime,
    fetchStart: entry.fetchStart,
    domainLookupStart: entry.domainLookupStart,
    domainLookupEnd: entry.domainLookupEnd,
    connectStart: entry.connectStart,
    connectEnd: entry.connectEnd,
    requestStart: entry.requestStart,
    responseStart: entry.responseStart,
    responseEnd: entry.responseEnd,
    domInteractive: entry.domInteractive,
    domContentLoadedEventStart: entry.domContentLoadedEventStart,
    domContentLoadedEventEnd: entry.domContentLoadedEventEnd,
    domComplete: entry.domComplete,
    loadEventStart: entry.loadEventStart,
    loadEventEnd: entry.loadEventEnd,

    // 计算的阶段时间
    phases: {
      redirect: entry.redirectEnd - entry.redirectStart,
      dns: entry.domainLookupEnd - entry.domainLookupStart,
      connect: entry.connectEnd - entry.connectStart,
      ttfb: entry.responseStart - entry.requestStart,
      download: entry.responseEnd - entry.responseStart,
      domParse: entry.domInteractive - entry.responseEnd,
      domContentLoaded: entry.domContentLoadedEventEnd - entry.domInteractive,
      load: entry.loadEventEnd - entry.domContentLoadedEventEnd
    }
  };
}
```

### 资源加载分析

```javascript
// 获取所有资源加载信息
const resources = performance.getEntriesByType('resource');

resources.forEach(resource => {
  console.log({

// 分析特定类型的资源
function analyzeResourcesByType(type) {
  const resources = performance.getEntriesByType('resource');
  const filtered = resources.filter(r => r.initiatorType === type);

  return {
    count: filtered.length,
    totalSize: filtered.reduce((sum, r) => sum + r.transferSize, 0),
    totalDuration: filtered.reduce((sum, r) => sum + r.duration, 0),
    avgDuration: filtered.reduce((sum, r) => sum + r.duration, 0) / filtered.length
  };
}

console.log('脚本资源:', analyzeResourcesByType('script'));
console.log('样式资源:', analyzeResourcesByType('link'));
console.log('图片资源:', analyzeResourcesByType('img'));

// 慢资源检测
function findSlowResources(threshold = 1000) {
  const resources = performance.getEntriesByType('resource');
  return resources
    .filter(r => r.duration > threshold)
    .map(r => ({
      name: r.name,
      duration: r.duration,
      type: r.initiatorType,
      size: r.transferSize
    }))
    .sort((a, b) => b.duration - a.duration);
}

// 大资源检测
function findLargeResources(threshold = 100000) {  // 100KB
  const resources = performance.getEntriesByType('resource');
  return resources
    .filter(r => r.transferSize > threshold)
    .map(r => ({
      name: r.name,
      size: r.transferSize,
      type: r.initiatorType
    }))
    .sort((a, b) => b.size - a.size);
}
```

## 用户计时

### 自定义性能测量

```javascript
// 简单计时器
class SimpleTimer {
  constructor(name) {
    this.name = name;
    this.startTime = null;
    this.endTime = null;
  }

  start() {
    this.startTime = performance.now();
    performance.mark(`${this.name}-start`);
  }

  end() {
    this.endTime = performance.now();
    performance.mark(`${this.name}-end`);
    performance.measure(this.name, `${this.name}-start`, `${this.name}-end`);

    const [measure] = performance.getEntriesByName(this.name, 'measure');
    return measure.duration;
  }

  get duration() {
    if (this.startTime && this.endTime) {
      return this.endTime - this.startTime;
    }
    return null;
  }
}

// 使用
const timer = new SimpleTimer('数据处理');
timer.start();
processData();
console.log(`耗时: ${timer.end()}ms`);

// 多个测量点
class MultiPointTimer {
  constructor(name) {
    this.name = name;
    this.points = [];
    performance.mark(`${name}-start`);
  }

  mark(pointName) {
    const time = performance.now();
    this.points.push({ name: pointName, time });
    performance.mark(`${this.name}-${pointName}`);
  }

  end() {
    performance.mark(`${this.name}-end`);
    performance.measure(this.name, `${this.name}-start`, `${this.name}-end`);

    return {
      totalTime: performance.getEntriesByName(this.name, 'measure')[0].duration,
      points: this.points
    };
  }
}

// 嵌套计时器
class NestedTimer {
  constructor() {
    this.stack = [];
  }

  start(name) {
    performance.mark(`${name}-start`);
    this.stack.push(name);
  }

  end() {
    const name = this.stack.pop();
    performance.mark(`${name}-end`);
    performance.measure(name, `${name}-start`, `${name}-end`);

    return performance.getEntriesByName(name, 'measure')[0].duration;
  }

  getResults() {
    return performance.getEntriesByType('measure');
  }
}
```

### 函数性能装饰器

```javascript
// 测量函数执行时间
function measure(target, propertyKey, descriptor) {
  const original = descriptor.value;

  descriptor.value = function(...args) {
    const start = performance.now();
    const result = original.apply(this, args);
    const end = performance.now();

    console.log(`${propertyKey} 执行时间: ${(end - start).toFixed(2)}ms`);
    return result;
  };

  return descriptor;
}

// 使用
class DataProcessor {
  @measure
  process(data) {
    // 处理数据
    return data;
  }
}

// 高阶函数版本
function measureFn(fn, name = fn.name) {
  return function(...args) {
    const start = performance.now();
    const result = fn.apply(this, args);
    const end = performance.now();

    console.log(`${name} 执行时间: ${(end - start).toFixed(2)}ms`);
    return result;
  };
}

// 异步函数测量
function measureAsyncFn(fn, name = fn.name) {
  return async function(...args) {
    const start = performance.now();
    const result = await fn.apply(this, args);
    const end = performance.now();

    console.log(`${name} 执行时间: ${(end - start).toFixed(2)}ms`);
    return result;
  };
}
```

```javascript
// 内存监控
class MemoryMonitor {
  constructor() {
    this.samples = [];
  }

  sample() {
    if (!performance.memory) {
      console.warn('performance.memory 不可用');
      return null;
    }

    const sample = {
      timestamp: Date.now(),
      usedJSHeapSize: performance.memory.usedJSHeapSize,
      totalJSHeapSize: performance.memory.totalJSHeapSize
    };

    this.samples.push(sample);
    return sample;
  }

  startMonitoring(interval = 1000) {
    this.intervalId = setInterval(() => this.sample(), interval);
  }

  stopMonitoring() {
    if (this.intervalId) {
      clearInterval(this.intervalId);
    }
  }

  getReport() {
    if (this.samples.length < 2) return null;

    const first = this.samples[0];
    const last = this.samples[this.samples.length - 1];

    return {
      duration: last.timestamp - first.timestamp,
      samples: this.samples.length,
      startMemory: first.usedJSHeapSize,
      endMemory: last.usedJSHeapSize,
      memoryGrowth: last.usedJSHeapSize - first.usedJSHeapSize,
      peakMemory: Math.max(...this.samples.map(s => s.usedJSHeapSize)),
      avgMemory: this.samples.reduce((sum, s) => sum + s.usedJSHeapSize, 0) / this.samples.length
    };
  }
}
```

## 性能监控实践

### 性能监控类

```javascript
class PerformanceMonitor {
  constructor(options = {}) {
    this.metrics = {};
    this.observers = [];
    this.reportingEndpoint = options.reportingEndpoint;
    this.sampleRate = options.sampleRate || 1;

    if (Math.random() < this.sampleRate) {
      this.init();
    }
  }

  init() {
    // 监听 Web Vitals
    this.observeLCP();
    this.observeFID();
    this.observeCLS();
    this.observeFCP();
    this.observeTTFB();

    // 页面卸载时上报
    window.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'hidden') {
        this.report();
      }
    });
  }

  observeLCP() {
    new PerformanceObserver((list) => {
      const entries = list.getEntries();
      const lastEntry = entries[entries.length - 1];
      this.metrics.LCP = lastEntry.startTime;
    }).observe({ type: 'largest-contentful-paint', buffered: true });
  }

  observeFID() {
    new PerformanceObserver((list) => {
      const entry = list.getEntries()[0];
      this.metrics.FID = entry.processingStart - entry.startTime;
    }).observe({ type: 'first-input', buffered: true });
  }

  observeCLS() {
    let clsValue = 0;
    new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        if (!entry.hadRecentInput) {
          clsValue += entry.value;
        }
      }
      this.metrics.CLS = clsValue;
    }).observe({ type: 'layout-shift', buffered: true });
  }

  observeFCP() {
    new PerformanceObserver((list) => {
      const entry = list.getEntries().find(e => e.name === 'first-contentful-paint');
      if (entry) {
        this.metrics.FCP = entry.startTime;
      }
    }).observe({ type: 'paint', buffered: true });
  }

  observeTTFB() {
    const entry = performance.getEntriesByType('navigation')[0];
    if (entry) {
      this.metrics.TTFB = entry.responseStart - entry.requestStart;
    }
  }

  // 自定义指标
  mark(name) {
    performance.mark(name);
  }

  measure(name, startMark, endMark) {
    performance.measure(name, startMark, endMark);
    const [entry] = performance.getEntriesByName(name, 'measure');
    if (entry) {
      this.metrics[name] = entry.duration;
    }
  }

  // 上报
  report() {
    const payload = {
      url: window.location.href,
      timestamp: Date.now(),
      metrics: this.metrics,
      userAgent: navigator.userAgent,
      connection: navigator.connection?.effectiveType
    };

    if (navigator.sendBeacon && this.reportingEndpoint) {
      navigator.sendBeacon(
        this.reportingEndpoint,
        JSON.stringify(payload)
      );
    }
  }

  // 获取当前指标
  getMetrics() {
    return { ...this.metrics };
  }
}

// 使用
const monitor = new PerformanceMonitor({
  reportingEndpoint: '/api/performance',
  sampleRate: 0.1  // 10% 采样率
});

// 自定义测量
monitor.mark('data-fetch-start');
await fetchData();
monitor.mark('data-fetch-end');
monitor.measure('data-fetch', 'data-fetch-start', 'data-fetch-end');
```

### 长任务监控

```javascript
// 监听长任务（超过 50ms）
const longTaskObserver = new PerformanceObserver((list) => {
  for (const entry of list.getEntries()) {
    console.warn('长任务检测:', {
      name: entry.name,
      duration: entry.duration.toFixed(2) + 'ms',
      startTime: entry.startTime.toFixed(2),
      attribution: entry.attribution  // 任务来源
    });

    // 上报长任务
    reportLongTask({
      duration: entry.duration,
      startTime: entry.startTime,
      name: entry.name
    });
  }
});

longTaskObserver.observe({ entryTypes: ['longtask'] });

// 长任务阈值
const LONG_TASK_THRESHOLD = 50;  // 50ms

function reportLongTask(task) {
  // 判断严重程度
  if (task.duration > 100) {
    console.error('严重长任务:', task.duration + 'ms');
  } else {
    console.warn('长任务:', task.duration + 'ms');
  }
}
```
