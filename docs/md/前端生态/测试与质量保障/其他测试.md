## 集成测试

### 概念与策略

```javascript
/**
 * 集成测试定义：
 * - 测试多个模块之间的协作
 * - 验证组件之间的接口
 * - 可以涉及真实的外部依赖
 *
 * 与单元测试的区别：
 * - 单元测试：隔离测试单个函数/模块
 * - 集成测试：测试多个模块的协作
 */

// 示例：测试 Service 与 Repository 的集成
// userService.js
class UserService {
  constructor(userRepository) {
    this.userRepository = userRepository;
  }

  async createUser(userData) {
    // 验证
    if (!userData.email || !userData.name) {
      throw new Error('Invalid user data');
    }

    // 检查是否已存在
    const existing = await this.userRepository.findByEmail(userData.email);
    if (existing) {
      throw new Error('User already exists');
    }

    // 创建用户
    const user = await this.userRepository.create(userData);
    return user;
  }
}

// userService.integration.test.js
import { UserService } from '../userService';
import { UserRepository } from '../userRepository';

describe('UserService Integration', () => {
  let service;
  let repository;

  beforeAll(async () => {
    // 使用真实数据库连接
    repository = new UserRepository(testDatabase);
    await repository.connect();
  });

  afterAll(async () => {
    await repository.disconnect();
  });

  beforeEach(async () => {
    service = new UserService(repository);
    await repository.clear();
  });

  test('creates user successfully', async () => {
    const userData = {
      name: 'John Doe',
      email: 'john@example.com'
    };

    const user = await service.createUser(userData);

    expect(user.id).toBeDefined();
    expect(user.name).toBe('John Doe');

    // 验证已保存到数据库
    const saved = await repository.findById(user.id);
    expect(saved).toEqual(user);
  });

  test('prevents duplicate users', async () => {
    const userData = {
      name: 'John Doe',
      email: 'john@example.com'
    };

    await service.createUser(userData);

    await expect(service.createUser(userData))
      .rejects.toThrow('User already exists');
  });
});
```

```javascript
// 使用测试数据库
import { MongoClient } from 'mongodb';

describe('Database Integration', () => {
  let connection;
  let db;
  let collection;

  beforeAll(async () => {
    connection = await MongoClient.connect(process.env.TEST_DB_URL);
    db = connection.db('test_db');
    collection = db.collection('users');
  });

  afterAll(async () => {
    await connection.close();
  });

  beforeEach(async () => {
    await collection.deleteMany({});
  });

  test('inserts and finds user', async () => {
    const user = { name: 'John', email: 'john@example.com' };

    const result = await collection.insertOne(user);

    const found = await collection.findOne({ _id: result.insertedId });
    expect(found.name).toBe('John');
  });

  test('updates user', async () => {
    const { insertedId } = await collection.insertOne({
      name: 'John',
      email: 'john@example.com'
    });

    await collection.updateOne(
      { _id: insertedId },
      { $set: { name: 'Jane' } }
    );

    const updated = await collection.findOne({ _id: insertedId });
    expect(updated.name).toBe('Jane');
  });

  test('transaction', async () => {
    const session = connection.startSession();

    try {
      await session.withTransaction(async () => {
        await collection.insertOne({ name: 'User1' }, { session });
        await collection.insertOne({ name: 'User2' }, { session });
      });

      const count = await collection.countDocuments();
      expect(count).toBe(2);
    } finally {
      await session.endSession();
    }
  });
});
```

### API 集成测试

```javascript
// 使用 supertest 测试 Express 应用
import request from 'supertest';
import express from 'express';
import { userRouter } from '../routes/users';

const app = express();
app.use(express.json());
app.use('/api/users', userRouter);

describe('User API Integration', () => {
  test('GET /api/users returns all users', async () => {
    const response = await request(app)
      .get('/api/users')
      .expect('Content-Type', /json/)
      .expect(200);

    expect(Array.isArray(response.body)).toBe(true);
  });

  test('POST /api/users creates a user', async () => {
    const userData = {
      name: 'John Doe',
      email: 'john@example.com'
    };

    const response = await request(app)
      .post('/api/users')
      .send(userData)
      .expect('Content-Type', /json/)
      .expect(201);

    expect(response.body.name).toBe('John Doe');
    expect(response.body.id).toBeDefined();
  });

  test('GET /api/users/:id returns specific user', async () => {
    // 先创建一个用户
    const createResponse = await request(app)
      .post('/api/users')
      .send({ name: 'John', email: 'john@example.com' });

    const userId = createResponse.body.id;

    // 然后获取该用户
    const response = await request(app)
      .get(`/api/users/${userId}`)
      .expect(200);

    expect(response.body.name).toBe('John');
  });

  test('PUT /api/users/:id updates user', async () => {
    const createResponse = await request(app)
      .post('/api/users')
      .send({ name: 'John', email: 'john@example.com' });

    const userId = createResponse.body.id;

    const response = await request(app)
      .put(`/api/users/${userId}`)
      .send({ name: 'Jane' })
      .expect(200);

    expect(response.body.name).toBe('Jane');
  });

  test('DELETE /api/users/:id removes user', async () => {
    const createResponse = await request(app)
      .post('/api/users')
      .send({ name: 'John', email: 'john@example.com' });

    const userId = createResponse.body.id;

    await request(app)
      .delete(`/api/users/${userId}`)
      .expect(204);

    // 验证已删除
    await request(app)
      .get(`/api/users/${userId}`)
      .expect(404);
  });

  test('validates request body', async () => {
    const response = await request(app)
      .post('/api/users')
      .send({}) // 空请求体
      .expect(400);

    expect(response.body.error).toBeDefined();
  });
});
```

```javascript
// 自定义性能测试
function measurePerformance(name, fn, iterations = 1000) {
  const start = performance.now();

  for (let i = 0; i < iterations; i++) {
    fn();
  }

  const end = performance.now();
  const duration = end - start;
  const opsPerSecond = (iterations / duration) * 1000;

  console.log(`${name}:`);
  console.log(`  总时间: ${duration.toFixed(2)}ms`);
  console.log(`  每次操作: ${(duration / iterations).toFixed(4)}ms`);
  console.log(`  每秒操作数: ${opsPerSecond.toFixed(0)}`);

  return { duration, opsPerSecond };
}

// Jest 性能测试
describe('Performance', () => {
  test('sorts large array within time limit', () => {
    const arr = Array.from({ length: 100000 }, () => Math.random());

    const start = performance.now();
    arr.sort((a, b) => a - b);
    const duration = performance.now() - start;

    expect(duration).toBeLessThan(100); // 100ms 内完成
  });
});
```

## 测试驱动开发 TDD

### TDD 流程

```javascript
/**
 * TDD 流程（红灯-绿灯-重构）：
 *
 * 1. Red（红灯）：先写一个失败的测试
 * 2. Green（绿灯）：编写最简单的代码使测试通过
 * 3. Refactor（重构）：优化代码，保持测试通过
 */

// 示例：开发一个计算器

// 第一步：红灯 - 先写测试
describe('Calculator', () => {
  test('adds two numbers', () => {
    const calc = new Calculator();
    expect(calc.add(2, 3)).toBe(5);
  });

  test('subtracts two numbers', () => {
    const calc = new Calculator();
    expect(calc.subtract(5, 3)).toBe(2);
  });

  test('multiplies two numbers', () => {
    const calc = new Calculator();
    expect(calc.multiply(4, 3)).toBe(12);
  });

  test('divides two numbers', () => {
    const calc = new Calculator();
    expect(calc.divide(10, 2)).toBe(5);
  });

  test('throws error when dividing by zero', () => {
    const calc = new Calculator();
    expect(() => calc.divide(10, 0)).toThrow('Division by zero');
  });
});

// 第二步：绿灯 - 实现最小代码
class Calculator {
  add(a, b) {
    return a + b;
  }

  subtract(a, b) {
    return a - b;
  }

  multiply(a, b) {
    return a * b;
  }

  divide(a, b) {
    if (b === 0) {
      throw new Error('Division by zero');
    }
    return a / b;
  }
}

// 第三步：重构 - 优化代码
class Calculator {
  operate(operation, a, b) {
    const operations = {
      add: (x, y) => x + y,
      subtract: (x, y) => x - y,
      multiply: (x, y) => x * y,
      divide: (x, y) => {
        if (y === 0) throw new Error('Division by zero');
        return x / y;
      }
    };

    return operations[operation](a, b);
  }

  add(a, b) {
    return this.operate('add', a, b);
  }

  subtract(a, b) {
    return this.operate('subtract', a, b);
  }

  multiply(a, b) {
    return this.operate('multiply', a, b);
  }

  divide(a, b) {
    return this.operate('divide', a, b);
  }
}
```
