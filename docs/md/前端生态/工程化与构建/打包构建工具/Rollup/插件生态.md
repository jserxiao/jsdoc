# 插件生态

Rollup 拥有丰富的插件生态，可以扩展其功能。

## 常用插件列表

| 插件 | 作用 |
|------|------|
| `@rollup/plugin-node-resolve` | 解析 node_modules 中的模块 |
| `@rollup/plugin-commonjs` | 转换 CommonJS 模块 |
| `@rollup/plugin-typescript` | 编译 TypeScript |
| `@rollup/plugin-json` | 导入 JSON 文件 |
| `@rollup/plugin-alias` | 路径别名 |
| `@rollup/plugin-image` | 导入图片 |
| `@rollup/plugin-url` | 导入文件为 URL |
| `@rollup/plugin-babel` | Babel 转换 |
| `rollup-plugin-terser` | 代码压缩 |
| `rollup-plugin-dts` | 类型声明文件合并 |
| `rollup-plugin-postcss` | CSS 处理 |
| `rollup-plugin-livereload` | 热更新 |
| `rollup-plugin-serve` | 开发服务器 |

## 基础插件组合

```javascript
import resolve from '@rollup/plugin-node-resolve'
import commonjs from '@rollup/plugin-commonjs'

export default {
  input: 'src/index.js',
  output: {
    file: 'dist/bundle.js',
    format: 'es'
  },
  plugins: [
    resolve(),    // 解析 node_modules
    commonjs()    // 转换 CommonJS
  ]
}
```

## TypeScript 支持

```javascript
import typescript from '@rollup/plugin-typescript'

export default {
  plugins: [
    typescript({
      tsconfig: './tsconfig.json',
      declaration: true,
      declarationDir: 'dist/types'
    })
  ]
}
```

## CSS 处理

```javascript
import postcss from 'rollup-plugin-postcss'
import autoprefixer from 'autoprefixer'
import cssnano from 'cssnano'

export default {
  plugins: [
    postcss({
      plugins: [
        autoprefixer(),
        cssnano()
      ],
      extract: true,           // 提取到单独文件
      minimize: true,          // 压缩
      sourceMap: true,
      extensions: ['.css', '.scss', '.less']
    })
  ]
}
```

## 代码压缩

```javascript
import { terser } from 'rollup-plugin-terser'

export default {
  plugins: [
    terser({
      compress: {
        drop_console: true,
        drop_debugger: true
      },
      format: {
        comments: false
      }
    })
  ]
}
```

## 路径别名

```javascript
import alias from '@rollup/plugin-alias'
import { resolve } from 'path'

export default {
  plugins: [
    alias({
      entries: [
        { find: '@', replacement: resolve(__dirname, 'src') },
        { find: '@components', replacement: resolve(__dirname, 'src/components') },
        { find: '@utils', replacement: resolve(__dirname, 'src/utils') }
      ]
    })
  ]
}
```

## 资源处理

```javascript
import image from '@rollup/plugin-image'
import url from '@rollup/plugin-url'

export default {
  plugins: [
    // 导入图片
    image(),
    
    // 文件作为 URL
    url({
      include: ['**/*.svg', '**/*.png', '**/*.jpg'],
      limit: 8192,  // 8KB 以下转 base64
      fileName: '[name][hash][extname]',
      outputPath: 'assets'
    })
  ]
}
```

## JSON 支持

```javascript
import json from '@rollup/plugin-json'

export default {
  plugins: [
    json({
      compact: true,
      preferConst: true,
      indent: '  '
    })
  ]
}

// 使用
import pkg from './package.json'
console.log(pkg.version)
```

## Babel 支持

```javascript
import babel from '@rollup/plugin-babel'

export default {
  plugins: [
    babel({
      babelHelpers: 'bundled',
      exclude: 'node_modules/**',
      extensions: ['.js', '.jsx', '.ts', '.tsx'],
      presets: [
        ['@babel/preset-env', { targets: '> 0.25%, not dead' }]
      ]
    })
  ]
}
```

## 开发服务器

```javascript
import serve from 'rollup-plugin-serve'
import livereload from 'rollup-plugin-livereload'

export default {
  plugins: [
    serve({
      open: true,
      contentBase: 'dist',
      host: 'localhost',
      port: 3000
    }),
    livereload({
      watch: 'dist'
    })
  ]
}
```

## 类型声明文件

```javascript
import dts from 'rollup-plugin-dts'

export default [
  // 主构建
  {
    input: 'src/index.ts',
    output: [
      { file: 'dist/index.js', format: 'es' }
    ],
    plugins: [typescript()]
  },
  // 类型声明
  {
    input: 'dist/types/index.d.ts',
    output: {
      file: 'dist/index.d.ts',
      format: 'es'
    },
    plugins: [dts()]
  }
]
```

## 自定义插件

```javascript
function myPlugin(options = {}) {
  return {
    name: 'my-plugin',
    
    // 构建开始
    buildStart() {
      console.log('构建开始')
    },
    
    // 解析模块 ID
    resolveId(source, importer) {
      if (source === 'my-virtual-module') {
        return source
      }
      return null
    },
    
    // 加载模块
    load(id) {
      if (id === 'my-virtual-module') {
        return 'export default "Hello from virtual module"'
      }
      return null
    },
    
    // 转换代码
    transform(code, id) {
      if (id.endsWith('.custom')) {
        return {
          code: `export default ${JSON.stringify(code)}`,
          map: null
        }
      }
    },
    
    // 构建结束
    buildEnd() {
      console.log('构建结束')
    }
  }
}

export default {
  plugins: [myPlugin()]
}
```
