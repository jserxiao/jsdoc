# 基础配置

## 配置文件

```javascript
// rollup.config.js
import resolve from '@rollup/plugin-node-resolve'
import commonjs from '@rollup/plugin-commonjs'
import { terser } from 'rollup-plugin-terser'

export default {
  // 入口文件
  input: 'src/index.js',
  
  // 输出配置
  output: {
    file: 'dist/bundle.js',
    format: 'es',
    sourcemap: true
  },
  
  // 插件
  plugins: [
    resolve(),
    commonjs(),
    terser()
  ]
}
```

## 完整配置示例

```javascript
import resolve from '@rollup/plugin-node-resolve'
import commonjs from '@rollup/plugin-commonjs'
import typescript from '@rollup/plugin-typescript'
import { terser } from 'rollup-plugin-terser'
import dts from 'rollup-plugin-dts'
import alias from '@rollup/plugin-alias'
import json from '@rollup/plugin-json'
import image from '@rollup/plugin-image'
import postcss from 'rollup-plugin-postcss'
import { defineConfig } from 'rollup'

export default defineConfig([
  // 主构建
  {
    input: 'src/index.ts',
    output: [
      {
        file: 'dist/index.cjs.js',
        format: 'cjs',
        sourcemap: true,
        exports: 'auto'
      },
      {
        file: 'dist/index.esm.js',
        format: 'es',
        sourcemap: true
      }
    ],
    external: ['vue', 'react'],
    plugins: [
      alias({
        entries: [
          { find: '@', replacement: 'src' }
        ]
      }),
      resolve({
        extensions: ['.ts', '.tsx', '.js', '.jsx', '.json']
      }),
      commonjs(),
      typescript({
        tsconfig: './tsconfig.json'
      }),
      json(),
      image(),
      postcss({
        extract: true,
        minimize: true
      }),
      terser()
    ]
  },
  // 类型声明文件
  {
    input: 'dist/types/index.d.ts',
    output: {
      file: 'dist/index.d.ts',
      format: 'es'
    },
    plugins: [dts()]
  }
])
```

## 输出选项详解

```javascript
export default {
  output: {
    // 文件路径
    file: 'dist/bundle.js',
    
    // 输出目录（多文件时使用）
    dir: 'dist',
    
    // 格式
    format: 'es', // 'amd' | 'cjs' | 'es' | 'iife' | 'umd' | 'system'
    
    // 全局变量名（UMD/IIFE）
    name: 'MyLibrary',
    
    // Sourcemap
    sourcemap: true,
    sourcemapFile: 'dist/bundle.js.map',
    
    // 全局变量映射
    globals: {
      vue: 'Vue',
      react: 'React'
    },
    
    // 导出方式
    exports: 'auto', // 'auto' | 'named' | 'default' | 'none'
    
    // 文件名模式
    entryFileNames: '[name].[format].js',
    chunkFileNames: '[name]-[hash].js',
    assetFileNames: 'assets/[name]-[hash][extname]',
    
    // 保留模块结构
    preserveModules: true,
    preserveModulesRoot: 'src',
    
    // 额外选项
    compact: true,       // 压缩输出
    freeze: true,        // 冻结对象
    interop: 'auto',     // 模块互操作
    strict: true         // 严格模式
  }
}
```

## 外部依赖配置

```javascript
export default {
  // 数组形式
  external: ['vue', 'react', 'react-dom'],
  
  // 函数形式
  external: (id) => {
    return id.includes('node_modules') || id.startsWith('vue')
  },
  
  // 正则形式
  external: [/^lodash/, /^@babel/],
  
  output: {
    globals: {
      vue: 'Vue',
      react: 'React',
      'react-dom': 'ReactDOM',
      'lodash/debounce': '_.debounce'
    }
  }
}
```

## 多入口配置

```javascript
export default {
  input: {
    index: 'src/index.ts',
    utils: 'src/utils/index.ts',
    components: 'src/components/index.ts'
  },
  output: {
    dir: 'dist',
    format: 'es',
    entryFileNames: '[name].js'
  }
}
```

## 条件配置

```javascript
import { defineConfig } from 'rollup'

export default defineConfig((commandLineArgs) => {
  const isProduction = commandLineArgs.configProduction
  
  return {
    input: 'src/index.js',
    output: {
      file: 'dist/bundle.js',
      format: 'es',
      sourcemap: !isProduction
    },
    plugins: isProduction ? [terser()] : []
  }
})
```
