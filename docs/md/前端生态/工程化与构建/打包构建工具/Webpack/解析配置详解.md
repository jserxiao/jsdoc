# 解析配置详解

## 概述

配置模块如何解析。webpack 提供合理的默认值，但是可能会更改一些解析的细节。

## resolve.alias

### 基本用法

```javascript
const path = require('path')

module.exports = {
  resolve: {
    alias: {
      '@': path.resolve(__dirname, 'src'),
      '@components': path.resolve(__dirname, 'src/components'),
      '@utils': path.resolve(__dirname, 'src/utils'),
      '@assets': path.resolve(__dirname, 'src/assets'),
      '@styles': path.resolve(__dirname, 'src/styles')
    }
  }
}

// 使用
import Button from '@/components/Button'
import { formatDate } from '@utils/date'
import logo from '@assets/logo.png'
```

### 精确匹配

```javascript
module.exports = {
  resolve: {
    alias: {
      // 精确匹配
      xyz$: path.resolve(__dirname, 'path/to/file.js')
    }
  }
}

// xyz$ 表示精确匹配 xyz
// import 'xyz' → path/to/file.js
// import 'xyz/file.js' → 不会替换
```

### 条件别名

```javascript
module.exports = {
  resolve: {
    alias: {
      // 根据环境使用不同实现
      api$: process.env.NODE_ENV === 'production'
        ? path.resolve(__dirname, 'src/api/prod.js')
        : path.resolve(__dirname, 'src/api/dev.js')
    }
  }
}
```

## resolve.extensions

### 基本用法

```javascript
module.exports = {
  resolve: {
    extensions: ['.js', '.json', '.ts', '.tsx', '.jsx', '.vue']
  }
}

// 使用
import file from './file'  // 自动尝试 .js, .json, .ts 等
```

### 优先级

```javascript
module.exports = {
  resolve: {
    extensions: ['.ts', '.js', '.json']  // 按顺序尝试
  }
}
```

### 性能优化

```javascript
module.exports = {
  resolve: {
    extensions: ['.js', '.json']  // 仅保留必要的扩展名
  }
}
```

## resolve.mainFields

### 基本用法

```javascript
module.exports = {
  resolve: {
    mainFields: ['browser', 'module', 'main']
  }
}
```

### 配置说明

| 字段 | 说明 |
|------|------|
| `browser` | 浏览器环境 |
| `module` | ES Module |
| `main` | CommonJS |

### 针对不同环境

```javascript
module.exports = {
  resolve: {
    mainFields: ['browser', 'module', 'main']  // 浏览器环境
    // mainFields: ['module', 'main']  // Node.js 环境
  }
}
```

## resolve.mainFiles

### 基本用法

```javascript
module.exports = {
  resolve: {
    mainFiles: ['index']  // 默认值
  }
}

// import './dir' 会解析为 './dir/index'
```

### 多文件名

```javascript
module.exports = {
  resolve: {
    mainFiles: ['index', 'main', 'src']
  }
}
```

## resolve.modules

### 基本用法

```javascript
module.exports = {
  resolve: {
    modules: ['node_modules']  // 默认值
  }
}
```

### 添加搜索路径

```javascript
module.exports = {
  resolve: {
    modules: [
      path.resolve(__dirname, 'src'),
      'node_modules'
    ]
  }
}

// 可以直接 import 'components/Button'
// 而不需要 import '@/components/Button'
```

### 绝对路径

```javascript
module.exports = {
  resolve: {
    modules: [
      path.resolve(__dirname, 'node_modules'),
      path.resolve(__dirname, 'src')
    ]
  }
}
```

## resolve.descriptionFiles

```javascript
module.exports = {
  resolve: {

  }

```

## resolve.enforceExtension

```javascript
module.exports = {
  resolve: {
    enforceExtension: true  // 必须使用扩展名
  }
}

// 必须写 import './file.js'
// 不能写 import './file'
```

## resolve.enforceModuleExtension

```javascript
module.exports = {
  resolve: {
    enforceModuleExtension: true  // 模块必须使用扩展名
  }
}
```

## resolve.unsafeCache

### 基本用法

```javascript
module.exports = {
  resolve: {
    unsafeCache: true  // 缓存所有解析
  }
}
```

### 正则匹配

```javascript
module.exports = {
  resolve: {
    unsafeCache: /node_modules/  // 仅缓存 node_modules
  }
}
```

### 函数形式

```javascript
module.exports = {
  resolve: {
    unsafeCache: (path) => {
      return path.includes('node_modules')
    }
  }
}
```

## resolve.cache

```javascript
module.exports = {
  resolve: {
    cache: true  // 启用安全缓存
  }
}
```

## resolve.plugins

```javascript
const TsconfigPathsPlugin = require('tsconfig-paths-webpack-plugin')

module.exports = {
  resolve: {
    plugins: [
      new TsconfigPathsPlugin({
        configFile: './tsconfig.json'
      })
    ]
  }
}
```

## resolve.symlinks

```javascript
module.exports = {
  resolve: {
    symlinks: true  // 解析符号链接
  }
}
```

## resolve.fullySpecified

```javascript
module.exports = {
  resolve: {
    fullySpecified: false  // 允许无扩展名导入
  }
}
```

## resolveLoader

### 基本配置

```javascript
module.exports = {
  resolveLoader: {
    modules: ['node_modules'],
    extensions: ['.js', '.json'],
    mainFields: ['loader', 'main']
  }
}
```

### 自定义 loader 搜索路径

```javascript
module.exports = {
  resolveLoader: {
    modules: [
      path.resolve(__dirname, 'loaders'),
      'node_modules'
    ]
  }
}
```

## 完整配置示例

### 基础配置

```javascript
const path = require('path')

module.exports = {
  resolve: {
    alias: {
      '@': path.resolve(__dirname, 'src'),
      '@components': path.resolve(__dirname, 'src/components'),
      '@utils': path.resolve(__dirname, 'src/utils'),
      '@assets': path.resolve(__dirname, 'src/assets'),
      '@styles': path.resolve(__dirname, 'src/styles')
    },
    extensions: ['.js', '.json', '.ts', '.tsx', '.jsx', '.vue'],
    mainFields: ['browser', 'module', 'main'],
    mainFiles: ['index'],
    modules: ['node_modules', path.resolve(__dirname, 'src')],
    unsafeCache: true
  }
}
```

### TypeScript 项目

```javascript
const TsconfigPathsPlugin = require('tsconfig-paths-webpack-plugin')

module.exports = {
  resolve: {
    extensions: ['.ts', '.tsx', '.js', '.json'],
    plugins: [
      new TsconfigPathsPlugin({
        configFile: './tsconfig.json'
      })
    ],
    alias: {
      '@': path.resolve(__dirname, 'src')
    }
  }
}
```

### React 项目

```javascript
module.exports = {
  resolve: {
    extensions: ['.js', '.jsx', '.ts', '.tsx', '.json'],
    alias: {
      '@': path.resolve(__dirname, 'src'),
      '@components': path.resolve(__dirname, 'src/components'),
      '@hooks': path.resolve(__dirname, 'src/hooks'),
      '@utils': path.resolve(__dirname, 'src/utils')
    },
    mainFields: ['browser', 'module', 'main']
  }
}
```

### Vue 项目

```javascript
module.exports = {
  resolve: {
    extensions: ['.js', '.vue', '.json', '.ts'],
    alias: {
      '@': path.resolve(__dirname, 'src'),
      'vue$': 'vue/dist/vue.esm.js'
    },
    mainFields: ['browser', 'module', 'main']
  }
}
```

### 性能优化配置

```javascript
module.exports = {
  resolve: {
    // 减少扩展名尝试
    extensions: ['.js', '.json'],

    // 缓存解析结果
    cache: true,
    unsafeCache: true,

    // 缩小搜索范围
    modules: [path.resolve(__dirname, 'node_modules')],

    // 明确主字段
    mainFields: ['main'],

    // 明确主文件
    mainFiles: ['index']
  }
}
```

## 最佳实践

### 1. 合理使用别名

```javascript
module.exports = {
  resolve: {
    alias: {
      // ✅ 好：使用语义化别名
      '@components': path.resolve(__dirname, 'src/components'),
      '@utils': path.resolve(__dirname, 'src/utils'),

      // ❌ 不好：过于简单的别名
      'c': path.resolve(__dirname, 'src/components'),
      'u': path.resolve(__dirname, 'src/utils')
    }
  }
}
```

### 2. 控制扩展名列表

```javascript
module.exports = {
  resolve: {
    // ✅ 好：只保留必要的扩展名
    extensions: ['.js', '.ts', '.json'],

    // ❌ 不好：太多扩展名
    extensions: ['.js', '.ts', '.tsx', '.jsx', '.json', '.vue', '.svelte']
  }
}
```

### 3. 优化模块搜索

```javascript
module.exports = {
  resolve: {
    // ✅ 好：明确搜索路径
    modules: [
      path.resolve(__dirname, 'src'),
      'node_modules'
    ],

    // ❌ 不好：使用相对路径
    modules: ['./src', 'node_modules']
  }
}
```
