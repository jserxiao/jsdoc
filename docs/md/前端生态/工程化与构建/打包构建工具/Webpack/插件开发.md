# Webpack 插件开发

Webpack 插件是一个带有 `apply` 方法的类，通过监听 Webpack 构建生命周期中的钩子来实现各种功能。

---

## 插件基础结构

### 最简插件

```javascript
// MyPlugin.js
class MyPlugin {
  constructor(options) {
    this.options = options
  }
  
  apply(compiler) {
    // compiler: Webpack 编译器实例
    // 通过 compiler.hooks 访问各种钩子
  }
}

module.exports = MyPlugin
```

### 使用插件

```javascript
// webpack.config.js
const MyPlugin = require('./MyPlugin')

module.exports = {
  plugins: [
    new MyPlugin({
      // 配置选项
      message: 'Hello Webpack!'
    })
  ]
}
```

## Compiler 和 Compilation

### Compiler 对象

Compiler 是 Webpack 的核心引擎，代表整个构建过程：

```javascript
apply(compiler) {
  // 常用属性
  console.log(compiler.options)      // webpack 配置
  console.log(compiler.context)      // 项目根目录
  console.log(compiler.inputFileSystem)   // 文件系统
  console.log(compiler.outputFileSystem)  // 输出文件系统
  
  // 常用方法
  compiler.run((err, stats) => {     // 执行构建
    // ...
  })
  
  compiler.watch({}, (err, stats) => { // 监听模式
    // ...
  })
}
```

### Compilation 对象

Compilation 代表一次具体的构建过程：

```javascript
compiler.hooks.compilation.tap('MyPlugin', (compilation) => {
  // 常用属性
  console.log(compilation.options)       // 配置
  console.log(compilation.modules)       // 所有模块
  console.log(compilation.chunks)        // 所有代码块
  console.log(compilation.assets)        // 生成资源
  console.log(compilation.warnings)      // 警告
  console.log(compilation.errors)        // 错误
  
  // 常用方法
  compilation.addModule(module)          // 添加模块
  compilation.emitAsset(name, source)    // 添加资源
  compilation.getStats()                  // 获取统计信息
})
```

## 钩子类型与注册方式

### 钩子类型

| 类型 | 说明 | 注册方法 |
|------|------|----------|
| `SyncHook` | 同步钩子 | `tap` |
| `SyncBailHook` | 同步熔断钩子，返回非 undefined 停止 | `tap` |
| `SyncWaterfallHook` | 同步瀑布钩子，传递返回值 | `tap` |
| `AsyncSeriesHook` | 异步串行钩子 | `tapAsync` / `tapPromise` |
| `AsyncParallelHook` | 异步并行钩子 | `tapAsync` / `tapPromise` |

### 注册方式

```javascript
// 同步钩子
compiler.hooks.done.tap('MyPlugin', (stats) => {
  console.log('Build done!')
})

// 异步钩子 - tapAsync
compiler.hooks.emit.tapAsync('MyPlugin', (compilation, callback) => {
  // 异步操作
  setTimeout(() => {
    callback() // 必须调用 callback
  }, 1000)
})

// 异步钩子 - tapPromise
compiler.hooks.emit.tapPromise('MyPlugin', (compilation) => {
  return new Promise((resolve) => {
    // 异步操作
    resolve()
  })
})

// 熔断钩子
compiler.hooks.shouldEmit.tap('MyPlugin', (compilation) => {
  // 返回 false 阻止输出
  if (compilation.errors.length > 0) {
    return false
  }
})
```

## Compiler 钩子详解

### 构建生命周期

```
┌─────────────────────────────────────────────────────────────┐
│                    Webpack 构建生命周期                       │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  environment                                                 │
│     ↓                                                       │
│  afterEnvironment                                           │
│     ↓                                                       │
│  entryOption ──→ 初始化入口                                   │
│     ↓                                                       │
│  afterPlugins                                                │
│     ↓                                                       │
│  afterResolvers                                              │
│     ↓                                                       │
│  initialize                                                  │
│     ↓                                                       │
│  run / watchRun ──→ 开始构建/监听                             │
│     ↓                                                       │
│  beforeRun                                                   │
│     ↓                                                       │
│  normalModuleFactory / contextModuleFactory                  │
│     ↓                                                       │
│  beforeCompile                                               │
│     ↓                                                       │
│  compile ──→ 开始编译                                         │
│     ↓                                                       │
│  thisCompilation ──→ 创建 compilation                        │
│     ↓                                                       │
│  compilation ──→ 编译过程                                     │
│     ↓                                                       │
│  make ──→ 构建模块依赖图                                      │
│     ↓                                                       │
│  finishMake                                                  │
│     ↓                                                       │
│  afterCompile                                                │
│     ↓                                                       │
│  shouldEmit ──→ 是否输出                                      │
│     ↓                                                       │
│  emit ──→ 输出资源到输出目录之前                               │
│     ↓                                                       │
│  afterEmit ──→ 输出资源到输出目录之后                          │
│     ↓                                                       │
│  done ──→ 构建完成                                            │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 常用 Compiler 钩子

```javascript
class MyPlugin {
  apply(compiler) {
    // 初始化
    compiler.hooks.environment.tap('MyPlugin', () => {
      console.log('Environment prepared')
    })
    
    // 构建开始
    compiler.hooks.run.tap('MyPlugin', (compiler) => {
      console.log('Webpack build starting...')
    })
    
    // 监听模式
    compiler.hooks.watchRun.tap('MyPlugin', (compiler) => {
      console.log('Watch mode starting...')
    })
    
    // 编译完成
    compiler.hooks.done.tap('MyPlugin', (stats) => {
      console.log('Build completed!')
      console.log('Errors:', stats.compilation.errors.length)
      console.log('Warnings:', stats.compilation.warnings.length)
    })
    
    // 输出前
    compiler.hooks.emit.tapAsync('MyPlugin', (compilation, callback) => {
      console.log('Assets about to be emitted:')
      console.log(Object.keys(compilation.assets))
      callback()
    })
    
    // 输出后
    compiler.hooks.afterEmit.tap('MyPlugin', (compilation) => {
      console.log('Assets emitted successfully')
    })
    
    // 失败时
    compiler.hooks.failed.tap('MyPlugin', (error) => {
      console.error('Build failed:', error)
    })
  }
}
```

## Compilation 钩子详解

### 模块处理钩子

```javascript
compiler.hooks.compilation.tap('MyPlugin', (compilation) => {
  // 模块构建成功
  compilation.hooks.succeedModule.tap('MyPlugin', (module) => {
    console.log('Module built:', module.resource)
  })
  
  // 模块构建失败
  compilation.hooks.failedModule.tap('MyPlugin', (module, error) => {
    console.error('Module failed:', module.resource, error)
  })
  
  // 模块构建完成
  compilation.hooks.finishModules.tap('MyPlugin', (modules) => {
    console.log('All modules built:', modules.size)
  })
  
  // 代码块生成
  compilation.hooks.afterOptimizeChunks.tap('MyPlugin', (chunks) => {
    chunks.forEach(chunk => {
      console.log('Chunk:', chunk.name, chunk.files)
    })
  })
})
```

### 资源处理钩子

```javascript
compiler.hooks.compilation.tap('MyPlugin', (compilation) => {
  // 处理资源 - 替换模板变量
  compilation.hooks.processAssets.tap(
    {
      name: 'MyPlugin',
      stage: compilation.PROCESS_ASSETS_STAGE_ADDITIONAL
    },
    (assets) => {
      Object.keys(assets).forEach(filename => {
        if (filename.endsWith('.js')) {
          const source = assets[filename].source()
          const newSource = source.replace(/__VERSION__/g, '1.0.0')
          
          compilation.updateAsset(filename, new RawSource(newSource))
        }
      })
    }
  )
})
```

### Process Assets 阶段

| 阶段 | 常量 | 说明 |
|------|------|------|
| 1 | `PROCESS_ASSETS_STAGE_PRE_PROCESS` | 基础预处理 |
| 2 | `PROCESS_ASSETS_STAGE_DERIVED` | 派生资源 |
| 3 | `PROCESS_ASSETS_STAGE_ADDITIONAL` | 添加额外资源 |
| 4 | `PROCESS_ASSETS_STAGE_OPTIMIZE` | 通用优化 |
| 5 | `PROCESS_ASSETS_STAGE_OPTIMIZE_COUNT` | 优化数量 |
| 6 | `PROCESS_ASSETS_STAGE_OPTIMIZE_COMPATIBILITY` | 兼容性优化 |
| 7 | `PROCESS_ASSETS_STAGE_OPTIMIZE_SIZE` | 优化大小 |
| 8 | `PROCESS_ASSETS_STAGE_DEV_TOOLING` | SourceMap |
| 9 | `PROCESS_ASSETS_STAGE_OPTIMIZE_INLINE` | 内联优化 |
| 10 | `PROCESS_ASSETS_STAGE_SUMMARIZE` | 汇总 |
| 11 | `PROCESS_ASSETS_STAGE_OPTIMIZE_INLINE` | 再次内联 |
| 12 | `PROCESS_ASSETS_STAGE_REPORT` | 报告 |
| 13 | `PROCESS_ASSETS_STAGE_FINAL_REPORT` | 最终报告 |

### 环境变量注入插件

```javascript
const { RawSource } = require('webpack-sources')

class EnvInjectPlugin {
  constructor(options = {}) {
    this.prefix = options.prefix || 'APP_'
    this.envFile = options.envFile || '.env'
  }
  
  apply(compiler) {
    // 读取环境变量
    const fs = require('fs')
    const path = require('path')
    const envPath = path.resolve(process.cwd(), this.envFile)
    
    let envVars = {}
    
    if (fs.existsSync(envPath)) {
      const content = fs.readFileSync(envPath, 'utf-8')
      content.split('\n').forEach(line => {
        const [key, ...values] = line.split('=')
        if (key && key.startsWith(this.prefix)) {
          envVars[key.trim()] = values.join('=').trim()
        }
      })
    }
    
    compiler.hooks.compilation.tap('EnvInjectPlugin', (compilation) => {
      compilation.hooks.processAssets.tap(
        {
          name: 'EnvInjectPlugin',
          stage: compilation.PROCESS_ASSETS_STAGE_PRE_PROCESS
        },
        (assets) => {
          Object.keys(assets).forEach(filename => {
            if (filename.endsWith('.js')) {
              const source = assets[filename].source()
              
              // 替换环境变量
              let modified = source
              Object.entries(envVars).forEach(([key, value]) => {
                const regex = new RegExp(`process\\.env\\.${key}`, 'g')
                modified = modified.replace(regex, JSON.stringify(value))
              })
              
              compilation.updateAsset(filename, new RawSource(modified))
            }
          })
        }
      )
    })
  }
}

module.exports = EnvInjectPlugin
```

### 压缩报告插件

```javascript
const { RawSource } = require('webpack-sources')
const zlib = require('zlib')

class CompressionReportPlugin {
  constructor(options = {}) {
    this.filename = options.filename || 'compression-report.json'
  }
  
  apply(compiler) {
    compiler.hooks.emit.tapAsync('CompressionReportPlugin', (compilation, callback) => {
      const report = {
        buildTime: new Date().toISOString(),
        files: []
      }
      
      const promises = Object.keys(compilation.assets).map(filename => {
        return new Promise((resolve) => {
          const source = compilation.assets[filename].source()
          
          // 计算原始大小
          const originalSize = Buffer.byteLength(source)
          
          // 计算 Gzip 大小
          zlib.gzip(source, (err, compressed) => {
            const gzipSize = compressed ? compressed.length : 0
            
            // 计算 Brotli 大小
            zlib.brotliCompress(source, (err, brotli) => {
              const brotliSize = brotli ? brotli.length : 0
              
              report.files.push({
                filename,
                originalSize,
                gzipSize,
                brotliSize,
                gzipRatio: (gzipSize / originalSize * 100).toFixed(2) + '%',
                brotliRatio: (brotliSize / originalSize * 100).toFixed(2) + '%'
              })
              
              resolve()
            })
          })
        })
      })
      
      Promise.all(promises).then(() => {
        compilation.emitAsset(
          this.filename,
          new RawSource(JSON.stringify(report, null, 2))
        )
        callback()
      })
    })
  }
}

module.exports = CompressionReportPlugin
```

### HTML 注入插件

```javascript
class HtmlInjectPlugin {
  constructor(options = {}) {
    this.tags = options.tags || []
  }
  
  apply(compiler) {
    compiler.hooks.compilation.tap('HtmlInjectPlugin', (compilation) => {
      compilation.hooks.processAssets.tap(
        {
          name: 'HtmlInjectPlugin',
          stage: compilation.PROCESS_ASSETS_STAGE_OPTIMIZE_INLINE
        },
        (assets) => {
          Object.keys(assets).forEach(filename => {
            if (filename.endsWith('.html')) {
              let html = assets[filename].source()
              
              // 注入标签
              this.tags.forEach(tag => {
                const tagStr = this.buildTag(tag)
                
                if (tag.position === 'head') {
                  html = html.replace('</head>', `${tagStr}</head>`)
                } else if (tag.position === 'body') {
                  html = html.replace('</body>', `${tagStr}</body>`)
                }
              })
              
              compilation.updateAsset(filename, new RawSource(html))
            }
          })
        }
      )
    })
  }
  
  buildTag(tag) {
    const attrs = Object.entries(tag.attrs || {})
      .map(([key, value]) => `${key}="${value}"`)
      .join(' ')
    
    if (tag.type === 'script') {
      if (tag.content) {
        return `<script ${attrs}>${tag.content}</script>`
      }
      return `<script ${attrs}></script>`
    } else if (tag.type === 'style') {
      return `<style ${attrs}>${tag.content || ''}</style>`
    } else if (tag.type === 'meta') {
      return `<meta ${attrs}>`
    } else if (tag.type === 'link') {
      return `<link ${attrs}>`
    }
    
    return ''
  }
}

module.exports = HtmlInjectPlugin

// 使用示例
new HtmlInjectPlugin({
  tags: [
    {
      type: 'meta',
      attrs: { name: 'version', content: '1.0.0' },
      position: 'head'
    },
    {
      type: 'script',
      content: 'console.log("Build time: ' + new Date().toISOString() + '")',
      position: 'body'
    }
  ]
})
```

### 构建通知插件

```javascript
class BuildNotifyPlugin {
  constructor(options = {}) {
    this.webhook = options.webhook
    this.title = options.title || 'Webpack Build'
  }
  
  apply(compiler) {
    compiler.hooks.done.tapPromise('BuildNotifyPlugin', async (stats) => {
      if (!this.webhook) return
      
      const { compilation } = stats
      
      const message = {
        text: `## ${this.title}`,
        attachments: [{
          color: compilation.errors.length > 0 ? 'danger' : 'good',
          fields: [
            {
              title: 'Status',
              value: compilation.errors.length > 0 ? 'Failed' : 'Success',
              short: true
            },
            {
              title: 'Duration',
              value: `${(compilation.endTime - compilation.startTime) / 1000}s`,
              short: true
            },
            {
              title: 'Assets',
              value: Object.keys(compilation.assets).length.toString(),
              short: true
            },
            {
              title: 'Modules',
              value: compilation.modules.size.toString(),
              short: true
            }
          ]
        }]
      }
      
      if (compilation.errors.length > 0) {
        message.attachments[0].fields.push({
          title: 'Errors',
          value: compilation.errors.slice(0, 5).map(e => e.message).join('\n'),
          short: false
        })
      }
      
      try {
        const https = require('https')
        const url = new URL(this.webhook)
        
        return new Promise((resolve) => {
          const req = https.request({
            hostname: url.hostname,
            path: url.pathname,
            method: 'POST',
            headers: { 'Content-Type': 'application/json' }
          }, () => resolve())
          
          req.write(JSON.stringify(message))
          req.end()
        })
      } catch (error) {
        console.error('Failed to send notification:', error)
      }
    })
  }
}

module.exports = BuildNotifyPlugin
```

### 清理控制台插件

```javascript
const { RawSource } = require('webpack-sources')

class CleanConsolePlugin {
  constructor(options = {}) {
    this.exclude = options.exclude || []
    this.include = options.include || []
  }
  
  apply(compiler) {
    const regex = /console\.(log|info|debug|warn)\([^)]*\);?/g
    
    compiler.hooks.compilation.tap('CleanConsolePlugin', (compilation) => {
      compilation.hooks.processAssets.tap(
        {
          name: 'CleanConsolePlugin',
          stage: compilation.PROCESS_ASSETS_STAGE_OPTIMIZE
        },
        (assets) => {
          Object.keys(assets).forEach(filename => {
            // 排除指定文件
            if (this.exclude.some(pattern => filename.includes(pattern))) {
              return
            }
            
            // 只处理包含的文件
            if (this.include.length > 0 && 
                !this.include.some(pattern => filename.includes(pattern))) {
              return
            }
            
            if (filename.endsWith('.js')) {
              const source = assets[filename].source()
              const cleaned = source.replace(regex, '')
              
              compilation.updateAsset(filename, new RawSource(cleaned))
            }
          })
        }
      )
    })
  }
}

module.exports = CleanConsolePlugin
```

### 5. 错误处理

class MyPlugin {

          // 添加编译错误

          return

          // 添加编译警告
          compilation.warnings.push(new Error('Warning message'))
        }
        
        callback()
      } catch (error) {
        // 回调错误，中断构建
        callback(error)
      }
    })
  }
}

```javascript
class MyPlugin {
  apply(compiler) {
    let startTime
    
    compiler.hooks.run.tap('MyPlugin', () => {
      startTime = Date.now()
    })
    
    compiler.hooks.emit.tapAsync('MyPlugin', (compilation, callback) => {
      // 避免重复处理
      if (compilation.__myPluginProcessed) {
        callback()
        return
      }
      
      // 处理逻辑
      this.processAssets(compilation)
      
      compilation.__myPluginProcessed = true
      
      console.log(`Plugin took ${Date.now() - startTime}ms`)
      callback()
    })
  }
}
```

### 2. 使用 Webpack Stats

```javascript
compiler.hooks.done.tap('MyPlugin', (stats) => {
  // 输出详细统计
  console.log(stats.toString({
    colors: true,
    modules: true,
    chunks: true,
    reasons: true
  }))
  
  // 获取模块信息
  stats.compilation.modules.forEach(module => {
    console.log('Module:', module.resource)
    console.log('  Dependencies:', module.dependencies.length)
    console.log('  Size:', module.size())
  })
})
```
