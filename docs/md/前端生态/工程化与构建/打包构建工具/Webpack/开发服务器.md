# 开发服务器详解

## 概述

webpack-dev-server 是一个开发环境服务器，提供了实时重载、热模块替换（HMR）、代理等功能，极大地提升了开发体验。

## 安装

```bash
npm install webpack-dev-server --save-dev
```

## 基础配置

### 最简配置

```javascript
// webpack.config.js
module.exports = {
  devServer: {
    static: './dist',
    port: 3000,
    open: true
  }
}
```

### 启动命令

```json
// package.json
{
  "scripts": {
    "dev": "webpack serve --mode development",
    "dev:open": "webpack serve --open",
    "dev:port": "webpack serve --port 4000"
  }
}
```

## 核心配置项

### devServer.static

配置静态文件服务。

```javascript
module.exports = {
  devServer: {
    static: {
      // 静态文件目录
      directory: path.join(__dirname, 'public'),

      // 公共路径
      publicPath: '/public',

      // 静态文件选项
      staticOptions: {
        redirect: true
      },

      // 监听配置
      watch: {
        ignored: /node_modules/,
        aggregateTimeout: 300,
        poll: 1000
      },

      // 服务器索引
      serveIndex: true
    }
  }
}

// 多个静态目录
module.exports = {
  devServer: {
    static: [
      {
        directory: path.join(__dirname, 'public'),
        publicPath: '/public'
      },
      {
        directory: path.join(__dirname, 'static'),
        publicPath: '/static'
      }
    ]
  }
}
```

### devServer.host 和 devServer.port

```javascript
module.exports = {
  devServer: {
    // 主机名
    host: '0.0.0.0',  // 允许外部访问
    host: 'localhost',  // 仅本地访问
    host: '192.168.1.100',  // 指定 IP

    // 端口号
    port: 3000,  // 固定端口
    port: 'auto',  // 自动选择可用端口

    // 严格端口
    strictPort: true,  // 端口被占用时报错
    strictPort: false  // 端口被占用时尝试下一个
  }
}
```

### devServer.hot

热模块替换配置。

```javascript
module.exports = {
  devServer: {
    hot: true,      // 启用 HMR
    hot: 'only',    // 编译失败时不刷新页面
    hot: false      // 禁用 HMR
  }
}
```

### devServer.open

自动打开浏览器。

```javascript
module.exports = {
  devServer: {
    open: true,  // 自动打开浏览器
    open: false, // 不自动打开

    // 指定浏览器
    open: {
      app: {
        name: 'Google Chrome',
        arguments: ['--incognito', '--new-window']
      }
    },

    // 指定打开路径
    open: ['/admin', '/dashboard']
  }
}
```

### devServer.compress

启用 gzip 压缩。

```javascript
module.exports = {
  devServer: {
    compress: true  // 启用 gzip 压缩所有服务
  }
}
```

## 代理配置详解

### 基础代理

```javascript
module.exports = {
  devServer: {
    proxy: {
      '/api': 'http://localhost:8080'
    }
  }
}

// 请求 /api/users → http://localhost:8080/api/users
```

### 高级代理配置

```javascript
module.exports = {
  devServer: {
    proxy: {
      '/api': {
        // 目标地址
        target: 'http://localhost:8080',

        // 修改请求头中的 host
        changeOrigin: true,

        // 路径重写
        pathRewrite: {
          '^/api': ''  // 移除 /api 前缀
        },

        // 是否验证 SSL 证书
        secure: false,

        // 请求头
        headers: {
          'X-Custom-Header': 'value',
          'Cookie': 'session=abc123'
        },

        // 超时时间
        timeout: 30000,

        // 代理响应延迟（调试用）
        proxyTimeout: 30000
      }
    }
  }
}
```

### 请求/响应拦截

```javascript
module.exports = {
  devServer: {
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,

        // 请求拦截
        onProxyReq(proxyReq, req, res) {
          console.log('[Proxy Request]', req.method, req.url)
          proxyReq.setHeader('X-Requested-With', 'XMLHttpRequest')
        },

        // 响应拦截
        onProxyRes(proxyRes, req, res) {
          console.log('[Proxy Response]', proxyRes.statusCode, req.url)
          proxyRes.headers['x-proxy'] = 'webpack-dev-server'
        },

        // 错误处理
        onError(err, req, res) {
          console.error('[Proxy Error]', err)
          res.writeHead(500, {
            'Content-Type': 'text/plain'
          })
          res.end('Proxy Error: ' + err.message)
        }
      }
    }
  }
}
```

### 多个代理配置

```javascript
module.exports = {
  devServer: {
    proxy: [
      {
        context: ['/api', '/auth'],
        target: 'http://localhost:8080',
        changeOrigin: true
      },
      {
        context: '/upload',
        target: 'http://localhost:9000',
        changeOrigin: true
      },
      {
        context: '/socket',
        target: 'ws://localhost:9090',
        ws: true  // WebSocket 代理
      }
    ]
  }
}
```

### WebSocket 代理

```javascript
module.exports = {
  devServer: {
    proxy: {
      '/socket': {
        target: 'ws://localhost:9090',
        ws: true,  // 启用 WebSocket 代理
        changeOrigin: true
      }
    }
  }
}
```

### 条件代理

```javascript
module.exports = {
  devServer: {
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        bypass(req, res, proxyOptions) {
          // 返回 false 跳过代理
          if (req.headers.accept.indexOf('html') !== -1) {
            return '/index.html'
          }

          // 返回路径重写请求
          return null
        }
      }
    }
  }
}
```

## HMR（热模块替换）

### 启用 HMR

```javascript
// webpack.config.js
const webpack = require('webpack')

module.exports = {
  devServer: {
    hot: true
  },
  plugins: [
    new webpack.HotModuleReplacementPlugin()
  ]
}
```

### 在代码中使用 HMR

```javascript
// JavaScript 模块
if (module.hot) {
  module.hot.accept('./module.js', () => {
    console.log('模块已更新')
    // 更新逻辑
  })

  module.hot.dispose(() => {
    // 清理逻辑
  })
}

// Vue 组件（vue-loader 自动处理）
// 无需额外配置

// React 组件
import React from 'react'

if (module.hot) {
  module.hot.accept()
}
```

### React HMR 完整配置

```javascript
// webpack.config.js
const ReactRefreshWebpackPlugin = require('@pmmmwh/react-refresh-webpack-plugin')

module.exports = {
  devServer: {
    hot: true
  },
  module: {
    rules: [
      {
        test: /\.jsx?$/,
        exclude: /node_modules/,
        use: {
          loader: 'babel-loader',
          options: {
            presets: ['@babel/preset-react'],
            plugins: ['react-refresh/babel']
          }
        }
      }
    ]
  },
  plugins: [
    new ReactRefreshWebpackPlugin()
  ]
}
```

### Vue HMR 配置

```javascript
// webpack.config.js
const { VueLoaderPlugin } = require('vue-loader')

module.exports = {
  devServer: {
    hot: true
  },
  module: {
    rules: [
      {
        test: /\.vue$/,
        use: 'vue-loader'
      }
    ]
  },
  plugins: [
    new VueLoaderPlugin()
  ]
}

// vue-loader 自动处理 HMR，无需额外配置
```

## History 路由配置

### 基础配置

```javascript
module.exports = {
  devServer: {
    historyApiFallback: true  // 所有 404 响应返回 index.html
  }
}
```

### 高级配置

```javascript
module.exports = {
  devServer: {
    historyApiFallback: {
      // 主入口
      index: '/index.html',

      // 路径重写
      rewrites: [
        { from: /^\/admin/, to: '/admin.html' },
        { from: /^\/dashboard/, to: '/dashboard.html' },
        { from: /\/page\/\d+/, to: '/index.html' }
      ],

      // 禁用 . 规则
      disableDotRule: true,

      // HTML 后缀
      htmlAcceptHeaders: ['text/html', 'application/xhtml+xml']
    }
  }
}
```

### 多页面应用路由

```javascript
module.exports = {
  entry: {
    index: './src/index.js',
    admin: './src/admin.js'
  },
  plugins: [
    new HtmlWebpackPlugin({
      template: './src/index.html',
      filename: 'index.html',
      chunks: ['index']
    }),
    new HtmlWebpackPlugin({
      template: './src/admin.html',
      filename: 'admin.html',
      chunks: ['admin']
    })
  ],
  devServer: {
    historyApiFallback: {
      rewrites: [
        { from: /^\/admin/, to: '/admin.html' },
        { from: /^\/$/, to: '/index.html' }
      ]
    }
  }
}
```

## HTTPS 配置

### 自动生成证书

```bash
# 安装
npm install --save-dev @pmmmwh/react-refresh-webpack-plugin

# 启动 HTTPS
npx webpack serve --https
```

### 自定义证书

```javascript
const fs = require('fs')

module.exports = {
  devServer: {
    https: {
      key: fs.readFileSync('server.key'),
      cert: fs.readFileSync('server.crt'),
      ca: fs.readFileSync('ca.pem'),
      pfx: fs.readFileSync('server.pfx'),
      passphrase: 'webpack-dev-server'
    }
  }
}
```

### mkcert 生成本地证书

```bash
# 安装 mkcert
brew install mkcert

# 生成证书
mkcert -install
mkcert localhost

# webpack.config.js
module.exports = {
  devServer: {
    https: {
      key: fs.readFileSync('localhost-key.pem'),
      cert: fs.readFileSync('localhost.pem')
    }
  }
}
```

## 控制台配置

### devServer.client

```javascript
module.exports = {
  devServer: {
    client: {
      // 日志级别
      logging: 'info',  // 'silent' | 'trace' | 'debug' | 'info' | 'warn' | 'error'

      // 浏览器控制台日志
      log: 'info',

      // 警告
      warnings: true,
      errors: true,

      // 进度显示
      progress: true,

      // 覆盖层配置
      overlay: {
        errors: true,     // 显示错误
        warnings: false   // 不显示警告
      },

      // 重新连接配置
      reconnect: true,
      reconnectAttempts: 10
    }
  }
}
```

### 自定义覆盖层

```javascript
module.exports = {
  devServer: {
    client: {
      overlay: {
        runtimeErrors: (error) => {
          const ignoreErrors = [
            "ResizeObserver loop completed with undelivered notifications"
          ]

          if (ignoreErrors.some(e => error.message.includes(e))) {
            return false
          }

          return true
        }
      }
    }
  }
}
```

## 开发工具配置

### devServer.devMiddleware

```javascript
module.exports = {
  devServer: {
    devMiddleware: {
      // 索引文件
      index: 'index.html',

      // 写入磁盘
      writeToDisk: true,  // 所有文件写入磁盘
      writeToDisk: (filePath) => {
        return /\.html$/.test(filePath)  // 只写 HTML 文件

      // MIME 类型
      mimeTypes: {
        'text/html': ['phtml']
      },

      // 服务端公共路径
      publicPath: '/assets/',

      // 服务器索引
      serverSideRender: true,

      // Stats 配置
      stats: 'errors-only'
    }
  }
}
```

## CORS 配置

```javascript
module.exports = {
  devServer: {
    // 启用 CORS
    cors: true,

    // 详细配置
    cors: {
      origin: '*',
      methods: ['GET', 'POST', 'PUT', 'DELETE'],
      allowedHeaders: ['Content-Type', 'Authorization'],
      credentials: true,
      maxAge: 86400
    },

    // 或通过 headers 配置
    headers: {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type, Authorization',
      'Access-Control-Allow-Credentials': 'true'
    }
  }
}
```

### devServer.onAfterSetupMiddleware 和 onBeforeSetupMiddleware

```javascript

  devServer: {
    onBeforeSetupMiddleware(devServer) {
      // 中间件之前执行
      devServer.app.get('/before', (req, res) => {
        res.json({ message: 'Before middleware' })
      })
    },

    onAfterSetupMiddleware(devServer) {
      // 中间件之后执行
      devServer.app.get('/after', (req, res) => {
        res.json({ message: 'After middleware' })
      })
    }
  }
}
```

## 完整配置示例

### 开发环境完整配置

```javascript
const path = require('path')
const webpack = require('webpack')
const HtmlWebpackPlugin = require('html-webpack-plugin')
const ReactRefreshWebpackPlugin = require('@pmmmwh/react-refresh-webpack-plugin')

module.exports = {
  mode: 'development',
  entry: './src/index.js',
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: '[name].js',
    publicPath: '/'
  },
  devServer: {
    // 基础配置
    host: '0.0.0.0',
    port: 3000,
    hot: true,
    open: true,
    compress: true,

    // 静态文件
    static: {
      directory: path.join(__dirname, 'public'),
      publicPath: '/public',
      watch: {
        ignored: /node_modules/
      }
    },

    // 代理
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,
        pathRewrite: { '^/api': '' }
      }
    },

    // History 路由
    historyApiFallback: true,

    // CORS
    cors: true,

    // 客户端配置
    client: {
      logging: 'info',
      overlay: {
        errors: true,
        warnings: false
      },
      progress: true
    },

    // 开发工具
    devMiddleware: {
      stats: 'errors-only'
    }
  },
  module: {
    rules: [
      {
        test: /\.jsx?$/,
        exclude: /node_modules/,
        use: {
          loader: 'babel-loader',
          options: {
            presets: ['@babel/preset-env', '@babel/preset-react'],
            plugins: ['react-refresh/babel']
          }
        }
      },
      {
        test: /\.css$/,
        use: ['style-loader', 'css-loader']
      }
    ]
  },
  plugins: [
    new HtmlWebpackPlugin({
      template: './public/index.html'
    }),
    new webpack.HotModuleReplacementPlugin(),
    new ReactRefreshWebpackPlugin()
  ],
  resolve: {
    extensions: ['.js', '.jsx', '.json'],
    alias: {
      '@': path.resolve(__dirname, 'src')
    }
  },
  devtool: 'eval-cheap-module-source-map'
}
```

## 常见问题解决

### 1. 端口被占用

```javascript
module.exports = {
  devServer: {
    port: 'auto',  // 自动选择可用端口
    strictPort: false  // 端口被占用时尝试下一个
  }
}
```

### 2. 跨域问题

```javascript
module.exports = {
  devServer: {
    // 方式一：设置 CORS
    headers: {
      'Access-Control-Allow-Origin': '*'
    },

    // 方式二：代理
    proxy: {
      '/api': {
        target: 'http://api.example.com',
        changeOrigin: true
      }
    }
  }
}
```

### 3. 热更新不生效

```javascript
// 检查配置
module.exports = {
  devServer: {
    hot: true,  // 启用 HMR
    inline: true  // 内联模式（webpack 4）
  },
  plugins: [
    new webpack.HotModuleReplacementPlugin()
  ]
}

// 检查代码
if (module.hot) {
  module.hot.accept()
}
```

### 4. 内存溢出

```javascript
// package.json
{
  "scripts": {
    "dev": "node --max-old-space-size=4096 node_modules/webpack/bin/webpack.js serve"
  }
}
```

### 5. 静态资源不更新

```javascript
module.exports = {
  devServer: {
    static: {
      watch: {
        ignored: /node_modules/,
        aggregateTimeout: 300,
        poll: 1000
      }
    }
  }
}
```
