# 优化配置详解

## 概述

从 webpack 4 开始，会根据你选择的 `mode` 来执行不同的优化，不过所有的优化还是可以手动配置和重写。

## optimization.minimize

```javascript
module.exports = {
  optimization: {
    minimize: true  // 启用压缩
  }
}
```

## optimization.minimizer

### 基本配置

```javascript
const TerserPlugin = require('terser-webpack-plugin')
const CssMinimizerPlugin = require('css-minimizer-webpack-plugin')

module.exports = {
  optimization: {
    minimize: true,
    minimizer: [
      new TerserPlugin({
        parallel: true,
        terserOptions: {
          compress: {
            drop_console: true
          }
        }
      }),
      new CssMinimizerPlugin()
    ]
  }
}
```

### TerserPlugin 配置

```javascript
module.exports = {
  optimization: {
    minimizer: [
      new TerserPlugin({
        parallel: true,  // 多进程
        extractComments: false,  // 不提取注释
        terserOptions: {
          parse: {
            ecma: 2015
          },
          compress: {
            ecma: 2015,
            warnings: true,
            comparisons: false,
            inline: 2,
            drop_console: true,
            drop_debugger: true,
            pure_funcs: ['console.log']
          },
          mangle: {
            safari10: true
          },
          output: {
            ecma: 2015,
            comments: false,
            ascii_only: true
          }
        }
      })
    ]
  }
}
```

## optimization.splitChunks

### 基本配置

```javascript
module.exports = {
  optimization: {
    splitChunks: {
      chunks: 'all'
    }
  }
}
```

### 完整配置

```javascript
module.exports = {
  optimization: {
    splitChunks: {
      chunks: 'all',  // 'initial' | 'async' | 'all'
      minSize: 20000,  // 最小尺寸
      minRemainingSize: 0,
      maxSize: 244000,  // 最大尺寸
      minChunks: 1,  // 最少引用次数
      maxAsyncRequests: 30,  // 异步最大请求数
      maxInitialRequests: 30,  // 入口最大请求数
      automaticNameDelimiter: '~',  // 名称分隔符
      enforceSizeThreshold: 50000,
      cacheGroups: {
        defaultVendors: {
          test: /[\\/]node_modules[\\/]/,
          priority: -10,
          reuseExistingChunk: true
        },
        default: {
          minChunks: 2,
          priority: -20,
          reuseExistingChunk: true
        }
      }
    }
  }
}
```

### cacheGroups 配置

```javascript
module.exports = {
  optimization: {
    splitChunks: {
      chunks: 'all',
      cacheGroups: {
        // 第三方库
        vendor: {
          test: /[\\/]node_modules[\\/]/,
          name: 'vendors',
          chunks: 'all',
          priority: 10
        },

        // UI 库
        elementUI: {
          test: /[\\/]node_modules[\\/]element-ui[\\/]/,
          name: 'element-ui',
          chunks: 'all',
          priority: 20
        },

        // 公共模块
        commons: {
          minChunks: 2,
          name: 'commons',
          chunks: 'initial',
          priority: 5,
          reuseExistingChunk: true
        },

        // 样式
        styles: {
          name: 'styles',
          test: /\.css$/,
          chunks: 'all',
          enforce: true
        }
      }
    }
  }
}
```

### 函数形式

```javascript
module.exports = {
  optimization: {
    splitChunks: {
      cacheGroups: {
        vendor: {
          test(module, chunks) {
            const moduleName = module.identifier()
            return moduleName.includes('node_modules')
          },
          name(module, chunks, cacheGroupKey) {
            const moduleName = module.identifier()
            const match = moduleName.match(/[\\/]node_modules[\\/](.*?)([\\/]|$)/)
            const packageName = match ? match[1] : 'vendor'
            return `${cacheGroupKey}-${packageName}`
          }
        }
      }
    }
  }
}
```

## optimization.runtimeChunk


## 完整优化配置示例

### 开发环境

```javascript
module.exports = {
  mode: 'development',
  optimization: {
    usedExports: true,
    splitChunks: {
      chunks: 'async'
    }
  }
}
```

### 生产环境

```javascript
const TerserPlugin = require('terser-webpack-plugin')
const CssMinimizerPlugin = require('css-minimizer-webpack-plugin')

module.exports = {
  mode: 'production',
  optimization: {
    minimize: true,
    minimizer: [
      new TerserPlugin({
        parallel: true,
        extractComments: false
      }),
      new CssMinimizerPlugin()
    ],
    splitChunks: {
      chunks: 'all',
      minSize: 20000,
      maxSize: 244000,
      minChunks: 1,
      maxAsyncRequests: 30,
      maxInitialRequests: 30,
      cacheGroups: {
        vendor: {
          test: /[\\/]node_modules[\\/]/,
          name: 'vendors',
          chunks: 'all',
          priority: 10
        }
      }
    },
    runtimeChunk: 'single',
    noEmitOnErrors: true,
    providedExports: true,
    usedExports: true,
    concatenateModules: true,
    sideEffects: true
  }
}
```

### 最佳实践配置

```javascript
module.exports = {
  optimization: {
    // 压缩
    minimize: true,
    minimizer: [
      '...',  // 保留默认压缩器
      new CssMinimizerPlugin()
    ],

    // 代码分割
    splitChunks: {
      chunks: 'all',
      minSize: 20000,
      minRemainingSize: 0,
      minChunks: 1,
      maxAsyncRequests: 30,
      maxInitialRequests: 30,
      enforceSizeThreshold: 50000,
      cacheGroups: {
        defaultVendors: {
          test: /[\\/]node_modules[\\/]/,
          priority: -10,
          reuseExistingChunk: true
        },
        default: {
          minChunks: 2,
          priority: -20,
          reuseExistingChunk: true
        }
      }
    },

    // 运行时代码
    runtimeChunk: 'single',

    // 模块优化
    providedExports: true,
    usedExports: true,
    concatenateModules: true,
    sideEffects: true,

    // Chunk 优化
    removeAvailableModules: true,
    removeEmptyChunks: true,
    mergeDuplicateChunks: true,
    flagIncludedChunks: true
  }
}
```
