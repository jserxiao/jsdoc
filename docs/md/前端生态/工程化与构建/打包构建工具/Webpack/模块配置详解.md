# 模块配置详解

## 概述

这些选项决定了如何处理项目中的不同类型的模块。

## module.rules

### 基本结构

```javascript
module.exports = {
  module: {
    rules: [
      {
        test: /\.css$/,
        use: ['style-loader', 'css-loader']
      }
    ]
  }
}
```

### 条件匹配

```javascript
module.exports = {
  module: {
    rules: [
      {
        // 1. test：匹配文件路径
        test: /\.js$/,

        // 2. include：包含的目录
        include: path.resolve(__dirname, 'src'),

        // 3. exclude：排除的目录
        exclude: /node_modules/,

        // 4. resource：资源条件
        resource: {
          test: /\.js$/,
          include: path.resolve(__dirname, 'src')
        },

        // 5. issuer：模块来源条件
        issuer: {
          test: /\.js$/
        },

        // 6. and：所有条件都满足
        and: [
          { test: /\.js$/ },
          { include: path.resolve(__dirname, 'src') }
        ],

        // 7. or：任一条件满足
        or: [
          { test: /\.js$/ },
          { test: /\.ts$/ }
        ],

        // 8. not：条件不满足
        not: [
          { test: /\.spec\.js$/ }
        ]
      }
    ]
  }
}
```

### use 配置

```javascript
module.exports = {
  module: {
    rules: [
      // 字符串形式
      {
        test: /\.css$/,
        use: 'css-loader'
      },

      // 数组形式
      {
        test: /\.css$/,
        use: ['style-loader', 'css-loader']
      },

      // 对象形式
      {
        test: /\.css$/,
        use: [
          {
            loader: 'style-loader'
          },
          {
            loader: 'css-loader',
            options: {
              modules: true
            }
          }
        ]
      },

      // 函数形式
      {
        test: /\.css$/,
        use: (info) => [
          'style-loader',
          {
            loader: 'css-loader',
            options: { modules: info.resource.includes('module') }
          }
        ]
      }
    ]
  }
}
```

### oneOf 规则

```javascript
module.exports = {
  module: {
    rules: [
      {
        test: /\.js$/,
        oneOf: [
          {
            // 只匹配一个规则
            test: /\.spec\.js$/,
            use: 'spec-loader'
          },
          {
            use: 'babel-loader'
          }
        ]
      }
    ]
  }
}
```

### use 配置选项

```javascript
module.exports = {
  module: {
    rules: [
      {
        test: /\.js$/,
        use: [
          {
            loader: 'babel-loader',
            options: {
              presets: ['@babel/preset-env'],
              cacheDirectory: true
            }
          }
        ],
        type: 'javascript/auto'  // 模块类型
      }
    ]
  }
}
```

## module.parser

### JavaScript 解析器

```javascript
module.exports = {
  module: {
    parser: {
      javascript: {
        commonjsMagicComments: true,
        exportsPresence: 'error',
        importExportsPresence: 'error',
        reexportExportsPresence: 'error',
        strictExportPresence: false,
        strictThisContextOnImports: false,
        unknownContextRequest: '.',
        unknownContextRecursive: true,
        unknownContextRegExp: /^\.\/.*$/,
        unknownContextCritical: true,
        exprContextRequest: '.',
        exprContextRegExp: /^\.\/.*$/,
        exprContextRecursive: true,
        exprContextCritical: true,
        wrappedContextRegExp: /.*/,
        wrappedContextRecursive: true,
        wrappedContextCritical: false
      }
    }
  }
}
```

## module.generator

### 资源生成器

```javascript
module.exports = {
  module: {
    generator: {
      asset: {
        dataUrl: {
          encoding: 'base64',
          mimeType: 'image/png'
        },
        filename: 'assets/[hash][ext][query]'
      },
      'asset/inline': {
        dataUrl: {
          encoding: 'base64',
          mimeType: 'image/svg+xml'
        }
      },
      'asset/resource': {
        filename: 'assets/[hash][ext][query]'
      }
    }
  }
}
```

## module.noParse

```javascript
module.exports = {
  module: {
    // 不解析的文件
    noParse: /jquery|lodash/,

    // 或使用函数
    noParse: (content) => {
      return /jquery|lodash/.test(content)
    },

    // 或使用数组
    noParse: [/jquery/, /lodash/]
  }
}
```

## module.unsafeCache

```javascript
module.exports = {
  module: {
    unsafeCache: false,  // 禁用缓存

    // 或指定缓存条件
    unsafeCache: (module) => {
      return module.resource.includes('node_modules')
    }
  }
}
```

## 资源模块 (Asset Modules)

### 类型说明

| 类型 | 说明 | 用途 |
|------|------|------|
| `asset/resource` | 发送单独文件 | 大文件 |
| `asset/inline` | 导出 data URI | 小文件 |
| `asset/source` | 导出源代码 | 文本文件 |
| `asset` | 自动选择 | 智能处理 |

### asset/resource

```javascript
module.exports = {
  module: {
    rules: [
      {
        test: /\.png$/,
        type: 'asset/resource',
        generator: {
          filename: 'images/[hash][ext][query]'
        }
      }
    ]
  }
}
```

### asset/inline

```javascript
module.exports = {
  module: {
    rules: [
      {
        test: /\.svg$/,
        type: 'asset/inline',
        generator: {
          dataUrl: {
            encoding: 'base64',
            mimeType: 'image/svg+xml'
          }
        }
      }
    ]
  }
}
```

### asset/source

```javascript
module.exports = {
  module: {
    rules: [
      {
        test: /\.txt$/,
        type: 'asset/source'
      }
    ]
  }
}
```

### asset（智能选择）

```javascript
module.exports = {
  module: {
    rules: [
      {
        test: /\.(png|jpg|gif)$/,
        type: 'asset',
        parser: {
          dataUrlCondition: {
            maxSize: 8 * 1024  // 8KB 以下内联
          }
        },
        generator: {
          filename: 'images/[hash][ext][query]'
        }
      }
    ]
  }
}
```

## 常见 Loader 配置

### JavaScript

```javascript
module.exports = {
  module: {
    rules: [
      // Babel
      {
        test: /\.js$/,
        exclude: /node_modules/,
        use: {
          loader: 'babel-loader',
          options: {
            presets: ['@babel/preset-env'],
            plugins: ['@babel/plugin-transform-runtime'],
            cacheDirectory: true
          }
        }
      },

      // TypeScript
      {
        test: /\.tsx?$/,
        use: 'ts-loader',
        exclude: /node_modules/
      },

      // esbuild
      {
        test: /\.js$/,
        loader: 'esbuild-loader',
        options: {
          loader: 'js',
          target: 'es2015'
        }
      }
    ]
  }
}
```

### CSS

```javascript
module.exports = {
  module: {
    rules: [
      // CSS
      {
        test: /\.css$/,
        use: [
          'style-loader',
          {
            loader: 'css-loader',
            options: {
              modules: true,
              sourceMap: true,
              importLoaders: 1
            }
          },
          'postcss-loader'
        ]
      },

      // Sass
      {
        test: /\.scss$/,
        use: [
          'style-loader',
          'css-loader',
          'postcss-loader',
          {
            loader: 'sass-loader',
            options: {
              sourceMap: true,
              sassOptions: {
                outputStyle: 'compressed'
              }
            }
          }
        ]
      },

      // Less
      {
        test: /\.less$/,
        use: [
          'style-loader',
          'css-loader',
          'postcss-loader',
          {
            loader: 'less-loader',
            options: {
              lessOptions: {
                modifyVars: {
                  'primary-color': '#1890ff'
                },
                javascriptEnabled: true
              }
            }
          }
        ]
      }
    ]
  }
}
```

### 图片和字体

```javascript
module.exports = {
  module: {
    rules: [
      // 图片
      {
        test: /\.(png|jpg|gif|svg|webp)$/,
        type: 'asset',
        parser: {
          dataUrlCondition: {
            maxSize: 8 * 1024
          }
        },
        generator: {
          filename: 'images/[name].[hash:8][ext]'
        }
      },

      // 字体
      {
        test: /\.(woff|woff2|eot|ttf|otf)$/,
        type: 'asset/resource',
        generator: {
          filename: 'fonts/[name].[hash:8][ext]'
        }
      }
    ]
  }
}
```

## 最佳实践

### 开发环境

```javascript
module.exports = {
  module: {
    rules: [
      {
        test: /\.js$/,
        exclude: /node_modules/,
        use: {
          loader: 'babel-loader',
          options: {
            cacheDirectory: true
          }
        }
      },
      {
        test: /\.css$/,
        use: ['style-loader', 'css-loader']
      }
    ]
  }
}
```

### 生产环境

```javascript
const MiniCssExtractPlugin = require('mini-css-extract-plugin')

module.exports = {
  module: {
    rules: [
      {
        test: /\.js$/,
        exclude: /node_modules/,
        use: 'babel-loader'
      },
      {
        test: /\.css$/,
        use: [
          MiniCssExtractPlugin.loader,
          'css-loader',
          'postcss-loader'
        ]
      },
      {
        test: /\.(png|jpg|gif|svg)$/,
        type: 'asset',
        parser: {
          dataUrlCondition: {
            maxSize: 4 * 1024
          }
        }
      }
    ]
  }
}
```
