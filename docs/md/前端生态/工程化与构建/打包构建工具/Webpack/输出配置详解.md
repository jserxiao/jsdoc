# 输出配置详解

## 概述

output 属性告诉 webpack 在哪里输出它所创建的 bundle，以及如何命名这些文件。主要包含输出路径、文件名、公共路径等配置。

## 基本配置

```javascript
const path = require('path')

module.exports = {
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'bundle.js'
  }
}
```

## path（输出路径）

### 基本用法

```javascript
module.exports = {
  output: {
    path: path.resolve(__dirname, 'dist')
  }
}
```

### 动态路径

```javascript
module.exports = (env) => {
  return {
    output: {
      path: path.resolve(__dirname, `dist/${env.NODE_ENV}`)
    }
  }
}
```

### 相对路径

```javascript
module.exports = {
  output: {
    path: path.resolve(__dirname, 'dist'),
    // 输出目录相对于项目根目录
  }
}
```

## filename（文件名）

### 单入口文件名

```javascript
module.exports = {
  output: {
    filename: 'bundle.js'
  }
}
```

### 多入口文件名

```javascript
module.exports = {
  entry: {
    app: './src/app.js',
    admin: './src/admin.js'
  },
  output: {
    filename: '[name].js'  // app.js, admin.js
  }
}
```

### 占位符

| 占位符 | 说明 | 示例 |
|--------|------|------|
| `[name]` | 模块名称 | `app.js` |
| `[id]` | 模块标识符 | `1.js` |
| `[hash]` | 模块标识符的 hash | `bundle.abc123.js` |
| `[chunkhash]` | chunk 内容的 hash | `bundle.def456.js` |
| `[contenthash]` | 内容 hash（推荐） | `bundle.ghi789.js` |
| `[fullhash]` | 编译的 hash | `bundle.jkl012.js` |

```javascript
module.exports = {
  output: {
    // 使用 contenthash
    filename: '[name].[contenthash:8].js',

    // 使用 chunkhash
    chunkFilename: '[name].[chunkhash:8].chunk.js'
  }
}
```

### Hash 长度控制

```javascript
module.exports = {
  output: {
    filename: '[name].[contenthash:8].js',     // 8位 hash
    filename: '[name].[contenthash:16].js',    // 16位 hash
    filename: '[name].[contenthash].js'        // 默认 20位 hash
  }
}
```

### 函数形式

```javascript
module.exports = {
  output: {
    filename: (pathData) => {
      return pathData.chunk.name === 'main'
        ? '[name].js'
        : '[name]/[name].[contenthash:8].js'
    }
  }
}
```

## chunkFilename（Chunk 文件名）

```javascript
module.exports = {
  output: {
    filename: '[name].[contenthash:8].js',
    chunkFilename: '[name].[contenthash:8].chunk.js'
  }
}
```

## publicPath（公共路径）

### 基本用法

```javascript
module.exports = {
  output: {
    publicPath: '/assets/'
  }
}

// 生成的 HTML
// <script src="/assets/bundle.js"></script>
```

### CDN 配置

```javascript
module.exports = {
  output: {
    publicPath: 'https://cdn.example.com/assets/'
  }
}
```

### 相对路径

```javascript
module.exports = {
  output: {
    publicPath: './'  // 相对于 HTML 文件
  }
}
```

### 动态 publicPath

```javascript
// 在运行时设置
__webpack_public_path__ = window.cdnUrl

// 或在入口文件中
import('./path/to/module').then(module => {
  // ...
})
```

## assetModuleFilename（资源文件名）

```javascript
module.exports = {
  output: {
    assetModuleFilename: 'images/[hash][ext][query]'
  }
}

// 或在 rules 中配置
module.exports = {
  module: {
    rules: [
      {
        test: /\.png$/,
        type: 'asset/resource',
        generator: {
          filename: 'images/[hash][ext][query]'
        }
      }
    ]
  }
}
```

## library（库配置）

### 基本用法

```javascript
module.exports = {
  output: {
    library: 'MyLibrary'
  }
}
```

### 完整配置

```javascript
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'umd',
      export: 'default',
      auxiliaryComment: {
        root: 'Root Comment',
        commonjs: 'CommonJS Comment',
        commonjs2: 'CommonJS2 Comment',
        amd: 'AMD Comment'
      }
    }
  }
}
```

### library.type（输出格式）

```javascript
// 1. var（默认）
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'var'
    }
  }
}
// 输出: var MyLibrary = xxx

// 2. assign
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'assign'
    }
  }
}
// 输出: MyLibrary = xxx

// 3. this
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'this'
    }
  }
}
// 输出: this["MyLibrary"] = xxx

// 4. window
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'window'
    }
  }
}
// 输出: window["MyLibrary"] = xxx

// 5. commonjs
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'commonjs'
    }
  }
}
// 输出: exports["MyLibrary"] = xxx

// 6. commonjs2
module.exports = {
  output: {
    library: {
      type: 'commonjs2'
    }
  }
}
// 输出: module.exports = xxx

// 7. amd
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'amd'
    }
  }
}
// 输出: define("MyLibrary", [], function() { return xxx })

// 8. umd（通用模块定义）
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'umd'
    }
  }
}
// 支持 AMD, CommonJS, 和全局变量

// 9. system
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'system'
    }
  }
}
// SystemJS 模块

// 10. module（ES Module）
module.exports = {
  output: {
    library: {
      type: 'module'
    }
  },
  experiments: {
    outputModule: true
  }
}
```

### library.export

```javascript
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'umd',
      export: 'default'  // 导出默认导出
    }
  }
}
```

## globalObject（全局对象）

```javascript
module.exports = {
  output: {
    library: {
      name: 'MyLibrary',
      type: 'umd'
    },
    globalObject: 'this'
  }
}

// 或使用 self
module.exports = {
  output: {
    globalObject: 'self'
  }
}
```

## clean（清空输出目录）

### 基本用法

```javascript
module.exports = {
  output: {
    clean: true
  }
}
```

### 高级配置

```javascript
module.exports = {
  output: {
    clean: {
      dry: true,  // 仅报告，不删除
      keep: /ignored\/dir\//,  // 保留的文件
      keepAsset: (asset) => {
        return asset.includes('ignored')
      }
    }
  }
}
```

## sourceMapFilename（Source Map 文件名）

```javascript
module.exports = {
  output: {
    sourceMapFilename: '[name].map',
    sourceMapFilename: '[file].map',
    sourceMapFilename: '[contenthash].map'
  }
}
```

## devtoolModuleFilenameTemplate

```javascript
module.exports = {
  output: {
    devtoolModuleFilenameTemplate: 'webpack://[namespace]/[resource-path]?[loaders]',
    devtoolModuleFilenameTemplate: info => {
      return `file:///${path.resolve(info.absoluteResourcePath)}`
    }
  }
}
```

## 其他配置选项

### pathinfo

```javascript
module.exports = {
  output: {
    pathinfo: true,  // 包含路径信息
    pathinfo: false  // 不包含（生产环境推荐）
  }
}
```

### hashDigest

```javascript
module.exports = {
  output: {
    hashDigest: 'hex',      // hash 编码
    hashDigestLength: 20,   // hash 长度
    hashFunction: 'md4',    // hash 算法
    hashSalt: 'my-salt'     // hash 盐值
  }
}
```

### strictModuleExceptionHandling

```javascript
module.exports = {
  output: {
    strictModuleExceptionHandling: true
  }
}
```

### environment

```javascript
module.exports = {
  output: {
    environment: {
      arrowFunction: true,
      bigIntLiteral: false,
      const: true,
      destructuring: true,
      dynamicImport: false,
      forOf: true,
      module: false
    }
  }
}
```

## 多输出配置

### 多入口输出

```javascript
module.exports = {
  entry: {
    app: './src/app.js',
    admin: './src/admin.js'
  },
  output: {
    filename: '[name].[contenthash:8].js',
    path: path.resolve(__dirname, 'dist')
  }
}
```

### 多目标输出

```javascript
module.exports = [
  {
    name: 'client',
    entry: './src/client.js',
    output: {
      path: path.resolve(__dirname, 'dist/client'),
      filename: '[name].js'
    }
  },
  {
    name: 'server',
    entry: './src/server.js',
    output: {
      path: path.resolve(__dirname, 'dist/server'),
      filename: '[name].js',
      libraryTarget: 'commonjs2'
    }
  }
]
```

## 输出配置最佳实践

### 开发环境

```javascript
module.exports = {
  mode: 'development',
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: '[name].js',
    chunkFilename: '[name].chunk.js',
    publicPath: '/',
    pathinfo: true
  }
}
```

### 生产环境

```javascript
module.exports = {
  mode: 'production',
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: '[name].[contenthash:8].js',
    chunkFilename: '[name].[contenthash:8].chunk.js',
    assetModuleFilename: 'assets/[hash][ext][query]',
    publicPath: '/',
    clean: true,
    pathinfo: false
  }
}
```

### CDN 部署

```javascript
module.exports = {
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: '[name].[contenthash:8].js',
    publicPath: 'https://cdn.example.com/v1.0.0/',
    crossOriginLoading: 'anonymous'
  }
}
```

### 库开发

```javascript
module.exports = {
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'my-library.js',
    library: {
      name: 'MyLibrary',
      type: 'umd'
    },
    globalObject: 'this',
    auxiliaryComment: 'My Library v1.0.0'
  }
}
```
