# 入口配置详解

## 概述

入口起点(entry point)指示 webpack 应该使用哪个模块，来作为构建其内部依赖图的开始。进入入口起点后，webpack 会找出有哪些模块和库是入口起点（直接和间接）依赖的。

## 单入口配置

### 简写语法

```javascript
// webpack.config.js
module.exports = {
  entry: './src/index.js'
}
```

### 完整语法

```javascript
module.exports = {
  entry: {
    main: './src/index.js'
  }
}
```

### 入口数组

```javascript
// 多个文件作为一个入口
module.exports = {
  entry: ['./src/index.js', './src/other.js']
}

// 或使用对象语法
module.exports = {
  entry: {
    main: ['./src/index.js', './src/other.js']
  }
}
```

### 入口数组的使用场景

```javascript
// 1. 多个依赖文件一起注入
module.exports = {
  entry: {
    app: ['./src/index.js', './src/polyfills.js']
  }
}

// 2. 开发环境热更新
module.exports = {
  entry: {
    app: [
      'webpack-hot-middleware/client',
      './src/index.js'
    ]
  }
}

// 3. React Hot Loader
module.exports = {
  entry: {
    app: [
      'react-hot-loader/patch',
      './src/index.js'
    ]
  }
}
```

## 多入口配置

### 多页面应用

```javascript
module.exports = {
  entry: {
    home: './src/pages/home/index.js',
    about: './src/pages/about/index.js',
    contact: './src/pages/contact/index.js'
  },
  output: {
    filename: '[name].bundle.js',
    path: path.resolve(__dirname, 'dist')
  }
}
```

### 分离 vendor

```javascript
// 方式一：显式声明
module.exports = {
  entry: {
    main: './src/index.js',
    vendor: ['react', 'react-dom', 'react-router']
  }
}

// 方式二：使用 optimization.splitChunks
module.exports = {
  entry: './src/index.js',
  optimization: {
    splitChunks: {
      cacheGroups: {
        vendor: {
          test: /[\\/]node_modules[\\/]/,
          name: 'vendors',
          chunks: 'all'
        }
      }
    }
  }
}
```

### 分离 polyfills

```javascript
module.exports = {
  entry: {
    polyfills: './src/polyfills.js',
    main: './src/index.js'
  }
}

// polyfills.js
import 'core-js/stable'
import 'regenerator-runtime/runtime'
```

## 动态入口配置

### 使用函数

```javascript
module.exports = {
  entry: () => './src/index.js'
}
```

### 使用 Promise

```javascript
module.exports = {
  entry: () => new Promise((resolve) => resolve('./src/index.js'))
}
```

### 根据环境动态配置

```javascript
module.exports = (env) => {
  const isProduction = env.NODE_ENV === 'production'

  return {
    entry: isProduction ? './src/index.js' : './src/dev.js'
  }
}
```

### 动态读取多入口

```javascript
// 使用 glob 读取
const glob = require('glob')

module.exports = {
  entry: glob.sync('./src/pages/*/index.js').reduce((entries, path) => {
    // 提取页面名称
    const match = path.match(/\/pages\/(.+)\/index\.js/)
    const name = match ? match[1] : path

    entries[name] = path
    return entries
  }, {})
}

// 结果示例：
// {
//   home: './src/pages/home/index.js',
//   about: './src/pages/about/index.js',
//   contact: './src/pages/contact/index.js'
// }
```

### 命令行动态入口

```javascript
// package.json
{
  "scripts": {
    "build:page": "webpack --env.page"
  }
}

// webpack.config.js
module.exports = (env) => {
  const page = env.page || 'index'

  return {
    entry: `./src/pages/${page}/index.js`
  }
}

// 运行
// npm run build:page --env.page=home
```

## 入口配置选项

### dependOn

```javascript
module.exports = {
  entry: {
    app: {
      import: './src/app.js',
      dependOn: 'shared'
    },
    admin: {
      import: './src/admin.js',
      dependOn: 'shared'
    },
    shared: ['react', 'react-dom']
  }
}
```

### runtime

```javascript
module.exports = {
  entry: {
    main: {
      import: './src/index.js',
      runtime: 'runtime'
    }
  }
}
```

### library

```javascript
module.exports = {
  entry: {
    myLib: {
      import: './src/index.js',
      library: {
        name: 'MyLib',
        type: 'umd'
      }
    }
  }
}
```

## 入口配置最佳实践

### 单页应用 (SPA)

```javascript
module.exports = {
  entry: {
    app: './src/index.js',
    vendor: ['react', 'react-dom', 'react-router']
  },
  optimization: {
    splitChunks: {
      cacheGroups: {
        vendor: {
          test: /[\\/]node_modules[\\/]/,
          name: 'vendors',
          chunks: 'all'
        }
      }
    }
  }
}
```

### 多页应用 (MPA)

```javascript
// 自动读取入口
const glob = require('glob')
const path = require('path')

const entries = glob.sync('./src/pages/*/index.js').reduce((acc, filePath) => {
  const name = path.basename(path.dirname(filePath))
  acc[name] = filePath
  return acc
}, {})

module.exports = {
  entry: entries,
  optimization: {
    splitChunks: {
      chunks: 'all',
      cacheGroups: {
        commons: {
          name: 'commons',
          chunks: 'initial',
          minChunks: 2
        }
      }
    }
  }
}
```

### 库开发

```javascript
module.exports = {
  entry: './src/index.js',
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'my-library.js',
    library: {
      name: 'MyLibrary',
      type: 'umd'
    },
    globalObject: 'this'
  },
  externals: {
    react: {
      commonjs: 'react',
      commonjs2: 'react',
      amd: 'react',
      root: 'React'
    }
  }
}
```

### 微前端应用

```javascript
const { ModuleFederationPlugin } = require('webpack').container

module.exports = {
  entry: './src/index.js',
  plugins: [
    new ModuleFederationPlugin({
      name: 'app1',
      filename: 'remoteEntry.js',
      exposes: {
        './Button': './src/components/Button'
      },
      shared: ['react', 'react-dom']
    })
  ]
}
```

## 入口配置注意事项

### 1. 默认值

```javascript
// 如果没有指定 entry，默认使用
module.exports = {
  entry: './src/index.js'
}

// 输出默认为
module.exports = {
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'main.js'
  }
}
```

### 2. 路径问题

```javascript
// ✅ 正确：使用绝对路径或相对路径
module.exports = {
  entry: './src/index.js'
}

// ❌ 错误：路径不存在
module.exports = {
  entry: './not-exist.js'
}

// ✅ 使用 path 模块
module.exports = {
  entry: path.resolve(__dirname, 'src/index.js')
}
```

### 3. chunk 名称冲突

```javascript
// ❌ 错误：名称冲突
module.exports = {
  entry: {
    main: './src/index.js',
    vendor: ['react']
  },
  optimization: {
    splitChunks: {
      cacheGroups: {
        vendor: {  // 与入口名称冲突
          test: /[\\/]node_modules[\\/]/,
          name: 'vendor',
          chunks: 'all'
        }
      }
    }
  }
}

// ✅ 正确：使用不同名称
module.exports = {
  entry: {
    main: './src/index.js',
    vendors: ['react']
  },
  optimization: {
    splitChunks: {
      cacheGroups: {
        vendor: {
          test: /[\\/]node_modules[\\/]/,
          name: 'vendor',
          chunks: 'all'
        }
      }
    }
  }
}
```

## 入口配置调试

### 查看入口信息

```javascript
// webpack.config.js
const webpack = require('webpack')

module.exports = {
  plugins: [
    new webpack.debug.ProfilingPlugin(),
    {
      apply(compiler) {
        compiler.hooks.entryOption.tap('MyPlugin', (context, entry) => {
          console.log('Entry:', entry)
        })
      }
    }
  ]
}
```
