# 库模式

Vite 提供了库模式配置，用于开发 JavaScript 库、UI 组件库等。

## 基础配置

```javascript
// vite.config.js
import { defineConfig } from 'vite'
import { resolve } from 'path'

export default defineConfig({
  build: {
    lib: {
      // 入口文件
      entry: resolve(__dirname, 'src/index.ts'),
      // 全局变量名（UMD 格式）
      name: 'MyLib',
      // 输出文件名
      fileName: (format) => `my-lib.${format}.js`
    },
    
    // 外部化依赖
    rollupOptions: {
      external: ['vue'],
      output: {
        globals: {
          vue: 'Vue'
        }
      }
    }
  }
})
```

## package.json 配置

```json
{
  "name": "my-lib",
  "version": "1.0.0",
  "type": "module",
  "files": ["dist"],
  "main": "./dist/my-lib.umd.cjs",
  "module": "./dist/my-lib.js",
  "exports": {
    ".": {
      "import": "./dist/my-lib.js",
      "require": "./dist/my-lib.umd.cjs"
    },
    "./style.css": "./dist/style.css"
  },
  "types": "./dist/index.d.ts",
  "sideEffects": ["**/*.css"],
  "peerDependencies": {
    "vue": "^3.0.0"
  },
  "keywords": ["vue", "component", "library"],
  "license": "MIT"
}
```

## 多入口配置

```javascript
// vite.config.js
import { defineConfig } from 'vite'
import { resolve } from 'path'

export default defineConfig({
  build: {
    lib: {
      entry: {
        index: resolve(__dirname, 'src/index.ts'),
        utils: resolve(__dirname, 'src/utils/index.ts'),
        components: resolve(__dirname, 'src/components/index.ts')
      },
      name: 'MyLib'
    },
    rollupOptions: {
      external: ['vue'],
      output: {
        globals: {
          vue: 'Vue'
        }
      }
    }
  }
})
```

```json
// package.json
{
  "exports": {
    ".": {
      "import": "./dist/index.js",
      "require": "./dist/index.cjs"
    },
    "./utils": {
      "import": "./dist/utils.js",
      "require": "./dist/utils.cjs"
    },
    "./components": {
      "import": "./dist/components.js",
      "require": "./dist/components.cjs"
    },
    "./style.css": "./dist/style.css"
  }
}
```

## 输出格式

```javascript
// vite.config.js
export default defineConfig({
  build: {
    lib: {
      entry: 'src/index.ts',
      name: 'MyLib',
      // 多格式输出
      fileName: (format) => {
        switch (format) {
          case 'es':
            return 'index.mjs'
          case 'cjs':
            return 'index.cjs'
          case 'umd':
            return 'index.umd.js'
          case 'iife':
            return 'index.iife.js'
        }
      },
      formats: ['es', 'cjs', 'umd']
    }
  }
})
```

### 格式对比

| 格式 | 说明 | 使用场景 |
|------|------|----------|
| `es` | ES Module | 现代构建工具、浏览器 |
| `cjs` | CommonJS | Node.js、旧版打包工具 |
| `umd` | Universal Module Definition | CDN、浏览器脚本标签 |
| `iife` | 立即执行函数 | 浏览器脚本标签 |

## TypeScript 类型生成

### 使用 vite-plugin-dts

```bash
npm install -D vite-plugin-dts
```

```javascript
// vite.config.js
import { defineConfig } from 'vite'
import dts from 'vite-plugin-dts'

export default defineConfig({
  plugins: [
    dts({
      include: ['src/**/*.ts'],
      outDir: 'dist',
      // 生成单个类型文件
      rollupTypes: true
    })
  ],
  build: {
    lib: {
      entry: 'src/index.ts',
      name: 'MyLib',
      fileName: (format) => `index.${format}.js`
    }
  }
})
```

## CSS 处理

### 分离 CSS

```javascript
// vite.config.js
export default defineConfig({
  build: {
    lib: {
      entry: 'src/index.ts',
      name: 'MyLib',
      fileName: 'my-lib'
    },
    // 不将 CSS 注入 JS
    cssCodeSplit: true,
    rollupOptions: {
      output: {
        // CSS 文件名
        assetFileNames: (assetInfo) => {
          if (assetInfo.name === 'style.css') {
            return 'style.css'
          }
          return assetInfo.name
        }
      }
    }
  }
})
```

### CSS 使用方式

```javascript
// 方式 1：自动导入
import 'my-lib/style.css'

// 方式 2：手动导入
import { Button } from 'my-lib'
import 'my-lib/dist/style.css'
```

## Vue 组件库配置

```javascript
// vite.config.js
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import dts from 'vite-plugin-dts'
import { resolve } from 'path'

export default defineConfig({
  plugins: [
    vue(),
    dts({
      include: ['src/**/*.ts', 'src/**/*.vue'],
      outDir: 'dist/types'
    })
  ],
  build: {
    lib: {
      entry: resolve(__dirname, 'src/index.ts'),
      name: 'MyUI',
      fileName: (format) => `my-ui.${format}.js`
    },
    rollupOptions: {
      external: ['vue'],
      output: {
        globals: {
          vue: 'Vue'
        },
        exports: 'named'
      }
    }
  }
})
```

```typescript
// src/index.ts
import type { App } from 'vue'
import Button from './components/Button.vue'
import Input from './components/Input.vue'

const components = {
  Button,
  Input
}

const install = (app: App) => {
  Object.entries(components).forEach(([name, component]) => {
    app.component(name, component)
  })
}

export { Button, Input }
export default { install }
```

## React 组件库配置

```javascript
// vite.config.js
import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'
import dts from 'vite-plugin-dts'
import { resolve } from 'path'

export default defineConfig({
  plugins: [
    react(),
    dts({ include: ['src'] })
  ],
  build: {
    lib: {
      entry: resolve(__dirname, 'src/index.ts'),
      name: 'MyUI',
      fileName: (format) => `index.${format}.js`
    },
    rollupOptions: {
      external: ['react', 'react-dom'],
      output: {
        globals: {
          react: 'React',
          'react-dom': 'ReactDOM'
        }
      }
    }
  }
})
```
