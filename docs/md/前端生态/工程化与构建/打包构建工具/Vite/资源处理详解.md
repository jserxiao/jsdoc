# Vite 资源处理详解

## 概述

Vite 提供了强大的资源处理能力，支持图片、字体、媒体文件等多种资源的导入和优化。

## 静态资源处理

### 导入静态资源

```typescript
// 导入图片
import logo from './logo.png'

// 导入 SVG
import svgUrl from './icon.svg'

// 导入字体
import fontUrl from './font.woff2'

// 导入 JSON
import data from './data.json'

// 导入 Worker
import Worker from './worker?worker'

// 导入原始内容
import rawContent from './file?raw'
```

### 静态资源分类

```typescript
// 显式导入为 URL
import imgAsUrl from './image.png?url'

// 显式导入为 base64
import imgAsBase64 from './image.png?base64'

// 导入为字符串
import svgAsString from './icon.svg?raw'

// 导入为 Worker
import MyWorker from './worker.js?worker'

// 内联 Worker
import InlineWorker from './worker.js?worker&inline'
```

## 图片处理

### 基础图片导入

```typescript
// TypeScript 定义
declare module '*.png' {
  const src: string
  export default src
}

declare module '*.jpg' {
  const src: string
  export default src

  const src: string
  export default src

```

### 图片优化

```bash
npm install -D vite-plugin-imagemin
```

```typescript
import { defineConfig } from 'vite'
import viteImagemin from 'vite-plugin-imagemin'

export default defineConfig({
  plugins: [
    viteImagemin({
      gifsicle: {
        optimizationLevel: 3,
        interlaced: false,
      },
      optipng: {
        optimizationLevel: 7,
      },
      mozjpeg: {
        quality: 80,
      },
      pngquant: {
        quality: [0.8, 0.9],
        speed: 4,
      },
      svgo: {
        plugins: [
          {
            name: 'removeViewBox',
          },
          {
            name: 'removeEmptyAttrs',
            active: false,
          },
        ],
      },
    }),
  ],
})
```

### SVG 组件化

```bash
npm install -D vite-plugin-svg-icons
```

```typescript
import { defineConfig } from 'vite'
import { createSvgIconsPlugin } from 'vite-plugin-svg-icons'
import path from 'path'

export default defineConfig({
  plugins: [
    createSvgIconsPlugin({
      iconDirs: [path.resolve(process.cwd(), 'src/icons/svg')],
      symbolId: 'icon-[dir]-[name]',
      inject: 'body-last',
      customDomId: '__svg__icons__dom__',
    }),
  ],
})
```

```typescript
// 使用
import 'virtual:svg-icons-register'

// 组件中
<svg>
  <use xlink:href="#icon-user" />
</svg>
```

### 图片 CDN

```typescript
import { defineConfig } from 'vite'

export default defineConfig({
  build: {
    rollupOptions: {
      output: {
        assetFileNames: (assetInfo) => {
          if (/\.(png|jpe?g|gif|svg|webp|ico)$/i.test(assetInfo.name)) {
            return 'images/[name]-[hash][extname]'
          }
          return 'assets/[name]-[hash][extname]'
        }
      }
    }
  }
})
```

## 字体处理

### 字体导入

```typescript
// CSS 中导入
@font-face {
  font-family: 'MyFont';
  src: url('@/fonts/my-font.woff2') format('woff2'),
       url('@/fonts/my-font.woff') format('woff');
  font-weight: normal;
  font-style: normal;
}
```

### 字体优化

```typescript
import { defineConfig } from 'vite'

export default defineConfig({
  build: {
    rollupOptions: {
      output: {
        assetFileNames: (assetInfo) => {

            return 'fonts/[name]-[hash][extname]'
          }
          return 'assets/[name]-[hash][extname]'
        }
      }
    }
  }
})
```

### 字体预加载

```html

<link rel="preload" href="/fonts/my-font.woff2" as="font" type="font/woff2" crossorigin>
```

```typescript
// vite.config.ts
import { defineConfig } from 'vite'

export default defineConfig({
```


## Worker 处理

### Web Worker

```typescript
// 导入 Worker
import MyWorker from './worker?worker'

const worker = new MyWorker()
worker.postMessage({ type: 'start' })
worker.onmessage = (e) => {
  console.log('Message from worker:', e.data)
}
```

### 内联 Worker

```typescript
// 导入为内联 Worker
import InlineWorker from './worker?worker&inline'

const worker = new InlineWorker()
```

### Worker 类型

```typescript
// worker.d.ts
declare module '*?worker' {
  const workerConstructor: {
    new (): Worker
  }
  export default workerConstructor
}

declare module '*?worker&inline' {
  const workerConstructor: {
    new (): Worker
  }
  export default workerConstructor
}
```

### Worker 示例

```typescript
// src/worker.ts
self.onmessage = (e) => {
  const { type, data } = e.data

  if (type === 'calculate') {
    const result = heavyCalculation(data)
    self.postMessage({ type: 'result', data: result })
  }
}

function heavyCalculation(data: any) {
  // 复杂计算
  return data
}
```

```typescript
// App.vue
import CalculatorWorker from './worker?worker'

const worker = new CalculatorWorker()

worker.postMessage({
  type: 'calculate',
  data: { numbers: [1, 2, 3, 4, 5] }
})

worker.onmessage = (e) => {
  if (e.data.type === 'result') {
    console.log('Result:', e.data.data)
  }
}
```

## assetsInlineLimit 配置

### 基础配置

```typescript
import { defineConfig } from 'vite'

export default defineConfig({
  build: {
    assetsInlineLimit: 4096,  // 4KB（默认）
    assetsInlineLimit: 8192,  // 8KB
    assetsInlineLimit: 0,     // 禁用内联
  }
})
```

### 强制内联或 URL

```typescript
// 强制内联
import img from './image.png?inline'

// 强制 URL
import img from './image.png?url'
```

### 条件内联

```typescript
import { defineConfig } from 'vite'

export default defineConfig({
  build: {
    assetsInlineLimit: (filePath, content) => {
      // SVG 始终内联
      if (filePath.endsWith('.svg')) {
        return true
      }
      // 大图片不内联
      if (content.length > 1024 * 10) {
        return false
      }
      return 4096
    }
  }
})
```

## 完整配置示例

### 图片优化配置

```typescript
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import viteImagemin from 'vite-plugin-imagemin'
import { createSvgIconsPlugin } from 'vite-plugin-svg-icons'
import path from 'path'

export default defineConfig({
  plugins: [
    vue(),

    // 图片压缩
    viteImagemin({
      mozjpeg: { quality: 80 },
      pngquant: { quality: [0.8, 0.9] },
      svgo: {
        plugins: [{ name: 'removeViewBox' }]
      }
    }),

    // SVG 图标
    createSvgIconsPlugin({
      iconDirs: [path.resolve(__dirname, 'src/icons/svg')],
      symbolId: 'icon-[name]'
    })
  ],

  build: {
    assetsInlineLimit: 4096,
    rollupOptions: {
      output: {
        assetFileNames: (assetInfo) => {
          if (/\.(png|jpe?g|gif|svg|webp|ico)$/i.test(assetInfo.name)) {
            return 'images/[name]-[hash][extname]'
          }
          if (/\.(woff2?|eot|ttf|otf)$/i.test(assetInfo.name)) {
            return 'fonts/[name]-[hash][extname]'
          }
          if (/\.css$/i.test(assetInfo.name)) {
            return 'css/[name]-[hash][extname]'
          }
          return 'assets/[name]-[hash][extname]'
        }
      }
    }
  }
})
```

### 资源加载优化

```typescript
import { defineConfig } from 'vite'

export default defineConfig({
  build: {
    // 资源内联阈值
    assetsInlineLimit: 4096,

    // 模块预加载
    modulePreload: {
      polyfill: true,
    },

    rollupOptions: {
      output: {
        // 资源文件名
        assetFileNames: 'assets/[name]-[hash][extname]',

        // 手动分块
        manualChunks(id) {
          // 大型资源单独分块
          if (id.includes('node_modules')) {
            if (id.includes('monaco-editor')) {
              return 'monaco-editor'
            }
            if (id.includes('echarts')) {
              return 'echarts'
            }
          }
        }
      }
    }
  }
})
```

## 常见问题

### 1. 图片路径错误

```typescript
// vite.config.ts
export default defineConfig({
  base: '/my-app/',  // 设置正确的基础路径
})
```

### 2. SVG 不显示

```typescript
// 确保正确导入
import svgUrl from './icon.svg'

// 或使用 SVG 组件
import { ReactComponent as Icon } from './icon.svg'  // React
```

### 3. 字体加载失败

```css
/* 确保正确的路径 */
@font-face {
  font-family: 'MyFont';
  src: url('@/fonts/my-font.woff2') format('woff2');
  font-display: swap;  /* 添加 font-display */
}
```

### 4. Worker 报错

```typescript
// 确保正确导入
import MyWorker from './worker?worker'

// TypeScript 定义
declare module '*?worker' {
  const workerConstructor: {
    new (): Worker
  }
  export default workerConstructor
}
```

### 5. 大文件内存溢出

```typescript
// 使用流式处理
export default defineConfig({
  build: {
    assetsInlineLimit: 0,  // 禁用大文件内联
    chunkSizeWarningLimit: 1000,
  }
})
```
