# Vite 服务器配置详解

## 概述

Vite 开发服务器提供了丰富的配置选项，包括端口、代理、HTTPS、CORS 等，可以灵活适应各种开发场景。

## 基础配置

### server.host

指定服务器应该监听哪个 IP 地址。

```typescript
import { defineConfig } from 'vite'

export default defineConfig({
  server: {
    host: '0.0.0.0',  // 允许外部访问
    host: 'localhost', // 仅本地访问（默认）
    host: true,        // 等同于 '0.0.0.0'
    host: '::',        // 监听所有 IPv6 地址
  }
})
```

### server.port

指定开发服务器端口。

```typescript
export default defineConfig({
  server: {
    port: 3000,        // 固定端口
    port: 0,           // 自动选择可用端口
    strictPort: true,  // 端口被占用时直接报错，不尝试下一个
    strictPort: false, // 端口被占用时自动尝试下一个可用端口（默认）
  }
})
```

## 代理配置

### 基础代理

```typescript
export default defineConfig({
  server: {
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,
      }
    }
  }
})

// 请求: /api/users
// 实际: http://localhost:8080/api/users
```

### 路径重写

```typescript
export default defineConfig({
  server: {
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,
        rewrite: (path) => path.replace(/^\/api/, '')
      }
    }
  }
})

// 请求: /api/users
// 实际: http://localhost:8080/users
```

### 高级代理配置

```typescript
export default defineConfig({
  server: {
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,
        secure: false,        // 允许自签名证书
        ws: true,             // 启用 WebSocket 代理
        rewrite: (path) => path.replace(/^\/api/, ''),

        // 请求头
        headers: {
          'X-Custom-Header': 'value'
        },

        // 配置请求拦截
        configure: (proxy, options) => {
          proxy.on('proxyReq', (proxyReq, req, res) => {
            console.log('Sending Request:', req.url)
          })

          proxy.on('proxyRes', (proxyRes, req, res) => {
            console.log('Received Response:', proxyRes.statusCode)
          })

          proxy.on('error', (err, req, res) => {
            console.error('Proxy Error:', err)
          })
        }
      }
    }
  }
})
```

### 多个代理

```typescript
export default defineConfig({
  server: {
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,
      },
      '/auth': {
        target: 'http://localhost:9000',
        changeOrigin: true,
      },
      '/socket': {
        target: 'ws://localhost:9090',
        ws: true,
      }
    }
  }
})
```

### 条件代理

```typescript
export default defineConfig({
  server: {
    proxy: {
      '^/api/.*': {
        target: 'http://localhost:8080',
        changeOrigin: true,
        bypass: (req, res, options) => {
          // 返回 false 跳过代理
          if (req.headers.accept?.indexOf('html') !== -1) {
            return '/index.html'
          }
        }
      }
    }
  }
})
```

## CORS 配置

### 启用 CORS

```typescript
export default defineConfig({
  server: {
    cors: true,  // 启用 CORS
  }
})
```

### 详细 CORS 配置

```typescript
export default defineConfig({
  server: {
    cors: {
      origin: '*',
      methods: ['GET', 'POST', 'PUT', 'DELETE', 'OPTIONS'],
      allowedHeaders: ['Content-Type', 'Authorization'],
      exposedHeaders: ['X-Custom-Header'],
      credentials: true,
      maxAge: 86400,
    }
  }
})
```

## HTTPS 配置

### 自动 HTTPS

```typescript
export default defineConfig({
  server: {
    https: true  // 自动生成证书
  }
})
```

### 自定义证书

```typescript
import fs from 'fs'

export default defineConfig({
  server: {
    https: {
      key: fs.readFileSync('server.key'),
      cert: fs.readFileSync('server.crt'),
      ca: fs.readFileSync('ca.pem'),
      pfx: fs.readFileSync('server.pfx'),
      passphrase: 'vite'
    }
  }
})
```

### 使用 @vitejs/plugin-basic-ssl

```bash
npm install -D @vitejs/plugin-basic-ssl
```

```typescript
import basicSsl from '@vitejs/plugin-basic-ssl'

export default defineConfig({
  plugins: [basicSsl()],
  server: {
    https: true
  }
})
```

## 浏览器配置

### server.open

自动打开浏览器。

```typescript
export default defineConfig({
  server: {
    open: true,  // 自动打开默认浏览器
    open: false, // 不自动打开

    // 指定浏览器
    open: '/docs/index.html',  // 打开指定路径
    open: {
      app: {
        name: 'Google Chrome',
        arguments: ['--incognito']
      }
    }
  }
})
```

## HMR 配置

### server.hmr

热模块替换配置。

```typescript
export default defineConfig({
  server: {
    hmr: true,  // 启用 HMR（默认）
    hmr: false, // 禁用 HMR

    hmr: {
      overlay: true,  // 显示错误覆盖层（默认）
      overlay: false, // 隐藏错误覆盖层

      // 自定义错误覆盖层
      overlay: {
        errors: true,
        warnings: false,
      },

      // 自定义 HMR 服务器
      host: 'localhost',
      port: 24678,
      protocol: 'ws',  // 'ws' | 'wss'
      clientPort: 24678,
      timeout: 30000,
    }
  }
})
```

### 禁用 HMR

```typescript
export default defineConfig({
  server: {
    hmr: false,
    watch: null  // 禁用文件监听
  }
})
```

## 文件监听配置

### server.watch

配置文件监听行为。

```typescript
export default defineConfig({
  server: {
    watch: {
      ignored: ['**/node_modules/**', '**/dist/**'],
      ignoreInitial: true,
      followSymlinks: true,
      usePolling: false,  // 在某些系统上可能需要启用
      interval: 100,
      binaryInterval: 300,
      alwaysStat: false,
      depth: 99,
      awaitWriteFinish: {
        stabilityThreshold: 2000,
        pollInterval: 100
      }
    }
  }
})
```

### 在 Docker 中使用

```typescript
export default defineConfig({
  server: {
    watch: {
      usePolling: true,  // 在 Docker 中需要启用轮询
      interval: 100
    },
    host: '0.0.0.0'  // Docker 中需要监听所有接口
  }
})
```

## 其他配置

### server.base

开发服务器的基础路径。

```typescript
export default defineConfig({
  server: {
    base: '/my-app/'  // 所有请求都添加 /my-app/ 前缀
  }
})
```

### server.headers

设置服务器响应头。

```typescript
export default defineConfig({
  server: {
    headers: {
      'X-Custom-Header': 'yes',
      'Cache-Control': 'no-store',
      'X-Frame-Options': 'DENY',
      'X-Content-Type-Options': 'nosniff'
    }
  }
})
```

### server.strictPort

端口被占用时的行为。

```typescript
export default defineConfig({
  server: {
    port: 3000,
    strictPort: true  // 端口被占用时直接报错退出
  }
})
```

### server.middlewareMode

中间件模式配置。

```typescript
export default defineConfig({
  server: {
    middlewareMode: true,  // 以中间件模式运行
    // 或
    middlewareMode: {
      // 在 Vite 的中间件之前添加自定义中间件
      // server.middlewares.use(customMiddleware)
    }
  }
})
```

### server.sourcemapIgnoreList

配置 Source Map 忽略列表。

```typescript
export default defineConfig({
  server: {
    sourcemapIgnoreList: (relativePath) => {
      return relativePath.includes('node_modules')
    }
  }
})
```

### server.warmup

预热常用文件以提升启动速度。

```typescript
export default defineConfig({
  server: {
    warmup: {
      clientFiles: [
        './src/main.ts',
        './src/App.vue',
        './src/router/index.ts',
        './src/stores/index.ts'
      ],
      ssrFiles: [
        './src/entry-server.ts'
      ]
    }
  }
})
```

## 完整配置示例

### 开发环境完整配置

```typescript
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import path from 'path'

export default defineConfig({
  plugins: [vue()],

  server: {
    // 基础配置
    host: '0.0.0.0',
    port: 3000,
    strictPort: false,
    open: true,

    // HTTPS
    https: false,

    // CORS
    cors: true,

    // 代理
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,
        rewrite: (path) => path.replace(/^\/api/, ''),
        configure: (proxy) => {
          proxy.on('proxyReq', (proxyReq, req) => {
            console.log('Request:', req.method, req.url)
          })
        }
      }
    },

    // HMR
    hmr: {
      overlay: true
    },

    // 文件监听
    watch: {
      ignored: ['**/node_modules/**', '**/dist/**']
    },

    // 响应头
    headers: {
      'X-Content-Type-Options': 'nosniff'
    },

    // 预热
    warmup: {
      clientFiles: [
        './src/main.ts',
        './src/App.vue'
      ]
    }
  },

  resolve: {
    alias: {
      '@': path.resolve(__dirname, 'src')
    }
  }
})
```

### 生产环境代理配置

```typescript
export default defineConfig(({ mode }) => {
  const isDev = mode === 'development'

  return {
    server: {
      proxy: {
        '/api': {
          target: isDev
            ? 'http://localhost:8080'
            : 'https://api.example.com',
          changeOrigin: true,
          rewrite: (path) => path.replace(/^\/api/, '')
        }
      }
    }
  }
})
```

## 常见问题

### 1. 端口被占用

```typescript
export default defineConfig({
  server: {
    port: 3000,
    strictPort: false  // 自动尝试下一个端口
  }
})
```

### 2. 跨域问题

```typescript
export default defineConfig({
  server: {
    cors: true,  // 启用 CORS
    proxy: {     // 或使用代理
      '/api': {
        target: 'http://api.example.com',
        changeOrigin: true
      }
    }
  }
})
```

### 3. Docker 中文件监听不生效

```typescript
export default defineConfig({
  server: {
    watch: {
      usePolling: true,
      interval: 100
    },
    host: '0.0.0.0'
  }
})
```

### 4. HMR 不工作

```typescript
export default defineConfig({
  server: {
    hmr: {
      host: 'localhost',
      port: 24678,
      overlay: true
    },
    watch: {
      ignored: ['**/node_modules/**']
    }
  }
})
```

### 5. 网络访问

```typescript
export default defineConfig({
  server: {
    host: '0.0.0.0',  // 允许局域网访问
    port: 3000
  }
})
```
