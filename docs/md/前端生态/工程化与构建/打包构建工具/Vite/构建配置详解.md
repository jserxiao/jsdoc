# Vite 构建配置详解

## 概述

Vite 生产构建使用 Rollup 进行打包，提供了丰富的构建配置选项来优化输出产物。

## 基础配置

### build.target

构建目标浏览器。

```typescript
import { defineConfig } from 'vite'

export default defineConfig({
  build: {
    target: 'modules',        // 支持原生 ESM 的浏览器（默认）
    target: 'es2015',         // ES2015+
    target: 'esnext',         // 最新特性
    target: ['es2015', 'edge14', 'firefox60', 'chrome61', 'safari11'],
  }
})
```

### build.outDir

输出目录。

```typescript
export default defineConfig({
  build: {
    outDir: 'dist',           // 默认
    outDir: 'build',          // 自定义输出目录
  }
})
```

### build.assetsDir

静态资源输出目录。

```typescript
export default defineConfig({
  build: {
    assetsDir: 'assets',      // 默认
    assetsDir: 'static',      // 自定义
  }
})
```

### build.base

公共基础路径。

```typescript
export default defineConfig({
  base: '/',                  // 默认
  base: '/my-app/',           // 自定义路径
  base: 'https://cdn.example.com/',  // CDN
})
```

## 入口配置

### build.lib

库模式配置。

```typescript
export default defineConfig({
  build: {
    lib: {
      entry: 'src/index.ts',          // 入口文件
      name: 'MyLib',                   // 全局变量名
      formats: ['es', 'cjs', 'umd'],   // 输出格式
      fileName: (format) => `my-lib.${format}.js`,
      entry: {
        main: 'src/index.ts',
        utils: 'src/utils.ts'
      }
    }
  }
})
```

### 多入口配置

```typescript
export default defineConfig({
  build: {
    rollupOptions: {
      input: {
        main: 'src/index.html',
        admin: 'src/admin.html',
        mobile: 'src/mobile.html'
      }
    }
  }
})
```

### build.ssr

SSR 构建。

```typescript
export default defineConfig({
  build: {
    ssr: 'src/entry-server.ts',  // SSR 入口
    ssr: true,                    // 自动检测
    ssrManifest: true,            // 生成 SSR manifest
  }
})
```

## 代码分割

### build.rollupOptions.output.manualChunks

手动配置代码分割。

```typescript
export default defineConfig({
  build: {
    rollupOptions: {
      output: {
        manualChunks: {
          // 将特定模块打包到单独文件
          'vendor': ['vue', 'vue-router', 'pinia'],
          'ui': ['element-plus'],
          'utils': ['lodash', 'dayjs']
        },

        // 函数形式
        manualChunks(id) {
          if (id.includes('node_modules')) {
            if (id.includes('vue')) {
              return 'vue-vendor'
            }
            if (id.includes('element-plus')) {
              return 'element-plus'
            }
            return 'vendor'
          }
        }
      }
    }
  }
})
```

### build.rollupOptions.output.chunkFileNames

自定义 chunk 文件名。

```typescript
export default defineConfig({
  build: {
    rollupOptions: {
      output: {
        chunkFileNames: 'js/[name]-[hash].js',
        entryFileNames: 'js/[name]-[hash].js',
        assetFileNames: (assetInfo) => {
          const info = assetInfo.name.split('.')
          let extType = info[info.length - 1]
          if (/\.(png|jpe?g|gif|svg|webp|ico)$/i.test(assetInfo.name)) {
            return 'images/[name]-[hash].[ext]'
          } else if (/\.(woff2?|eot|ttf|otf)$/i.test(assetInfo.name)) {
            return 'fonts/[name]-[hash].[ext]'
          } else if (/\.css$/i.test(assetInfo.name)) {
            return 'css/[name]-[hash].[ext]'
          }
          return 'assets/[name]-[hash].[ext]'
        }
      }
    }
  }
})
```

## 压缩配置

### build.minify

配置压缩方式。

```typescript
export default defineConfig({
  build: {
    minify: 'esbuild',        // 使用 esbuild（默认，更快）
    minify: 'terser',         // 使用 terser（压缩率更高）
    minify: false,            // 不压缩
  }
})
```

### build.terserOptions

Terser 压缩选项。

```typescript
export default defineConfig({
  build: {
    minify: 'terser',
    terserOptions: {
      compress: {
        drop_console: true,      // 移除 console
        drop_debugger: true,     // 移除 debugger
        pure_funcs: ['console.log'],
      },
      format: {
        comments: false,         // 移除注释
      },
      mangle: {
        safari10: true,
      },
    }
  }
})
```

## Source Map 配置

### build.sourcemap

配置 Source Map。

```typescript
export default defineConfig({
  build: {
    sourcemap: true,          // 生成 source map
    sourcemap: false,         // 不生成（默认）
    sourcemap: 'inline',      // 内联 source map
    sourcemap: 'hidden',      // 隐藏 source map
  }
})
```

### build.sourcemapBaseUrl

Source Map 基础 URL。

```typescript
export default defineConfig({
  build: {
    sourcemap: true,
    sourcemapBaseUrl: 'https://example.com/maps/'
  }
})
```

### build.sourcemapFile

Source Map 文件路径。

```typescript
export default defineConfig({
  build: {
    sourcemap: true,
    sourcemapFile: 'dist/bundle.js.map'
  }
})
```

## CSS 配置

### build.cssCodeSplit

CSS 代码分割。

```typescript
export default defineConfig({
  build: {
    cssCodeSplit: true,       // 启用 CSS 代码分割（默认）
    cssCodeSplit: false,      // 禁用，所有 CSS 合并为一个文件
  }
})
```

### build.cssMinify

CSS 压缩配置。

```typescript
export default defineConfig({
  build: {
    cssMinify: true,          // 启用 CSS 压缩（默认）
    cssMinify: false,         // 禁用 CSS 压缩
    cssMinify: 'esbuild',     // 使用 esbuild 压缩
    cssMinify: 'lightningcss', // 使用 lightningcss 压缩
  }
})
```

### build.cssTarget

CSS 构建目标。

```typescript
export default defineConfig({
  build: {
    cssTarget: 'chrome61',    // CSS 降级目标
    cssTarget: ['chrome61', 'firefox60', 'safari11'],
  }
})
```

## 资源处理

### build.assetsInlineLimit

内联资源阈值。

```typescript
export default defineConfig({
  build: {
    assetsInlineLimit: 4096,  // 4KB 以下转为 base64（默认）
    assetsInlineLimit: 8192,  // 8KB 以下
    assetsInlineLimit: 0,     // 禁用内联
  }
})
```

### build.modulePreload

模块预加载配置。

```typescript
export default defineConfig({
  build: {
    modulePreload: {
      polyfill: true,         // 自动注入预加载 polyfill（默认）
      polyfill: false,        // 禁用
      resolveDependencies: (filePath, deps, context) => {
        // 自定义依赖解析
        return deps.filter(dep => !dep.includes('polyfill'))
      }
    }
  }
})
```

## 构建优化

### build.rollupOptions

Rollup 配置。

```typescript
export default defineConfig({
  build: {
    rollupOptions: {
      // 外部依赖
      external: ['vue', 'vue-router', 'axios'],

      // 全局变量
      output: {
        globals: {
          vue: 'Vue',
          'vue-router': 'VueRouter',
          axios: 'axios'
        }
      },

      // 插件
      plugins: [
        // 自定义 Rollup 插件
      ]
    }
  }
})
```

### build.commonjsOptions

CommonJS 配置。

```typescript
export default defineConfig({
  build: {
    commonjsOptions: {
      include: [/node_modules/],
      extensions: ['.js', '.cjs'],
      transformMixedEsModules: true,
      defaultIsModuleExports: 'auto',
      ignoreDynamicRequires: true,
    }
  }
})
```

### build.dynamicImportVarsOptions

动态导入配置。

```typescript
export default defineConfig({
  build: {
    dynamicImportVarsOptions: {
      include: ['**/*.js', '**/*.ts'],
      exclude: ['**/node_modules/**'],
    }
  }
})
```

## 清理和报告

### build.emptyOutDir

构建前清理输出目录。

```typescript
export default defineConfig({
  build: {
    emptyOutDir: true,        // 清理（默认）
    emptyOutDir: false,       // 不清理
  }
})
```

### build.reportCompressedSize

报告压缩后大小。

```typescript
export default defineConfig({
  build: {
    reportCompressedSize: true,  // 报告（默认）
    reportCompressedSize: false, // 不报告，加快构建速度
  }
})
```

### build.chunkSizeWarningLimit

chunk 大小警告阈值。

```typescript
export default defineConfig({
  build: {
    chunkSizeWarningLimit: 500,  // 500KB（默认）
    chunkSizeWarningLimit: 1000, // 1000KB
  }
})
```

## 完整配置示例

### 生产环境完整配置

```typescript
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import path from 'path'

export default defineConfig({
  plugins: [vue()],

  build: {
    // 目标浏览器
    target: 'es2015',

    // 输出目录
    outDir: 'dist',
    assetsDir: 'assets',

    // 清理输出目录
    emptyOutDir: true,

    // 代码分割
    rollupOptions: {
      output: {
        chunkFileNames: 'js/[name]-[hash].js',
        entryFileNames: 'js/[name]-[hash].js',
        assetFileNames: (assetInfo) => {
          if (/\.(png|jpe?g|gif|svg|webp)$/i.test(assetInfo.name)) {
            return 'images/[name]-[hash].[ext]'
          }
          if (/\.css$/i.test(assetInfo.name)) {
            return 'css/[name]-[hash].[ext]'
          }
          return 'assets/[name]-[hash].[ext]'
        },
        manualChunks: {
          vue: ['vue', 'vue-router', 'pinia'],
          ui: ['element-plus'],
          utils: ['lodash', 'dayjs', 'axios']
        }
      }
    },

    // 压缩
    minify: 'terser',
    terserOptions: {
      compress: {
        drop_console: true,
        drop_debugger: true,
        pure_funcs: ['console.log']
      },
      format: {
        comments: false
      }
    },

    // Source Map
    sourcemap: false,

    // CSS
    cssCodeSplit: true,
    cssMinify: true,

    // 资源内联
    assetsInlineLimit: 4096,

    // 报告
    reportCompressedSize: true,
    chunkSizeWarningLimit: 500
  },

  resolve: {
    alias: {
      '@': path.resolve(__dirname, 'src')
    }
  }
})
```

### 库模式配置

```typescript
import { defineConfig } from 'vite'
```

## 常见问题

### 1. 构建产物过大

```typescript
export default defineConfig({
  build: {
    rollupOptions: {
      output: {
        manualChunks: {
          // 分离第三方库
          vendor: ['vue', 'vue-router', 'pinia'],
          // 分离大型依赖
          'element-plus': ['element-plus'],
        }
      }
    },
    chunkSizeWarningLimit: 1000
  }
})
```

### 2. 构建速度慢

```typescript
export default defineConfig({
  build: {
    minify: 'esbuild',          // 使用更快的压缩器
    sourcemap: false,           // 不生成 source map
    reportCompressedSize: false // 不报告大小
  }
})
```

### 3. 资源路径错误

```typescript
export default defineConfig({
  base: '/my-app/',  // 设置正确的基础路径
  build: {
    assetsDir: 'assets'
  }
})
```

### 4. 兼容旧浏览器

```typescript
import legacy from '@vitejs/plugin-legacy'

export default defineConfig({
  plugins: [
    legacy({
      targets: ['ie >= 11', 'chrome >= 52']
    })
  ],
  build: {
    target: 'es2015',
    cssTarget: 'chrome52'
  }
})
```

### 5. 静态资源未内联

```typescript
export default defineConfig({
  build: {
    assetsInlineLimit: 8192,  // 提高阈值
    rollupOptions: {
      output: {
        assetFileNames: (assetInfo) => {
          // 指定哪些资源应该内联
          if (assetInfo.name.endsWith('.svg')) {
            return 'assets/[name]-[hash].[ext]'
          }
        }
      }
    }
  }
})
```
