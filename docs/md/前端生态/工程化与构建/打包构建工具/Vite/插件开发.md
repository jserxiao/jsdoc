# Vite 插件开发

Vite 插件基于 Rollup 插件接口扩展而来，提供了更多 Vite 特有的钩子和功能。

---

## 插件基础结构

### 最简插件

```javascript
// vite-plugin-hello.js
export default function helloPlugin() {
  return {
    name: 'vite-plugin-hello',
    
    // 插件执行顺序
    enforce: 'pre', // 'pre' | 'post'
    
    // 应用阶段
    apply: 'serve', // 'serve' | 'build' | 函数
    
    // 各种钩子...
  }
}
```

### 使用插件

```javascript
// vite.config.js
import helloPlugin from './vite-plugin-hello'

export default defineConfig({
  plugins: [
    helloPlugin(),
    // 带配置的插件
    helloPlugin({ greeting: 'Hi' })
  ]
})
```

## 插件钩子详解

### 通用钩子（Rollup 兼容）

| 钩子 | 类型 | 说明 |
|------|------|------|
| `options` | 同步/异步 | 修改 Rollup 配置 |
| `buildStart` | 异步 | 构建开始 |
| `buildEnd` | 异步 | 构建结束 |
| `resolveId` | 同步/异步 | 自定义模块解析 |
| `load` | 同步/异步 | 自定义模块加载 |
| `transform` | 同步/异步 | 转换模块代码 |
| `renderChunk` | 异步 | 转换代码块 |
| `generateBundle` | 异步 | 生成产物前 |
| `writeBundle` | 异步 | 写入产物后 |

### Vite 特有钩子

| 钩子 | 说明 |
|------|------|
| `config` | 修改 Vite 配置 |
| `configResolved` | 配置解析完成 |
| `configureServer` | 配置开发服务器 |
| `transformIndexHtml` | 转换 HTML |
| `handleHotUpdate` | 处理 HMR |

## config 钩子

修改 Vite 配置：

```javascript
export default function myPlugin() {
  return {
    name: 'my-plugin',
    
    config(config, { command, mode }) {
      // config: 用户配置
      // command: 'serve' | 'build'
      // mode: 'development' | 'production'
      
      // 返回部分配置，会与用户配置合并
      return {
        define: {
          __MY_PLUGIN__: JSON.stringify(true)
        },
        resolve: {
          alias: {
            '@plugin': '/src/plugin'
          }
        }
      }
    }
  }
}
```

## configResolved 钩子

读取最终配置：

```javascript
export default function myPlugin() {
  let config
  
  return {
    name: 'my-plugin',
    
    configResolved(resolvedConfig) {
      // 存储最终配置
      config = resolvedConfig
      
      console.log('Root:', config.root)
      console.log('Mode:', config.mode)
      console.log('Command:', config.command)
    }
  }
}
```

## configureServer 钩子

配置开发服务器：

```javascript
export default function myPlugin() {
  return {
    name: 'my-plugin',
    
    configureServer(server) {
      // server: ViteDevServer 实例
      
      // 添加自定义中间件
      server.middlewares.use((req, res, next) => {
        if (req.url === '/api/hello') {
          res.end(JSON.stringify({ message: 'Hello from plugin!' }))
          return
        }
        next()
      })
      
      // 返回一个函数，会在内部中间件之后执行
      return () => {
        server.middlewares.use((req, res, next) => {
          // 后置中间件
          next()
        })
      }
    }
  }
}
```

### ViteDevServer 实例

```javascript
configureServer(server) {
  const { 
    middlewares,      // Connect 服务器实例
    httpServer,       // HTTP 服务器
    watcher,          // Chokidar 文件监听器
    moduleGraph,      // 模块依赖图
    ws,               // WebSocket 服务器
    transformRequest, // 转换请求函数
    transformIndexHtml,
    ssrLoadModule     // SSR 加载模块
  } = server
  
  // 监听文件变化
  watcher.on('change', (file) => {
    console.log('File changed:', file)
  })
  
  // 发送 WebSocket 消息
  server.ws.send({
    type: 'custom',
    event: 'my-event',
    data: { message: 'Hello' }
  })
}
```

## transformIndexHtml 钩子

转换 HTML 文件：

```javascript
export default function myPlugin() {
  return {
    name: 'my-plugin',
    
    transformIndexHtml(html, ctx) {
      // html: 原始 HTML
      // ctx: { filename, path, server }
      
      // 方式一：返回转换后的 HTML
      return html.replace(
        '</head>',
        `<script>console.log('Injected by plugin')</script></head>`
      )
      
      // 方式二：返回标签描述符数组
      return [
        // 注入到 head
        {
          tag: 'script',
          attrs: { src: '/my-script.js' },
          injectTo: 'head'
        },
        // 注入 meta 标签
        {
          tag: 'meta',
          attrs: { name: 'plugin', content: 'my-plugin' },
          injectTo: 'head-prepend'
        }
      ]
      
      // 方式三：组合使用
      return {
        html: html.replace('<!--placeholder-->', 'replaced'),
        tags: [
          { tag: 'script', children: 'console.log("injected")', injectTo: 'body' }
        ]
      }
    }
  }
}
```

### 注入位置

| 位置 | 说明 |
|------|------|
| `head` | `</head>` 之前 |
| `head-prepend` | `<head>` 之后 |
| `body` | `</body>` 之前 |
| `body-prepend` | `<body>` 之后 |

## handleHotUpdate 钩子

处理热更新：

```javascript
export default function myPlugin() {
  return {
    name: 'my-plugin',
    
    handleHotUpdate({ file, server, modules, read }) {
      // file: 变化的文件路径
      // server: ViteDevServer 实例
      // modules: 受影响的模块数组
      // read: 读取文件内容的函数
      
      if (file.endsWith('.custom')) {
        // 自定义处理逻辑
        const content = await read()
        
        // 手动触发更新
        server.ws.send({
          type: 'custom',
          event: 'custom-update',
          data: { file, content }
        })
        
        // 返回空数组，跳过默认 HMR
        return []
      }
      
      // 返回需要更新的模块
      return modules
    }
  }
}
```

### 自定义 HMR 事件

```javascript
// 插件端
handleHotUpdate({ file, server }) {
  server.ws.send({
    type: 'custom',
    event: 'my-update',
    data: { file }
  })
}

// 客户端
if (import.meta.hot) {
  import.meta.hot.on('my-update', (data) => {
    console.log('Custom update:', data.file)
  })
}
```

## resolveId 钩子

自定义模块解析：

```javascript
export default function myPlugin() {
  return {
    name: 'my-plugin',
    
    resolveId(source, importer, options) {
      // source: 导入标识符
      // importer: 导入者文件路径
      // options: { ssr, scan, isEntry }
      
      if (source === 'virtual:my-module') {
        // 返回解析后的 ID
        return source
      }
      
      // 返回 null，交给其他插件处理
      return null
    }
  }
}
```

## transform 钩子

转换模块代码：

```javascript
export default function myPlugin() {
  return {
    name: 'my-plugin',
    
    transform(code, id) {
      // code: 模块源代码
      // id: 模块完整路径
      
      // 只处理特定文件
      if (!id.endsWith('.md')) return null
      
      // 转换 Markdown 为组件
      const html = markdownToHtml(code)
      
      return {
        code: `export default ${JSON.stringify(html)}`,
        map: null // 提供 sourcemap
      }
    }
  }
}
```

### 带 SourceMap 的转换

transform(code, id) {

  return {
    code: result.code,
    map: map.toString()
  }
}

## 完整插件示例

### 虚拟模块插件

```javascript
/**
 * 创建虚拟模块插件
 * 支持从配置文件生成运行时配置
 */
export default function virtualConfigPlugin(options = {}) {
  const virtualId = 'virtual:config'
  const resolvedVirtualId = '\0' + virtualId
  
  let config = options.config || {}
  
  return {
    name: 'vite-plugin-virtual-config',
    
    resolveId(id) {
      if (id === virtualId) {
        return resolvedVirtualId
      }
    },
    
    load(id) {
      if (id === resolvedVirtualId) {
        return `export default ${JSON.stringify(config)}`
      }
    },
    
    // 支持 HMR
    handleHotUpdate({ file, server }) {
      if (file.includes('config')) {
        // 重新加载配置
        config = loadConfig()
        
        // 触发虚拟模块更新
        const module = server.moduleGraph.getModuleById(resolvedVirtualId)
        if (module) {
          server.moduleGraph.invalidateModule(module)
        }
        
        return [module]
      }
    }
  }
}
```

### Markdown 插件

```javascript
import { marked } from 'marked'
import { dirname, relative } from 'path'

export default function markdownPlugin(options = {}) {
  return {
    name: 'vite-plugin-markdown',
    
    enforce: 'pre',
    
    transform(code, id) {
      if (!id.endsWith('.md')) return null
      
      // 解析 Markdown
      const html = marked.parse(code)
      
      // 提取 frontmatter
      const frontmatter = extractFrontmatter(code)
      
      // 生成模块代码
      const componentCode = `
        import { h, defineComponent } from 'vue'
        
        const frontmatter = ${JSON.stringify(frontmatter)}
        
        export default defineComponent({
          name: 'MarkdownComponent',
          setup() {
            return () => h('div', {
              innerHTML: ${JSON.stringify(html)},
              class: 'markdown-body'
            })
          }
        })
        
        export { frontmatter }
      `
      
      return {
        code: componentCode,
        map: null
      }
    }
  }
}
```

### 自定义导入插件

```javascript
/**
 * 支持 import xxx from 'xxx?type=raw'
 */
export default function importTypePlugin() {
  return {
    name: 'vite-plugin-import-type',
    
    enforce: 'pre',
    
    resolveId(source, importer) {
      // 处理 ?type=raw 后缀
      if (source.includes('?type=raw')) {
        const [path] = source.split('?')
        return this.resolve(path, importer).then(resolved => {
          return resolved ? { ...resolved, id: source } : null
        })
      }
    },
    
    load(id) {
      if (id.includes('?type=raw')) {
        const [path] = id.split('?')
        const content = require('fs').readFileSync(path, 'utf-8')
        return `export default ${JSON.stringify(content)}`
      }
    }
  }
}

// 使用
import shaderCode from './shader.glsl?type=raw'
```

### CSS 预处理插件

```javascript
export default function themeCssPlugin(options = {}) {
  const { themes = {} } = options
  let currentTheme = 'default'
  
  return {
    name: 'vite-plugin-theme-css',
    
    config() {
      return {
        css: {
          preprocessorOptions: {
            scss: {
              additionalData: () => {
                const theme = themes[currentTheme] || {}
                const vars = Object.entries(theme)
                  .map(([key, value]) => `$${key}: ${value};`)
                  .join('\n')
                return vars
              }
            }
          }
        }
      }
    },
    
    configureServer(server) {
      // API 切换主题
      server.middlewares.use('/api/theme', (req, res) => {
        if (req.method === 'POST') {
          let body = ''
          req.on('data', chunk => body += chunk)
          req.on('end', () => {
            const { theme } = JSON.parse(body)
            currentTheme = theme
            
            // 触发所有样式文件更新
            server.ws.send({ type: 'full-reload' })
            
            res.end(JSON.stringify({ success: true }))
          })
        }
      })
    }
  }
}
```

### 图片优化插件

```javascript
import sharp from 'sharp'
import { createHash } from 'crypto'

export default function imageOptimizePlugin(options = {}) {
  const { 
    quality = 80,
    format = 'webp',
    sizes = [320, 640, 960, 1280]
  } = options
  
  return {
    name: 'vite-plugin-image-optimize',
    
    enforce: 'pre',
    
    async load(id) {
      if (!/\.(png|jpe?g)$/i.test(id)) return null
      
      const buffer = await sharp(id)
        .webp({ quality })
        .toBuffer()
      
      const hash = createHash('md5')
        .update(buffer)
        .digest('hex')
        .slice(0, 8)
      
      const base64 = buffer.toString('base64')
      
      return {
        code: `
          const src = 'data:image/webp;base64,${base64}'
          const hash = '${hash}'
          export default { src, hash }
        `,
        map: null
      }
    }
  }
}
```

### 代码分析插件

```javascript
export default function analyzePlugin() {
  const modules = new Map()
  
  return {
    name: 'vite-plugin-analyze',
    
    buildStart() {
      modules.clear()
    },
    
    transform(code, id) {
      modules.set(id, {
        size: code.length,
        imports: extractImports(code)
      })
      return null
    },
    
    buildEnd() {
      const report = {
        totalSize: 0,
        modules: []
      }
      
      modules.forEach((info, id) => {
        report.totalSize += info.size
        report.modules.push({ id, ...info })
      })
      
      // 输出报告
      require('fs').writeFileSync(
        'analyze-report.json',
        JSON.stringify(report, null, 2)
      )
    }
  }
}

function extractImports(code) {
  const imports = []
  const regex = /import\s+.*?\s+from\s+['"]([^'"]+)['"]/g
  let match
  
  while ((match = regex.exec(code)) !== null) {
    imports.push(match[1])
  }
  
  return imports
}
```

## 插件开发最佳实践

### 1. 插件命名

```javascript
// 推荐：vite-plugin-<name>
name: 'vite-plugin-my-feature'

// 作用域包
name: '@scope/vite-plugin-my-feature'
```

### 2. TypeScript 支持

```typescript
// vite-plugin-my-feature.ts
import type { Plugin, ResolvedConfig } from 'vite'

interface MyPluginOptions {
  include?: string[]
  exclude?: string[]
}

export default function myPlugin(options: MyPluginOptions = {}): Plugin {
  let config: ResolvedConfig
  
  return {
    name: 'vite-plugin-my-feature',
    
    configResolved(resolvedConfig) {
      config = resolvedConfig
    },
    
    transform(code, id) {
      // ...
    }
  }
}
```

### 3. 条件执行

```typescript
export default function myPlugin(): Plugin {
  return {
    name: 'vite-plugin-my-feature',
    
    apply: 'build', // 只在生产构建时生效
    
    // 或使用函数
    apply(config, { command }) {
      return command === 'build' && !config.build.ssr
    }
  }
}
```

### 4. 错误处理

```typescript
transform(code, id) {
  try {
    // 转换逻辑
  } catch (error) {
    // 使用 this.error 报告错误
    this.error(`Failed to transform ${id}: ${error.message}`)
    
    // 或警告
    this.warn(`Warning for ${id}`)
  }
}
```

### 5. 调试技巧

```typescript
import debug from 'debug'

const log = debug('vite-plugin-my-feature')

export default function myPlugin() {
  return {
    name: 'vite-plugin-my-feature',

      // ...
    }
  }
}

// 使用：DEBUG=vite-plugin-my-feature vite
```
