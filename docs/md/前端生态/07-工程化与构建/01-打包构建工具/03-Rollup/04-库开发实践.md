## 目录结构

```
my-library/
├── src/
│   ├── index.ts           # 主入口
│   ├── components/        # 组件目录
│   │   ├── index.ts
│   │   ├── Button.ts
│   │   └── Input.ts
│   ├── utils/             # 工具函数
│   │   └── index.ts
│   └── styles/            # 样式文件
│       └── index.css
├── dist/                  # 构建产物
│   ├── index.js           # ESM
│   ├── index.cjs          # CommonJS
│   ├── index.d.ts         # 类型声明
│   └── style.css          # 样式
├── rollup.config.js       # Rollup 配置
├── tsconfig.json          # TypeScript 配置
└── package.json
```

## 入口文件设计

```typescript
// src/index.ts
// 组件导出
export { Button } from './components/Button'
export { Input } from './components/Input'
export type { ButtonProps } from './components/Button'
export type { InputProps } from './components/Input'

// 工具函数导出
export { debounce, throttle } from './utils'

// 默认导出
import { Button, Input } from './components'
export default { Button, Input }

// 安装函数（Vue 插件模式）
import type { App } from 'vue'
import { Button, Input } from './components'

export const install = (app: App) => {
  app.component('MyButton', Button)
  app.component('MyInput', Input)
}

export default { install }
```

## Rollup 配置

```javascript
// rollup.config.js
import resolve from '@rollup/plugin-node-resolve'
import commonjs from '@rollup/plugin-commonjs'
import typescript from '@rollup/plugin-typescript'
import { terser } from 'rollup-plugin-terser'
import dts from 'rollup-plugin-dts'
import postcss from 'rollup-plugin-postcss'
import { defineConfig } from 'rollup'

const config = defineConfig([
  // 主构建
  {
    input: 'src/index.ts',
    output: [
      {
        file: 'dist/index.js',
        format: 'es',
        sourcemap: true,
        exports: 'named'
      },
      {
        file: 'dist/index.cjs',
        format: 'cjs',
        sourcemap: true,
        exports: 'named'
      },
      {
        file: 'dist/index.umd.js',
        format: 'umd',
        name: 'MyLibrary',
        sourcemap: true,
        globals: {
          vue: 'Vue'
        }
      }
    ],
    external: ['vue'],
    plugins: [
      resolve(),
      commonjs(),
      typescript({
        tsconfig: './tsconfig.json'
      }),
      postcss({
        extract: 'style.css',
        minimize: true
      }),
      terser()
    ]
  },
  // 类型声明
  {
    input: 'dist/types/index.d.ts',
    output: [
      { file: 'dist/index.d.ts', format: 'es' },
      { file: 'dist/index.d.cts', format: 'commonjs' }
    ],
    plugins: [dts()]
  }
])

export default config
```

## 发布检查清单

### package.json 检查

- [ ] `name` - 包名是否正确
- [ ] `version` - 版本号是否更新
- [ ] `main` - CommonJS 入口
- [ ] `module` - ESM 入口
- [ ] `types` - 类型声明入口
- [ ] `exports` - 导出配置是否完整
- [ ] `files` - 发布文件列表
- [ ] `sideEffects` - 副作用标记
- [ ] `peerDependencies` - 对等依赖

### 构建产物检查

- [ ] ESM 格式输出正常
- [ ] CommonJS 格式输出正常
- [ ] 类型声明文件生成
- [ ] CSS 文件提取
- [ ] Sourcemap 生成
- [ ] Tree Shaking 生效

### 文档检查

- [ ] README.md 完整
- [ ] 使用示例
- [ ] API 文档
- [ ] CHANGELOG.md
- [ ] LICENSE 文件

## 发布流程

```bash
# 1. 更新版本
npm version patch  # 或 minor / major

# 2. 构建
npm run build

# 3. 测试
npm test

# 4. 发布
npm publish

# 或发布 beta 版本
npm publish --tag beta
```
