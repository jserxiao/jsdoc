# 基础配置

## 命令行使用

```bash
# 基础构建
npx esbuild src/index.js --bundle --outfile=dist/bundle.js

# 完整示例
npx esbuild src/index.ts \
  --bundle \
  --outfile=dist/bundle.js \
  --minify \
  --sourcemap \
  --format=esm \
  --platform=browser \
  --target=es2020
```

### 命令行参数

| 参数 | 说明 |
|------|------|
| `--bundle` | 打包依赖 |
| `--outfile` | 输出文件 |
| `--outdir` | 输出目录 |
| `--format` | 输出格式 (iife/cjs/esm) |
| `--platform` | 目标平台 (browser/node/neutral) |
| `--minify` | 压缩代码 |
| `--sourcemap` | 生成 sourcemap |
| `--target` | 目标环境 |
| `--define` | 定义变量 |
| `--external` | 外部依赖 |
| `--loader` | 文件加载器 |
| `--splitting` | 代码分割 |
| `--watch` | 监听模式 |
| `--serve` | 开发服务器 |

## JavaScript API

### 同步构建

```javascript
const esbuild = require('esbuild')

const result = esbuild.buildSync({
  entryPoints: ['src/index.js'],
  bundle: true,
  outfile: 'dist/bundle.js',
  minify: true,
  sourcemap: true
})

console.log('Build complete')
```

### 异步构建

```javascript
const esbuild = require('esbuild')

async function build() {
  try {
    const result = await esbuild.build({
      entryPoints: ['src/index.js'],
      bundle: true,
      outfile: 'dist/bundle.js'
    })
    console.log('Build complete')
  } catch (error) {
    console.error('Build failed:', error)
    process.exit(1)
  }
}

build()
```

## 完整配置示例

```javascript
const esbuild = require('esbuild')

const buildOptions = {
  // 入口文件
  entryPoints: ['src/index.tsx'],
  
  // 打包依赖
  bundle: true,
  
  // 输出文件
  outfile: 'dist/bundle.js',
  
  // 输出格式
  format: 'esm',
  
  // 目标环境
  target: ['es2020', 'chrome80', 'firefox75', 'safari13'],
  
  // 压缩
  minify: true,
  
  // Sourcemap
  sourcemap: true,
  
  // 平台
  platform: 'browser',
  
  // 外部依赖
  external: ['react', 'react-dom'],
  
  // 定义变量
  define: {
    'process.env.NODE_ENV': '"production"'
  },
  
  // Loader 配置
  loader: {
    '.png': 'dataurl',
    '.woff': 'file',
    '.svg': 'file'
  },
  
  // 代码分割
  splitting: false,
  
  // Tree Shaking
  treeShaking: true
}

esbuild.build(buildOptions)
  .then(() => console.log('Build success!'))
  .catch(() => process.exit(1))
```

## 增量构建

```javascript
const esbuild = require('esbuild')

async function watch() {
  const ctx = await esbuild.context({
    entryPoints: ['src/index.js'],
    bundle: true,
    outfile: 'dist/bundle.js'
  })
  
  // 启用监听
  await ctx.watch()
  
  console.log('Watching...')
}

watch()
```

## 开发服务器

```javascript
const esbuild = require('esbuild')

async function serve() {
  const ctx = await esbuild.context({
    entryPoints: ['src/index.js'],
    bundle: true,
    outfile: 'www/bundle.js'
  })
  
  const { host, port } = await ctx.serve({
    servedir: 'www'
  })
  
  console.log(`Server running at http://${host}:${port}`)
}

serve()
```

## 多入口配置

```javascript
esbuild.build({
  entryPoints: {
    main: 'src/main.js',
    admin: 'src/admin.js',
    mobile: 'src/mobile.js'
  },
  bundle: true,
  outdir: 'dist',
  format: 'esm',
  splitting: true
})
```

## 与 Webpack 集成

```javascript
// webpack.config.js
module.exports = {
  module: {
    rules: [
      {
        test: /\.js$/,
        use: {
          loader: 'esbuild-loader',
          options: {
            loader: 'js',
            target: 'es2015'
          }
        }
      },
      {
        test: /\.tsx?$/,
        use: {
          loader: 'esbuild-loader',
          options: {
            loader: 'tsx',
            target: 'es2015'
          }
        }
      }
    ]
  }
}
```
