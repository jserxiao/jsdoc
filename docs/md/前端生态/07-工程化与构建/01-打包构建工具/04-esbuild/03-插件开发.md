# 插件开发

esbuild 提供了简洁的插件 API，用于扩展功能。

## 插件结构

```javascript
const myPlugin = {
  name: 'my-plugin',
  setup(build) {
    // 插件逻辑
  }
}

// 使用插件
esbuild.build({
  entryPoints: ['src/index.js'],
  bundle: true,
  outfile: 'dist/bundle.js',
  plugins: [myPlugin]
})
```

## 钩子函数

| 钩子 | 说明 |
|------|------|
| `onStart` | 构建开始 |
| `onEnd` | 构建结束 |
| `onResolve` | 解析模块路径 |
| `onLoad` | 加载模块内容 |

## 构建生命周期钩子

```javascript
const lifecyclePlugin = {
  name: 'lifecycle-plugin',
  setup(build) {
    // 构建开始
    build.onStart(() => {
      console.log('Build started')
    })
    
    // 构建结束
    build.onEnd(result => {
      console.log('Build ended')
      console.log('Warnings:', result.warnings)
      console.log('Errors:', result.errors)
    })
  }
}
```

## 解析钩子 (onResolve)

```javascript
const aliasPlugin = {
  name: 'alias-plugin',
  setup(build) {
    build.onResolve({ filter: /^@\/(.*)$/ }, args => {
      // @/utils → src/utils
      return { path: `src/${args.path.slice(2)}` }
    })
  }
}
```

## 加载钩子 (onLoad)

```javascript
const textPlugin = {
  name: 'text-plugin',
  setup(build) {
    build.onLoad({ filter: /\.txt$/ }, async args => {
      const text = await fs.promises.readFile(args.path, 'utf8')
      return {
        contents: `export default ${JSON.stringify(text)}`,
        loader: 'js'
      }
    })
  }
}
```

## 虚拟模块

```javascript
const virtualPlugin = {
  name: 'virtual-plugin',
  setup(build) {
    const virtualModules = {
      'virtual:config': JSON.stringify({ api: 'https://api.example.com' })
    }
    
    // 解析虚拟模块
    build.onResolve({ filter: /^virtual:/ }, args => {
      return { path: args.path, namespace: 'virtual' }
    })
    
    // 加载虚拟模块内容
    build.onLoad({ filter: /.*/, namespace: 'virtual' }, args => {
      const contents = virtualModules[args.path]
      return { contents, loader: 'json' }
    })
  }
}

// 使用
import config from 'virtual:config'
console.log(config.api)
```

## HTTP 插件

```javascript
const httpPlugin = {
  name: 'http-plugin',
  setup(build) {
    // 拦截 https? 导入
    build.onResolve({ filter: /^https?:\/\// }, args => ({
      path: args.path,
      namespace: 'http-url'
    }))
    
    // 加载远程模块
    build.onLoad({ filter: /.*/, namespace: 'http-url' }, async args => {
      const res = await fetch(args.path)
      const contents = await res.text()
      return { contents, loader: 'js' }
    })
  }
}

// 使用
import confetti from 'https://cdn.skypack.dev/canvas-confetti'
confetti()
```

## 环境变量插件

```javascript
const envPlugin = {
  name: 'env-plugin',
  setup(build) {
    build.onResolve({ filter: /^env$/ }, args => ({
      path: args.path,
      namespace: 'env-ns'
    }))
    
    build.onLoad({ filter: /.*/, namespace: 'env-ns' }, () => ({
      contents: JSON.stringify(process.env),
      loader: 'json'
    }))
  }
}

// 使用
import env from 'env'
console.log(env.NODE_ENV)
```

## Sass 插件

```javascript
const sassPlugin = {
  name: 'sass-plugin',
  setup(build) {
    const sass = require('sass')
    
    build.onLoad({ filter: /\.scss$/ }, async args => {
      const result = sass.compile(args.path, {
        loadPaths: ['node_modules']
      })
      
      return {
        contents: result.css,
        loader: 'css'
      }
    })
  }
}
```

## 完整插件示例

```javascript
const fs = require('fs')
const path = require('path')

const imagePlugin = (options = {}) => ({
  name: 'image-plugin',
  setup(build) {
    const { limit = 8192 } = options
    
    build.onLoad({ filter: /\.(png|jpg|gif|svg)$/ }, async args => {
      const buffer = await fs.promises.readFile(args.path)
      const ext = path.extname(args.path)
      
      if (buffer.length < limit) {
        // 转换为 base64
        const base64 = buffer.toString('base64')
        const mimeType = getMimeType(ext)
        return {
          contents: `export default "data:${mimeType};base64,${base64}"`,
          loader: 'js'
        }
      } else {
        // 复制文件
        const hash = createHash(buffer)
        const filename = `${hash}${ext}`
        const outputPath = path.join('dist', 'images', filename)
        
        await fs.promises.mkdir(path.dirname(outputPath), { recursive: true })
        await fs.promises.writeFile(outputPath, buffer)
        
        return {
          contents: `export default "/images/${filename}"`,
          loader: 'js'
        }
      }
    })
  }
})

function getMimeType(ext) {
  const types = {
    '.png': 'image/png',
    '.jpg': 'image/jpeg',
    '.gif': 'image/gif',
    '.svg': 'image/svg+xml'
  }
  return types[ext] || 'application/octet-stream'
}

function createHash(buffer) {
  return require('crypto').createHash('md5').update(buffer).digest('hex').slice(0, 8)
}

// 使用
esbuild.build({
  entryPoints: ['src/index.js'],
  bundle: true,
  outfile: 'dist/bundle.js',
  plugins: [imagePlugin({ limit: 10240 })]
})
```
