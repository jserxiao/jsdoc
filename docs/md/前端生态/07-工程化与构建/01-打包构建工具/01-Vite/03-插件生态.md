# 插件生态

Vite 拥有丰富的插件生态，可以扩展其功能。

## 官方插件

```javascript
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'           // Vue 3
import vueJsx from '@vitejs/plugin-vue-jsx'    // Vue JSX
import react from '@vitejs/plugin-react'       // React
import reactSwc from '@vitejs/plugin-react-swc' // React + SWC
import legacy from '@vitejs/plugin-legacy'     // 浏览器兼容

export default defineConfig({
  plugins: [
    vue(),
    vueJsx(),
    react(),
    reactSwc(),
    legacy({
      targets: ['ie >= 11']
    })
  ]
})
```


### 开发体验

```javascript
import { defineConfig } from 'vite'
import Components from 'unplugin-vue-components/vite'     // 组件自动导入
import AutoImport from 'unplugin-auto-import/vite'       // API 自动导入
import Pages from 'vite-plugin-pages'                    // 文件系统路由
import Layouts from 'vite-plugin-vue-layouts'            // 布局系统

export default defineConfig({
  plugins: [
    // 自动导入 API
    AutoImport({
      imports: ['vue', 'vue-router', 'pinia'],
      dts: 'src/auto-imports.d.ts'
    }),
    
    // 自动注册组件
    Components({
      dirs: ['src/components'],
      extensions: ['vue'],
      dts: 'src/components.d.ts'
    }),
    
    // 文件系统路由
    Pages({
      dirs: 'src/pages',
      extensions: ['vue']
    }),
    
    // 布局系统
    Layouts({
      layoutsDirs: 'src/layouts'
    })
  ]
})
```

## 自定义插件

### 插件结构

```javascript
// vite-plugin-my-plugin.js
export default function myPlugin(options = {}) {
  return {
    name: 'vite-plugin-my-plugin',
    
    // 插件类型
    enforce: 'pre', // 'pre' | 'post'
    
    // 应用阶段
    apply: 'build', // 'build' | 'serve'
    
    // 配置解析后
    configResolved(config) {
      console.log('配置已解析:', config)
    },
    
    // 转换模块
    transform(code, id) {
      if (id.endsWith('.custom')) {
        return {
          code: `export default ${JSON.stringify(code)}`,
          map: null
        }
      }
    },
    
    // 构建开始
    buildStart() {
      console.log('构建开始')
    },
    
    // 构建结束
    buildEnd() {
      console.log('构建结束')
    }
  }
}
```

### 虚拟模块插件

```javascript
export default function virtualModulePlugin() {
  const virtualModuleId = 'virtual:my-module'
  const resolvedVirtualModuleId = '\0' + virtualModuleId
  
  return {
    name: 'virtual-module-plugin',
    
    resolveId(id) {
      if (id === virtualModuleId) {
        return resolvedVirtualModuleId
      }
    },
    
    load(id) {
      if (id === resolvedVirtualModuleId) {
        return `export const msg = "Hello from virtual module"`
      }
    }
  }
}

// 使用
import { msg } from 'virtual:my-module'
console.log(msg) // "Hello from virtual module"
```

### 自定义转换插件

```javascript
export default function transformPlugin() {
  return {
    name: 'transform-plugin',
    
    transform(code, id) {
      // 只处理 .vue 文件
      if (!id.endsWith('.vue')) return null
      
      // 转换代码
      const transformed = code.replace(
        /console\.log\(/g,
        '// console.log('
      )
      
      return {
        code: transformed,
        map: null
      }
    }
  }
}
```

### HTML 转换插件

```javascript
export default function htmlTransformPlugin() {
  return {
    name: 'html-transform-plugin',
    
    transformIndexHtml(html) {
      return html.replace(
        /<title>(.*?)<\/title>/,
        '<title>My App - $1</title>'
      ).replace(
        '</head>',
        `<script>
          console.log('Injected script')
        </script>
        </head>`
      )
    }
  }
}
```

## 插件开发 API

### 钩子函数

| 钩子 | 类型 | 说明 |
|------|------|------|
| `config` | 同步/异步 | 修改配置 |
| `configResolved` | 同步/异步 | 配置解析完成 |
| `options` | 同步/异步 | 修改 Rollup 选项 |
| `buildStart` | 同步/异步 | 构建开始 |
| `buildEnd` | 同步/异步 | 构建结束 |
| `transform` | 同步/异步 | 转换模块 |
| `load` | 同步/异步 | 加载模块 |
| `resolveId` | 同步/异步 | 解析模块 ID |

### 插件顺序

```javascript
export default defineConfig({
  plugins: [
    {
      name: 'pre-plugin',
      enforce: 'pre'  // 在核心插件之前
    },
    {
      name: 'normal-plugin'
      // 默认顺序
    },
    {
      name: 'post-plugin',
      enforce: 'post' // 在构建插件之后
    }
  ]
})
```
