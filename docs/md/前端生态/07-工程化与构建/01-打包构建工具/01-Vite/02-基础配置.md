# 基础配置

## 配置文件

```javascript
// vite.config.js
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import { resolve } from 'path'

export default defineConfig({
  // 插件配置
  plugins: [vue()],
  
  // 路径别名
  resolve: {
    alias: {
      '@': resolve(__dirname, 'src')
    }
  },
  
  // 开发服务器
  server: {
    port: 3000,
    host: true,
    open: true
  },
  
  // 构建配置
  build: {
    outDir: 'dist'
  }
})
```

## 开发服务器配置

```javascript
export default defineConfig({
  server: {
    // 端口号
    port: 3000,

    host: true,

    open: true,

    https: {
```

## 构建配置

```javascript
export default defineConfig({
  build: {
    // 输出目录
    outDir: 'dist',
    
    // 静态资源目录
    assetsDir: 'assets',
    
    // 静态资源文件名
    assetsInlineLimit: 4096, // 4KB
    
    // CSS 代码分割
    cssCodeSplit: true,
    
    // Sourcemap
    sourcemap: true,
    
    // 压缩
    minify: 'esbuild', // 'terser' | 'esbuild'
    
    // 移除 console
    terserOptions: {
      compress: {
        drop_console: true,
        drop_debugger: true
      }
    },
    
    // Rollup 配置
    rollupOptions: {
      input: {
        main: resolve(__dirname, 'index.html'),
        admin: resolve(__dirname, 'admin.html')
      },
      output: {
        // 文件名格式
        entryFileNames: 'js/[name]-[hash].js',
        chunkFileNames: 'js/[name]-[hash].js',
        assetFileNames: '[ext]/[name]-[hash].[ext]',
        
        // 手动分割
        manualChunks: {
          vendor: ['vue', 'vue-router', 'pinia'],
          ui: ['element-plus'],
          utils: ['lodash-es', 'dayjs']
        },
        
        // 或使用函数
        manualChunks(id) {
          if (id.includes('node_modules')) {
            return 'vendor'
          }
        }
      },
      
      // 外部依赖
      external: ['some-external-lib'],
      
      // 全局变量
      globals: {
        'some-external-lib': 'SomeLib'
      }
    },
    
    // 块大小警告限制
    chunkSizeWarningLimit: 1000,
    
    // 构建后清空目录
    emptyOutDir: true
  }
})
```

## CSS 配置

```javascript
export default defineConfig({
  css: {
    // CSS Modules 配置
    modules: {
      generateScopedName: '[name]__[local]__[hash:base64:5]'
    },
    
    // PostCSS 配置
    postcss: {
      plugins: [
        require('autoprefixer'),
        require('postcss-nested')
      ]
    },
    
    // CSS 预处理器配置
    preprocessorOptions: {
      scss: {
        additionalData: `@import "@/styles/variables.scss";`
      },
      less: {
        modifyVars: {
          'primary-color': '#1890ff'
        }
      }
    },
    
    // 开发时注入的 CSS
    devSourcemap: true
  }
})
```


## 环境特定配置

```javascript
// vite.config.js
export default defineConfig(({ mode }) => {
  if (mode === 'development') {
    return {
      // 开发配置
      server: {
        proxy: { /* ... */ }
      }
    }
  } else if (mode === 'production') {
    return {
      // 生产配置
      build: {
        minify: 'terser'
      }
    }
  }
})
```
