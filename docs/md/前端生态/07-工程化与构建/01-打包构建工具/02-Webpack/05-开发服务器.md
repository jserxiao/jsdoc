# 开发服务器

Webpack Dev Server 提供了一个开发环境服务器，支持热更新、代理等功能。

## 基础配置

```javascript
// webpack.config.js
module.exports = {
  devServer: {
    // 静态文件目录
    static: {
      directory: path.join(__dirname, 'public'),
      publicPath: '/public'
    },
    
    // 端口号
    port: 3000,
    
    // 自动打开浏览器
    open: true,
    
    // 启用压缩
    compress: true,
    
    // 启用 HMR
    hot: true,
    
    // 显示进度
    progress: true
  }
}
```

## 完整配置

```javascript
module.exports = {
  devServer: {
    // 基础配置
    host: '0.0.0.0',      // 允许外部访问
    port: 3000,            // 端口号
    hot: true,             // 热更新
    open: true,            // 自动打开浏览器
    compress: true,        // gzip 压缩
    
    // 静态文件服务
    static: {
      directory: path.join(__dirname, 'public'),
      staticOptions: {},
      publicPath: '/public',
      watch: {
        ignored: /node_modules/,
        aggregateTimeout: 300,
        poll: 1000
      }
    },
    
    // 代理配置
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,
        pathRewrite: { '^/api': '' },
        secure: false,
        headers: {
          'X-Custom-Header': 'value'
        }
      }
    },
    
    // CORS
    cors: true,
    
    // history 路由
    historyApiFallback: {
      index: '/index.html',
      rewrites: [
        { from: /^\/admin/, to: '/admin.html' }
      ]
    },
    
    // HTTPS
    https: {
      key: fs.readFileSync('key.pem'),
      cert: fs.readFileSync('cert.pem')
    },
    
    // 控制台输出
    client: {
      logging: 'info',      // 'silent' | 'trace' | 'debug' | 'info' | 'warn' | 'error'
      overlay: {
        errors: true,
        warnings: false
      },
      progress: true
    },
    
    // 开发工具配置
    devMiddleware: {
      index: true,
      writeToDisk: true     // 写入磁盘
    }
  }
}
```

## 代理配置

### 基础代理

```javascript
module.exports = {
  devServer: {
    proxy: {
      '/api': 'http://localhost:8080'
    }
  }
}

// 请求 /api/users → http://localhost:8080/api/users
```

### 高级代理

```javascript
module.exports = {
  devServer: {
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,    // 修改请求头中的 host
        pathRewrite: {
          '^/api': ''          // 移除 /api 前缀
        },
        secure: false,         // 允许自签名证书
        
        // 请求拦截
        onProxyReq(proxyReq, req, res) {
          proxyReq.setHeader('X-Custom-Header', 'value')
        },
        
        // 响应拦截
        onProxyRes(proxyRes, req, res) {
          console.log('Response from:', req.url)
        },
        
        // 错误处理
        onError(err, req, res) {
          console.error('Proxy error:', err)
        }
      }
    }
  }
}
```

### 多个代理

```javascript
module.exports = {
  devServer: {
    proxy: [
      {
        context: ['/api', '/auth'],
        target: 'http://localhost:8080',
        changeOrigin: true
      },
      {
        context: '/upload',
        target: 'http://localhost:9000'
      }
    ]
  }
}
```

## HMR（热模块替换）

### 启用 HMR

```javascript
module.exports = {
  devServer: {
    hot: true,
    // 或更细粒度控制
    hot: 'only'  // 编译失败时不刷新页面
  }
}
```

### 在代码中使用

```javascript
// React
import module from './module'

if (module.hot) {
  module.hot.accept('./module', () => {
    // 模块更新后的处理
  })
}

// Vue（vue-loader 自动处理）
// 无需额外配置
```

### React HMR 配置

```javascript
// webpack.config.js
const ReactRefreshWebpackPlugin = require('@pmmmwh/react-refresh-webpack-plugin')

module.exports = {
  devServer: {
    hot: true
  },
  plugins: [
    new ReactRefreshWebpackPlugin()
  ],
  module: {
    rules: [
      {
        test: /\.js$/,
        use: {
          loader: 'babel-loader',
          options: {
            plugins: ['react-refresh/babel']
          }
        }
      }
    ]
  }
}
```

## History 路由

```javascript
module.exports = {
  devServer: {
    historyApiFallback: true  // 简单配置
  }
}

// 或高级配置
module.exports = {
  devServer: {
    historyApiFallback: {
      index: '/index.html',
      rewrites: [
        { from: /^\/admin/, to: '/admin.html' },
        { from: /\/page\/\d+/, to: '/index.html' }
      ],
      disableDotRule: true
    }
  }
}
```

## HTTPS 配置

### 自动生成证书

```bash
# 安装
npm install -D @pmmmwh/react-refresh-webpack-plugin

# 启动
npx webpack serve --https
```

### 自定义证书

```javascript
module.exports = {
  devServer: {
    https: {
      key: fs.readFileSync('server.key'),
      cert: fs.readFileSync('server.crt'),
      ca: fs.readFileSync('ca.pem')
    }
  }
}
```

## 启动脚本

```json
// package.json
{
  "scripts": {
    "dev": "webpack serve --mode development",
    "dev:open": "webpack serve --open",
    "dev:port": "webpack serve --port 4000",
    "dev:https": "webpack serve --https"
  }
}
```

## 常见问题

### 端口被占用

```javascript
module.exports = {
  devServer: {
    port: 'auto',  // 自动选择可用端口
    // 或
    port: 3000,
    strictPort: false  // 端口被占用时尝试下一个
  }
}
```

### 跨域问题

```javascript
module.exports = {
  devServer: {
    // 方式一：设置 CORS
    headers: {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE'
    },
    
    // 方式二：代理
    proxy: {
      '/api': {
        target: 'http://api.example.com',
        changeOrigin: true
      }
    }
  }
}
```

### 静态资源不更新

```javascript
module.exports = {
  devServer: {
    static: {
      watch: {
        ignored: /node_modules/,
        aggregateTimeout: 300,
        poll: 1000
      }
    }
  }
}
```
