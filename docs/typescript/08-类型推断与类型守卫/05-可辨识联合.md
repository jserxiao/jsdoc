# 可辨识联合

## 概述

可辨识联合(Discriminated Union)使用公共属性来区分类型。

## 一、基本概念

```typescript
interface Circle {
  kind: 'circle'
  radius: number
}

interface Square {
  kind: 'square'
  sideLength: number
}

type Shape = Circle | Square

function area(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2
    case 'square':
      return shape.sideLength ** 2
  }
}
```

## 二、穷尽检查

```typescript
function getArea(shape: Shape): number {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2
    case 'square':
      return shape.sideLength ** 2
    default:
      const exhaustive: never = shape  // 如果有遗漏会报错
      throw new Error(`Unknown shape: ${exhaustive}`)
  }
}
```

## 三、实战示例

```typescript
// 状态管理
interface Pending {
  status: 'pending'
}

interface Loading {
  status: 'loading'
  progress: number
}

interface Success {
  status: 'success'
  data: string
}

interface Failure {
  status: 'failure'
  error: string
}

type Request = Pending | Loading | Success | Failure

function handleRequest(req: Request) {
  switch (req.status) {
    case 'pending':
      console.log('Pending...')
      break
    case 'loading':
      console.log(`Loading: ${req.progress}%`)
      break
    case 'success':
      console.log('Success:', req.data)
      break
    case 'failure':
      console.error('Error:', req.error)
      break
  }
}
```

## 四、Redux Action 示例

```typescript
interface AddTodoAction {
  type: 'ADD_TODO'
  payload: { text: string }
}

interface RemoveTodoAction {
  type: 'REMOVE_TODO'
  payload: { id: number }
}

type TodoAction = AddTodoAction | RemoveTodoAction

function todoReducer(state: Todo[], action: TodoAction): Todo[] {
  switch (action.type) {
    case 'ADD_TODO':
      return [...state, { id: Date.now(), text: action.payload.text }]
    case 'REMOVE_TODO':
      return state.filter(todo => todo.id !== action.payload.id)
  }
}
```

## 五、最佳实践

```typescript
// ✅ 好: 使用字面量类型作为辨识属性
interface User {
  type: 'user'
  name: string
}

interface Admin {
  type: 'admin'
  name: string
  permissions: string[]
}

// ❌ 不好: 使用不同属性名
interface User2 {
  isUser: true
  name: string
}

interface Admin2 {
  isAdmin: true
  name: string
}
```

## 六、总结

- 使用公共字面量属性区分类型
- switch 语句实现类型缩小
- never 类型实现穷尽检查
- 适用于状态管理和 Action 类型
