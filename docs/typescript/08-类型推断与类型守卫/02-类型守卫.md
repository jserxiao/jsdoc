# 类型守卫

## 概述

类型守卫是运行时检查,用于缩小类型范围。

## 一、typeof 守卫

```typescript
function process(value: string | number) {
  if (typeof value === 'string') {
    return value.toUpperCase()  // string
  } else {
    return value * 2           // number
  }
}
```

## 二、instanceof 守卫

```typescript
function process(value: Date | string) {
  if (value instanceof Date) {
    return value.toISOString()  // Date
  } else {
    return value.toUpperCase()  // string
  }
}
```

## 三、in 操作符守卫

```typescript
interface User {
  name: string
  email: string
}

interface Admin {
  name: string
  permissions: string[]
}

function process(user: User | Admin) {
  if ('email' in user) {
    console.log(user.email)  // User
  } else {
    console.log(user.permissions)  // Admin
  }
}
```

## 四、自定义类型守卫

```typescript
interface User {
  name: string
  age: number
}

// 类型谓词
function isUser(obj: any): obj is User {
  return obj &&
    typeof obj === 'object' &&
    typeof obj.name === 'string' &&
    typeof obj.age === 'number'
}

function process(value: unknown) {
  if (isUser(value)) {
    console.log(value.name)  // User
  }
}
```

## 五、字面量守卫

```typescript
type Direction = 'up' | 'down' | 'left' | 'right'

function move(dir: Direction) {
  switch (dir) {
    case 'up':
      // dir: 'up'
      break
    case 'down':
      // dir: 'down'
      break
  }
}
```

## 六、实战示例

```typescript
interface Success {
  success: true
  data: string
}

interface Failure {
  success: false
  error: string
}

type Result = Success | Failure

function handleResult(result: Result) {
  if (result.success) {
    console.log('Success:', result.data)
  } else {
    console.error('Error:', result.error)
  }
}
```

## 七、总结

- `typeof`: 检查原始类型
- `instanceof`: 检查类实例
- `in`: 检查属性存在
- 自定义守卫: `obj is Type`
