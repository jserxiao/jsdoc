# 特殊类型

## 概述

TypeScript 提供了一些特殊类型来处理特定场景,包括 any、unknown、never、void 等。

## 一、any 类型

### 1.1 基本概念

`any` 类型表示任意类型,会跳过类型检查:

```typescript
let anything: any

anything = 'hello'
anything = 42
anything = { name: 'Alice' }
anything = [1, 2, 3]

// 任何操作都允许
anything.toUpperCase()
anything.toFixed()
anything.nonExistentMethod()
```

### 1.2 使用场景

```typescript
// 1. 渐进式迁移
let data: any = fetchSomeData()  // 暂时不检查类型

// 2. 第三方库没有类型定义
declare const jQuery: any
jQuery('.my-element').hide()

// 3. 类型断言绕过检查
const result = someValue as any as OtherType
```

### 1.3 隐式 any

```typescript
// noImplicitAny: false (默认)
function process(data) {  // data: any
  return data.value
}

// noImplicitAny: true (推荐)
function process(data: any) {  // 显式标注
  return data.value
}
```

### 1.4 危险性

```typescript
let a: any = 'hello'
let b: number = a  // ✅ 编译通过,运行时错误

a.nonExistentMethod()  // ✅ 编译通过,运行时错误

// any 会污染其他类型
let c: string = a
let d: number = a
let e: boolean = a
```

## 二、unknown 类型

### 2.1 基本概念

`unknown` 是类型安全的 any:

```typescript
let value: unknown

value = 'hello'
value = 42
value = { name: 'Alice' }

// ❌ 不能直接使用
// value.toUpperCase()
// value.toFixed()
// value.name

// ✅ 需要先进行类型检查
if (typeof value === 'string') {
  console.log(value.toUpperCase())
}

if (typeof value === 'number') {
  console.log(value.toFixed(2))
}
```

### 2.2 类型收窄

```typescript
function process(value: unknown) {
  // typeof 检查
  if (typeof value === 'string') {
    return value.toUpperCase()  // string
  }

  // instanceof 检查
  if (value instanceof Date) {
    return value.toISOString()  // string
  }

  // 类型谓词
  if (isUser(value)) {
    return value.name  // User
  }

  // 类型断言
  const str = value as string
  return str.toUpperCase()
}

interface User {
  name: string
}

function isUser(value: unknown): value is User {
  return typeof value === 'object' && value !== null && 'name' in value
}
```

### 2.3 unknown vs any

```typescript
let anyValue: any
let unknownValue: unknown

// any - 不安全
let a1: string = anyValue  // ✅
let a2: number = anyValue  // ✅
anyValue.method()          // ✅

// unknown - 安全
let u1: string = unknownValue   // ❌ 编译错误
let u2: number = unknownValue   // ❌ 编译错误
unknownValue.method()           // ❌ 编译错误

// unknown 必须先收窄
if (typeof unknownValue === 'string') {
  let u3: string = unknownValue  // ✅
}
```

### 2.4 实际应用

```typescript
// API 响应
async function fetchData(): Promise<unknown> {
  const response = await fetch('/api/data')
  return response.json()
}

// 使用
const data = await fetchData()
if (isUserArray(data)) {
  data.forEach(user => console.log(user.name))
}

function isUserArray(value: unknown): value is User[] {
  return Array.isArray(value) && value.every(isUser)
}
```

## 三、never 类型

### 3.1 基本概念

`never` 表示永不存在的值的类型:

```typescript
// 永不返回的函数
function error(message: string): never {
  throw new Error(message)
}

function infiniteLoop(): never {
  while (true) {}
}

// 无法到达的代码
function fail() {
  return error('Something failed')
}
```

### 3.2 类型收窄

```typescript
function process(value: string | number) {
  if (typeof value === 'string') {
    // value: string
  } else if (typeof value === 'number') {
    // value: number
  } else {
    // value: never
    const exhaustive: never = value
    throw new Error(`Unknown type: ${value}`)
  }
}
```

### 3.3 可辨识联合

```typescript
interface Square {
  kind: 'square'
  size: number
}

interface Circle {
  kind: 'circle'
  radius: number
}

type Shape = Square | Circle

function area(shape: Shape): number {
  switch (shape.kind) {
    case 'square':
      return shape.size ** 2
    case 'circle':
      return Math.PI * shape.radius ** 2
    default:
      // 如果新增类型但忘记处理,这里会报错
      const exhaustive: never = shape
      throw new Error(`Unknown shape: ${shape}`)
  }
}
```

### 3.4 never 的用途

```typescript
// 1. 不可能的交叉
type StringAndNumber = string & number  // never

// 2. 排除类型
type NonNullable<T> = T extends null | undefined ? never : T

type Result = NonNullable<string | null | undefined>  // string

// 3. 空数组
const empty: never[] = []

// 4. 不可能的函数返回值
function impossible(): never {
  throw new Error('Never returns')
}
```

## 四、void 类型

### 4.1 基本概念

`void` 表示没有返回值:

```typescript
function log(message: string): void {
  console.log(message)
  // 没有显式返回值
}

const result = log('hello')  // result: void
```

### 4.2 void vs undefined

```typescript
function returnUndefined(): undefined {
  console.log('hello')
  return undefined  // 必须返回 undefined
}

function returnVoid(): void {
  console.log('hello')
  // 可以返回任何值,也可以不返回
}

// void 接受任何返回值
function fn(): void {
  return 42  // ✅
}
```

### 4.3 函数类型

```typescript
type Callback = () => void

function execute(callback: Callback) {
  callback()
}

execute(() => console.log('done'))
execute(() => 42)  // ✅ 允许返回值
```

### 4.4 void vs never

```typescript
// void - 返回但不返回值
function log(): void {
  console.log('log')
}

// never - 永不返回
function error(): never {
  throw new Error('error')
}

// void 可以到达函数末尾
// never 永远无法到达函数末尾
```

## 五、null 和 undefined

### 5.1 基本用法

```typescript
let u: undefined = undefined
let n: null = null

// 严格模式下
let str: string = undefined  // ❌ 编译错误
let num: number = null      // ❌ 编译错误

// 联合类型
let strOrNull: string | null = null
let numOrUndefined: number | undefined = undefined
```

### 5.2 可选参数

```typescript
function greet(name?: string) {
  // name: string | undefined
  if (name) {
    console.log(`Hello, ${name}`)
  } else {
    console.log('Hello, stranger')
  }
}

greet('Alice')
greet()  // name is undefined
```

### 5.3 非空断言

```typescript
let value: string | null = 'hello'

// 非空断言
const length1 = value!.length  // string

// 可选链
const length2 = value?.length  // number | undefined

// 空值合并
const str = value ?? 'default'
```

## 六、类型对比总结

### 6.1 对比表

| 类型 | 含义 | 安全性 | 用途 |
|------|------|--------|------|
| any | 任意类型 | ❌ 不安全 | 兼容旧代码,绕过检查 |
| unknown | 未知类型 | ✅ 安全 | 类型安全的 any |
| never | 永不存在 | ✅ 安全 | 不可能的类型, exhaustive check |
| void | 无返回值 | ✅ 安全 | 函数返回类型 |
| null | 空值 | ⚠️ 需检查 | 表示空 |
| undefined | 未定义 | ⚠️ 需检查 | 表示未定义 |

### 6.2 类型层级

```
any (最宽泛)
  ↓
unknown (安全但宽泛)
  ↓
具体类型 (string, number, object 等)
  ↓
字面量类型 ('hello', 42 等)
  ↓
never (最窄,不可能存在)
```

## 七、最佳实践

### 1. 避免使用 any

```typescript
// ❌ 不好
function process(data: any) {
  return data.value
}

// ✅ 好
function process(data: unknown) {
  if (typeof data === 'object' && data !== null && 'value' in data) {
    return data.value
  }
  return undefined
}
```

### 2. 使用 unknown 替代 any

```typescript
// ❌ 不好
const json: any = JSON.parse(data)

// ✅ 好
const json: unknown = JSON.parse(data)
if (isExpectedType(json)) {
  // 使用 json
}
```

### 3. 使用 never 进行穷尽检查

```typescript
function assertNever(value: never): never {
  throw new Error(`Unexpected value: ${value}`)
}

function process(value: 'a' | 'b') {
  switch (value) {
    case 'a':
      return 'A'
    case 'b':
      return 'B'
    default:
      return assertNever(value)  // 如果新增值会报错
  }
}
```

### 4. 明确处理 null 和 undefined

```typescript
// ✅ 好: 明确处理空值
function greet(name: string | null | undefined) {
  return `Hello, ${name ?? 'stranger'}`
}

// ✅ 好: 使用可选参数
function greet2(name?: string) {
  return `Hello, ${name ?? 'stranger'}`
}
```

## 八、总结

### 核心要点
- **any**: 放弃类型检查,尽量少用
- **unknown**: 类型安全的 any,必须先收窄
- **never**: 不可能的类型,用于 exhaustive check
- **void**: 无返回值的函数
- **null/undefined**: 空值,需要严格检查

### 类型安全原则
1. 优先使用 unknown 而非 any
2. 使用 never 确保类型穷尽
3. 严格处理 null 和 undefined
4. 明确函数的返回类型
